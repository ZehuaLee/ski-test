00096 Event *
00097 Event::insertBefore(Event *event, Event *curr)
00098 {
00099     // Either way, event will be the top element in the 'in bin' list
00100     // which is the pointer we need in order to look into the list, so
00101     // we need to insert that into the bin list.
00102     if (!curr || *event < *curr) {
00103         // Insert the event before the current list since it is in the future.
00104         event->nextBin = curr;
00105         event->nextInBin = NULL;
00106     } else {
00107         // Since we're on the correct list, we need to point to the next list
00108         event->nextBin = curr->nextBin;  // curr->nextBin can now become stale
00109 
00110         // Insert event at the top of the stack
00111         event->nextInBin = curr;
00112     }
00113 
00114     return event;
00115 }

00117 void
00118 EventQueue::insert(Event *event)
00119 {
00120     // Deal with the head case
00121     if (!head || *event <= *head) {
00122         head = Event::insertBefore(event, head);
00123         return;
00124     }
00125 
00126     // Figure out either which 'in bin' list we are on, or where a new list
00127     // needs to be inserted
00128     Event *prev = head;
00129     Event *curr = head->nextBin;
00130     while (curr && *curr < *event) {
00131         prev = curr;
00132         curr = curr->nextBin;
00133     }
00134 
00135     // Note: this operation may render all nextBin pointers on the
00136     // prev 'in bin' list stale (except for the top one)
00137     prev->nextBin = Event::insertBefore(event, curr);
00138 }

00140 Event *
00141 Event::removeItem(Event *event, Event *top)
00142 {
00143     Event *curr = top;
00144     Event *next = top->nextInBin;
00145 
00146     // if we removed the top item, we need to handle things specially
00147     // and just remove the top item, fixing up the next bin pointer of
00148     // the new top item
00149     if (event == top) {
00150         if (!next)
00151             return top->nextBin;
00152         next->nextBin = top->nextBin;
00153         return next;
00154     }
00155 
00156     // Since we already checked the current element, we're going to
00157     // keep checking event against the next element.
00158     while (event != next) {
00159         if (!next)
00160             panic("event not found!");
00161 
00162         curr = next;
00163         next = next->nextInBin;
00164     }
00165 
00166     // remove next from the 'in bin' list since it's what we're looking for
00167     curr->nextInBin = next->nextInBin;
00168     return top;
00169 }

00171 void
00172 EventQueue::remove(Event *event)
00173 {
00174     if (head == NULL)
00175         panic("event not found!");
00176 
00177     assert(event->queue == this);
00178 
00179     // deal with an event on the head's 'in bin' list (event has the same
00180     // time as the head)
00181     if (*head == *event) {
00182         head = Event::removeItem(event, head);
00183         return;
00184     }
00185 
00186     // Find the 'in bin' list that this event belongs on
00187     Event *prev = head;
00188     Event *curr = head->nextBin;
00189     while (curr && *curr < *event) {
00190         prev = curr;
00191         curr = curr->nextBin;
00192     }
00193 
00194     if (!curr || *curr != *event)
00195         panic("event not found!");
00196 
00197     // curr points to the top item of the the correct 'in bin' list, when
00198     // we remove an item, it returns the new top item (which may be
00199     // unchanged)
00200     prev->nextBin = Event::removeItem(event, curr);

00203 Event *
00204 EventQueue::serviceOne()
00205 {
00206     std::lock_guard<EventQueue> lock(*this);
00207     Event *event = head;
00208     Event *next = head->nextInBin;
00209     event->flags.clear(Event::Scheduled);
00210 
00211     if (next) {
00212         // update the next bin pointer since it could be stale
00213         next->nextBin = head->nextBin;
00214 
00215         // pop the stack
00216         head = next;
00217     } else {
00218         // this was the only element on the 'in bin' list, so get rid of
00219         // the 'in bin' list and point to the next bin list
00220         head = head->nextBin;
00221     }
00222 
00223     // handle action
00224     if (!event->squashed()) {
00225         // forward current cycle to the time when this event occurs.
00226         setCurTick(event->when());
00227 
00228         event->process();
00229         if (event->isExitEvent()) {
00230             assert(!event->flags.isSet(Event::AutoDelete) ||
00231                    !event->flags.isSet(Event::IsMainQueue)); // would be silly
00232             return event;
00233         }
00234     } else {
00235         event->flags.clear(Event::Squashed);
00236     }
00237 
00238     if (event->flags.isSet(Event::AutoDelete) && !event->scheduled())
00239         delete event;
00240

00258 void
00259 Event::unserialize(Checkpoint *cp, const string &section, EventQueue *eventq)
00260 {
00261     if (scheduled())
00262         eventq->deschedule(this);
00263 
00264     UNSERIALIZE_SCALAR(_when);
00265     UNSERIALIZE_SCALAR(_priority);
00266 
00267     short _flags;
00268     UNSERIALIZE_SCALAR(_flags);
00269 
00270     // Old checkpoints had no concept of the Initialized flag
00271     // so restoring from old checkpoints always fail.
00272     // Events are initialized on construction but original code 
00273     // "flags = _flags" would just overwrite the initialization. 
00274     // So, read in the checkpoint flags, but then set the Initialized 
00275     // flag on top of it in order to avoid failures.

00279 
00280     // need to see if original event was in a scheduled, unsquashed
00281     // state, but don't want to restore those flags in the current
00282     // object itself (since they aren't immediately true)
00283     bool wasScheduled = flags.isSet(Scheduled) && !flags.isSet(Squashed);
00284     flags.clear(Squashed | Scheduled);
00285 
00286     if (wasScheduled) {
00287         DPRINTF(Config, "rescheduling at %d\n", _when);
00288         eventq->schedule(this, _when);

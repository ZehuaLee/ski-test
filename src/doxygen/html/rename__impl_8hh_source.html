00323 template <class Impl>
00324 void
00325 DefaultRename<Impl>::squash(const InstSeqNum &squash_seq_num, ThreadID tid)
00326 {
00327     DPRINTF(Rename, "[tid:%u]: Squashing instructions.\n",tid);
00328 
00329     // Clear the stall signal if rename was blocked or unblocking before.
00330     // If it still needs to block, the blocking should happen the next
00331     // cycle and there should be space to hold everything due to the squash.
00332     if (renameStatus[tid] == Blocked ||
00333         renameStatus[tid] == Unblocking) {
00334         toDecode->renameUnblock[tid] = 1;
00335 
00336         resumeSerialize = false;
00337         serializeInst[tid] = NULL;
00338     } else if (renameStatus[tid] == SerializeStall) {
00339         if (serializeInst[tid]->seqNum <= squash_seq_num) {
00340             DPRINTF(Rename, "Rename will resume serializing after squash\n");
00341             resumeSerialize = true;
00342             assert(serializeInst[tid]);
00343         } else {
00344             resumeSerialize = false;
00345             toDecode->renameUnblock[tid] = 1;
00346 
00347             serializeInst[tid] = NULL;
00348         }
00349     }
00350 
00351     // Set the status to Squashing.
00352     renameStatus[tid] = Squashing;
00353 
00354     // Squash any instructions from decode.
00355     for (int i=0; i<fromDecode->size; i++) {
00356         if (fromDecode->insts[i]->threadNumber == tid &&
00357             fromDecode->insts[i]->seqNum > squash_seq_num) {
00358             fromDecode->insts[i]->setSquashed();
00359             wroteToTimeBuffer = true;
00360         }
00361 
00362     }
00363 
00364     // Clear the instruction list and skid buffer in case they have any
00365     // insts in them.
00366     insts[tid].clear();
00367 
00368     // Clear the skid buffer in case it has any data in it.
00369     skidBuffer[tid].clear();
00370 
00371     doSquash(squash_seq_num, tid);
00372 }

00438 template<class Impl>
00439 void
00440 DefaultRename<Impl>::rename(bool &status_change, ThreadID tid)
00441 {
00442     // If status is Running or idle,
00443     //     call renameInsts()
00444     // If status is Unblocking,
00445     //     buffer any instructions coming from decode
00446     //     continue trying to empty skid buffer
00447     //     check if stall conditions have passed
00448 
00449     if (renameStatus[tid] == Blocked) {
00450         ++renameBlockCycles;
00451     } else if (renameStatus[tid] == Squashing) {
00452         ++renameSquashCycles;
00453     } else if (renameStatus[tid] == SerializeStall) {
00454         ++renameSerializeStallCycles;
00455         // If we are currently in SerializeStall and resumeSerialize
00456         // was set, then that means that we are resuming serializing
00457         // this cycle.  Tell the previous stages to block.
00458         if (resumeSerialize) {
00459             resumeSerialize = false;
00460             block(tid);
00461             toDecode->renameUnblock[tid] = false;
00462         }
00463     } else if (renameStatus[tid] == Unblocking) {
00464         if (resumeUnblocking) {
00465             block(tid);
00466             resumeUnblocking = false;
00467             toDecode->renameUnblock[tid] = false;
00468         }
00469     }
00470 
00471     if (renameStatus[tid] == Running ||
00472         renameStatus[tid] == Idle) {
00473         DPRINTF(Rename, "[tid:%u]: Not blocked, so attempting to run "
00474                 "stage.\n", tid);
00475 
00476         renameInsts(tid);
00477     } else if (renameStatus[tid] == Unblocking) {
00478         renameInsts(tid);
00479 
00480         if (validInsts()) {
00481             // Add the current inputs to the skid buffer so they can be
00482             // reprocessed when this stage unblocks.
00483             skidInsert(tid);
00484         }
00485 
00486         // If we switched over to blocking, then there's a potential for
00487         // an overall status change.
00488         status_change = unblock(tid) || status_change || blockThisCycle;
00489     }
00490 }

00501     // Check the decode queue to see if instructions are available.
00502     // If there are no available instructions to rename, then do nothing.
00503     if (insts_available == 0) {
00504         DPRINTF(Rename, "[tid:%u]: Nothing to do, breaking out early.\n",
00505                 tid);
00506         // Should I change status to idle?
00507         ++renameIdleCycles;
00508         return;
00509     } else if (renameStatus[tid] == Unblocking) {
00510         ++renameUnblockCycles;
00511     } else if (renameStatus[tid] == Running) {
00512         ++renameRunCycles;
00513     }
00514 
00515     DynInstPtr inst;
00516 
00517     // Will have to do a different calculation for the number of free
00518     // entries.
00519     int free_rob_entries = calcFreeROBEntries(tid);
00520     int free_iq_entries  = calcFreeIQEntries(tid);
00521     int min_free_entries = free_rob_entries;
00522 
00523     FullSource source = ROB;
00524 
00525     if (free_iq_entries < min_free_entries) {
00526         min_free_entries = free_iq_entries;
00527         source = IQ;
00528     }
00529 
00530     // Check if there's any space left.
00531     if (min_free_entries <= 0) {
00532         DPRINTF(Rename, "[tid:%u]: Blocking due to no free ROB/IQ/ "
00533                 "entries.\n"
00534                 "ROB has %i free entries.\n"
00535                 "IQ has %i free entries.\n",
00536                 tid,
00537                 free_rob_entries,
00538                 free_iq_entries);

00541

00546         return;

00551                 tid, insts_available, min_free_entries);

00554 
00555         blockThisCycle = true;
00556 
00557         incrFullStat(source);
00558     }

00561         skidBuffer[tid] : insts[tid];

00564             "send iew.\n", tid, insts_available);

00571     if (serializeOnNextInst[tid]) {

00599                 }
00600         }
00601 
00602         if (inst->isStore()) {
00603                 if(calcFreeSQEntries(tid) <= 0) {
00604                         DPRINTF(Rename, "[tid:%u]: Cannot rename due to no free SQ\n");
00605                         source = SQ;
00606                         incrFullStat(source);
00607                         break;
00608                 }
00609         }
00610

00612 
00613         if (renameStatus[tid] == Unblocking) {

00641             blockThisCycle = true;
00642             insts_to_rename.push_front(inst);
00643             ++renameFullRegistersEvents;
00644 
00645             break;
00646         }
00647 
00648         // Handle serializeAfter/serializeBefore instructions.
00649         // serializeAfter marks the next instruction as serializeBefore.
00650         // serializeBefore makes the instruction wait in rename until the ROB
00651         // is empty.
00652 
00653         // In this model, IPR accesses are serialize before
00654         // instructions, and store conditionals are serialize after
00655         // instructions.  This is mainly due to lack of support for
00656         // out-of-order operations of either of those classes of
00657         // instructions.
00658         if ((inst->isIprAccess() || inst->isSerializeBefore()) &&
00659             !inst->isSerializeHandled()) {
00660             DPRINTF(Rename, "Serialize before instruction encountered.\n");
00661 
00662             if (!inst->isTempSerializeBefore()) {
00663                 renamedSerializing++;
00664                 inst->setSerializeHandled();
00665             } else {
00666                 renamedTempSerializing++;
00667             }
00668 
00669             // Change status over to SerializeStall so that other stages know
00670             // what this is blocked on.
00671             renameStatus[tid] = SerializeStall;
00672 
00673             serializeInst[tid] = inst;
00674 
00675             blockThisCycle = true;
00676 
00677             break;
00678         } else if ((inst->isStoreConditional() || inst->isSerializeAfter()) &&
00679                    !inst->isSerializeHandled()) {
00680             DPRINTF(Rename, "Serialize after instruction encountered.\n");
00681 
00682             renamedSerializing++;
00683 
00684             inst->setSerializeHandled();
00685 
00686             serializeAfter(insts_to_rename, tid);
00687         }
00688 
00689         renameSrcRegs(inst, inst->threadNumber);
00690 
00691         renameDestRegs(inst, inst->threadNumber);
00692 
00693         if (inst->isLoad()) {
00694                 loadsInProgress[tid]++;
00695         }
00696         if (inst->isStore()) {
00697                 storesInProgress[tid]++;
00698         }
00699         ++renamed_insts;
00700 
00701 
00702         // Put instruction in rename queue.
00703         toIEW->insts[toIEWIndex] = inst;
00704         ++(toIEW->size);
00705 
00706         // Increment which instruction we're on.
00707         ++toIEWIndex;
00708 
00709         // Decrement how many instructions are available.
00710         --insts_available;
00711     }
00712 
00713     instsInProgress[tid] += renamed_insts;
00714     renameRenamedInsts += renamed_insts;
00715 
00716     // If we wrote to the time buffer, record this.
00717     if (toIEWIndex) {
00718         wroteToTimeBuffer = true;
00719     }
00720 
00721     // Check if there's any instructions left that haven't yet been renamed.
00722     // If so then block.
00723     if (insts_available) {
00724         blockThisCycle = true;
00725     }
00726 
00727     if (blockThisCycle) {
00728         block(tid);
00729         toDecode->renameUnblock[tid] = false;
00730     }
00731 }

00733 template<class Impl>
00734 void
00735 DefaultRename<Impl>::skidInsert(ThreadID tid)
00736 {
00737     DynInstPtr inst = NULL;
00738 
00739     while (!insts[tid].empty()) {
00740         inst = insts[tid].front();
00741 
00742         insts[tid].pop_front();
00743 
00744         assert(tid == inst->threadNumber);
00745 
00746         DPRINTF(Rename, "[tid:%u]: Inserting [sn:%lli] PC: %s into Rename "
00747                 "skidBuffer\n", tid, inst->seqNum, inst->pcState());
00748 
00749         ++renameSkidInsts;
00750 
00751         skidBuffer[tid].push_back(inst);
00752     }
00753 
00754     if (skidBuffer[tid].size() > skidBufferMax)
00755     {
00756         typename InstQueue::iterator it;
00757         warn("Skidbuffer contents:\n");
00758         for(it = skidBuffer[tid].begin(); it != skidBuffer[tid].end(); it++)
00759         {
00760             warn("[tid:%u]: %s [sn:%i].\n", tid,
00761                     (*it)->staticInst->disassemble(inst->instAddr()),
00762                     (*it)->seqNum);
00763         }
00764         panic("Skidbuffer Exceeded Max Size");
00765     }
00766 }

00801 template<class Impl>
00802 void
00803 DefaultRename<Impl>::updateStatus()
00804 {
00805     bool any_unblocking = false;
00806 
00807     list<ThreadID>::iterator threads = activeThreads->begin();
00808     list<ThreadID>::iterator end = activeThreads->end();
00809 
00810     while (threads != end) {
00811         ThreadID tid = *threads++;
00812 
00813         if (renameStatus[tid] == Unblocking) {
00814             any_unblocking = true;
00815             break;
00816         }
00817     }
00818 
00819     // Rename will have activity if it's unblocking.
00820     if (any_unblocking) {
00821         if (_status == Inactive) {
00822             _status = Active;
00823 
00824             DPRINTF(Activity, "Activating stage.\n");
00825 
00826             cpu->activateStage(O3CPU::RenameIdx);
00827         }
00828     } else {
00829         // If it's not unblocking, then rename will not have any internal
00830         // activity.  Switch it to inactive.
00831         if (_status == Active) {
00832             _status = Inactive;
00833             DPRINTF(Activity, "Deactivating stage.\n");
00834 
00835             cpu->deactivateStage(O3CPU::RenameIdx);
00836         }
00837     }

00840 template <class Impl>
00841 bool
00842 DefaultRename<Impl>::block(ThreadID tid)
00843 {
00844     DPRINTF(Rename, "[tid:%u]: Blocking.\n", tid);
00845 
00846     // Add the current inputs onto the skid buffer, so they can be
00847     // reprocessed when this stage unblocks.
00848     skidInsert(tid);
00849 
00850     // Only signal backwards to block if the previous stages do not think
00851     // rename is already blocked.
00852     if (renameStatus[tid] != Blocked) {
00853         // If resumeUnblocking is set, we unblocked during the squash,
00854         // but now we're have unblocking status. We need to tell earlier
00855         // stages to block.
00856         if (resumeUnblocking || renameStatus[tid] != Unblocking) {
00857             toDecode->renameBlock[tid] = true;
00858             toDecode->renameUnblock[tid] = false;
00859             wroteToTimeBuffer = true;
00860         }
00861 
00862         // Rename can not go from SerializeStall to Blocked, otherwise
00863         // it would not know to complete the serialize stall.
00864         if (renameStatus[tid] != SerializeStall) {
00865             // Set status to Blocked.
00866             renameStatus[tid] = Blocked;
00867             return true;
00868         }
00869     }
00870 
00871     return false;
00872 }

00874 template <class Impl>
00875 bool
00876 DefaultRename<Impl>::unblock(ThreadID tid)
00877 {
00878     DPRINTF(Rename, "[tid:%u]: Trying to unblock.\n", tid);
00879 
00880     // Rename is done unblocking if the skid buffer is empty.
00881     if (skidBuffer[tid].empty() && renameStatus[tid] != SerializeStall) {
00882 
00883         DPRINTF(Rename, "[tid:%u]: Done unblocking.\n", tid);
00884 
00885         toDecode->renameUnblock[tid] = true;
00886         wroteToTimeBuffer = true;
00887 
00888         renameStatus[tid] = Running;
00889         return true;
00890     }
00891 
00892     return false;
00893 }

00895 template <class Impl>
00896 void
00897 DefaultRename<Impl>::doSquash(const InstSeqNum &squashed_seq_num, ThreadID tid)
00898 {
00899     typename std::list<RenameHistory>::iterator hb_it =
00900         historyBuffer[tid].begin();
00901 
00902     // After a syscall squashes everything, the history buffer may be empty
00903     // but the ROB may still be squashing instructions.
00904     if (historyBuffer[tid].empty()) {
00905         return;
00906     }
00907 
00908     // Go through the most recent instructions, undoing the mappings
00909     // they did and freeing up the registers.
00910     while (!historyBuffer[tid].empty() &&
00911            hb_it->instSeqNum > squashed_seq_num) {
00912         assert(hb_it != historyBuffer[tid].end());
00913 
00914         DPRINTF(Rename, "[tid:%u]: Removing history entry with sequence "
00915                 "number %i.\n", tid, hb_it->instSeqNum);
00916 
00917         // Undo the rename mapping only if it was really a change.
00918         // Special regs that are not really renamed (like misc regs
00919         // and the zero reg) can be recognized because the new mapping
00920         // is the same as the old one.  While it would be merely a
00921         // waste of time to update the rename table, we definitely
00922         // don't want to put these on the free list.
00923         if (hb_it->newPhysReg != hb_it->prevPhysReg) {
00924             // Tell the rename map to set the architected register to the
00925             // previous physical register that it was renamed to.
00926             renameMap[tid]->setEntry(hb_it->archReg, hb_it->prevPhysReg);
00927 
00928             // Put the renamed physical register back on the free list.
00929             freeList->addReg(hb_it->newPhysReg);
00930         }
00931 
00932         historyBuffer[tid].erase(hb_it++);
00933 
00934         ++renameUndoneMaps;
00935     }
00936 }

00938 template<class Impl>
00939 void
00940 DefaultRename<Impl>::removeFromHistory(InstSeqNum inst_seq_num, ThreadID tid)
00941 {
00942     DPRINTF(Rename, "[tid:%u]: Removing a committed instruction from the "
00943             "history buffer %u (size=%i), until [sn:%lli].\n",
00944             tid, tid, historyBuffer[tid].size(), inst_seq_num);
00945 
00946     typename std::list<RenameHistory>::iterator hb_it =
00947         historyBuffer[tid].end();
00948 
00949     --hb_it;
00950 
00951     if (historyBuffer[tid].empty()) {
00952         DPRINTF(Rename, "[tid:%u]: History buffer is empty.\n", tid);
00953         return;
00954     } else if (hb_it->instSeqNum > inst_seq_num) {
00955         DPRINTF(Rename, "[tid:%u]: Old sequence number encountered.  Ensure "
00956                 "that a syscall happened recently.\n", tid);
00957         return;
00958     }
00959 
00960     // Commit all the renames up until (and including) the committed sequence
00961     // number. Some or even all of the committed instructions may not have
00962     // rename histories if they did not have destination registers that were
00963     // renamed.
00964     while (!historyBuffer[tid].empty() &&
00965            hb_it != historyBuffer[tid].end() &&
00966            hb_it->instSeqNum <= inst_seq_num) {
00967 
00968         DPRINTF(Rename, "[tid:%u]: Freeing up older rename of reg %i, "
00969                 "[sn:%lli].\n",
00970                 tid, hb_it->prevPhysReg, hb_it->instSeqNum);
00971 
00972         // Don't free special phys regs like misc and zero regs, which
00973         // can be recognized because the new mapping is the same as
00974         // the old one.
00975         if (hb_it->newPhysReg != hb_it->prevPhysReg) {
00976             freeList->addReg(hb_it->prevPhysReg);
00977         }
00978 
00979         ++renameCommittedMaps;
00980 
00981         historyBuffer[tid].erase(hb_it--);
00982     }
00983 }

00985 template <class Impl>
00986 inline void
00987 DefaultRename<Impl>::renameSrcRegs(DynInstPtr &inst, ThreadID tid)
00988 {
00989     ThreadContext *tc = inst->tcBase();
00990     RenameMap *map = renameMap[tid];
00991     unsigned num_src_regs = inst->numSrcRegs();
00992 
00993     // Get the architectual register numbers from the source and
00994     // operands, and redirect them to the right physical register.
00995     for (int src_idx = 0; src_idx < num_src_regs; src_idx++) {
00996         RegIndex src_reg = inst->srcRegIdx(src_idx);
00997         RegIndex rel_src_reg;
00998         RegIndex flat_rel_src_reg;
00999         PhysRegIndex renamed_reg;
01000 
01001         switch (regIdxToClass(src_reg, &rel_src_reg)) {
01002           case IntRegClass:
01003             flat_rel_src_reg = tc->flattenIntIndex(rel_src_reg);
01004             renamed_reg = map->lookupInt(flat_rel_src_reg);
01005             intRenameLookups++;
01006             break;
01007 
01008           case FloatRegClass:
01009             flat_rel_src_reg = tc->flattenFloatIndex(rel_src_reg);
01010             renamed_reg = map->lookupFloat(flat_rel_src_reg);
01011             fpRenameLookups++;
01012             break;
01013 
01014           case CCRegClass:
01015             flat_rel_src_reg = tc->flattenCCIndex(rel_src_reg);
01016             renamed_reg = map->lookupCC(flat_rel_src_reg);
01017             break;
01018 
01019           case MiscRegClass:
01020             // misc regs don't get flattened
01021             flat_rel_src_reg = rel_src_reg;
01022             renamed_reg = map->lookupMisc(flat_rel_src_reg);
01023             break;
01024 
01025           default:
01026             panic("Reg index is out of bound: %d.", src_reg);
01027         }
01028 
01029         DPRINTF(Rename, "[tid:%u]: Looking up %s arch reg %i (flattened %i), "
01030                 "got phys reg %i\n", tid, RegClassStrings[regIdxToClass(src_reg)],
01031                 (int)src_reg, (int)flat_rel_src_reg, (int)renamed_reg);
01032 
01033         inst->renameSrcReg(src_idx, renamed_reg);
01034 
01035         // See if the register is ready or not.
01036         if (scoreboard->getReg(renamed_reg)) {
01037             DPRINTF(Rename, "[tid:%u]: Register %d is ready.\n",
01038                     tid, renamed_reg);
01039 
01040             inst->markSrcRegReady(src_idx);
01041         } else {
01042             DPRINTF(Rename, "[tid:%u]: Register %d is not ready.\n",
01043                     tid, renamed_reg);
01044         }
01045 
01046         ++renameRenameLookups;
01047     }
01048 }

01050 template <class Impl>
01051 inline void
01052 DefaultRename<Impl>::renameDestRegs(DynInstPtr &inst, ThreadID tid)
01053 {
01054     ThreadContext *tc = inst->tcBase();
01055     RenameMap *map = renameMap[tid];
01056     unsigned num_dest_regs = inst->numDestRegs();
01057 
01058     // Rename the destination registers.
01059     for (int dest_idx = 0; dest_idx < num_dest_regs; dest_idx++) {
01060         RegIndex dest_reg = inst->destRegIdx(dest_idx);
01061         RegIndex rel_dest_reg;
01062         RegIndex flat_rel_dest_reg;
01063         RegIndex flat_uni_dest_reg;
01064         typename RenameMap::RenameInfo rename_result;
01065 
01066         switch (regIdxToClass(dest_reg, &rel_dest_reg)) {
01067           case IntRegClass:
01068             flat_rel_dest_reg = tc->flattenIntIndex(rel_dest_reg);
01069             rename_result = map->renameInt(flat_rel_dest_reg);
01070             flat_uni_dest_reg = flat_rel_dest_reg;  // 1:1 mapping
01071             break;
01072 
01073           case FloatRegClass:
01074             flat_rel_dest_reg = tc->flattenFloatIndex(rel_dest_reg);
01075             rename_result = map->renameFloat(flat_rel_dest_reg);
01076             flat_uni_dest_reg = flat_rel_dest_reg + TheISA::FP_Reg_Base;
01077             break;
01078 
01079           case CCRegClass:
01080             flat_rel_dest_reg = tc->flattenCCIndex(rel_dest_reg);
01081             rename_result = map->renameCC(flat_rel_dest_reg);
01082             flat_uni_dest_reg = flat_rel_dest_reg + TheISA::CC_Reg_Base;
01083             break;
01084 
01085           case MiscRegClass:
01086             // misc regs don't get flattened
01087             flat_rel_dest_reg = rel_dest_reg;
01088             rename_result = map->renameMisc(flat_rel_dest_reg);
01089             flat_uni_dest_reg = flat_rel_dest_reg + TheISA::Misc_Reg_Base;
01090             break;
01091 
01092           default:
01093             panic("Reg index is out of bound: %d.", dest_reg);
01094         }
01095 
01096         inst->flattenDestReg(dest_idx, flat_uni_dest_reg);
01097 
01098         // Mark Scoreboard entry as not ready
01099         scoreboard->unsetReg(rename_result.first);
01100 
01101         DPRINTF(Rename, "[tid:%u]: Renaming arch reg %i to physical "
01102                 "reg %i.\n", tid, (int)flat_rel_dest_reg,
01103                 (int)rename_result.first);
01104 
01105         // Record the rename information so that a history can be kept.
01106         RenameHistory hb_entry(inst->seqNum, flat_uni_dest_reg,
01107                                rename_result.first,
01108                                rename_result.second);
01109 
01110         historyBuffer[tid].push_front(hb_entry);
01111 
01112         DPRINTF(Rename, "[tid:%u]: Adding instruction to history buffer "
01113                 "(size=%i), [sn:%lli].\n",tid,
01114                 historyBuffer[tid].size(),
01115                 (*historyBuffer[tid].begin()).instSeqNum);
01116 
01117         // Tell the instruction to rename the appropriate destination
01118         // register (dest_idx) to the new physical register
01119         // (rename_result.first), and record the previous physical
01120         // register that the same logical register was renamed to
01121         // (rename_result.second).
01122         inst->renameDestReg(dest_idx,
01123                             rename_result.first,
01124                             rename_result.second);
01125 
01126         ++renameRenamedOperands;
01127     }
01128 }

01206 template <class Impl>

01209 {

01225         DPRINTF(Rename,"[tid:%i]: Stall: RenameMap has 0 free entries.\n", tid);
01226         ret_val = true;
01227     } else if (renameStatus[tid] == SerializeStall &&
01228                (!emptyROB[tid] || instsInProgress[tid])) {
01229         DPRINTF(Rename,"[tid:%i]: Stall: Serialize stall and ROB is not "
01230                 "empty.\n",
01231                 tid);

01268 template <class Impl>
01269 bool
01270 DefaultRename<Impl>::checkSignalsAndUpdate(ThreadID tid)
01271 {
01272     // Check if there's a squash signal, squash if there is
01273     // Check stall signals, block if necessary.
01274     // If status was blocked
01275     //     check if stall conditions have passed
01276     //         if so then go to unblocking
01277     // If status was Squashing
01278     //     check if squashing is not high.  Switch to running this cycle.
01279     // If status was serialize stall
01280     //     check if ROB is empty and no insts are in flight to the ROB
01281 
01282     readFreeEntries(tid);
01283     readStallSignals(tid);
01284 
01285     if (fromCommit->commitInfo[tid].squash) {
01286         DPRINTF(Rename, "[tid:%u]: Squashing instructions due to squash from "
01287                 "commit.\n", tid);
01288 
01289         squash(fromCommit->commitInfo[tid].doneSeqNum, tid);
01290 
01291         return true;
01292     }
01293 
01294     if (checkStall(tid)) {
01295         return block(tid);
01296     }
01297 
01298     if (renameStatus[tid] == Blocked) {
01299         DPRINTF(Rename, "[tid:%u]: Done blocking, switching to unblocking.\n",
01300                 tid);
01301 
01302         renameStatus[tid] = Unblocking;
01303 
01304         unblock(tid);
01305 
01306         return true;
01307     }
01308 
01309     if (renameStatus[tid] == Squashing) {
01310         // Switch status to running if rename isn't being told to block or
01311         // squash this cycle.
01312         if (resumeSerialize) {
01313             DPRINTF(Rename, "[tid:%u]: Done squashing, switching to serialize.\n",
01314                     tid);
01315 
01316             renameStatus[tid] = SerializeStall;
01317             return true;
01318         } else if (resumeUnblocking) {
01319             DPRINTF(Rename, "[tid:%u]: Done squashing, switching to unblocking.\n",
01320                     tid);
01321             renameStatus[tid] = Unblocking;
01322             return true;
01323         } else {
01324             DPRINTF(Rename, "[tid:%u]: Done squashing, switching to running.\n",
01325                     tid);
01326 
01327             renameStatus[tid] = Running;
01328             return false;
01329         }
01330     }
01331 
01332     if (renameStatus[tid] == SerializeStall) {
01333         // Stall ends once the ROB is free.
01334         DPRINTF(Rename, "[tid:%u]: Done with serialize stall, switching to "
01335                 "unblocking.\n", tid);
01336 
01337         DynInstPtr serial_inst = serializeInst[tid];
01338 
01339         renameStatus[tid] = Unblocking;
01340 
01341         unblock(tid);
01342 
01343         DPRINTF(Rename, "[tid:%u]: Processing instruction [%lli] with "
01344                 "PC %s.\n", tid, serial_inst->seqNum, serial_inst->pcState());
01345 
01346         // Put instruction into queue here.
01347         serial_inst->clearSerializeBefore();
01348 
01349         if (!skidBuffer[tid].empty()) {
01350             skidBuffer[tid].push_front(serial_inst);
01351         } else {
01352             insts[tid].push_front(serial_inst);
01353         }
01354 
01355         DPRINTF(Rename, "[tid:%u]: Instruction must be processed by rename."
01356                 " Adding to front of list.\n", tid);
01357 
01358         serializeInst[tid] = NULL;
01359 
01360         return true;
01361     }
01362 
01363     // If we've reached this point, we have not gotten any signals that
01364     // cause rename to change its status.  Rename remains the same as before.
01365     return false;
01366 }

02320             for (off_type i = 0; i < size; ++i)
02321                 vresult[i] = op(lvec[0], rvec[i]);
02322         } else if (rvec.size() == 1) {
02323             size_type size = lvec.size();
02324             vresult.resize(size);
02325             for (off_type i = 0; i < size; ++i)

02327         } else if (rvec.size() == lvec.size()) {
02328             size_type size = rvec.size();
02329             vresult.resize(size);
02330             for (off_type i = 0; i < size; ++i)

02355                 lsum += lvec[i];
02356                 rsum += rvec[i];

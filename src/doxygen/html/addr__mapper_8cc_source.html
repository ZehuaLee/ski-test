00114 bool
00115 AddrMapper::recvTimingReq(PacketPtr pkt)
00116 {
00117     Addr orig_addr = pkt->getAddr();
00118     bool needsResponse = pkt->needsResponse();
00119     bool memInhibitAsserted = pkt->memInhibitAsserted();
00120 
00121     if (needsResponse && !memInhibitAsserted) {
00122         pkt->pushSenderState(new AddrMapperSenderState(orig_addr));
00123     }
00124 
00125     pkt->setAddr(remapAddr(orig_addr));
00126 
00127     // Attempt to send the packet (always succeeds for inhibited
00128     // packets)
00129     bool successful = masterPort.sendTimingReq(pkt);
00130 
00131     // If not successful, restore the address and sender state
00132     if (!successful) {
00133         pkt->setAddr(orig_addr);
00134 
00135         if (needsResponse) {
00136             delete pkt->popSenderState();
00137         }
00138     }
00139 
00140     return successful;
00141 }

00143 bool
00144 AddrMapper::recvTimingResp(PacketPtr pkt)
00145 {
00146     AddrMapperSenderState* receivedState =
00147         dynamic_cast<AddrMapperSenderState*>(pkt->senderState);
00148 
00149     // Restore initial sender state
00150     if (receivedState == NULL)
00151         panic("AddrMapper %s got a response without sender state\n",
00152               name());
00153 
00154     Addr remapped_addr = pkt->getAddr();
00155 
00156     // Restore the state and address
00157     pkt->senderState = receivedState->predecessor;
00158     pkt->setAddr(receivedState->origAddr);
00159 
00160     // Attempt to send the packet
00161     bool successful = slavePort.sendTimingResp(pkt);
00162 
00163     // If packet successfully sent, delete the sender state, otherwise
00164     // restore state
00165     if (successful) {
00166         delete receivedState;
00167     } else {
00168         // Don't delete anything and let the packet look like we did
00169         // not touch it
00170         pkt->senderState = receivedState;
00171         pkt->setAddr(remapped_addr);
00172     }
00173     return successful;
00174 }

00236 Addr
00237 RangeAddrMapper::remapAddr(Addr addr) const
00238 {
00239     for (int i = 0; i < originalRanges.size(); ++i) {
00240         if (originalRanges[i].contains(addr)) {
00241             Addr offset = addr - originalRanges[i].start();
00242             return offset + remappedRanges[i].start();
00243         }
00244     }
00245 
00246     return addr;
00247 }

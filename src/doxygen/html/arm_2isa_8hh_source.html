00316                     warn("User mode does not have SPSR\n");
00317                     flat_idx = MISCREG_SPSR;
00318                     break;
00319                   case MODE_FIQ:
00320                     flat_idx = MISCREG_SPSR_FIQ;

00323                     flat_idx = MISCREG_SPSR_IRQ;
00324                     break;
00325                   case MODE_SVC:
00326                     flat_idx = MISCREG_SPSR_SVC;
00327                     break;
00328                   case MODE_MON:
00329                     flat_idx = MISCREG_SPSR_MON;
00330                     break;
00331                   case MODE_ABORT:
00332                     flat_idx = MISCREG_SPSR_ABT;
00333                     break;
00334                   case MODE_HYP:
00335                     flat_idx = MISCREG_SPSR_HYP;
00336                     break;
00337                   case MODE_UNDEFINED:
00338                     flat_idx = MISCREG_SPSR_UND;
00339                     break;
00340                   default:

00342                          cpsr.mode);
00343                     flat_idx = MISCREG_SPSR;
00344                     break;
00345                 }
00346             } else if (miscRegInfo[reg][MISCREG_MUTEX]) {
00347                 // Mutually exclusive CP15 register
00348                 switch (reg) {
00349                   case MISCREG_PRRR_MAIR0:
00350                   case MISCREG_PRRR_MAIR0_NS:
00351                   case MISCREG_PRRR_MAIR0_S:
00352                     {
00353                         TTBCR ttbcr = readMiscRegNoEffect(MISCREG_TTBCR);
00354                         // If the muxed reg has been flattened, work out the

00357                         if (ttbcr.eae)
00358                             flat_idx = flattenMiscIndex(MISCREG_MAIR0 +

00360                         else
00361                             flat_idx = flattenMiscIndex(MISCREG_PRRR +
00362                                                         idxOffset);
00363                     }
00364                     break;
00365                   case MISCREG_NMRR_MAIR1:
00366                   case MISCREG_NMRR_MAIR1_NS:
00367                   case MISCREG_NMRR_MAIR1_S:
00368                     {
00369                         TTBCR ttbcr = readMiscRegNoEffect(MISCREG_TTBCR);
00370                         // If the muxed reg has been flattened, work out the
00371                         // offset and apply it to the unmuxed reg
00372                         int idxOffset = reg - MISCREG_NMRR_MAIR1;
00373                         if (ttbcr.eae)
00374                             flat_idx = flattenMiscIndex(MISCREG_MAIR1 +
00375                                                         idxOffset);

00380                     break;
00381                   case MISCREG_PMXEVTYPER_PMCCFILTR:

00383                         PMSELR pmselr = miscRegs[MISCREG_PMSELR];

00394             } else {
00395                 if (miscRegInfo[reg][MISCREG_BANKED]) {
00396                     bool secureReg = haveSecurity &&
00397                                      inSecureState(miscRegs[MISCREG_SCR],
00398                                                    miscRegs[MISCREG_CPSR]);

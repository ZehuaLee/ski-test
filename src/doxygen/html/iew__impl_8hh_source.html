00504 template<class Impl>
00505 void
00506 DefaultIEW<Impl>::squashDueToMemOrder(DynInstPtr &inst, ThreadID tid)
00507 {
00508     DPRINTF(IEW, "[tid:%i]: Memory violation, squashing violator and younger "
00509             "insts, PC: %s [sn:%i].\n", tid, inst->pcState(), inst->seqNum);
00510     // Need to include inst->seqNum in the following comparison to cover the
00511     // corner case when a branch misprediction and a memory violation for the
00512     // same instruction (e.g. load PC) are detected in the same cycle.  In this
00513     // case the memory violator should take precedence over the branch
00514     // misprediction because it requires the violator itself to be included in
00515     // the squash.
00516     if (!toCommit->squash[tid] ||
00517             inst->seqNum <= toCommit->squashedSeqNum[tid]) {
00518         toCommit->squash[tid] = true;
00519 
00520         toCommit->squashedSeqNum[tid] = inst->seqNum;
00521         toCommit->pc[tid] = inst->pcState();
00522         toCommit->mispredictInst[tid] = NULL;
00523 
00524         // Must include the memory violator in the squash.
00525         toCommit->includeSquashInst[tid] = true;
00526 
00527         wroteToTimeBuffer = true;
00528     }

00602 template<class Impl>
00603 void
00604 DefaultIEW<Impl>::instToCommit(DynInstPtr &inst)
00605 {
00606     // This function should not be called after writebackInsts in a
00607     // single cycle.  That will cause problems with an instruction
00608     // being added to the queue to commit without being processed by
00609     // writebackInsts prior to being sent to commit.
00610 
00611     // First check the time slot that this instruction will write
00612     // to.  If there are free write ports at the time, then go ahead
00613     // and write the instruction to that time.  If there are not,
00614     // keep looking back to see where's the first time there's a
00615     // free slot.
00616     while ((*iewQueue)[wbCycle].insts[wbNumInst]) {
00617         ++wbNumInst;
00618         if (wbNumInst == wbWidth) {
00619             ++wbCycle;
00620             wbNumInst = 0;
00621         }
00622     }
00623 
00624     DPRINTF(IEW, "Current wb cycle: %i, width: %i, numInst: %i\nwbActual:%i\n",
00625             wbCycle, wbWidth, wbNumInst, wbCycle * wbWidth + wbNumInst);
00626     // Add finished instruction to queue to commit.
00627     (*iewQueue)[wbCycle].insts[wbNumInst] = inst;
00628     (*iewQueue)[wbCycle].size++;
00629 }

00706 {

00708

00710     list<ThreadID>::iterator end = activeThreads->end();

00726         !ldstQueue.willWB() && !any_unblocking) {

00733                                        ldstQueue.willWB() ||
00734                                        any_unblocking)) {
00735         // Otherwise there is internal activity.  Set to active.
00736         DPRINTF(IEW, "IEW switching to active\n");
00737

00740         _status = Active;

00769 template <class Impl>
00770 void
00771 DefaultIEW<Impl>::checkSignalsAndUpdate(ThreadID tid)
00772 {
00773     // Check if there's a squash signal, squash if there is
00774     // Check stall signals, block if there is.
00775     // If status was Blocked
00776     //     if so then go to unblocking
00777     // If status was Squashing
00778     //     check if squashing is not high.  Switch to running this cycle.
00779 
00780     if (fromCommit->commitInfo[tid].squash) {
00781         squash(tid);
00782 
00783         if (dispatchStatus[tid] == Blocked ||
00784             dispatchStatus[tid] == Unblocking) {
00785             toRename->iewUnblock[tid] = true;
00786             wroteToTimeBuffer = true;
00787         }
00788 
00789         dispatchStatus[tid] = Squashing;
00790         fetchRedirect[tid] = false;
00791         return;
00792     }
00793 
00794     if (fromCommit->commitInfo[tid].robSquashing) {
00795         DPRINTF(IEW, "[tid:%i]: ROB is still squashing.\n", tid);
00796 
00797         dispatchStatus[tid] = Squashing;
00798         emptyRenameInsts(tid);
00799         wroteToTimeBuffer = true;
00800     }
00801 
00802     if (checkStall(tid)) {
00803         block(tid);
00804         dispatchStatus[tid] = Blocked;
00805         return;
00806     }
00807 
00808     if (dispatchStatus[tid] == Blocked) {
00809         // Status from previous cycle was blocked, but there are no more stall
00810         // conditions.  Switch over to unblocking.
00811         DPRINTF(IEW, "[tid:%i]: Done blocking, switching to unblocking.\n",
00812                 tid);
00813 
00814         dispatchStatus[tid] = Unblocking;
00815 
00816         unblock(tid);
00817 
00818         return;
00819     }
00820 
00821     if (dispatchStatus[tid] == Squashing) {
00822         // Switch status to running if rename isn't being told to block or
00823         // squash this cycle.
00824         DPRINTF(IEW, "[tid:%i]: Done squashing, switching to running.\n",
00825                 tid);
00826 
00827         dispatchStatus[tid] = Running;
00828 
00829         return;
00830     }
00831 }

00899 template<class Impl>
00900 void
00901 DefaultIEW<Impl>::dispatch(ThreadID tid)
00902 {
00903     // If status is Running or idle,
00904     //     call dispatchInsts()
00905     // If status is Unblocking,
00906     //     buffer any instructions coming from rename
00907     //     continue trying to empty skid buffer
00908     //     check if stall conditions have passed
00909 
00910     if (dispatchStatus[tid] == Blocked) {
00911         ++iewBlockCycles;
00912 
00913     } else if (dispatchStatus[tid] == Squashing) {
00914         ++iewSquashCycles;
00915     }
00916 
00917     // Dispatch should try to dispatch as many instructions as its bandwidth
00918     // will allow, as long as it is not currently blocked.
00919     if (dispatchStatus[tid] == Running ||
00920         dispatchStatus[tid] == Idle) {
00921         DPRINTF(IEW, "[tid:%i] Not blocked, so attempting to run "
00922                 "dispatch.\n", tid);
00923 
00924         dispatchInsts(tid);
00925     } else if (dispatchStatus[tid] == Unblocking) {
00926         // Make sure that the skid buffer has something in it if the
00927         // status is unblocking.
00928         assert(!skidsEmpty());
00929 
00930         // If the status was unblocking, then instructions from the skid
00931         // buffer were used.  Remove those instructions and handle
00932         // the rest of unblocking.
00933         dispatchInsts(tid);
00934 
00935         ++iewUnblockCycles;
00936 
00937         if (validInstsFromRename()) {
00938             // Add the current inputs to the skid buffer so they can be
00939             // reprocessed when this stage unblocks.
00940             skidInsert(tid);
00941         }
00942 
00943         unblock(tid);
00944     }
00945 }

00947 template <class Impl>
00948 void
00949 DefaultIEW<Impl>::dispatchInsts(ThreadID tid)
00950 {
00951     // Obtain instructions from skid buffer if unblocking, or queue from rename
00952     // otherwise.
00953     std::queue<DynInstPtr> &insts_to_dispatch =
00954         dispatchStatus[tid] == Unblocking ?
00955         skidBuffer[tid] : insts[tid];
00956 
00957     int insts_to_add = insts_to_dispatch.size();
00958 
00959     DynInstPtr inst;
00960     bool add_to_iq = false;
00961     int dis_num_inst = 0;
00962 
00963     // Loop through the instructions, putting them in the instruction
00964     // queue.
00965     for ( ; dis_num_inst < insts_to_add &&
00966               dis_num_inst < dispatchWidth;
00967           ++dis_num_inst)
00968     {
00969         inst = insts_to_dispatch.front();
00970 
00971         if (dispatchStatus[tid] == Unblocking) {
00972             DPRINTF(IEW, "[tid:%i]: Issue: Examining instruction from skid "
00973                     "buffer\n", tid);
00974         }
00975 
00976         // Make sure there's a valid instruction there.
00977         assert(inst);
00978 
00979         DPRINTF(IEW, "[tid:%i]: Issue: Adding PC %s [sn:%lli] [tid:%i] to "
00980                 "IQ.\n",
00981                 tid, inst->pcState(), inst->seqNum, inst->threadNumber);
00982 
00983         // Be sure to mark these instructions as ready so that the
00984         // commit stage can go ahead and execute them, and mark
00985         // them as issued so the IQ doesn't reprocess them.
00986 
00987         // Check for squashed instructions.
00988         if (inst->isSquashed()) {
00989             DPRINTF(IEW, "[tid:%i]: Issue: Squashed instruction encountered, "
00990                     "not adding to IQ.\n", tid);
00991 
00992             ++iewDispSquashedInsts;
00993 
00994             insts_to_dispatch.pop();
00995 
00996             //Tell Rename That An Instruction has been processed
00997             if (inst->isLoad()) {
00998                 toRename->iewInfo[tid].dispatchedToLQ++;
00999             }
01000             if (inst->isStore()) {
01001                 toRename->iewInfo[tid].dispatchedToSQ++;
01002             }
01003 
01004             toRename->iewInfo[tid].dispatched++;
01005 
01006             continue;
01007         }
01008 
01009         // Check for full conditions.
01010         if (instQueue.isFull(tid)) {
01011             DPRINTF(IEW, "[tid:%i]: Issue: IQ has become full.\n", tid);
01012 
01013             // Call function to start blocking.
01014             block(tid);
01015 
01016             // Set unblock to false. Special case where we are using
01017             // skidbuffer (unblocking) instructions but then we still
01018             // get full in the IQ.
01019             toRename->iewUnblock[tid] = false;
01020 
01021             ++iewIQFullEvents;
01022             break;
01023         }
01024 
01025         // Check LSQ if inst is LD/ST
01026         if ((inst->isLoad() && ldstQueue.lqFull(tid)) ||
01027             (inst->isStore() && ldstQueue.sqFull(tid))) {
01028             DPRINTF(IEW, "[tid:%i]: Issue: %s has become full.\n",tid,
01029                     inst->isLoad() ? "LQ" : "SQ");
01030 
01031             // Call function to start blocking.
01032             block(tid);
01033 
01034             // Set unblock to false. Special case where we are using
01035             // skidbuffer (unblocking) instructions but then we still
01036             // get full in the IQ.
01037             toRename->iewUnblock[tid] = false;
01038 
01039             ++iewLSQFullEvents;
01040             break;
01041         }
01042 
01043         // Otherwise issue the instruction just fine.
01044         if (inst->isLoad()) {
01045             DPRINTF(IEW, "[tid:%i]: Issue: Memory instruction "
01046                     "encountered, adding to LSQ.\n", tid);
01047 
01048             // Reserve a spot in the load store queue for this
01049             // memory access.
01050             ldstQueue.insertLoad(inst);
01051 
01052             ++iewDispLoadInsts;
01053 
01054             add_to_iq = true;
01055 
01056             toRename->iewInfo[tid].dispatchedToLQ++;
01057         } else if (inst->isStore()) {
01058             DPRINTF(IEW, "[tid:%i]: Issue: Memory instruction "
01059                     "encountered, adding to LSQ.\n", tid);
01060 
01061             ldstQueue.insertStore(inst);
01062 
01063             ++iewDispStoreInsts;
01064 
01065             if (inst->isStoreConditional()) {
01066                 // Store conditionals need to be set as "canCommit()"
01067                 // so that commit can process them when they reach the
01068                 // head of commit.
01069                 // @todo: This is somewhat specific to Alpha.
01070                 inst->setCanCommit();
01071                 instQueue.insertNonSpec(inst);
01072                 add_to_iq = false;
01073 
01074                 ++iewDispNonSpecInsts;
01075             } else {
01076                 add_to_iq = true;
01077             }
01078 
01079             toRename->iewInfo[tid].dispatchedToSQ++;
01080         } else if (inst->isMemBarrier() || inst->isWriteBarrier()) {
01081             // Same as non-speculative stores.
01082             inst->setCanCommit();
01083             instQueue.insertBarrier(inst);
01084             add_to_iq = false;
01085         } else if (inst->isNop()) {
01086             DPRINTF(IEW, "[tid:%i]: Issue: Nop instruction encountered, "
01087                     "skipping.\n", tid);
01088 
01089             inst->setIssued();
01090             inst->setExecuted();
01091             inst->setCanCommit();
01092 
01093             instQueue.recordProducer(inst);
01094 
01095             iewExecutedNop[tid]++;
01096 
01097             add_to_iq = false;
01098         } else {
01099             assert(!inst->isExecuted());
01100             add_to_iq = true;
01101         }
01102 
01103         if (inst->isNonSpeculative()) {
01104             DPRINTF(IEW, "[tid:%i]: Issue: Nonspeculative instruction "
01105                     "encountered, skipping.\n", tid);
01106 
01107             // Same as non-speculative stores.
01108             inst->setCanCommit();
01109 
01110             // Specifically insert it as nonspeculative.
01111             instQueue.insertNonSpec(inst);
01112 
01113             ++iewDispNonSpecInsts;
01114 
01115             add_to_iq = false;
01116         }
01117 
01118         // If the instruction queue is not full, then add the
01119         // instruction.
01120         if (add_to_iq) {
01121             instQueue.insert(inst);
01122         }
01123 
01124         insts_to_dispatch.pop();
01125 
01126         toRename->iewInfo[tid].dispatched++;
01127 
01128         ++iewDispatchedInsts;
01129 
01130 #if TRACING_ON
01131         inst->dispatchTick = curTick() - inst->fetchTick;
01132 #endif
01133         ppDispatch->notify(inst);
01134     }
01135 
01136     if (!insts_to_dispatch.empty()) {
01137         DPRINTF(IEW,"[tid:%i]: Issue: Bandwidth Full. Blocking.\n", tid);
01138         block(tid);
01139         toRename->iewUnblock[tid] = false;
01140     }
01141 
01142     if (dispatchStatus[tid] == Idle && dis_num_inst) {
01143         dispatchStatus[tid] = Running;
01144 
01145         updatedQueues = true;
01146     }
01147 
01148     dis_num_inst = 0;
01149 }

01174 template <class Impl>
01175 void
01176 DefaultIEW<Impl>::executeInsts()
01177 {
01178     wbNumInst = 0;
01179     wbCycle = 0;
01180 
01181     list<ThreadID>::iterator threads = activeThreads->begin();
01182     list<ThreadID>::iterator end = activeThreads->end();
01183 
01184     while (threads != end) {
01185         ThreadID tid = *threads++;
01186         fetchRedirect[tid] = false;
01187     }
01188 
01189     // Uncomment this if you want to see all available instructions.
01190     // @todo This doesn't actually work anymore, we should fix it.
01191 //    printAvailableInsts();
01192 
01193     // Execute/writeback any instructions that are available.
01194     int insts_to_execute = fromIssue->size;
01195     int inst_num = 0;
01196     for (; inst_num < insts_to_execute;
01197           ++inst_num) {
01198 
01199         DPRINTF(IEW, "Execute: Executing instructions from IQ.\n");
01200 
01201         DynInstPtr inst = instQueue.getInstToExecute();
01202 
01203         DPRINTF(IEW, "Execute: Processing PC %s, [tid:%i] [sn:%i].\n",
01204                 inst->pcState(), inst->threadNumber,inst->seqNum);
01205 
01206         // Check if the instruction is squashed; if so then skip it
01207         if (inst->isSquashed()) {
01208             DPRINTF(IEW, "Execute: Instruction was squashed. PC: %s, [tid:%i]"
01209                          " [sn:%i]\n", inst->pcState(), inst->threadNumber,
01210                          inst->seqNum);
01211 
01212             // Consider this instruction executed so that commit can go
01213             // ahead and retire the instruction.
01214             inst->setExecuted();
01215 
01216             // Not sure if I should set this here or just let commit try to
01217             // commit any squashed instructions.  I like the latter a bit more.
01218             inst->setCanCommit();
01219 
01220             ++iewExecSquashedInsts;
01221 
01222             continue;
01223         }
01224 
01225         Fault fault = NoFault;
01226 
01227         // Execute instruction.
01228         // Note that if the instruction faults, it will be handled
01229         // at the commit stage.
01230         if (inst->isMemRef()) {
01231             DPRINTF(IEW, "Execute: Calculating address for memory "
01232                     "reference.\n");
01233 
01234             // Tell the LDSTQ to execute this instruction (if it is a load).
01235             if (inst->isLoad()) {
01236                 // Loads will mark themselves as executed, and their writeback
01237                 // event adds the instruction to the queue to commit
01238                 fault = ldstQueue.executeLoad(inst);
01239 
01240                 if (inst->isTranslationDelayed() &&
01241                     fault == NoFault) {
01242                     // A hw page table walk is currently going on; the
01243                     // instruction must be deferred.
01244                     DPRINTF(IEW, "Execute: Delayed translation, deferring "
01245                             "load.\n");
01246                     instQueue.deferMemInst(inst);
01247                     continue;
01248                 }
01249 
01250                 if (inst->isDataPrefetch() || inst->isInstPrefetch()) {
01251                     inst->fault = NoFault;
01252                 }
01253             } else if (inst->isStore()) {
01254                 fault = ldstQueue.executeStore(inst);
01255 
01256                 if (inst->isTranslationDelayed() &&
01257                     fault == NoFault) {
01258                     // A hw page table walk is currently going on; the
01259                     // instruction must be deferred.
01260                     DPRINTF(IEW, "Execute: Delayed translation, deferring "
01261                             "store.\n");
01262                     instQueue.deferMemInst(inst);
01263                     continue;
01264                 }
01265 
01266                 // If the store had a fault then it may not have a mem req
01267                 if (fault != NoFault || !inst->readPredicate() ||
01268                         !inst->isStoreConditional()) {
01269                     // If the instruction faulted, then we need to send it along
01270                     // to commit without the instruction completing.
01271                     // Send this instruction to commit, also make sure iew stage
01272                     // realizes there is activity.
01273                     inst->setExecuted();
01274                     instToCommit(inst);
01275                     activityThisCycle();
01276                 }
01277 
01278                 // Store conditionals will mark themselves as
01279                 // executed, and their writeback event will add the
01280                 // instruction to the queue to commit.
01281             } else {
01282                 panic("Unexpected memory type!\n");
01283             }
01284 
01285         } else {
01286             // If the instruction has already faulted, then skip executing it.
01287             // Such case can happen when it faulted during ITLB translation.
01288             // If we execute the instruction (even if it's a nop) the fault
01289             // will be replaced and we will lose it.
01290             if (inst->getFault() == NoFault) {
01291                 inst->execute();
01292                 if (!inst->readPredicate())
01293                     inst->forwardOldRegs();
01294             }
01295 
01296             inst->setExecuted();
01297 
01298             instToCommit(inst);
01299         }
01300 
01301         updateExeInstStats(inst);
01302 
01303         // Check if branch prediction was correct, if not then we need
01304         // to tell commit to squash in flight instructions.  Only
01305         // handle this if there hasn't already been something that
01306         // redirects fetch in this group of instructions.
01307 
01308         // This probably needs to prioritize the redirects if a different
01309         // scheduler is used.  Currently the scheduler schedules the oldest
01310         // instruction first, so the branch resolution order will be correct.
01311         ThreadID tid = inst->threadNumber;
01312 
01313         if (!fetchRedirect[tid] ||
01314             !toCommit->squash[tid] ||
01315             toCommit->squashedSeqNum[tid] > inst->seqNum) {
01316 
01317             // Prevent testing for misprediction on load instructions,
01318             // that have not been executed.
01319             bool loadNotExecuted = !inst->isExecuted() && inst->isLoad();
01320 
01321             if (inst->mispredicted() && !loadNotExecuted) {
01322                 fetchRedirect[tid] = true;
01323 
01324                 DPRINTF(IEW, "Execute: Branch mispredict detected.\n");
01325                 DPRINTF(IEW, "Predicted target was PC: %s.\n",
01326                         inst->readPredTarg());
01327                 DPRINTF(IEW, "Execute: Redirecting fetch to PC: %s.\n",
01328                         inst->pcState());
01329                 // If incorrect, then signal the ROB that it must be squashed.
01330                 squashDueToBranch(inst, tid);
01331 
01332                 ppMispredict->notify(inst);
01333 
01334                 if (inst->readPredTaken()) {
01335                     predictedTakenIncorrect++;
01336                 } else {
01337                     predictedNotTakenIncorrect++;
01338                 }
01339             } else if (ldstQueue.violation(tid)) {
01340                 assert(inst->isMemRef());
01341                 // If there was an ordering violation, then get the
01342                 // DynInst that caused the violation.  Note that this
01343                 // clears the violation signal.
01344                 DynInstPtr violator;
01345                 violator = ldstQueue.getMemDepViolator(tid);
01346 
01347                 DPRINTF(IEW, "LDSTQ detected a violation. Violator PC: %s "
01348                         "[sn:%lli], inst PC: %s [sn:%lli]. Addr is: %#x.\n",
01349                         violator->pcState(), violator->seqNum,
01350                         inst->pcState(), inst->seqNum, inst->physEffAddr);
01351 
01352                 fetchRedirect[tid] = true;
01353 
01354                 // Tell the instruction queue that a violation has occured.
01355                 instQueue.violation(inst, violator);
01356 
01357                 // Squash.
01358                 squashDueToMemOrder(violator, tid);
01359 
01360                 ++memOrderViolationEvents;
01361             }
01362         } else {
01363             // Reset any state associated with redirects that will not
01364             // be used.
01365             if (ldstQueue.violation(tid)) {
01366                 assert(inst->isMemRef());
01367 
01368                 DynInstPtr violator = ldstQueue.getMemDepViolator(tid);
01369 
01370                 DPRINTF(IEW, "LDSTQ detected a violation.  Violator PC: "
01371                         "%s, inst PC: %s.  Addr is: %#x.\n",
01372                         violator->pcState(), inst->pcState(),
01373                         inst->physEffAddr);
01374                 DPRINTF(IEW, "Violation will not be handled because "
01375                         "already squashing\n");
01376 
01377                 ++memOrderViolationEvents;
01378             }

01400 template <class Impl>
01401 void
01402 DefaultIEW<Impl>::writebackInsts()
01403 {
01404     // Loop through the head of the time buffer and wake any
01405     // dependents.  These instructions are about to write back.  Also
01406     // mark scoreboard that this instruction is finally complete.
01407     // Either have IEW have direct access to scoreboard, or have this
01408     // as part of backwards communication.
01409     for (int inst_num = 0; inst_num < wbWidth &&
01410              toCommit->insts[inst_num]; inst_num++) {
01411         DynInstPtr inst = toCommit->insts[inst_num];
01412         ThreadID tid = inst->threadNumber;
01413 
01414         DPRINTF(IEW, "Sending instructions to commit, [sn:%lli] PC %s.\n",
01415                 inst->seqNum, inst->pcState());
01416 
01417         iewInstsToCommit[tid]++;
01418 
01419         // Some instructions will be sent to commit without having
01420         // executed because they need commit to handle them.
01421         // E.g. Strictly ordered loads have not actually executed when they
01422         // are first sent to commit.  Instead commit must tell the LSQ
01423         // when it's ready to execute the strictly ordered load.
01424         if (!inst->isSquashed() && inst->isExecuted() && inst->getFault() == NoFault) {
01425             int dependents = instQueue.wakeDependents(inst);
01426 
01427             for (int i = 0; i < inst->numDestRegs(); i++) {
01428                 //mark as Ready
01429                 DPRINTF(IEW,"Setting Destination Register %i\n",
01430                         inst->renamedDestRegIdx(i));
01431                 scoreboard->setReg(inst->renamedDestRegIdx(i));
01432             }
01433 
01434             if (dependents) {
01435                 producerInst[tid]++;
01436                 consumerInst[tid]+= dependents;
01437             }
01438             writebackCount[tid]++;
01439         }
01440     }
01441 }

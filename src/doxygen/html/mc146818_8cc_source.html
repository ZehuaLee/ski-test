00137 void
00138 MC146818::writeData(const uint8_t addr, const uint8_t data)
00139 {
00140     bool panic_unsupported(false);
00141 
00142     if (addr < RTC_STAT_REGA) {
00143         clock_data[addr] = data;
00144         curTime.tm_sec = unbcdize(sec);
00145         curTime.tm_min = unbcdize(min);
00146         curTime.tm_hour = unbcdize(hour);
00147         curTime.tm_mday = unbcdize(mday);
00148         curTime.tm_mon = unbcdize(mon) - 1;
00149         curTime.tm_year = ((unbcdize(year) + 50) % 100) + 1950;
00150         curTime.tm_wday = unbcdize(wday) - 1;
00151     } else {
00152         switch (addr) {
00153           case RTC_STAT_REGA: {
00154               RtcRegA old_rega(stat_regA);
00155               stat_regA = data;
00156               // The "update in progress" bit is read only.
00157               stat_regA.uip = old_rega;
00158 
00159               if (!rega_dv_disabled(stat_regA) &&
00160                   stat_regA.dv != RTCA_DV_32768HZ) {
00161                   inform("RTC: Unimplemented divider configuration: %i\n",
00162                         stat_regA.dv);
00163                   panic_unsupported = true;
00164               }
00165 
00166               if (stat_regA.rs != RTCA_RS_1024HZ) {
00167                   inform("RTC: Unimplemented interrupt rate: %i\n",
00168                         stat_regA.rs);
00169                   panic_unsupported = true;
00170               }
00171 
00172               if (rega_dv_disabled(stat_regA)) {
00173                   // The divider is disabled, make sure that we don't
00174                   // schedule any ticks.
00175                   if (tickEvent.scheduled())
00176                       deschedule(tickEvent);
00177               } else if (rega_dv_disabled(old_rega))  {
00178                   // According to the specification, the next tick
00179                   // happens after 0.5s when the divider chain goes
00180                   // from reset to active. So, we simply schedule the
00181                   // tick after 0.5s.
00182                   assert(!tickEvent.scheduled());
00183                   schedule(tickEvent, curTick() + SimClock::Int::s / 2);
00184               }
00185           } break;
00186           case RTC_STAT_REGB:
00187             stat_regB = data;
00188             if (stat_regB.aie || stat_regB.uie) {
00189                 inform("RTC: Unimplemented interrupt configuration: %s %s\n",
00190                       stat_regB.aie ? "alarm" : "",
00191                       stat_regB.uie ? "update" : "");
00192                 panic_unsupported = true;
00193             }
00194 
00195             if (stat_regB.dm) {
00196                 inform("RTC: The binary interface is not fully implemented.\n");
00197                 panic_unsupported = true;
00198             }
00199 
00200             if (!stat_regB.format24h) {
00201                 inform("RTC: The 12h time format not supported.\n");
00202                 panic_unsupported = true;
00203             }
00204 
00205             if (stat_regB.dse) {
00206                 inform("RTC: Automatic daylight saving time not supported.\n");
00207                 panic_unsupported = true;
00208             }
00209 
00210             if (stat_regB.pie) {
00211                 if (!event.scheduled())
00212                     event.scheduleIntr();
00213             } else {
00214                 if (event.scheduled())
00215                     deschedule(event);
00216             }
00217             break;
00218           case RTC_STAT_REGC:
00219           case RTC_STAT_REGD:
00220             panic("RTC status registers C and D are not implemented.\n");
00221             break;
00222         }
00223     }
00224 
00225     if (panic_unsupported)
00226         panic("Unimplemented RTC configuration!\n");
00227 
00228 }

00230 uint8_t
00231 MC146818::readData(uint8_t addr)
00232 {
00233     if (addr < RTC_STAT_REGA)
00234         return clock_data[addr];
00235     else {
00236         switch (addr) {
00237           case RTC_STAT_REGA:
00238             // toggle UIP bit for linux
00239             stat_regA.uip = !stat_regA.uip;
00240             return stat_regA;
00241             break;
00242           case RTC_STAT_REGB:
00243             return stat_regB;
00244             break;
00245           case RTC_STAT_REGC:
00246           case RTC_STAT_REGD:
00247             return 0x00;
00248             break;
00249           default:
00250             panic("Shouldn't be here");
00251         }
00252     }
00253 }

00053 // Shift Rm by an immediate value
00054 int32_t
00055 ArmStaticInst::shift_rm_imm(uint32_t base, uint32_t shamt,
00056                                 uint32_t type, uint32_t cfval) const
00057 {
00058     assert(shamt < 32);
00059     ArmShiftType shiftType;
00060     shiftType = (ArmShiftType)type;
00061 
00062     switch (shiftType)
00063     {
00064       case LSL:
00065         return base << shamt;
00066       case LSR:
00067         if (shamt == 0)
00068             return 0;
00069         else
00070             return base >> shamt;
00071       case ASR:
00072         if (shamt == 0)
00073             return (base >> 31) | -((base & (1 << 31)) >> 31);
00074         else
00075             return (base >> shamt) | -((base & (1 << 31)) >> shamt);
00076       case ROR:
00077         if (shamt == 0)
00078             return (cfval << 31) | (base >> 1); // RRX
00079         else
00080             return (base << (32 - shamt)) | (base >> shamt);
00081       default:
00082         ccprintf(std::cerr, "Unhandled shift type\n");
00083         exit(1);
00084         break;
00085     }
00086     return 0;
00087 }

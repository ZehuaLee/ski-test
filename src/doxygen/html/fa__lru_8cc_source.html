00057 FALRU::FALRU(const Params *p)
00058     : BaseTags(p), cacheBoundaries(nullptr)
00059 {
00060     if (!isPowerOf2(blkSize))
00061         fatal("cache block size (in bytes) `%d' must be a power of two",
00062               blkSize);
00063     if (!isPowerOf2(size))
00064         fatal("Cache Size must be power of 2 for now");
00065 
00066     // Track all cache sizes from 128K up by powers of 2
00067     numCaches = floorLog2(size) - 17;
00068     if (numCaches >0){
00069         cacheBoundaries = new FALRUBlk *[numCaches];
00070         cacheMask = (1 << numCaches) - 1;
00071     } else {
00072         cacheMask = 0;
00073     }
00074 
00075     warmupBound = size/blkSize;
00076     numBlocks = size/blkSize;
00077 
00078     blks = new FALRUBlk[numBlocks];
00079     head = &(blks[0]);
00080     tail = &(blks[numBlocks-1]);
00081 
00082     head->prev = NULL;
00083     head->next = &(blks[1]);
00084     head->inCache = cacheMask;
00085 
00086     tail->prev = &(blks[numBlocks-2]);
00087     tail->next = NULL;
00088     tail->inCache = 0;
00089 
00090     unsigned index = (1 << 17) / blkSize;
00091     unsigned j = 0;
00092     int flags = cacheMask;
00093     for (unsigned i = 1; i < numBlocks - 1; i++) {
00094         blks[i].inCache = flags;
00095         if (i == index - 1){
00096             cacheBoundaries[j] = &(blks[i]);
00097             flags &= ~ (1<<j);
00098             ++j;
00099             index = index << 1;
00100         }
00101         blks[i].prev = &(blks[i-1]);
00102         blks[i].next = &(blks[i+1]);
00103         blks[i].isTouched = false;
00104     }
00105     assert(j == numCaches);
00106     assert(index == numBlocks);
00107     //assert(check());
00108 }

00256 void
00257 FALRU::moveToHead(FALRUBlk *blk)
00258 {
00259     int updateMask = blk->inCache ^ cacheMask;
00260     for (unsigned i = 0; i < numCaches; i++){
00261         if ((1<<i) & updateMask) {
00262             cacheBoundaries[i]->inCache &= ~(1<<i);
00263             cacheBoundaries[i] = cacheBoundaries[i]->prev;
00264         } else if (cacheBoundaries[i] == blk) {
00265             cacheBoundaries[i] = blk->prev;
00266         }
00267     }
00268     blk->inCache = cacheMask;
00269     if (blk != head) {
00270         if (blk == tail){
00271             assert(blk->next == NULL);
00272             tail = blk->prev;
00273             tail->next = NULL;
00274         } else {
00275             blk->prev->next = blk->next;
00276             blk->next->prev = blk->prev;
00277         }
00278         blk->next = head;
00279         blk->prev = NULL;
00280         head->prev = blk;

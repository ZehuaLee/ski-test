00121     ForwardInstData &insts_out = *out.inputWire;
00122 
00123     assert(insts_out.isBubble());
00124 
00125     blocked = false;
00126 
00127     if (!nextStageReserve.canReserve()) {
00128         blocked = true;
00129     } else {
00130         const ForwardInstData *insts_in = getInput();
00131 
00132         unsigned int output_index = 0;
00133 
00134         /* Pack instructions into the output while we can.  This may involve
00135          * using more than one input line */

00137            inputIndex < insts_in->width() && /* Still more input */
00138            output_index < outputWidth /* Still more output to fill */)
00139         {
00140             MinorDynInstPtr inst = insts_in->insts[inputIndex];
00141 
00142             if (inst->isBubble()) {
00143                 /* Skip */
00144                 inputIndex++;
00145                 inMacroop = false;
00146             } else {
00147                 StaticInstPtr static_inst = inst->staticInst;
00148                 /* Static inst of a macro-op above the output_inst */
00149                 StaticInstPtr parent_static_inst = NULL;
00150                 MinorDynInstPtr output_inst = inst;
00151 
00152                 if (inst->isFault()) {
00153                     DPRINTF(Decode, "Fault being passed: %d\n",
00154                         inst->fault->name());
00155 
00156                     inputIndex++;
00157                     inMacroop = false;
00158                 } else if (static_inst->isMacroop()) {
00159                     /* Generate a new micro-op */
00160                     StaticInstPtr static_micro_inst;
00161 
00162                     /* Set up PC for the next micro-op emitted */
00163                     if (!inMacroop) {
00164                         microopPC = inst->pc;
00165                         inMacroop = true;
00166                     }
00167 
00168                     /* Get the micro-op static instruction from the
00169                      * static_inst. */
00170                     static_micro_inst =
00171                         static_inst->fetchMicroop(microopPC.microPC());
00172 
00173                     output_inst = new MinorDynInst(inst->id);
00174                     output_inst->pc = microopPC;
00175                     output_inst->staticInst = static_micro_inst;
00176                     output_inst->fault = NoFault;
00177 
00178                     /* Allow a predicted next address only on the last
00179                      *  microop */
00180                     if (static_micro_inst->isLastMicroop()) {
00181                         output_inst->predictedTaken = inst->predictedTaken;
00182                         output_inst->predictedTarget = inst->predictedTarget;
00183                     }
00184 
00185                     DPRINTF(Decode, "Microop decomposition inputIndex:"
00186                         " %d output_index: %d lastMicroop: %s microopPC:"
00187                         " %d.%d inst: %d\n",
00188                         inputIndex, output_index,
00189                         (static_micro_inst->isLastMicroop() ?
00190                             "true" : "false"),
00191                         microopPC.instAddr(), microopPC.microPC(),
00192                         *output_inst);
00193 
00194                     /* Acknowledge that the static_inst isn't mine, it's my
00195                      * parent macro-op's */
00196                     parent_static_inst = static_inst;
00197 
00198                     static_micro_inst->advancePC(microopPC);
00199 
00200                     /* Step input if this is the last micro-op */
00201                     if (static_micro_inst->isLastMicroop()) {
00202                         inputIndex++;
00203                         inMacroop = false;
00204                     }
00205                 } else {
00206                     /* Doesn't need decomposing, pass on instruction */
00207                     DPRINTF(Decode, "Passing on inst: %s inputIndex:"
00208                         " %d output_index: %d\n",
00209                         *output_inst, inputIndex, output_index);
00210 
00211                     parent_static_inst = static_inst;
00212 
00213                     /* Step input */
00214                     inputIndex++;
00215                     inMacroop = false;
00216                 }
00217 
00218                 /* Set execSeqNum of output_inst */
00219                 output_inst->id.execSeqNum = execSeqNum;
00220                 /* Add tracing */
00221 #if TRACING_ON
00222                 dynInstAddTracing(output_inst, parent_static_inst, cpu);
00223 #endif
00224 
00225                 /* Step to next sequence number */
00226                 execSeqNum++;
00227 
00228                 /* Correctly size the output before writing */
00229                 if(output_index == 0) insts_out.resize(outputWidth);
00230                 /* Push into output */
00231                 insts_out.insts[output_index] = output_inst;
00232                 output_index++;
00233             }
00234 
00235             /* Have we finished with the input? */
00236             if (inputIndex == insts_in->width()) {
00237                 /* If we have just been producing micro-ops, we *must* have
00238                  * got to the end of that for inputIndex to be pushed past
00239                  * insts_in->width() */
00240                 assert(!inMacroop);
00241                 popInput();
00242                 insts_in = NULL;
00243 
00244                 if (processMoreThanOneInput) {
00245                     DPRINTF(Decode, "Wrapping\n");
00246                     insts_in = getInput();
00247                 }
00248             }
00249         }
00250 
00251         /* The rest of the output (if any) should already have been packed
00252          *  with bubble instructions by insts_out's initialisation
00253          *
00254          *  for (; output_index < outputWidth; output_index++)
00255          *      assert(insts_out.insts[output_index]->isBubble());
00256          */
00257     }
00258 
00259     /* If we generated output, reserve space for the result in the next stage
00260      *  and mark the stage as being active this cycle */
00261     if (!insts_out.isBubble()) {
00262         /* Note activity of following buffer */
00263         cpu.activityRecorder->activity();
00264         nextStageReserve.reserve();
00265     }
00266 
00267     /* If we still have input to process and somewhere to put it,
00268      *  mark stage as active */
00269     if (getInput() && nextStageReserve.canReserve())
00270         cpu.activityRecorder->activateStage(Pipeline::DecodeStageId);
00271 
00272     /* Make sure the input (if any left) is pushed */
00273     inputBuffer.pushTail();
00274 }

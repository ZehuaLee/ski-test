00158 VirtIO9PBase::sendRMsg(const P9MsgHeader &header, const uint8_t *data, size_t size)
00159 {
00160     DPRINTF(VIO9P, "Sending RMsg\n");
00161     dumpMsg(header, data, size);
00162     DPRINTF(VIO9P, "\tPending transactions: %i\n", pendingTransactions.size());
00163     assert(header.len >= sizeof(header));
00164 
00165     VirtDescriptor *main_desc(pendingTransactions[header.tag]);
00166     pendingTransactions.erase(header.tag);
00167 
00168     // Find the first output descriptor
00169     VirtDescriptor *out_desc(main_desc);
00170     while (out_desc && !out_desc->isOutgoing())
00171         out_desc = out_desc->next();
00172     if (!out_desc)
00173         panic("sendRMsg: Framing error, no output descriptor.\n");
00174 
00175     P9MsgHeader header_out(htop9(header));
00176     header_out.len = htop9(sizeof(P9MsgHeader) + size);
00177 
00178     out_desc->chainWrite(0, (uint8_t *)&header_out, sizeof(header_out));
00179     out_desc->chainWrite(sizeof(header_out), data, size);
00180 
00181     queue.produceDescriptor(main_desc, sizeof(P9MsgHeader) + size);
00182     kick();

00231                         const uint8_t *data, size_t size)
00232 {
00233     assert(header.len == sizeof(header) + size);
00234     // While technically not needed, we send the packet as one

00329         panic("Fork failed: %i\n", errno);
00330     } else if (diod_pid == 0) {
00331         close(STDIN_FILENO);
00332

00334             dup2(pipe_wfd[1], DIOD_WFD) == -1) {

00336             panic("Failed to setup read/write pipes: %i\n",
00337                   errno);
00338         }
00339

00341                "-f", // start in foreground
00342                "-r", "3", // setup read FD
00343                "-w", "4", // setup write FD

00345                "-n", // disable security
00346                "-S", // squash all users
00347                (char *)NULL);

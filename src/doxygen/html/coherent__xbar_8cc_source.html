00141     // determine the source port based on the id
00142     SlavePort *src_port = slavePorts[slave_port_id];
00143 
00144     // remember if the packet is an express snoop
00145     bool is_express_snoop = pkt->isExpressSnoop();
00146     bool is_inhibited = pkt->memInhibitAsserted();
00147     // for normal requests, going downstream, the express snoop flag
00148     // and the inhibited flag should always be the same
00149     assert(is_express_snoop == is_inhibited);
00150 
00151     // determine the destination based on the address
00152     PortID master_port_id = findPort(pkt->getAddr());
00153 
00154     // test if the crossbar should be considered occupied for the current
00155     // port, and exclude express snoops from the check
00156     if (!is_express_snoop && !reqLayers[master_port_id]->tryTiming(src_port)) {
00157         DPRINTF(CoherentXBar, "recvTimingReq: src %s %s 0x%x BUSY\n",
00158                 src_port->name(), pkt->cmdString(), pkt->getAddr());
00159         return false;
00160     }
00161 
00162     DPRINTF(CoherentXBar, "recvTimingReq: src %s %s expr %d 0x%x\n",
00163             src_port->name(), pkt->cmdString(), is_express_snoop,
00164             pkt->getAddr());
00165 
00166     // store size and command as they might be modified when
00167     // forwarding the packet
00168     unsigned int pkt_size = pkt->hasData() ? pkt->getSize() : 0;
00169     unsigned int pkt_cmd = pkt->cmdToIndex();
00170 
00171     // store the old header delay so we can restore it if needed
00172     Tick old_header_delay = pkt->headerDelay;
00173 
00174     // a request sees the frontend and forward latency
00175     Tick xbar_delay = (frontendLatency + forwardLatency) * clockPeriod();
00176 
00177     // set the packet header and payload delay
00178     calcPacketTiming(pkt, xbar_delay);
00179 
00180     // determine how long to be crossbar layer is busy
00181     Tick packetFinishTime = clockEdge(Cycles(1)) + pkt->payloadDelay;
00182 
00183     if (!system->bypassCaches()) {
00184         // the packet is a memory-mapped request and should be
00185         // broadcasted to our snoopers but the source
00186         if (snoopFilter) {
00187             // check with the snoop filter where to forward this packet
00188             auto sf_res = snoopFilter->lookupRequest(pkt, *src_port);
00189             // If SnoopFilter is enabled, the total time required by a packet
00190             // to be delivered through the xbar has to be charged also with
00191             // to lookup latency of the snoop filter (sf_res.second).
00192             pkt->headerDelay += sf_res.second * clockPeriod();
00193             packetFinishTime += sf_res.second * clockPeriod();
00194             DPRINTF(CoherentXBar, "recvTimingReq: src %s %s 0x%x"\
00195                     " SF size: %i lat: %i\n", src_port->name(),
00196                     pkt->cmdString(), pkt->getAddr(), sf_res.first.size(),
00197                     sf_res.second);
00198             forwardTiming(pkt, slave_port_id, sf_res.first);
00199         } else {
00200             forwardTiming(pkt, slave_port_id);
00201         }
00202     }
00203 
00204     // remember if the packet will generate a snoop response
00205     const bool expect_snoop_resp = !is_inhibited && pkt->memInhibitAsserted();
00206     const bool expect_response = pkt->needsResponse() &&
00207         !pkt->memInhibitAsserted();
00208 
00209     // Note: Cannot create a copy of the full packet, here.
00210     MemCmd orig_cmd(pkt->cmd);
00211 
00212     // since it is a normal request, attempt to send the packet
00213     bool success = masterPorts[master_port_id]->sendTimingReq(pkt);
00214 
00215     if (snoopFilter && !system->bypassCaches()) {
00216         // The packet may already be overwritten by the sendTimingReq function.
00217         // The snoop filter needs to see the original request *and* the return
00218         // status of the send operation, so we need to recreate the original
00219         // request.  Atomic mode does not have the issue, as there the send
00220         // operation and the response happen instantaneously and don't need two
00221         // phase tracking.
00222         MemCmd tmp_cmd(pkt->cmd);
00223         pkt->cmd = orig_cmd;
00224         // Let the snoop filter know about the success of the send operation
00225         snoopFilter->updateRequest(pkt, *src_port, !success);
00226         pkt->cmd = tmp_cmd;
00227     }
00228 
00229     // check if we were successful in sending the packet onwards
00230     if (!success)  {
00231         // express snoops and inhibited packets should never be forced
00232         // to retry
00233         assert(!is_express_snoop);
00234         assert(!pkt->memInhibitAsserted());
00235 
00236         // restore the header delay
00237         pkt->headerDelay = old_header_delay;
00238 
00239         DPRINTF(CoherentXBar, "recvTimingReq: src %s %s 0x%x RETRY\n",
00240                 src_port->name(), pkt->cmdString(), pkt->getAddr());
00241 
00242         // update the layer state and schedule an idle event
00243         reqLayers[master_port_id]->failedTiming(src_port,
00244                                                 clockEdge(Cycles(1)));
00245     } else {
00246         // express snoops currently bypass the crossbar state entirely
00247         if (!is_express_snoop) {
00248             // if this particular request will generate a snoop
00249             // response
00250             if (expect_snoop_resp) {
00251                 // we should never have an exsiting request outstanding
00252                 assert(outstandingSnoop.find(pkt->req) ==
00253                        outstandingSnoop.end());
00254                 outstandingSnoop.insert(pkt->req);
00255 
00256                 // basic sanity check on the outstanding snoops
00257                 panic_if(outstandingSnoop.size() > 512,
00258                          "Outstanding snoop requests exceeded 512\n");
00259             }
00260 
00261             // remember where to route the normal response to
00262             if (expect_response || expect_snoop_resp) {
00263                 assert(routeTo.find(pkt->req) == routeTo.end());
00264                 routeTo[pkt->req] = slave_port_id;
00265 
00266                 panic_if(routeTo.size() > 512,
00267                          "Routing table exceeds 512 packets\n");
00268             }
00269 
00270             // update the layer state and schedule an idle event
00271             reqLayers[master_port_id]->succeededTiming(packetFinishTime);
00272         }
00273 
00274         // stats updates only consider packets that were successfully sent
00275         pktCount[slave_port_id][master_port_id]++;
00276         pktSize[slave_port_id][master_port_id] += pkt_size;
00277         transDist[pkt_cmd]++;
00278 
00279         if (is_express_snoop)
00280             snoops++;
00281     }
00282 
00283     return success;
00284 }

00403     assert(route_lookup != routeTo.end());
00404     const PortID dest_port_id = route_lookup->second;
00405     assert(dest_port_id != InvalidPortID);
00406 
00407     // determine if the response is from a snoop request we
00408     // created as the result of a normal request (in which case it
00409     // should be in the outstandingSnoop), or if we merely forwarded
00410     // someone else's snoop request

00412         outstandingSnoop.end();

00415     // current port, note that the check is bypassed if the response
00416     // is being passed on as a normal response since this is occupying
00417     // the response layer rather than the snoop response layer
00418     if (forwardAsSnoop) {

00420         if (!snoopLayers[dest_port_id]->tryTiming(src_port)) {
00421             DPRINTF(CoherentXBar, "recvTimingSnoopResp: src %s %s 0x%x BUSY\n",
00422                     src_port->name(), pkt->cmdString(), pkt->getAddr());
00423             return false;
00424         }

00427         MasterPort* snoop_port = snoopRespPorts[slave_port_id];

00430             DPRINTF(CoherentXBar, "recvTimingSnoopResp: src %s %s 0x%x BUSY\n",

00432             return false;

00536     assert(!system->bypassCaches());
00537 
00538     unsigned fanout = 0;
00539 
00540     for (const auto& p: dests) {
00541         // we could have gotten this request from a snooping master

00544         // from

00546             p->getId() != exclude_slave_port_id) {
00547             // cache is not allowed to refuse snoop
00548             p->sendTimingSnoopReq(pkt);
00549             fanout++;
00550         }
00551     }
00552

00554     snoopFanout.sample(fanout);

00681 std::pair<MemCmd, Tick>
00682 CoherentXBar::forwardAtomic(PacketPtr pkt, PortID exclude_slave_port_id,
00683                            PortID source_master_port_id,
00684                            const std::vector<SlavePort*>& dests)
00685 {
00686     // the packet may be changed on snoops, record the original
00687     // command to enable us to restore it between snoops so that
00688     // additional snoops can take place properly
00689     MemCmd orig_cmd = pkt->cmd;
00690     MemCmd snoop_response_cmd = MemCmd::InvalidCmd;
00691     Tick snoop_response_latency = 0;
00692 
00693     // snoops should only happen if the system isn't bypassing caches
00694     assert(!system->bypassCaches());
00695 
00696     unsigned fanout = 0;
00697 
00698     for (const auto& p: dests) {
00699         // we could have gotten this request from a snooping master
00700         // (corresponding to our own slave port that is also in
00701         // snoopPorts) and should not send it back to where it came
00702         // from
00703         if (exclude_slave_port_id != InvalidPortID &&
00704             p->getId() == exclude_slave_port_id)
00705             continue;
00706 
00707         Tick latency = p->sendAtomicSnoop(pkt);
00708         fanout++;
00709 
00710         // in contrast to a functional access, we have to keep on
00711         // going as all snoopers must be updated even if we get a
00712         // response
00713         if (!pkt->isResponse())
00714             continue;
00715 
00716         // response from snoop agent
00717         assert(pkt->cmd != orig_cmd);
00718         assert(pkt->memInhibitAsserted());
00719         // should only happen once
00720         assert(snoop_response_cmd == MemCmd::InvalidCmd);
00721         // save response state
00722         snoop_response_cmd = pkt->cmd;
00723         snoop_response_latency = latency;
00724 
00725         if (snoopFilter) {
00726             // Handle responses by the snoopers and differentiate between
00727             // responses to requests from above and snoops from below
00728             if (source_master_port_id != InvalidPortID) {
00729                 // Getting a response for a snoop from below
00730                 assert(exclude_slave_port_id == InvalidPortID);
00731                 snoopFilter->updateSnoopForward(pkt, *p,
00732                              *masterPorts[source_master_port_id]);
00733             } else {
00734                 // Getting a response for a request from above
00735                 assert(source_master_port_id == InvalidPortID);
00736                 snoopFilter->updateSnoopResponse(pkt, *p,
00737                              *slavePorts[exclude_slave_port_id]);
00738             }
00739         }
00740         // restore original packet state for remaining snoopers
00741         pkt->cmd = orig_cmd;
00742     }
00743 
00744     // Stats for fanout
00745     snoopFanout.sample(fanout);
00746 
00747     // the packet is restored as part of the loop and any potential
00748     // snoop response is part of the returned pair
00749     return std::make_pair(snoop_response_cmd, snoop_response_latency);
00750 }

00097 bool
00098 SimpleMemory::recvTimingReq(PacketPtr pkt)
00099 {
00102     for (int x = 0; x < pendingDelete.size(); x++)
00103         delete pendingDelete[x];
00104     pendingDelete.clear();
00105 
00106     if (pkt->memInhibitAsserted()) {
00107         // snooper will supply based on copy of packet
00108         // still target's responsibility to delete packet
00109         pendingDelete.push_back(pkt);
00110         return true;
00111     }
00112 
00113     // we should never get a new request after committing to retry the
00114     // current one, the bus violates the rule as it simply sends a
00115     // retry to the next one waiting on the retry list, so simply
00116     // ignore it
00117     if (retryReq)
00118         return false;
00119 
00120     // if we are busy with a read or write, remember that we have to
00121     // retry
00122     if (isBusy) {
00123         retryReq = true;
00124         return false;
00125     }
00126 
00127     // @todo someone should pay for this
00128     pkt->headerDelay = pkt->payloadDelay = 0;
00129 
00130     // update the release time according to the bandwidth limit, and
00131     // do so with respect to the time it takes to finish this request
00132     // rather than long term as it is the short term data rate that is
00133     // limited for any real memory
00134 
00135     // only look at reads and writes when determining if we are busy,
00136     // and for how long, as it is not clear what to regulate for the
00137     // other types of commands
00138     if (pkt->isRead() || pkt->isWrite()) {
00139         // calculate an appropriate tick to release to not exceed
00140         // the bandwidth limit
00141         Tick duration = pkt->getSize() * bandwidth;
00142 
00143         // only consider ourselves busy if there is any need to wait
00144         // to avoid extra events being scheduled for (infinitely) fast
00145         // memories
00146         if (duration != 0) {
00147             schedule(releaseEvent, curTick() + duration);
00148             isBusy = true;
00149         }
00150     }
00151 
00152     // go ahead and deal with the packet and put the response in the
00153     // queue if there is one
00154     bool needsResponse = pkt->needsResponse();
00155     recvAtomic(pkt);
00156     // turn packet around to go back to requester if response expected
00157     if (needsResponse) {
00158         // recvAtomic() should already have turned packet into
00159         // atomic response
00160         assert(pkt->isResponse());
00161         // to keep things simple (and in order), we put the packet at
00162         // the end even if the latency suggests it should be sent
00163         // before the packet(s) before it
00164         packetQueue.emplace_back(DeferredPacket(pkt, curTick() + getLatency()));
00165         if (!retryResp && !dequeueEvent.scheduled())
00166             schedule(dequeueEvent, packetQueue.back().tick);
00167     } else {
00168         pendingDelete.push_back(pkt);
00169     }
00170 
00171     return true;
00172 }

00185 void
00186 SimpleMemory::dequeue()
00187 {
00188     assert(!packetQueue.empty());
00189     DeferredPacket deferred_pkt = packetQueue.front();
00190 
00191     retryResp = !port.sendTimingResp(deferred_pkt.pkt);
00192 
00193     if (!retryResp) {
00194         packetQueue.pop_front();
00195 
00196         // if the queue is not empty, schedule the next dequeue event,
00197         // otherwise signal that we are drained if we were asked to do so
00198         if (!packetQueue.empty()) {
00199             // if there were packets that got in-between then we
00200             // already have an event scheduled, so use re-schedule
00201             reschedule(dequeueEvent,
00202                        std::max(packetQueue.front().tick, curTick()), true);
00203         } else if (drainManager) {
00204             DPRINTF(Drain, "Drainng of SimpleMemory complete\n");
00205             drainManager->signalDrainDone();
00206             drainManager = NULL;
00207         }
00208     }

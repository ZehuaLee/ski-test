00099 void
00100 StridePrefetcher::calculatePrefetch(const PacketPtr &pkt,
00101                                     std::vector<Addr> &addresses)
00102 {
00103     if (!pkt->req->hasPC()) {
00104         DPRINTF(HWPrefetch, "Ignoring request with no PC.\n");
00105         return;
00106     }
00107 
00108     // Get required packet info
00109     Addr pkt_addr = pkt->getAddr();
00110     Addr pc = pkt->req->getPC();
00111     bool is_secure = pkt->isSecure();
00112     MasterID master_id = useMasterId ? pkt->req->masterId() : 0;
00113 
00114     // Lookup pc-based information
00115     StrideEntry *entry;
00116 
00117     if(pcTableHit(pc, is_secure, master_id, entry)) {
00118         // Hit in table
00119         int new_stride = pkt_addr - entry->lastAddr;
00120         bool stride_match = (new_stride == entry->stride);
00121 
00122         // Adjust confidence for stride entry
00123         if (stride_match && new_stride != 0) {
00124             if (entry->confidence < maxConf)
00125                 entry->confidence++;
00126         } else {
00127             if (entry->confidence > minConf)
00128                 entry->confidence--;
00129             // If confidence has dropped below the threshold, train new stride
00130             if (entry->confidence < threshConf)
00131                 entry->stride = new_stride;
00132         }
00133 
00134         DPRINTF(HWPrefetch, "Hit: PC %x pkt_addr %x (%s) stride %d (%s), "
00135                 "conf %d\n", pc, pkt_addr, is_secure ? "s" : "ns", new_stride,
00136                 stride_match ? "match" : "change",
00137                 entry->confidence);
00138 
00139         entry->lastAddr = pkt_addr;
00140 
00141         // Abort prefetch generation if below confidence threshold
00142         if (entry->confidence < threshConf)
00143             return;
00144 
00145         // Generate up to degree prefetches
00146         for (int d = 1; d <= degree; d++) {
00147             // Round strides up to atleast 1 cacheline
00148             int prefetch_stride = new_stride;
00149             if (abs(new_stride) < blkSize) {
00150                 prefetch_stride = (new_stride < 0) ? -blkSize : blkSize;
00151             }
00152 
00153             Addr new_addr = pkt_addr + d * prefetch_stride;
00154             if (samePage(pkt_addr, new_addr)) {
00155                 DPRINTF(HWPrefetch, "Queuing prefetch to %#x.\n", new_addr);
00156                 addresses.push_back(new_addr);
00157             } else {
00158                 // Record the number of page crossing prefetches generated
00159                 pfSpanPage += degree - d + 1;
00160                 DPRINTF(HWPrefetch, "Ignoring page crossing prefetch.\n");
00161                 return;
00162             }
00163         }
00164     } else {
00165         // Miss in table
00166         DPRINTF(HWPrefetch, "Miss: PC %x pkt_addr %x (%s)\n", pc, pkt_addr,
00167                 is_secure ? "s" : "ns");
00168 
00169         StrideEntry* entry = pcTableVictim(pc, master_id);
00170         entry->instAddr = pc;
00171         entry->lastAddr = pkt_addr;
00172         entry->isSecure= is_secure;
00173         entry->stride = 0;
00174         entry->confidence = startConf;
00175     }
00176 }

00178 inline Addr
00179 StridePrefetcher::pcHash(Addr pc) const
00180 {
00181     Addr hash1 = pc >> 1;
00182     Addr hash2 = hash1 >> floorLog2(pcTableSets);
00183     return (hash1 ^ hash2) & (Addr)(pcTableSets - 1);

00197 inline bool
00198 StridePrefetcher::pcTableHit(Addr pc, bool is_secure, int master_id,
00199                              StrideEntry* &entry)
00200 {
00201     int set = pcHash(pc);
00202     StrideEntry* set_entries = pcTable[master_id][set];
00203     for (int way = 0; way < pcTableAssoc; way++) {
00204         // Search ways for match
00205         if (set_entries[way].instAddr == pc &&
00206             set_entries[way].isSecure == is_secure) {
00207             DPRINTF(HWPrefetch, "Lookup hit table[%d][%d].\n", set, way);
00208             entry = &set_entries[way];
00209             return true;
00210         }
00211     }
00212     return false;
00213 }

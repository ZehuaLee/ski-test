00068 MSHR *
00069 MSHRQueue::findMatch(Addr blk_addr, bool is_secure) const
00070 {
00071     for (const auto& mshr : allocatedList) {
00072         // we ignore any MSHRs allocated for uncacheable accesses and
00073         // simply ignore them when matching, in the cache we never
00074         // check for matches when adding new uncacheable entries, and
00075         // we do not want normal cacheable accesses being added to an
00076         // MSHR serving an uncacheable access
00077         if (!mshr->isUncacheable() && mshr->blkAddr == blk_addr &&
00078             mshr->isSecure == is_secure) {
00079             return mshr;
00080         }
00081     }
00082     return NULL;
00083 }

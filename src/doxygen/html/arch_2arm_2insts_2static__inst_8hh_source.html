00193     static inline uint32_t
00194     cpsrWriteByInstr(CPSR cpsr, uint32_t val, SCR scr, NSACR nsacr,
00195             uint8_t byteMask, bool affectState, bool nmfi, ThreadContext *tc)
00196     {
00197         bool privileged   = (cpsr.mode != MODE_USER);
00198         bool haveVirt     = ArmSystem::haveVirtualization(tc);
00199         bool haveSecurity = ArmSystem::haveSecurity(tc);
00200         bool isSecure     = inSecureState(scr, cpsr) || !haveSecurity;
00201 
00202         uint32_t bitMask = 0;
00203 
00204         if (bits(byteMask, 3)) {
00205             unsigned lowIdx = affectState ? 24 : 27;
00206             bitMask = bitMask | mask(31, lowIdx);
00207         }
00208         if (bits(byteMask, 2)) {
00209             bitMask = bitMask | mask(19, 16);
00210         }
00211         if (bits(byteMask, 1)) {
00212             unsigned highIdx = affectState ? 15 : 9;
00213             unsigned lowIdx = (privileged && (isSecure || scr.aw || haveVirt))
00214                             ? 8 : 9;
00215             bitMask = bitMask | mask(highIdx, lowIdx);
00216         }
00217         if (bits(byteMask, 0)) {
00218             if (privileged) {
00219                 bitMask |= 1 << 7;
00220                 if ( (!nmfi || !((val >> 6) & 0x1)) &&
00221                      (isSecure || scr.fw || haveVirt) ) {
00222                     bitMask |= 1 << 6;
00223                 }
00224                 // Now check the new mode is allowed
00225                 OperatingMode newMode = (OperatingMode) (val & mask(5));
00226                 OperatingMode oldMode = (OperatingMode)(uint32_t)cpsr.mode;
00227                 if (!badMode(newMode)) {
00228                     bool validModeChange = true;
00229                     // Check for attempts to enter modes only permitted in
00230                     // Secure state from Non-secure state. These are Monitor
00231                     // mode ('10110'), and FIQ mode ('10001') if the Security
00232                     // Extensions have reserved it.
00233                     if (!isSecure && newMode == MODE_MON)
00234                         validModeChange = false;
00235                     if (!isSecure && newMode == MODE_FIQ && nsacr.rfr == '1')
00236                         validModeChange = false;
00237                     // There is no Hyp mode ('11010') in Secure state, so that
00238                     // is UNPREDICTABLE
00239                     if (scr.ns == '0' && newMode == MODE_HYP)
00240                         validModeChange = false;
00241                     // Cannot move into Hyp mode directly from a Non-secure
00242                     // PL1 mode
00243                     if (!isSecure && oldMode != MODE_HYP && newMode == MODE_HYP)
00244                         validModeChange = false;
00245                     // Cannot move out of Hyp mode with this function except
00246                     // on an exception return
00247                     if (oldMode == MODE_HYP && newMode != MODE_HYP && !affectState)
00248                         validModeChange = false;
00249                     // Must not change to 64 bit when running in 32 bit mode
00250                     if (!opModeIs64(oldMode) && opModeIs64(newMode))
00251                         validModeChange = false;
00252 
00253                     // If we passed all of the above then set the bit mask to
00254                     // copy the mode accross
00255                     if (validModeChange) {
00256                         bitMask = bitMask | mask(5);
00257                     } else {
00258                         warn_once("Illegal change to CPSR mode attempted\n");
00259                     }
00260                 } else {
00261                     warn_once("Ignoring write of bad mode to CPSR.\n");
00262                 }
00263             }
00264             if (affectState)
00265                 bitMask = bitMask | (1 << 5);
00266         }
00267 
00268         return ((uint32_t)cpsr & ~bitMask) | (val & bitMask);
00269     }

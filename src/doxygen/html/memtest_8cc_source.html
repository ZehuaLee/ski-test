00136 {
00137     Request *req = pkt->req;
00138     assert(req->getSize() == 1);
00139 
00140     // this address is no longer outstanding
00141     auto remove_addr = outstandingAddrs.find(req->getPaddr());
00142     assert(remove_addr != outstandingAddrs.end());
00143     outstandingAddrs.erase(remove_addr);
00144 
00145     DPRINTF(MemTest, "Completing %s at address %x (blk %x) %s\n",
00146             pkt->isWrite() ? "write" : "read",
00147             req->getPaddr(), blockAlign(req->getPaddr()),
00148             pkt->isError() ? "error" : "success");
00149 
00150     const uint8_t *pkt_data = pkt->getConstPtr<uint8_t>();
00151 
00152     if (pkt->isError()) {
00153         if (!functional || !suppressFuncWarnings) {
00154             warn("%s access failed at %#x\n",
00155                  pkt->isWrite() ? "Write" : "Read", req->getPaddr());
00156         }
00157     } else {
00158         if (pkt->isRead()) {
00159             uint8_t ref_data = referenceData[req->getPaddr()];
00160             if (pkt_data[0] != ref_data) {
00161                 panic("%s: read of %x (blk %x) @ cycle %d "
00162                       "returns %x, expected %x\n", name(),
00163                       req->getPaddr(), blockAlign(req->getPaddr()), curTick(),
00164                       pkt_data[0], ref_data);
00165             }
00166 
00167             numReads++;
00168             numReadsStat++;
00169 
00170             if (numReads == (uint64_t)nextProgressMessage) {
00171                 ccprintf(cerr, "%s: completed %d read, %d write accesses @%d\n",
00172                          name(), numReads, numWrites, curTick());
00173                 nextProgressMessage += progressInterval;
00174             }
00175 
00176             if (maxLoads != 0 && numReads >= maxLoads)
00177                 exitSimLoop("maximum number of loads reached");
00178         } else {
00179             assert(pkt->isWrite());
00180 
00181             // update the reference data
00182             referenceData[req->getPaddr()] = pkt_data[0];
00183             numWrites++;
00184             numWritesStat++;
00185         }
00186     }
00187 
00188     delete pkt->req;
00189 
00190     // the packet will delete the data
00191     delete pkt;
00192 
00193     // finally shift the response timeout forward
00194     reschedule(noResponseEvent, clockEdge(progressCheck), true);
00195 }

00247

00251     panic_if(outstandingAddrs.size() > 100,
00252              "Tester %s has more than 100 outstanding requests\n", name());

00256 
00257     if (cmd < percentReads) {
00258         // start by ensuring there is a reference value if we have not
00259         // seen this address before
00260         uint8_t M5_VAR_USED ref_data = 0;
00261         auto ref = referenceData.find(req->getPaddr());
00262         if (ref == referenceData.end()) {
00263             referenceData[req->getPaddr()] = 0;

00272

00279 
00280         pkt = new Packet(req, MemCmd::WriteReq);

00282         pkt_data[0] = data;
00283     }
00284 
00285     // there is no point in ticking if we are waiting for a retry
00286     bool keep_ticking = true;
00287     if (do_functional) {
00288         pkt->setSuppressFuncError();
00289         port.sendFunctional(pkt);
00290         completeRequest(pkt, true);
00291     } else {
00292         keep_ticking = sendPkt(pkt);
00293     }

00295     if (keep_ticking) {
00296         // schedule the next tick
00297         schedule(tickEvent, clockEdge(interval));
00298 
00299         // finally shift the timeout for sending of requests forwards
00300         // as we have successfully sent a packet
00301         reschedule(noRequestEvent, clockEdge(progressCheck), true);

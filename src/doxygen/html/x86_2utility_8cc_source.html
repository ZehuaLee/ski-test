00053 getArgument(ThreadContext *tc, int &number, uint16_t size, bool fp)
00054 {
00055     if (fp) {
00056         panic("getArgument(): Floating point arguments not implemented\n");
00057     } else if (size != 8) {
00058         panic("getArgument(): Can only handle 64-bit arguments.\n");
00059     }
00060 
00061     // The first 6 integer arguments are passed in registers, the rest
00062     // are passed on the stack.
00063     const int int_reg_map[] = {
00064         INTREG_RDI, INTREG_RSI, INTREG_RDX,
00065         INTREG_RCX, INTREG_R8, INTREG_R9
00066     };
00067     if (number < sizeof(int_reg_map) / sizeof(*int_reg_map)) {
00068         return tc->readIntReg(int_reg_map[number]);
00069     } else {
00070         panic("getArgument(): Don't know how to handle stack arguments.\n");
00071     }

00213 void
00214 copyMiscRegs(ThreadContext *src, ThreadContext *dest)
00215 {
00216     // This function assumes no side effects other than TLB invalidation
00217     // need to be considered while copying state. That will likely not be
00218     // true in the future.
00219     for (int i = 0; i < NUM_MISCREGS; ++i) {
00220         if ( ( i != MISCREG_CR1 &&
00221              !(i > MISCREG_CR4 && i < MISCREG_CR8) &&
00222              !(i > MISCREG_CR8 && i <= MISCREG_CR15) ) == false) {
00223              continue;
00224         }
00225         dest->setMiscRegNoEffect(i, src->readMiscRegNoEffect(i));
00226     }
00227 
00228     // The TSC has to be updated with side-effects if the CPUs in a
00229     // CPU switch have different frequencies.
00230     dest->setMiscReg(MISCREG_TSC, src->readMiscReg(MISCREG_TSC));
00231 
00232     dest->getITBPtr()->flushAll();
00233     dest->getDTBPtr()->flushAll();

00258 uint64_t
00259 getRFlags(ThreadContext *tc)
00260 {
00261     const uint64_t ncc_flags(tc->readMiscRegNoEffect(MISCREG_RFLAGS));
00262     const uint64_t cc_flags(tc->readCCReg(X86ISA::CCREG_ZAPS));
00263     const uint64_t cfof_bits(tc->readCCReg(X86ISA::CCREG_CFOF));
00264     const uint64_t df_bit(tc->readCCReg(X86ISA::CCREG_DF));
00265     // ecf (PSEUDO(3)) & ezf (PSEUDO(4)) are only visible to
00266     // microcode, so we can safely ignore them.
00267 
00268     // Reconstruct the real rflags state, mask out internal flags, and
00269     // make sure reserved bits have the expected values.
00270     return ((ncc_flags | cc_flags | cfof_bits | df_bit) & 0x3F7FD5)
00271         | 0x2;
00272 }

00290 uint8_t
00291 convX87TagsToXTags(uint16_t ftw)
00292 {
00293     uint8_t ftwx(0);
00294     for (int i = 0; i < 8; ++i) {
00295         // Extract the tag for the current element on the FP stack
00296         const unsigned tag((ftw >> (2 * i)) & 0x3);
00297 
00298         /*
00299          * Check the type of the current FP element. Valid values are:
00300          * 0 == Valid
00301          * 1 == Zero
00302          * 2 == Special (Nan, unsupported, infinity, denormal)
00303          * 3 == Empty
00304          */
00305         // The xsave version of the tag word only keeps track of
00306         // whether the element is empty or not. Set the corresponding
00307         // bit in the ftwx if it's not empty,
00308         if (tag != 0x3)
00309             ftwx |= 1 << i;
00310     }
00311 
00312     return ftwx;
00313 }

00336 uint16_t
00337 genX87Tags(uint16_t ftw, uint8_t top, int8_t spm)
00338 {
00339     const uint8_t new_top((top + spm + 8) % 8);
00340 
00341     if (spm > 0) {
00342         // Removing elements from the stack. Flag the elements as empty.
00343         for (int i = top; i != new_top; i = (i + 1 + 8) % 8)
00344             ftw |= 0x3 << (2 * i);
00345     } else if (spm < 0) {
00346         // Adding elements to the stack. Flag the new elements as
00347         // valid. We should ideally decode them and "do the right
00348         // thing".
00349         for (int i = new_top; i != top; i = (i + 1 + 8) % 8)
00350             ftw &= ~(0x3 << (2 * i));
00351     }
00352 
00353     return ftw;
00354 }

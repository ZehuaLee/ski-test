00069 void
00070 Stage2LookUp::mergeTe(RequestPtr req, BaseTLB::Mode mode)
00071 {
00072     // Since we directly requested the table entry (which we need later on to
00073     // merge the attributes) then we've skipped some stage 2 permissinos
00074     // checking. So call translate on stage 2 to do the checking. As the entry
00075     // is now in the TLB this should always hit the cache.
00076     if (fault == NoFault) {
00077         fault = stage2Tlb->checkPermissions(stage2Te, req, mode);
00078     }
00079 
00080     // Check again that we haven't got a fault
00081     if (fault == NoFault) {
00082         assert(stage2Te != NULL);
00083 
00084         // Now we have the table entries for both stages of translation
00085         // merge them and insert the result into the stage 1 TLB. See
00086         // CombineS1S2Desc() in pseudocode
00087         stage1Te.N             = stage2Te->N;
00088         stage1Te.nonCacheable |= stage2Te->nonCacheable;
00089         stage1Te.xn           |= stage2Te->xn;
00090 
00091         if (stage1Te.size > stage2Te->size) {
00092             // Size mismatch also implies vpn mismatch (this is shifted by
00093             // sizebits!).
00094             stage1Te.vpn  = s1Req->getVaddr() / (stage2Te->size+1);
00095             stage1Te.pfn  = stage2Te->pfn;
00096             stage1Te.size = stage2Te->size;
00097         } else if (stage1Te.size < stage2Te->size) {
00098             // Guest 4K could well be section-backed by host hugepage!  In this
00099             // case a 4K entry is added but pfn needs to be adjusted.  New PFN =
00100             // offset into section PFN given by stage2 IPA treated as a stage1
00101             // page size.
00102             stage1Te.pfn = (stage2Te->pfn * ((stage2Te->size+1) / (stage1Te.size+1))) +
00103                            (stage2Te->vpn / (stage1Te.size+1));
00104             // Size remains smaller of the two.
00105         } else {
00106             // Matching sizes
00107             stage1Te.pfn = stage2Te->pfn;
00108         }
00109 
00110         if (stage2Te->mtype == TlbEntry::MemoryType::StronglyOrdered ||
00111             stage1Te.mtype  == TlbEntry::MemoryType::StronglyOrdered) {
00112             stage1Te.mtype  =  TlbEntry::MemoryType::StronglyOrdered;
00113         } else if (stage2Te->mtype == TlbEntry::MemoryType::Device ||
00114                    stage1Te.mtype  == TlbEntry::MemoryType::Device) {
00115             stage1Te.mtype = TlbEntry::MemoryType::Device;
00116         } else {
00117             stage1Te.mtype = TlbEntry::MemoryType::Normal;
00118         }
00119 
00120         if (stage1Te.mtype == TlbEntry::MemoryType::Normal) {
00121 
00122             if (stage2Te->innerAttrs == 0 ||
00123                 stage1Te.innerAttrs  == 0) {
00124                 // either encoding Non-cacheable
00125                 stage1Te.innerAttrs = 0;
00126             } else if (stage2Te->innerAttrs == 2 ||
00127                        stage1Te.innerAttrs  == 2) {
00128                 // either encoding Write-Through cacheable
00129                 stage1Te.innerAttrs = 2;
00130             } else {
00131                 // both encodings Write-Back
00132                 stage1Te.innerAttrs = 3;
00133             }
00134 
00135             if (stage2Te->outerAttrs == 0 ||
00136                 stage1Te.outerAttrs  == 0) {
00137                 // either encoding Non-cacheable
00138                 stage1Te.outerAttrs = 0;
00139             } else if (stage2Te->outerAttrs == 2 ||
00140                        stage1Te.outerAttrs  == 2) {
00141                 // either encoding Write-Through cacheable
00142                 stage1Te.outerAttrs = 2;
00143             } else {
00144                 // both encodings Write-Back
00145                 stage1Te.outerAttrs = 3;
00146             }
00147 
00148             stage1Te.shareable       |= stage2Te->shareable;
00149             stage1Te.outerShareable |= stage2Te->outerShareable;
00150             if (stage1Te.innerAttrs == 0 &&
00151                 stage1Te.outerAttrs == 0) {
00152                 // something Non-cacheable at each level is outer shareable
00153                 stage1Te.shareable       = true;
00154                 stage1Te.outerShareable = true;
00155             }
00156         } else {
00157             stage1Te.shareable       = true;
00158             stage1Te.outerShareable = true;
00159         }
00160         stage1Te.updateAttributes();
00161     }
00162 
00163     // if there's a fault annotate it,
00164     if (fault != NoFault) {
00165         // If the second stage of translation generated a fault add the
00166         // details of the original stage 1 virtual address
00167         reinterpret_cast<ArmFault *>(fault.get())->annotate(ArmFault::OVA,
00168             s1Req->getVaddr());
00169     }
00170     complete = true;
00171 }

00173 void
00174 Stage2LookUp::finish(const Fault &_fault, RequestPtr req,
00175     ThreadContext *tc, BaseTLB::Mode mode)
00176 {
00177     fault = _fault;
00178     // if we haven't got the table entry get it now
00179     if ((fault == NoFault) && (stage2Te == NULL)) {
00180         fault = stage2Tlb->getTE(&stage2Te, req, tc, mode, this,
00181             timing, functional, false, tranType);
00182     }
00183 
00184     // Now we have the stage 2 table entry we need to merge it with the stage
00185     // 1 entry we were given at the start
00186     mergeTe(req, mode);
00187 
00188     if (fault != NoFault) {
00189         transState->finish(fault, req, tc, mode);
00190     } else if (timing) {
00191         // Now notify the original stage 1 translation that we finally have
00192         // a result
00193         stage1Tlb->translateComplete(s1Req, tc, transState, mode, tranType, true);
00194     }
00195     // if we have been asked to delete ourselfs do it now

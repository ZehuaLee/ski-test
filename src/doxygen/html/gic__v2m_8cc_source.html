00081 Gicv2m::Gicv2m(const Params *p)
00082     : PioDevice(p), pioDelay(p->pio_delay), frames(p->frames), gic(p->gic)
00083 {
00084     // Assert SPI ranges start at 32
00085     for (int i = 0; i < frames.size(); i++) {
00086         if (frames[i]->spi_base < 32)
00087             fatal("Gicv2m: Frame %d's SPI base (%d) is not in SPI space\n",
00088                   i, frames[i]->spi_base);
00089     }
00090     unsigned int x = frames.size();
00091     fatal_if(!isPowerOf2(x), "Gicv2m: The v2m shim must be configured with "
00092               "a power-of-two number of frames\n");
00093     log2framenum = floorLog2(x);
00094 }

00106 Tick
00107 Gicv2m::read(PacketPtr pkt)
00108 {
00109     int frame = frameFromAddr(pkt->getAddr());
00110 
00111     assert(frame >= 0);
00112 
00113     Addr offset = pkt->getAddr() - frames[frame]->addr;
00114 
00115     switch (offset) {
00116       case MSI_TYPER:
00117         pkt->set<uint32_t>((frames[frame]->spi_base << 16) |
00118                            frames[frame]->spi_len);
00119         break;
00120 
00121       case PER_ID4:
00122         pkt->set<uint32_t>(0x4 | ((4+log2framenum) << 4));
00123         // Nr of 4KB blocks used by component.  This is messy as frames are 64K
00124         // (16, ie 2^4) and we should assert we're given a Po2 number of frames.
00125         break;
00126       default:
00127         DPRINTF(GICV2M, "GICv2m: Read of unk reg %#x\n", offset);
00128         pkt->set<uint32_t>(0);
00129     };
00130 
00131     pkt->makeAtomicResponse();
00132 
00133     return pioDelay;
00134 }

00137 Gicv2m::write(PacketPtr pkt)
00138 {
00139     int frame = frameFromAddr(pkt->getAddr());
00140 
00141     assert(frame >= 0);
00142 
00143     Addr offset = pkt->getAddr() - frames[frame]->addr;
00144 
00145     if (offset == MSI_SETSPI_NSR) {
00146         /* Is payload SPI number within range? */
00147         uint32_t m = pkt->get<uint32_t>();
00148         if (m >= frames[frame]->spi_base &&
00149             m < (frames[frame]->spi_base + frames[frame]->spi_len)) {
00150             DPRINTF(GICV2M, "GICv2m: Frame %d raising MSI %d\n", frame, m);
00151             gic->sendInt(m);
00152         }
00153     } else {
00154         DPRINTF(GICV2M, "GICv2m: Write of unk reg %#x\n", offset);
00155     }

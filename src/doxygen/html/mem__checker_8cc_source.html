00073 MemChecker::WriteCluster::completeWrite(MemChecker::Serial serial, Tick _complete)
00074 {
00075     auto it = writes.find(serial);
00076 
00077     if (it == writes.end()) {
00078         warn("Could not locate write transaction: serial = %d, complete = %d\n",
00079              serial, _complete);
00080         return;
00081     }
00082 
00083     // Record completion time of the write
00084     assert(it->second.complete == TICK_FUTURE);
00085     it->second.complete = _complete;
00086 
00087     // Update max completion time for the cluster
00088     if (completeMax < _complete) {
00089         completeMax = _complete;
00090     }
00091 
00092     if (--numIncomplete == 0) {
00093         // All writes have completed, this cluster is now complete and will be
00094         // assigned the max of completion tick values among all writes.
00095         //
00096         // Note that we cannot simply keep updating complete, because that would
00097         // count the cluster as closed already.  Instead, we keep TICK_FUTURE
00098         // until all writes have completed.
00099         complete = completeMax;
00100     }

00129 bool
00130 MemChecker::ByteTracker::inExpectedData(Tick start, Tick complete, uint8_t data)
00131 {
00132     _lastExpectedData.clear();
00133 
00134     bool wc_overlap = true;
00135 
00136     // Find the last value read from the location
00137     const Transaction& last_obs =
00138         *lastCompletedTransaction(&readObservations, start);
00139     bool last_obs_valid = (last_obs.complete != TICK_INITIAL);
00140 
00141     // Scan backwards through the write clusters to find the closest younger
00142     // preceding & overlapping writes.
00143     for (auto cluster = writeClusters.rbegin();
00144          cluster != writeClusters.rend() && wc_overlap; ++cluster) {
00145         for (const auto& addr_write : cluster->writes) {
00146             const Transaction& write = addr_write.second;
00147 
00148             if (write.complete < last_obs.start) {
00149                 // If this write transaction completed before the last
00150                 // observation, we ignore it as the last_observation has the
00151                 // correct value
00152                 continue;
00153             }
00154 
00155             if (write.data == data) {
00156                 // Found a match, end search.
00157                 return true;
00158             }
00159 
00160             // Record possible, but non-matching data for debugging
00161             _lastExpectedData.push_back(write.data);
00162 
00163             if (write.complete > start) {
00164                 // This write overlapped with the transaction we want to check
00165                 // -> continue checking the overlapping write cluster
00166                 continue;
00167             }
00168 
00169             // This write cluster has writes that have completed before the
00170             // checked transaction. There is no need to check an earlier
00171             // write-cluster -> set the exit condition for the outer loop
00172             wc_overlap = false;
00173 
00174             if (last_obs.complete < write.start) {
00175                 // We found a write which started after the last observed read,
00176                 // therefore we can not longer consider the value seen by the
00177                 // last observation as a valid expected value.
00178                 //
00179                 // Once all writes have been iterated through, we can check if
00180                 // the last observation is still valid to compare against.
00181                 last_obs_valid = false;
00182             }
00183         }
00184     }
00185 
00186     // We have not found any matching write, so far; check other sources of
00187     // confirmation
00188     if (last_obs_valid) {
00189         // The last observation is not outdated according to the writes we have
00190         // seen so far.
00191         assert(last_obs.complete <= start);
00192         if (last_obs.data == data) {
00193             // Matched data from last observation -> all good
00194             return true;
00195         }
00196         // Record non-matching, but possible value
00197         _lastExpectedData.push_back(last_obs.data);
00198     } else {
00199         // We have not seen any valid observation, and the only writes
00200         // observed are overlapping, so anything (in particular the
00201         // initialisation value) goes
00202         // NOTE: We can overlap with multiple write clusters, here
00203         if (!writeClusters.empty() && wc_overlap) {
00204             // ensure that all write clusters really overlap this read
00205             assert(writeClusters.begin()->start < complete &&
00206                    writeClusters.rbegin()->complete > start);
00207             return true;
00208         }
00209     }
00210 
00211     if (_lastExpectedData.empty()) {
00212         assert(last_obs.complete == TICK_INITIAL);
00213         // We have not found any possible (non-matching data). Can happen in
00214         // initial system state
00215         DPRINTF(MemChecker, "no last observation nor write! start = %d, "\
00216                 "complete = %d, data = %#x\n", start, complete, data);
00217         return true;
00218     }
00219     return false;
00220 }

00279 {
00280     // Obtain tick of first outstanding read. If there are no outstanding
00281     // reads, we use curTick(), i.e. we will remove all readObservation except
00282     // the most recent one.

00327                              tracker->lastExpectedData()[j],

00331         }
00332     }
00333 
00334     if (!result) {

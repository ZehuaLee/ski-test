00223     offset64()
00224     {
00225         if (toEL == fromEL) {
00226             if (opModeIsT(fromMode))
00227                 return vals.currELTOffset;
00228             return vals.currELHOffset;
00229         } else {
00230             if (from64)
00231                 return vals.lowerEL64Offset;
00232             return vals.lowerEL32Offset;
00233         }
00234     }
00235 
00236     OperatingMode nextMode() { return vals.nextMode; }
00237     virtual bool routeToMonitor(ThreadContext *tc) const { return false; }
00238     uint8_t armPcOffset(bool isHyp)   { return isHyp ? vals.armPcElrOffset
00239                                                      : vals.armPcOffset; }
00240     uint8_t thumbPcOffset(bool isHyp) { return isHyp ? vals.thumbPcElrOffset
00241                                                      : vals.thumbPcOffset; }

00243     uint8_t thumbPcElrOffset() { return vals.thumbPcElrOffset; }

00088 {
00089     bool simFIQ(interrupts->checkRaw(INT_FIQ));
00090     bool simIRQ(interrupts->checkRaw(INT_IRQ));
00091 
00092     if (fiqAsserted != simFIQ) {
00093         fiqAsserted = simFIQ;

00101     }
00102 
00103     return BaseKvmCPU::kvmRun(ticks);

00106 const BaseArmKvmCPU::RegIndexVector &
00107 BaseArmKvmCPU::getRegList() const
00108 {
00109     // Do we need to request a list of registers from the kernel?
00110     if (_regIndexList.size() == 0) {
00111         // Start by probing for the size of the list. We do this
00112         // calling the ioctl with a struct size of 0. The kernel will
00113         // return the number of elements required to hold the list.
00114         kvm_reg_list regs_probe;
00115         regs_probe.n = 0;
00116         getRegList(regs_probe);
00117 
00118         // Request the actual register list now that we know how many
00119         // register we need to allocate space for.
00120         std::unique_ptr<struct kvm_reg_list> regs;
00121         const size_t size(sizeof(struct kvm_reg_list) +
00122                           regs_probe.n * sizeof(uint64_t));
00123         regs.reset((struct kvm_reg_list *)operator new(size));
00124         regs->n = regs_probe.n;
00125         if (!getRegList(*regs))
00126             panic("Failed to determine register list size.\n");
00127 
00128         _regIndexList.assign(regs->reg, regs->reg + regs->n);
00129     }
00130 
00131     return _regIndexList;
00132 }

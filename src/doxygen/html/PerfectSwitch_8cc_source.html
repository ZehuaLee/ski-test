00121         m_round_robin_start = 0;

00125         // for all input ports, use round robin scheduling

00127             // Round robin scheduling
00128             incoming++;

00130                 incoming = 0;

00140             }

00143             if (buffer == nullptr) {
00144                 continue;

00160                 // zero-destination messages, so this assert isn't valid
00161                 // assert(msg_dsts.count() > 0);

00163                 assert(m_link_order.size() == m_routing_table.size());

00168                         // Don't adaptively route

00171                             m_link_order[out].m_value = 0;

00178                                 out_queue_length += m_out[out][v]->getSize();

00184                             m_link_order[out].m_value = value;
00185                         }

00187                         // Look at the most empty link first
00188                         sort(m_link_order.begin(), m_link_order.end());

00190                 }
00191

00193                     // pick the next link to look at
00194                     int link = m_link_order[i].m_link;
00195                     NetDest dst = m_routing_table[link];
00196                     DPRINTF(RubyNetwork, "dst: %s\n", dst);
00197 
00198                     if (!msg_dsts.intersectionIsNotEmpty(dst))
00199                         continue;

00283 void
00284 PerfectSwitch::wakeup()
00285 {
00286     // Give the highest numbered link priority most of the time
00287     m_wakeups_wo_switch++;
00288     int highest_prio_vnet = m_virtual_networks-1;
00289     int lowest_prio_vnet = 0;
00290     int decrementer = 1;
00291 
00292     // invert priorities to avoid starvation seen in the component network
00293     if (m_wakeups_wo_switch > PRIORITY_SWITCH_LIMIT) {
00294         m_wakeups_wo_switch = 0;
00295         highest_prio_vnet = 0;
00296         lowest_prio_vnet = m_virtual_networks-1;
00297         decrementer = -1;
00298     }
00299 
00300     // For all components incoming queues
00301     for (int vnet = highest_prio_vnet;
00302          (vnet * decrementer) >= (decrementer * lowest_prio_vnet);
00303          vnet -= decrementer) {
00304         operateVnet(vnet);
00305     }
00306 }

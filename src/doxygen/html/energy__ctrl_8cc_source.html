00062 Tick
00063 EnergyCtrl::read(PacketPtr pkt)
00064 {
00065     assert(pkt->getAddr() >= pioAddr && pkt->getAddr() < pioAddr + pioSize);
00066     assert(pkt->getSize() == 4);
00067 
00068     Addr daddr = pkt->getAddr() - pioAddr;
00069     assert((daddr & 3) == 0);
00070     Registers reg = Registers(daddr / 4);
00071 
00072     if (!dvfsHandler->isEnabled()) {
00073         // NB: Zero is a good response if the handler is disabled
00074         pkt->set<uint32_t>(0);
00075         warn_once("EnergyCtrl: Disabled handler, ignoring read from reg %i\n",
00076                   reg);
00077         DPRINTF(EnergyCtrl, "dvfs handler disabled, return 0 for read from "\
00078                 "reg %i\n", reg);
00079         pkt->makeAtomicResponse();
00080         return pioDelay;
00081     }
00082 
00083     uint32_t result = 0;
00084     Tick period;
00085     double voltage;
00086 
00087     switch(reg) {
00088       case DVFS_HANDLER_STATUS:
00089         result = 1;
00090         DPRINTF(EnergyCtrl, "dvfs handler enabled\n");
00091         break;
00092       case DVFS_NUM_DOMAINS:
00093         result = dvfsHandler->numDomains();
00094         DPRINTF(EnergyCtrl, "reading number of domains %d\n", result);
00095         break;
00096       case DVFS_DOMAINID_AT_INDEX:
00097         result = dvfsHandler->domainID(domainIDIndexToRead);
00098         DPRINTF(EnergyCtrl, "reading domain id at index %d as %d\n",
00099                 domainIDIndexToRead, result);
00100         break;
00101       case DVFS_HANDLER_TRANS_LATENCY:
00102         // Return transition latency in nanoseconds
00103         result = dvfsHandler->transLatency() / SimClock::Int::ns;
00104         DPRINTF(EnergyCtrl, "reading dvfs handler trans latency %d ns\n",
00105                 result);
00106         break;
00107       case DOMAIN_ID:
00108         result = domainID;
00109         DPRINTF(EnergyCtrl, "reading domain id:%d\n", result);
00110         break;
00111       case PERF_LEVEL:
00112         result = dvfsHandler->perfLevel(domainID);
00113         DPRINTF(EnergyCtrl, "reading domain %d perf level: %d\n",
00114                 domainID, result);
00115         break;
00116       case PERF_LEVEL_ACK:
00117         result = perfLevelAck;
00118         DPRINTF(EnergyCtrl, "reading ack:%d\n", result);
00119         // Signal is set for a single read only
00120         if (result == 1)
00121             perfLevelAck = 0;
00122         break;
00123       case NUM_OF_PERF_LEVELS:
00124         result = dvfsHandler->numPerfLevels(domainID);
00125         DPRINTF(EnergyCtrl, "reading num of perf level:%d\n", result);
00126         break;
00127       case FREQ_AT_PERF_LEVEL:
00128         period = dvfsHandler->clkPeriodAtPerfLevel(domainID, perfLevelToRead);
00129         result = ticksTokHz(period);
00130         DPRINTF(EnergyCtrl, "reading freq %d KHz at perf level: %d\n",
00131                 result, perfLevelToRead);
00132         break;
00133       case VOLT_AT_PERF_LEVEL:
00134         voltage = dvfsHandler->voltageAtPerfLevel(domainID, perfLevelToRead);
00135         result = toMicroVolt(voltage);
00136         DPRINTF(EnergyCtrl, "reading voltage %d u-volt at perf level: %d\n",
00137                 result, perfLevelToRead);
00138         break;
00139       default:
00140         panic("Tried to read EnergyCtrl at offset %#x / reg %i\n", daddr,
00141               reg);
00142     }
00143     pkt->set<uint32_t>(result);
00144     pkt->makeAtomicResponse();
00145     return pioDelay;
00146 }

00148 Tick
00149 EnergyCtrl::write(PacketPtr pkt)
00150 {
00151     assert(pkt->getAddr() >= pioAddr && pkt->getAddr() < pioAddr + pioSize);
00152     assert(pkt->getSize() == 4);
00153 
00154     uint32_t data;
00155     data = pkt->get<uint32_t>();
00156 
00157     Addr daddr = pkt->getAddr() - pioAddr;
00158     assert((daddr & 3) == 0);
00159     Registers reg = Registers(daddr / 4);
00160 
00161     if (!dvfsHandler->isEnabled()) {
00162         // Ignore writes to a disabled controller
00163         warn_once("EnergyCtrl: Disabled handler, ignoring write %u to "\
00164                   "reg %i\n", data, reg);
00165         DPRINTF(EnergyCtrl, "dvfs handler disabled, ignoring write %u to "\
00166                 "reg %i\n", data, reg);
00167         pkt->makeAtomicResponse();
00168         return pioDelay;
00169     }
00170 
00171     switch(reg) {
00172       case DVFS_DOMAINID_AT_INDEX:
00173         domainIDIndexToRead = data;
00174         DPRINTF(EnergyCtrl, "writing domain id index:%d\n",
00175                 domainIDIndexToRead);
00176         break;
00177       case DOMAIN_ID:
00178         // Extra check to ensure that a valid domain ID is being queried
00179         if (dvfsHandler->validDomainID(data)) {
00180             domainID = data;
00181             DPRINTF(EnergyCtrl, "writing domain id:%d\n", domainID);
00182         } else {
00183            DPRINTF(EnergyCtrl, "invalid domain id:%d\n", domainID);
00184         }
00185         break;
00186       case PERF_LEVEL:
00187         if (dvfsHandler->perfLevel(domainID, data)) {
00188             if (updateAckEvent.scheduled()) {
00189                 // The OS driver is trying to change the perf level while
00190                 // another change is in flight.  This is fine, but only a
00191                 // single acknowledgment will be sent.
00192                 DPRINTF(EnergyCtrl, "descheduling previous pending ack "\
00193                         "event\n");
00194                 deschedule(updateAckEvent);
00195             }
00196             schedule(updateAckEvent, curTick() + dvfsHandler->transLatency());
00197             DPRINTF(EnergyCtrl, "writing domain %d perf level: %d\n",
00198                     domainID, data);
00199         } else {
00200             DPRINTF(EnergyCtrl, "invalid / ineffective perf level:%d for "\
00201                     "domain:%d\n", data, domainID);
00202         }
00203         break;
00204       case PERF_LEVEL_TO_READ:
00205         perfLevelToRead = data;
00206         DPRINTF(EnergyCtrl, "writing perf level to read opp at: %d\n",
00207                 data);
00208         break;
00209       default:
00210         panic("Tried to write EnergyCtrl at offset %#x\n", daddr);
00211         break;
00212     }
00213 
00214     pkt->makeAtomicResponse();
00215     return pioDelay;
00216 }

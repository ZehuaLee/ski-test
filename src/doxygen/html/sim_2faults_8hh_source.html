00044 // Each class has it's name statically define in _name,
00045 // and has a virtual function to access it's name.
00046 // The function is necessary because otherwise, all objects
00047 // which are being accessed cast as a FaultBase * (namely
00048 // all faults returned using the Fault type) will use the
00049 // generic FaultBase name.
00050 
00051 class FaultBase
00052 {
00053   public:
00054     virtual FaultName name() const = 0;
00055     virtual void invoke(ThreadContext * tc, const StaticInstPtr &inst =
00056                         StaticInst::nullStaticInstPtr);
00057 };

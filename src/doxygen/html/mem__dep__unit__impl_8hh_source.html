00172 template <class MemDepPred, class Impl>
00173 void
00174 MemDepUnit<MemDepPred, Impl>::insert(DynInstPtr &inst)
00175 {
00176     ThreadID tid = inst->threadNumber;
00177 
00178     MemDepEntryPtr inst_entry = std::make_shared<MemDepEntry>(inst);
00179 
00180     // Add the MemDepEntry to the hash.
00181     memDepHash.insert(
00182         std::pair<InstSeqNum, MemDepEntryPtr>(inst->seqNum, inst_entry));
00183 #ifdef DEBUG
00184     MemDepEntry::memdep_insert++;
00185 #endif
00186 
00187     instList[tid].push_back(inst);
00188 
00189     inst_entry->listIt = --(instList[tid].end());
00190 
00191     // Check any barriers and the dependence predictor for any
00192     // producing memrefs/stores.
00193     InstSeqNum producing_store;
00194     if (inst->isLoad() && loadBarrier) {
00195         DPRINTF(MemDepUnit, "Load barrier [sn:%lli] in flight\n",
00196                 loadBarrierSN);
00197         producing_store = loadBarrierSN;
00198     } else if (inst->isStore() && storeBarrier) {
00199         DPRINTF(MemDepUnit, "Store barrier [sn:%lli] in flight\n",
00200                 storeBarrierSN);
00201         producing_store = storeBarrierSN;
00202     } else {
00203         producing_store = depPred.checkInst(inst->instAddr());
00204     }
00205 
00206     MemDepEntryPtr store_entry = NULL;
00207 
00208     // If there is a producing store, try to find the entry.
00209     if (producing_store != 0) {
00210         DPRINTF(MemDepUnit, "Searching for producer\n");
00211         MemDepHashIt hash_it = memDepHash.find(producing_store);
00212 
00213         if (hash_it != memDepHash.end()) {
00214             store_entry = (*hash_it).second;
00215             DPRINTF(MemDepUnit, "Proucer found\n");
00216         }
00217     }
00218 
00219     // If no store entry, then instruction can issue as soon as the registers
00220     // are ready.
00221     if (!store_entry) {
00222         DPRINTF(MemDepUnit, "No dependency for inst PC "
00223                 "%s [sn:%lli].\n", inst->pcState(), inst->seqNum);
00224 
00225         inst_entry->memDepReady = true;
00226 
00227         if (inst->readyToIssue()) {
00228             inst_entry->regsReady = true;
00229 
00230             moveToReady(inst_entry);
00231         }
00232     } else {
00233         // Otherwise make the instruction dependent on the store/barrier.
00234         DPRINTF(MemDepUnit, "Adding to dependency list; "
00235                 "inst PC %s is dependent on [sn:%lli].\n",
00236                 inst->pcState(), producing_store);
00237 
00238         if (inst->readyToIssue()) {
00239             inst_entry->regsReady = true;
00240         }
00241 
00242         // Clear the bit saying this instruction can issue.
00243         inst->clearCanIssue();
00244 
00245         // Add this instruction to the list of dependents.
00246         store_entry->dependInsts.push_back(inst_entry);
00247 
00248         if (inst->isLoad()) {
00249             ++conflictingLoads;
00250         } else {
00251             ++conflictingStores;
00252         }
00253     }
00254 
00255     if (inst->isStore()) {
00256         DPRINTF(MemDepUnit, "Inserting store PC %s [sn:%lli].\n",
00257                 inst->pcState(), inst->seqNum);
00258 
00259         depPred.insertStore(inst->instAddr(), inst->seqNum, inst->threadNumber);
00260 
00261         ++insertedStores;
00262     } else if (inst->isLoad()) {
00263         ++insertedLoads;
00264     } else {
00265         panic("Unknown type! (most likely a barrier).");
00266     }
00267 }

00451 void
00452 MemDepUnit<MemDepPred, Impl>::wakeDependents(DynInstPtr &inst)
00453 {
00454     // Only stores and barriers have dependents.
00455     if (!inst->isStore() && !inst->isMemBarrier() && !inst->isWriteBarrier()) {
00456         return;
00457     }
00458 
00459     MemDepEntryPtr inst_entry = findInHash(inst);
00460 
00461     for (int i = 0; i < inst_entry->dependInsts.size(); ++i ) {
00462         MemDepEntryPtr woken_inst = inst_entry->dependInsts[i];
00463 
00464         if (!woken_inst->inst) {
00465             // Potentially removed mem dep entries could be on this list
00466             continue;

00468 
00469         DPRINTF(MemDepUnit, "Waking up a dependent inst, "
00470                 "[sn:%lli].\n",
00471                 woken_inst->inst->seqNum);
00472 
00473         if (woken_inst->regsReady && !woken_inst->squashed) {
00474             moveToReady(woken_inst);

00483 template <class MemDepPred, class Impl>
00484 void
00485 MemDepUnit<MemDepPred, Impl>::squash(const InstSeqNum &squashed_num,
00486                                      ThreadID tid)
00487 {
00488     if (!instsToReplay.empty()) {
00489         ListIt replay_it = instsToReplay.begin();
00490         while (replay_it != instsToReplay.end()) {
00491             if ((*replay_it)->threadNumber == tid &&
00492                 (*replay_it)->seqNum > squashed_num) {
00493                 instsToReplay.erase(replay_it++);
00494             } else {
00495                 ++replay_it;
00496             }
00497         }
00498     }
00499 
00500     ListIt squash_it = instList[tid].end();
00501     --squash_it;
00502 
00503     MemDepHashIt hash_it;
00504 
00505     while (!instList[tid].empty() &&
00506            (*squash_it)->seqNum > squashed_num) {
00507 
00508         DPRINTF(MemDepUnit, "Squashing inst [sn:%lli]\n",
00509                 (*squash_it)->seqNum);
00510 
00511         if ((*squash_it)->seqNum == loadBarrierSN)
00512               loadBarrier = false;
00513 
00514         if ((*squash_it)->seqNum == storeBarrierSN)
00515               storeBarrier = false;
00516 
00517         hash_it = memDepHash.find((*squash_it)->seqNum);
00518 
00519         assert(hash_it != memDepHash.end());
00520 
00521         (*hash_it).second->squashed = true;
00522 
00523         (*hash_it).second = NULL;
00524 
00525         memDepHash.erase(hash_it);
00526 #ifdef DEBUG
00527         MemDepEntry::memdep_erase++;
00528 #endif
00529 
00530         instList[tid].erase(squash_it--);
00531     }
00532 
00533     // Tell the dependency predictor to squash as well.
00534     depPred.squash(squashed_num, tid);
00535 }

00073 Data::MemTimingSpec
00074 DRAMPower::getTimingParams(const DRAMCtrlParams* p)
00075 {
00076     // Set the values that are used for power calculations and ignore
00077     // the ones only used by the controller functionality in DRAMPower
00078 
00079     // All DRAMPower timings are in clock cycles
00080     Data::MemTimingSpec timingSpec;
00081     timingSpec.RC = divCeil((p->tRAS + p->tRP), p->tCK);
00082     timingSpec.RCD = divCeil(p->tRCD, p->tCK);
00083     timingSpec.RL = divCeil(p->tCL, p->tCK);
00084     timingSpec.RP = divCeil(p->tRP, p->tCK);
00085     timingSpec.RFC = divCeil(p->tRFC, p->tCK);
00086     timingSpec.RAS = divCeil(p->tRAS, p->tCK);
00087     // Write latency is read latency - 1 cycle
00088     // Source: B.Jacob Memory Systems Cache, DRAM, Disk
00089     timingSpec.WL = timingSpec.RL - 1;
00090     timingSpec.DQSCK = 0; // ignore for now
00091     timingSpec.RTP = divCeil(p->tRTP, p->tCK);
00092     timingSpec.WR = divCeil(p->tWR, p->tCK);
00093     timingSpec.XP = divCeil(p->tXP, p->tCK);
00094     timingSpec.XPDLL = divCeil(p->tXPDLL, p->tCK);
00095     timingSpec.XS = divCeil(p->tXS, p->tCK);
00096     timingSpec.XSDLL = divCeil(p->tXSDLL, p->tCK);
00097 
00098     // Clock period in ns
00099     timingSpec.clkPeriod = (p->tCK / (double)(SimClock::Int::ns));
00100     assert(timingSpec.clkPeriod != 0);
00101     timingSpec.clkMhz = (1 / timingSpec.clkPeriod) * 1000;
00102     return timingSpec;
00103 }

00153 uint8_t
00154 DRAMPower::getDataRate(const DRAMCtrlParams* p)
00155 {
00156     uint32_t burst_cycles = divCeil(p->tBURST, p->tCK);
00157     uint8_t data_rate = p->burst_length / burst_cycles;
00158     // 4 for GDDR5
00159     if (data_rate != 1 && data_rate != 2 && data_rate != 4)
00160         fatal("Got unexpected data rate %d, should be 1 or 2 or 4\n");
00161     return data_rate;
00162 }

00169 /*
00170  * Determine if the mapping at va..(va+len) is valid.
00171  */
00172 bool
00173 RemoteGDB::acc(Addr va, size_t len)
00174 {
00175     if (FullSystem) {
00176         for (ChunkGenerator gen(va, len, PageBytes); !gen.done(); gen.next()) {
00177             if (!virtvalid(context, gen.addr())) {
00178                 DPRINTF(GDBAcc, "acc:   %#x mapping is invalid\n", va);
00179                 return false;
00180             }
00181         }
00182 
00183         DPRINTF(GDBAcc, "acc:   %#x mapping is valid\n", va);
00184         return true;
00185     } else {
00186         TlbEntry entry;
00187         //Check to make sure the first byte is mapped into the processes address
00188         //space.
00189         if (context->getProcessPtr()->pTable->lookup(va, entry))
00190             return true;
00191         return false;
00192     }
00193 }

00196  * Translate the kernel debugger register format into the GDB register

00199 void
00200 RemoteGDB::getregs()
00201 {
00202     DPRINTF(GDBAcc, "getregs in remotegdb \n");
00203 
00204     memset(gdbregs.regs, 0, gdbregs.bytes());
00205 
00206     if (inAArch64(context)) {  // AArch64
00207         // x0-x30
00208         for (int i = 0; i < 31; ++i)
00209             gdbregs.regs64[GDB64_X0 + i] = context->readIntReg(INTREG_X0 + i);
00210         gdbregs.regs64[GDB64_SPX] = context->readIntReg(INTREG_SPX);
00211         // pc
00212         gdbregs.regs64[GDB64_PC] = context->pcState().pc();
00213         // cpsr
00214         gdbregs.regs64[GDB64_CPSR] =
00215             context->readMiscRegNoEffect(MISCREG_CPSR);
00216         // v0-v31
00217         for (int i = 0; i < 128; i += 4) {
00218             int gdboff = GDB64_V0_32 + i;
00219             gdbregs.regs32[gdboff + 0] = context->readFloatRegBits(i + 2);
00220             gdbregs.regs32[gdboff + 1] = context->readFloatRegBits(i + 3);
00221             gdbregs.regs32[gdboff + 2] = context->readFloatRegBits(i + 0);
00222             gdbregs.regs32[gdboff + 3] = context->readFloatRegBits(i + 1);
00223         }
00224     } else {  // AArch32
00225         // R0-R15 supervisor mode
00226         gdbregs.regs32[GDB32_R0 + 0] = context->readIntReg(INTREG_R0);
00227         gdbregs.regs32[GDB32_R0 + 1] = context->readIntReg(INTREG_R1);
00228         gdbregs.regs32[GDB32_R0 + 2] = context->readIntReg(INTREG_R2);
00229         gdbregs.regs32[GDB32_R0 + 3] = context->readIntReg(INTREG_R3);
00230         gdbregs.regs32[GDB32_R0 + 4] = context->readIntReg(INTREG_R4);
00231         gdbregs.regs32[GDB32_R0 + 5] = context->readIntReg(INTREG_R5);
00232         gdbregs.regs32[GDB32_R0 + 6] = context->readIntReg(INTREG_R6);
00233         gdbregs.regs32[GDB32_R0 + 7] = context->readIntReg(INTREG_R7);

00235         gdbregs.regs32[GDB32_R0 + 9] = context->readIntReg(INTREG_R9);
00236         gdbregs.regs32[GDB32_R0 + 10] = context->readIntReg(INTREG_R10);

00256 /*
00257  * Translate the GDB register format into the kernel debugger register
00258  * format.
00259  */
00260 void
00261 RemoteGDB::setregs()
00262 {
00263 
00264     DPRINTF(GDBAcc, "setregs in remotegdb \n");
00265     if (inAArch64(context)) {  // AArch64
00266         // x0-x30
00267         for (int i = 0; i < 31; ++i)
00268             context->setIntReg(INTREG_X0 + i, gdbregs.regs64[GDB64_X0 + i]);
00269         // pc
00270         context->pcState(gdbregs.regs64[GDB64_PC]);
00271         // cpsr
00272         context->setMiscRegNoEffect(MISCREG_CPSR, gdbregs.regs64[GDB64_CPSR]);
00273         // Update the stack pointer. This should be done after
00274         // updating CPSR/PSTATE since that might affect how SPX gets
00275         // mapped.
00276         context->setIntReg(INTREG_SPX, gdbregs.regs64[GDB64_SPX]);
00277         // v0-v31
00278         for (int i = 0; i < 128; i += 4) {
00279             int gdboff = GDB64_V0_32 + i;
00280             context->setFloatRegBits(i + 2, gdbregs.regs32[gdboff + 0]);
00281             context->setFloatRegBits(i + 3, gdbregs.regs32[gdboff + 1]);
00282             context->setFloatRegBits(i + 0, gdbregs.regs32[gdboff + 2]);
00283             context->setFloatRegBits(i + 1, gdbregs.regs32[gdboff + 3]);
00284         }
00285     } else {  // AArch32
00286         // R0-R15 supervisor mode
00287         // arm registers are 32 bits wide, gdb registers are 64 bits wide
00288         // two arm registers are packed into one gdb register (little endian)
00289         context->setIntReg(INTREG_R0, gdbregs.regs32[GDB32_R0 + 0]);
00290         context->setIntReg(INTREG_R1, gdbregs.regs32[GDB32_R0 + 1]);
00291         context->setIntReg(INTREG_R2, gdbregs.regs32[GDB32_R0 + 2]);
00292         context->setIntReg(INTREG_R3, gdbregs.regs32[GDB32_R0 + 3]);
00293         context->setIntReg(INTREG_R4, gdbregs.regs32[GDB32_R0 + 4]);
00294         context->setIntReg(INTREG_R5, gdbregs.regs32[GDB32_R0 + 5]);
00295         context->setIntReg(INTREG_R6, gdbregs.regs32[GDB32_R0 + 6]);
00296         context->setIntReg(INTREG_R7, gdbregs.regs32[GDB32_R0 + 7]);
00297         context->setIntReg(INTREG_R8, gdbregs.regs32[GDB32_R0 + 8]);
00298         context->setIntReg(INTREG_R9, gdbregs.regs32[GDB32_R0 + 9]);
00299         context->setIntReg(INTREG_R10, gdbregs.regs32[GDB32_R0 + 10]);
00300         context->setIntReg(INTREG_R11, gdbregs.regs32[GDB32_R0 + 11]);
00301         context->setIntReg(INTREG_R12, gdbregs.regs32[GDB32_R0 + 12]);
00302         context->setIntReg(INTREG_SP, gdbregs.regs32[GDB32_R0 + 13]);
00303         context->setIntReg(INTREG_LR, gdbregs.regs32[GDB32_R0 + 14]);
00304         context->pcState(gdbregs.regs32[GDB32_R0 + 7]);
00305 
00306         //CPSR
00307         context->setMiscRegNoEffect(MISCREG_CPSR, gdbregs.regs32[GDB32_CPSR]);
00308 
00309         //vfpv3/neon floating point registers (32 double or 64 float)
00310         for (int i = 0; i < NumFloatV7ArchRegs; ++i)
00311             context->setFloatRegBits(i, gdbregs.regs32[GDB32_F0 + i]);
00312 
00313         //FPSCR
00314         context->setMiscReg(MISCREG_FPSCR, gdbregs.regs32[GDB32_FPSCR]);
00315     }

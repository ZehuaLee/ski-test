00122 void
00123 Pipeline::evaluate()
00124 {
00125     /* Note that it's important to evaluate the stages in order to allow
00126      *  'immediate', 0-time-offset TimeBuffer activity to be visible from
00127      *  later stages to earlier ones in the same cycle */
00128     execute.evaluate();
00129     decode.evaluate();
00130     fetch2.evaluate();
00131     fetch1.evaluate();
00132 
00133     if (DTRACE(MinorTrace))
00134         minorTrace();
00135 
00136     /* Update the time buffers after the stages */
00137     f1ToF2.evaluate();
00138     f2ToF1.evaluate();
00139     f2ToD.evaluate();
00140     dToE.evaluate();
00141     eToF1.evaluate();
00142 
00143     /* The activity recorder must be be called after all the stages and
00144      *  before the idler (which acts on the advice of the activity recorder */
00145     activityRecorder.evaluate();
00146 
00147     if (allow_idling) {
00148         /* Become idle if we can but are not draining */
00149         if (!activityRecorder.active() && !needToSignalDrained) {
00150             DPRINTF(Quiesce, "Suspending as the processor is idle\n");
00151             stop();
00152         }
00153 
00154         /* Deactivate all stages.  Note that the stages *could*
00155          *  activate and deactivate themselves but that's fraught
00156          *  with additional difficulty.
00157          *  As organised herre */
00158         activityRecorder.deactivateStage(Pipeline::CPUStageId);
00159         activityRecorder.deactivateStage(Pipeline::Fetch1StageId);
00160         activityRecorder.deactivateStage(Pipeline::Fetch2StageId);
00161         activityRecorder.deactivateStage(Pipeline::DecodeStageId);
00162         activityRecorder.deactivateStage(Pipeline::ExecuteStageId);
00163     }
00164 
00165     if (needToSignalDrained) /* Must be draining */
00166     {
00167         DPRINTF(Drain, "Still draining\n");
00168         if (isDrained()) {
00169             DPRINTF(Drain, "Signalling end of draining\n");
00170             cpu.signalDrainDone();
00171             needToSignalDrained = false;
00172             stop();
00173         }
00174     }
00175 }

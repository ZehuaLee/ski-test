00359     //
00360     // If we had to stall the MemSlavePorts, wake them up because the sequencer
00361     // likely has free resources now.

00375         for (auto i = curRetryList.begin(); i != curRetryList.end(); ++i) {
00376             DPRINTF(RubyPort,
00377                     "Sequencer may now be free.  SendRetry to port %s\n",

00379             (*i)->sendRetryReq();

00461 void
00462 RubyPort::MemSlavePort::hitCallback(PacketPtr pkt)
00463 {
00464     bool needsResponse = pkt->needsResponse();
00465 
00466     // Unless specified at configuraiton, all responses except failed SC 
00467     // and Flush operations access M5 physical memory.
00468     bool accessPhysMem = access_backing_store;
00469 
00470     if (pkt->isLLSC()) {
00471         if (pkt->isWrite()) {
00472             if (pkt->req->getExtraData() != 0) {
00473                 //
00474                 // Successful SC packets convert to normal writes
00475                 //
00476                 pkt->convertScToWrite();
00477             } else {
00478                 //
00479                 // Failed SC packets don't access physical memory and thus
00480                 // the RubyPort itself must convert it to a response.
00481                 //
00482                 accessPhysMem = false;
00483             }
00484         } else {
00485             //
00486             // All LL packets convert to normal loads so that M5 PhysMem does
00487             // not lock the blocks.
00488             //
00489             pkt->convertLlToRead();
00490         }
00491     }
00492 
00493     // Flush requests don't access physical memory
00494     if (pkt->isFlush()) {
00495         accessPhysMem = false;
00496     }
00497 
00498     DPRINTF(RubyPort, "Hit callback needs response %d\n", needsResponse);
00499 
00500     if (accessPhysMem) {
00501         ruby_system->getPhysMem()->access(pkt);
00502     } else if (needsResponse) {
00503         pkt->makeResponse();
00504     }
00505 
00506     // turn packet around to go back to requester if response expected
00507     if (needsResponse) {
00508         DPRINTF(RubyPort, "Sending packet back over port\n");
00509         // send next cycle
00510         schedTimingResp(pkt, curTick() + g_system_ptr->clockPeriod());
00511     } else {
00512         delete pkt;
00513     }
00514 
00515     DPRINTF(RubyPort, "Hit callback done!\n");

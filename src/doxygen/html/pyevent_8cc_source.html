00048 void
00049 PythonEvent::process()
00050 {
00051     PyObject *args = PyTuple_New(0);
00052     PyObject *result = PyObject_Call(eventCode, args, NULL);
00053     Py_DECREF(args);
00054 
00055     if (result) {
00056         // Nothing to do just decrement the reference count
00057         Py_DECREF(result);
00058     } else {
00059         // Somethign should be done to signal back to the main interpreter
00060         // that there's been an exception.
00061         async_event = true;
00062         async_exception = true;
00063         /* Wake up some event queue to handle event */
00064         getEventQueue(0)->wakeup();
00065     }
00066 
00067     // Since the object has been removed from the event queue, its
00068     // reference count must be decremented.
00069     Py_DECREF(object);
00070 }

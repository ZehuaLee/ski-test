00053 void BaseGlobalEvent::schedule(Tick when)
00054 {
00055     // This function is scheduling a global event, which actually is a
00056     // set of local events, one event on each eventq. Global events need
00057     // to have a total order. A thread cannot start executing events that
00058     // follow a global event till all other threads have executed that global
00059     // event as well. If global events were not in a total order, a deadlock
00060     // would occur for there will be two threads who would be waiting for
00061     // each other to execute the global events they themselves have executed.
00062     //
00063     // To ensure this total order, we do two things.
00064     // First, before scheduling any global event, a thread needs to acquire
00065     // the lock globalQMutex. This ensures that only one thread can schedule
00066     // global events at any given time.
00067     // Second, the local events corresponding to a global event are always
00068     // first inserted in to the asyncq, irrespective of whether or not the
00069     // thread scheduling the event owns the eventq on which the event is
00070     // being scheduled. Thus global events have the same order in the asyncq
00071     // of each thread. When they are inserted in the actual eventq, the
00072     // comparators in the Event class ensure that the total order is
00073     // maintained.
00074 
00075     globalQMutex.lock();
00076 
00077     for (int i = 0; i < numMainEventQueues; ++i) {
00078         mainEventQueue[i]->schedule(barrierEvent[i], when, true);
00079     }
00080 
00081     globalQMutex.unlock();
00082 }

00084 void BaseGlobalEvent::deschedule()

00087     for (uint32_t i = 0; i < numMainEventQueues; ++i) {

00092     }

00097 void BaseGlobalEvent::reschedule(Tick when)
00098 {
00099     // Read the comment in the schedule() function above.
00100     globalQMutex.lock();
00101 
00102     for (uint32_t i = 0; i < numMainEventQueues; ++i) {
00103         if (barrierEvent[i]->scheduled())
00104             mainEventQueue[i]->reschedule(barrierEvent[i], when);
00105         else
00106             mainEventQueue[i]->schedule(barrierEvent[i], when, true);
00107     }
00108 
00109     globalQMutex.unlock();

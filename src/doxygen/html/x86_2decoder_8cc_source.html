00070 void
00071 Decoder::process()
00072 {
00073     //This function drives the decoder state machine.
00074 
00075     //Some sanity checks. You shouldn't try to process more bytes if
00076     //there aren't any, and you shouldn't overwrite an already
00077     //decoder ExtMachInst.
00078     assert(!outOfBytes);
00079     assert(!instDone);
00080 
00081     if (state == ResetState)
00082         state = doResetState();
00083     if (state == FromCacheState) {
00084         state = doFromCacheState();
00085     } else {
00086         instBytes->chunks.push_back(fetchChunk);
00087     }
00088 
00089     //While there's still something to do...
00090     while (!instDone && !outOfBytes) {
00091         uint8_t nextByte = getNextByte();
00092         switch (state) {
00093           case PrefixState:
00094             state = doPrefixState(nextByte);
00095             break;
00096           case OneByteOpcodeState:
00097             state = doOneByteOpcodeState(nextByte);
00098             break;
00099           case TwoByteOpcodeState:
00100             state = doTwoByteOpcodeState(nextByte);
00101             break;
00102           case ThreeByte0F38OpcodeState:
00103             state = doThreeByte0F38OpcodeState(nextByte);
00104             break;
00105           case ThreeByte0F3AOpcodeState:
00106             state = doThreeByte0F3AOpcodeState(nextByte);
00107             break;
00108           case ModRMState:
00109             state = doModRMState(nextByte);
00110             break;
00111           case SIBState:
00112             state = doSIBState(nextByte);
00113             break;
00114           case DisplacementState:
00115             state = doDisplacementState();
00116             break;
00117           case ImmediateState:
00118             state = doImmediateState();
00119             break;
00120           case ErrorState:
00121             panic("Went to the error state in the decoder.\n");
00122           default:
00123             panic("Unrecognized state! %d\n", state);
00124         }
00125     }

00128 Decoder::State
00129 Decoder::doFromCacheState()
00130 {
00131     DPRINTF(Decoder, "Looking at cache state.\n");
00132     if ((fetchChunk & instBytes->masks[chunkIdx]) !=
00133             instBytes->chunks[chunkIdx]) {
00134         DPRINTF(Decoder, "Decode cache miss.\n");
00135         // The chached chunks didn't match what was fetched. Fall back to the
00136         // predecoder.
00137         instBytes->chunks[chunkIdx] = fetchChunk;
00138         instBytes->chunks.resize(chunkIdx + 1);
00139         instBytes->si = NULL;
00140         chunkIdx = 0;
00141         fetchChunk = instBytes->chunks[0];
00142         offset = origPC % sizeof(MachInst);
00143         basePC = origPC - offset;
00144         return PrefixState;
00145     } else if (chunkIdx == instBytes->chunks.size() - 1) {
00146         // We matched the cache, so use its value.
00147         instDone = true;
00148         offset = instBytes->lastOffset;
00149         if (offset == sizeof(MachInst))
00150             outOfBytes = true;
00151         return ResetState;
00152     } else {
00153         // We matched so far, but need to check more chunks.
00154         chunkIdx++;
00155         outOfBytes = true;
00156         return FromCacheState;
00157     }
00158 }

00167     // REX prefixes are only recognized in 64 bit mode.
00168     if (prefix == RexPrefix && emi.mode.submode != SixtyFourBitMode)
00169         prefix = 0;

00180         DPRINTF(Decoder, "Found address size override prefix.\n");
00181         emi.legacy.addr = true;

00184       case CSOverride:
00185       case DSOverride:
00186       case ESOverride:
00187       case FSOverride:
00188       case GSOverride:
00189       case SSOverride:

00194         DPRINTF(Decoder, "Found lock prefix.\n");

00196         break;
00197       case Rep:
00198         DPRINTF(Decoder, "Found rep prefix.\n");

00202         DPRINTF(Decoder, "Found repne prefix.\n");

00204         break;

00206         DPRINTF(Decoder, "Found Rex prefix %#x.\n", nextByte);
00207         emi.rex = nextByte;
00208         break;
00209       case 0:
00210         nextState = OneByteOpcodeState;
00211         break;
00212       default:

00290 // Generic opcode processing which determines the immediate size, and whether
00291 // or not there's a modrm byte.
00292 Decoder::State
00293 Decoder::processOpcode(ByteTable &immTable, ByteTable &modrmTable,
00294                        bool addrSizedImm)
00295 {
00296     State nextState = ErrorState;
00297     const uint8_t opcode = emi.opcode.op;
00298 
00299     //Figure out the effective operand size. This can be overriden to
00300     //a fixed value at the decoder level.
00301     int logOpSize;
00302     if (emi.rex.w)
00303         logOpSize = 3; // 64 bit operand size
00304     else if (emi.legacy.op)
00305         logOpSize = altOp;
00306     else
00307         logOpSize = defOp;
00308 
00309     //Set the actual op size
00310     emi.opSize = 1 << logOpSize;
00311 
00312     //Figure out the effective address size. This can be overriden to
00313     //a fixed value at the decoder level.
00314     int logAddrSize;
00315     if(emi.legacy.addr)
00316         logAddrSize = altAddr;
00317     else
00318         logAddrSize = defAddr;
00319 
00320     //Set the actual address size
00321     emi.addrSize = 1 << logAddrSize;
00322 
00323     //Figure out the effective stack width. This can be overriden to
00324     //a fixed value at the decoder level.
00325     emi.stackSize = 1 << stack;
00326 
00327     //Figure out how big of an immediate we'll retreive based
00328     //on the opcode.
00329     int immType = immTable[opcode];
00330     if (addrSizedImm)
00331         immediateSize = SizeTypeToSize[logAddrSize - 1][immType];
00332     else
00333         immediateSize = SizeTypeToSize[logOpSize - 1][immType];
00334 
00335     //Determine what to expect next
00336     if (modrmTable[opcode]) {
00337         nextState = ModRMState;
00338     } else {
00339         if(immediateSize) {
00340             nextState = ImmediateState;
00341         } else {
00342             instDone = true;
00343             nextState = ResetState;
00344         }
00345     }
00346     return nextState;
00347 }

00349 //Get the ModRM byte and determine what displacement, if any, there is.
00350 //Also determine whether or not to get the SIB byte, displacement, or
00351 //immediate next.
00352 Decoder::State
00353 Decoder::doModRMState(uint8_t nextByte)
00354 {
00355     State nextState = ErrorState;
00356     ModRM modRM;
00357     modRM = nextByte;
00358     DPRINTF(Decoder, "Found modrm byte %#x.\n", nextByte);
00359     if (defOp == 1) {
00360         //figure out 16 bit displacement size
00361         if ((modRM.mod == 0 && modRM.rm == 6) || modRM.mod == 2)
00362             displacementSize = 2;
00363         else if (modRM.mod == 1)
00364             displacementSize = 1;
00365         else
00366             displacementSize = 0;
00367     } else {
00368         //figure out 32/64 bit displacement size
00369         if ((modRM.mod == 0 && modRM.rm == 5) || modRM.mod == 2)
00370             displacementSize = 4;
00371         else if (modRM.mod == 1)
00372             displacementSize = 1;
00373         else
00374             displacementSize = 0;
00375     }
00376 
00377     // The "test" instruction in group 3 needs an immediate, even though
00378     // the other instructions with the same actual opcode don't.
00379     if (emi.opcode.type == OneByteOpcode && (modRM.reg & 0x6) == 0) {
00380        if (emi.opcode.op == 0xF6)
00381            immediateSize = 1;
00382        else if (emi.opcode.op == 0xF7)
00383            immediateSize = (emi.opSize == 8) ? 4 : emi.opSize;
00384     }
00385 
00386     //If there's an SIB, get that next.
00387     //There is no SIB in 16 bit mode.
00388     if (modRM.rm == 4 && modRM.mod != 3) {
00389             // && in 32/64 bit mode)
00390         nextState = SIBState;
00391     } else if(displacementSize) {
00392         nextState = DisplacementState;
00393     } else if(immediateSize) {
00394         nextState = ImmediateState;
00395     } else {
00396         instDone = true;
00397         nextState = ResetState;
00398     }
00399     //The ModRM byte is consumed no matter what
00400     consumeByte();
00401     emi.modRM = modRM;
00402     return nextState;
00403 }

00538 Decoder::decode(PCState &nextPC)
00539 {
00540     if (!instDone)
00541         return NULL;
00542     instDone = false;
00543     updateNPC(nextPC);
00544 
00545     StaticInstPtr &si = instBytes->si;
00546     if (si)
00547         return si;
00548 
00549     // We didn't match in the AddrMap, but we still populated an entry. Fix
00550     // up its byte masks.
00551     const int chunkSize = sizeof(MachInst);
00552 
00553     instBytes->lastOffset = offset;
00554 
00555     Addr firstBasePC = basePC - (instBytes->chunks.size() - 1) * chunkSize;
00556     Addr firstOffset = origPC - firstBasePC;
00557     Addr totalSize = instBytes->lastOffset - firstOffset +
00558         (instBytes->chunks.size() - 1) * chunkSize;
00559     int start = firstOffset;
00560     instBytes->masks.clear();
00561 
00562     while (totalSize) {
00563         int end = start + totalSize;
00564         end = (chunkSize < end) ? chunkSize : end;
00565         int size = end - start;
00566         int idx = instBytes->masks.size();
00567 
00568         MachInst maskVal = mask(size * 8) << (start * 8);
00569         assert(maskVal);
00570 
00571         instBytes->masks.push_back(maskVal);
00572         instBytes->chunks[idx] &= instBytes->masks[idx];
00573         totalSize -= size;
00574         start = 0;
00575     }
00576 
00577     si = decode(emi, origPC);
00578     return si;

00084 {
00085     // Find the entry with the lowest (and hence least recently updated)
00086     // sequence number.
00087 
00088     unsigned lru = 0;
00089     for (unsigned i = 1; i < size; i++) {
00090         if (tlb[i].lruSeq < tlb[lru].lruSeq)

00096     tlb[lru].trieHandle = NULL;

00177 TLB::translateInt(RequestPtr req, ThreadContext *tc)
00178 {
00179     DPRINTF(TLB, "Addresses references internal memory.\n");
00180     Addr vaddr = req->getVaddr();
00181     Addr prefix = (vaddr >> 3) & IntAddrPrefixMask;
00182     if (prefix == IntAddrPrefixCPUID) {
00183         panic("CPUID memory space not yet implemented!\n");
00184     } else if (prefix == IntAddrPrefixMSR) {
00185         vaddr = (vaddr >> 3) & ~IntAddrPrefixMask;
00186         req->setFlags(Request::MMAPPED_IPR);
00187 
00188         MiscRegIndex regNum;
00189         if (!msrAddrToIndex(regNum, vaddr))
00190             return std::make_shared<GeneralProtection>(0);
00191 
00192         //The index is multiplied by the size of a MiscReg so that
00193         //any memory dependence calculations will not see these as
00194         //overlapping.
00195         req->setPaddr((Addr)regNum * sizeof(MiscReg));
00196         return NoFault;
00197     } else if (prefix == IntAddrPrefixIO) {
00198         // TODO If CPL > IOPL or in virtual mode, check the I/O permission
00199         // bitmap in the TSS.
00200 
00201         Addr IOPort = vaddr & ~IntAddrPrefixMask;
00202         // Make sure the address fits in the expected 16 bit IO address
00203         // space.
00204         assert(!(IOPort & ~0xFFFF));
00205         if (IOPort == 0xCF8 && req->getSize() == 4) {
00206             req->setFlags(Request::MMAPPED_IPR);
00207             req->setPaddr(MISCREG_PCI_CONFIG_ADDRESS * sizeof(MiscReg));
00208         } else if ((IOPort & ~mask(2)) == 0xCFC) {
00209             req->setFlags(Request::UNCACHEABLE | Request::STRICT_ORDER);
00210             Addr configAddress =
00211                 tc->readMiscRegNoEffect(MISCREG_PCI_CONFIG_ADDRESS);
00212             if (bits(configAddress, 31, 31)) {
00213                 req->setPaddr(PhysAddrPrefixPciConfig |
00214                         mbits(configAddress, 30, 2) |
00215                         (IOPort & mask(2)));
00216             } else {
00217                 req->setPaddr(PhysAddrPrefixIO | IOPort);
00218             }
00219         } else {
00220             req->setFlags(Request::UNCACHEABLE | Request::STRICT_ORDER);
00221             req->setPaddr(PhysAddrPrefixIO | IOPort);
00222         }
00223         return NoFault;
00224     } else {
00225         panic("Access to unrecognized internal address space %#x.\n",
00226                 prefix);
00227     }

00275         Mode mode, bool &delayedResponse, bool timing)
00276 {
00277     uint32_t flags = req->getFlags();
00278     int seg = flags & SegmentFlagMask;
00279     bool storeCheck = flags & (StoreCheck << FlagShift);
00280 
00281     delayedResponse = false;
00282 
00283     // If this is true, we're dealing with a request to a non-memory address
00284     // space.
00285     if (seg == SEGMENT_REG_MS) {
00286         return translateInt(req, tc);
00287     }
00288 
00289     Addr vaddr = req->getVaddr();
00290     DPRINTF(TLB, "Translating vaddr %#x.\n", vaddr);
00291 
00292     HandyM5Reg m5Reg = tc->readMiscRegNoEffect(MISCREG_M5_REG);
00293 
00294     // If protected mode has been enabled...
00295     if (m5Reg.prot) {
00296         DPRINTF(TLB, "In protected mode.\n");
00297         // If we're not in 64-bit mode, do protection/limit checks
00298         if (m5Reg.mode != LongMode) {
00299             DPRINTF(TLB, "Not in long mode. Checking segment protection.\n");
00300             // Check for a NULL segment selector.
00301             if (!(seg == SEGMENT_REG_TSG || seg == SYS_SEGMENT_REG_IDTR ||
00302                         seg == SEGMENT_REG_HS || seg == SEGMENT_REG_LS)
00303                     && !tc->readMiscRegNoEffect(MISCREG_SEG_SEL(seg)))
00304                 return std::make_shared<GeneralProtection>(0);
00305             bool expandDown = false;
00306             SegAttr attr = tc->readMiscRegNoEffect(MISCREG_SEG_ATTR(seg));
00307             if (seg >= SEGMENT_REG_ES && seg <= SEGMENT_REG_HS) {
00308                 if (!attr.writable && (mode == Write || storeCheck))
00309                     return std::make_shared<GeneralProtection>(0);
00310                 if (!attr.readable && mode == Read)
00311                     return std::make_shared<GeneralProtection>(0);
00312                 expandDown = attr.expandDown;
00313 
00314             }
00315             Addr base = tc->readMiscRegNoEffect(MISCREG_SEG_BASE(seg));
00316             Addr limit = tc->readMiscRegNoEffect(MISCREG_SEG_LIMIT(seg));
00317             bool sizeOverride = (flags & (AddrSizeFlagBit << FlagShift));
00318             unsigned logSize = sizeOverride ? (unsigned)m5Reg.altAddr
00319                                             : (unsigned)m5Reg.defAddr;
00320             int size = (1 << logSize) * 8;
00321             Addr offset = bits(vaddr - base, size - 1, 0);
00322             Addr endOffset = offset + req->getSize() - 1;
00323             if (expandDown) {
00324                 DPRINTF(TLB, "Checking an expand down segment.\n");
00325                 warn_once("Expand down segments are untested.\n");
00326                 if (offset <= limit || endOffset <= limit)
00327                     return std::make_shared<GeneralProtection>(0);
00328             } else {
00329                 if (offset > limit || endOffset > limit)
00330                     return std::make_shared<GeneralProtection>(0);
00331             }
00332         }
00333         if (m5Reg.submode != SixtyFourBitMode ||
00334                 (flags & (AddrSizeFlagBit << FlagShift)))
00335             vaddr &= mask(32);
00336         // If paging is enabled, do the translation.
00337         if (m5Reg.paging) {
00338             DPRINTF(TLB, "Paging enabled.\n");
00339             // The vaddr already has the segment base applied.
00340             TlbEntry *entry = lookup(vaddr);
00341             if (!entry) {
00342                 if (FullSystem) {
00343                     Fault fault = walker->start(tc, translation, req, mode);
00344                     if (timing || fault != NoFault) {
00345                         // This gets ignored in atomic mode.
00346                         delayedResponse = true;
00347                         return fault;
00348                     }
00349                     entry = lookup(vaddr);
00350                     assert(entry);
00351                 } else {
00352                     DPRINTF(TLB, "Handling a TLB miss for "
00353                             "address %#x at pc %#x.\n",
00354                             vaddr, tc->instAddr());
00355 
00356                     Process *p = tc->getProcessPtr();
00357                     TlbEntry newEntry;
00358                     bool success = p->pTable->lookup(vaddr, newEntry);
00359                     if (!success && mode != Execute) {
00360                         // Check if we just need to grow the stack.
00361                         if (p->fixupStackFault(vaddr)) {
00362                             // If we did, lookup the entry for the new page.
00363                             success = p->pTable->lookup(vaddr, newEntry);
00364                         }
00365                     }
00366                     if (!success) {
00367                         return std::make_shared<PageFault>(vaddr, true, mode,
00368                                                            true, false);
00369                     } else {
00370                         Addr alignedVaddr = p->pTable->pageAlign(vaddr);
00371                         DPRINTF(TLB, "Mapping %#x to %#x\n", alignedVaddr,
00372                                 newEntry.pageStart());
00373                         entry = insert(alignedVaddr, newEntry);
00374                     }
00375                     DPRINTF(TLB, "Miss was serviced.\n");
00376                 }
00377             }
00378 
00379             DPRINTF(TLB, "Entry found with paddr %#x, "
00380                     "doing protection checks.\n", entry->paddr);
00381             // Do paging protection checks.
00382             bool inUser = (m5Reg.cpl == 3 &&
00383                     !(flags & (CPL0FlagBit << FlagShift)));
00384             CR0 cr0 = tc->readMiscRegNoEffect(MISCREG_CR0);
00385             bool badWrite = (!entry->writable && (inUser || cr0.wp));
00386             if ((inUser && !entry->user) || (mode == Write && badWrite)) {
00387                 // The page must have been present to get into the TLB in
00388                 // the first place. We'll assume the reserved bits are
00389                 // fine even though we're not checking them.
00390                 return std::make_shared<PageFault>(vaddr, true, mode, inUser,
00391                                                    false);
00392             }
00393             if (storeCheck && badWrite) {
00394                 // This would fault if this were a write, so return a page
00395                 // fault that reflects that happening.
00396                 return std::make_shared<PageFault>(vaddr, true, Write, inUser,
00397                                                    false);
00398             }
00399 
00400             Addr paddr = entry->paddr | (vaddr & mask(entry->logBytes));
00401             DPRINTF(TLB, "Translated %#x -> %#x.\n", vaddr, paddr);
00402             req->setPaddr(paddr);
00403             if (entry->uncacheable)
00404                 req->setFlags(Request::UNCACHEABLE | Request::STRICT_ORDER);
00405         } else {
00406             //Use the address which already has segmentation applied.
00407             DPRINTF(TLB, "Paging disabled.\n");
00408             DPRINTF(TLB, "Translated %#x -> %#x.\n", vaddr, vaddr);
00409             req->setPaddr(vaddr);
00410         }
00411     } else {
00412         // Real mode
00413         DPRINTF(TLB, "In real mode.\n");
00414         DPRINTF(TLB, "Translated %#x -> %#x.\n", vaddr, vaddr);
00415         req->setPaddr(vaddr);
00416     }
00417 
00418     return finalizePhysical(req, tc, mode);
00419 }

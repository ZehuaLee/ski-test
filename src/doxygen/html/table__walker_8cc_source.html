00369     // @TODO Should this always be the TLB or should we look in the stage2 TLB?
00370     TlbEntry* te = tlb->lookup(currState->vaddr, currState->asid,
00371             currState->vmid, currState->isHyp, currState->isSecure, true, false,

00374     // Check if we still need to have a walk for this request. If the requesting
00375     // instruction has been squashed, or a previous walk has filled the TLB with
00376     // a match, we just want to get rid of the walk. The latter could happen
00377     // when there are multiple outstanding misses to a single page and a
00378     // previous request has been successfully translated.
00379     if (!currState->transState->squashed() && !te) {
00380         // We've got a valid request, lets process it
00381         pending = true;
00382         pendingQueue.pop_front();

00384         WalkerState *curr_state_copy = currState;
00385         Fault f;
00386         if (currState->aarch64)
00387             f = processWalkAArch64();
00388         else if ((_haveLPAE && currState->ttbcr.eae) || currState->isHyp || isStage2)
00389             f = processWalkLPAE();
00390         else
00391             f = processWalk();

00393         if (f != NoFault) {
00394             curr_state_copy->transState->finish(f, curr_state_copy->req,
00395                     curr_state_copy->tc, curr_state_copy->mode);
00396 
00397             delete curr_state_copy;
00398         }
00399         return;
00400     }
00401 
00402 
00403     // If the instruction that we were translating for has been
00404     // squashed we shouldn't bother.
00405     unsigned num_squashed = 0;

00413         DPRINTF(TLB, "Squashing table walk for address %#x\n",
00414                       currState->vaddr_tainted);
00415

00418             currState->transState->finish(
00419                 std::make_shared<UnimpFault>("Squashed Inst"),
00420                 currState->req, currState->tc, currState->mode);

00422             // translate the request now that we know it will work
00423             statWalkServiceTime.sample(curTick() - currState->startTime);

00425                         currState->transState, currState->mode);
00426 
00427         }
00428

00431

01204 void
01205 TableWalker::memAttrsLPAE(ThreadContext *tc, TlbEntry &te,
01206     LongDescriptor &lDescriptor)
01207 {
01208     assert(_haveLPAE);
01209 
01210     uint8_t attr;
01211     uint8_t sh = lDescriptor.sh();
01212     // Different format and source of attributes if this is a stage 2
01213     // translation
01214     if (isStage2) {
01215         attr = lDescriptor.memAttr();
01216         uint8_t attr_3_2 = (attr >> 2) & 0x3;
01217         uint8_t attr_1_0 =  attr       & 0x3;
01218 
01219         DPRINTF(TLBVerbose, "memAttrsLPAE MemAttr:%#x sh:%#x\n", attr, sh);
01220 
01221         if (attr_3_2 == 0) {
01222             te.mtype        = attr_1_0 == 0 ? TlbEntry::MemoryType::StronglyOrdered
01223                                             : TlbEntry::MemoryType::Device;
01224             te.outerAttrs   = 0;
01225             te.innerAttrs   = attr_1_0 == 0 ? 1 : 3;
01226             te.nonCacheable = true;
01227         } else {
01228             te.mtype        = TlbEntry::MemoryType::Normal;
01229             te.outerAttrs   = attr_3_2 == 1 ? 0 :
01230                               attr_3_2 == 2 ? 2 : 1;
01231             te.innerAttrs   = attr_1_0 == 1 ? 0 :
01232                               attr_1_0 == 2 ? 6 : 5;
01233             te.nonCacheable = (attr_3_2 == 1) || (attr_1_0 == 1);
01234         }
01235     } else {
01236         uint8_t attrIndx = lDescriptor.attrIndx();
01237 
01238         // LPAE always uses remapping of memory attributes, irrespective of the
01239         // value of SCTLR.TRE
01240         MiscRegIndex reg = attrIndx & 0x4 ? MISCREG_MAIR1 : MISCREG_MAIR0;
01241         int reg_as_int = flattenMiscRegNsBanked(reg, currState->tc,
01242                                                 !currState->isSecure);
01243         uint32_t mair = currState->tc->readMiscReg(reg_as_int);
01244         attr = (mair >> (8 * (attrIndx % 4))) & 0xff;
01245         uint8_t attr_7_4 = bits(attr, 7, 4);
01246         uint8_t attr_3_0 = bits(attr, 3, 0);
01247         DPRINTF(TLBVerbose, "memAttrsLPAE AttrIndx:%#x sh:%#x, attr %#x\n", attrIndx, sh, attr);
01248 
01249         // Note: the memory subsystem only cares about the 'cacheable' memory
01250         // attribute. The other attributes are only used to fill the PAR register
01251         // accordingly to provide the illusion of full support
01252         te.nonCacheable = false;
01253 
01254         switch (attr_7_4) {
01255           case 0x0:
01256             // Strongly-ordered or Device memory
01257             if (attr_3_0 == 0x0)
01258                 te.mtype = TlbEntry::MemoryType::StronglyOrdered;
01259             else if (attr_3_0 == 0x4)
01260                 te.mtype = TlbEntry::MemoryType::Device;
01261             else
01262                 panic("Unpredictable behavior\n");
01263             te.nonCacheable = true;
01264             te.outerAttrs   = 0;
01265             break;
01266           case 0x4:
01267             // Normal memory, Outer Non-cacheable
01268             te.mtype = TlbEntry::MemoryType::Normal;
01269             te.outerAttrs = 0;
01270             if (attr_3_0 == 0x4)
01271                 // Inner Non-cacheable
01272                 te.nonCacheable = true;
01273             else if (attr_3_0 < 0x8)
01274                 panic("Unpredictable behavior\n");
01275             break;
01276           case 0x8:
01277           case 0x9:
01278           case 0xa:
01279           case 0xb:
01280           case 0xc:
01281           case 0xd:
01282           case 0xe:
01283           case 0xf:
01284             if (attr_7_4 & 0x4) {
01285                 te.outerAttrs = (attr_7_4 & 1) ? 1 : 3;
01286             } else {
01287                 te.outerAttrs = 0x2;
01288             }
01289             // Normal memory, Outer Cacheable
01290             te.mtype = TlbEntry::MemoryType::Normal;
01291             if (attr_3_0 != 0x4 && attr_3_0 < 0x8)
01292                 panic("Unpredictable behavior\n");
01293             break;
01294           default:
01295             panic("Unpredictable behavior\n");
01296             break;
01297         }
01298 
01299         switch (attr_3_0) {
01300           case 0x0:
01301             te.innerAttrs = 0x1;
01302             break;
01303           case 0x4:
01304             te.innerAttrs = attr_7_4 == 0 ? 0x3 : 0;
01305             break;
01306           case 0x8:
01307           case 0x9:
01308           case 0xA:
01309           case 0xB:
01310             te.innerAttrs = 6;
01311             break;
01312           case 0xC:
01313           case 0xD:
01314           case 0xE:
01315           case 0xF:
01316             te.innerAttrs = attr_3_0 & 1 ? 0x5 : 0x7;
01317             break;
01318           default:
01319             panic("Unpredictable behavior\n");
01320             break;
01321         }
01322     }
01323 
01324     te.outerShareable = sh == 2;
01325     te.shareable       = (sh & 0x2) ? true : false;
01326     te.setAttributes(true);
01327     te.attributes |= (uint64_t) attr << 56;
01328 }

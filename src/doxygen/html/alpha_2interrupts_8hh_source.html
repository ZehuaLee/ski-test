00143     Fault
00144     getInterrupt(ThreadContext *tc)
00145     {
00146         uint64_t ipl = 0;
00147         uint64_t summary = 0;
00148 
00149         if (tc->readMiscRegNoEffect(IPR_ASTRR))
00150             panic("asynchronous traps not implemented\n");
00151 
00152         if (tc->readMiscRegNoEffect(IPR_SIRR)) {
00153             for (uint64_t i = INTLEVEL_SOFTWARE_MIN;
00154                  i < INTLEVEL_SOFTWARE_MAX; i++) {
00155                 if (tc->readMiscRegNoEffect(IPR_SIRR) & (ULL(1) << i)) {
00156                     // See table 4-19 of 21164 hardware reference
00157                     ipl = (i - INTLEVEL_SOFTWARE_MIN) + 1;
00158                     summary |= (ULL(1) << i);
00159                 }
00160             }
00161         }
00162 
00163         if (intstatus) {
00164             for (uint64_t i = INTLEVEL_EXTERNAL_MIN;
00165                  i < INTLEVEL_EXTERNAL_MAX; i++) {
00166                 if (intstatus & (ULL(1) << i)) {
00167                     // See table 4-19 of 21164 hardware reference
00168                     ipl = i;
00169                     summary |= (ULL(1) << i);
00170                 }
00171             }
00172         }
00173 
00174         if (ipl && ipl > tc->readMiscRegNoEffect(IPR_IPLR)) {
00175             newIpl = ipl;
00176             newSummary = summary;
00177             newInfoSet = true;
00178             DPRINTF(Flow, "Interrupt! IPLR=%d ipl=%d summary=%x\n",
00179                     tc->readMiscRegNoEffect(IPR_IPLR), ipl, summary);
00180 
00181             return std::make_shared<InterruptFault>();
00182         } else {
00183             return NoFault;
00184         }
00185     }

00058 void
00059 Trace::ExeTracerRecord::traceInst(const StaticInstPtr &inst, bool ran)
00060 {
00061     ostream &outs = Trace::output();
00062 
00063     if (!Debug::ExecUser || !Debug::ExecKernel) {
00064         bool in_user_mode = TheISA::inUserMode(thread);
00065         if (in_user_mode && !Debug::ExecUser) return;
00066         if (!in_user_mode && !Debug::ExecKernel) return;
00067     }
00068 
00069     if (Debug::ExecTicks)
00070         dumpTicks(outs);
00071 
00072     outs << thread->getCpuPtr()->name() << " ";
00073 
00074     if (Debug::ExecAsid)
00075         outs << "A" << dec << TheISA::getExecutingAsid(thread) << " ";
00076 
00077     if (Debug::ExecThread)
00078         outs << "T" << thread->threadId() << " : ";
00079 
00080     std::string sym_str;
00081     Addr sym_addr;
00082     Addr cur_pc = pc.instAddr();
00083     if (debugSymbolTable && Debug::ExecSymbol &&
00084             (!FullSystem || !inUserMode(thread)) &&
00085             debugSymbolTable->findNearestSymbol(cur_pc, sym_str, sym_addr)) {
00086         if (cur_pc != sym_addr)
00087             sym_str += csprintf("+%d",cur_pc - sym_addr);
00088         outs << "@" << sym_str;
00089     } else {
00090         outs << "0x" << hex << cur_pc;
00091     }
00092 
00093     if (inst->isMicroop()) {
00094         outs << "." << setw(2) << dec << pc.microPC();
00095     } else {
00096         outs << "   ";
00097     }
00098 
00099     outs << " : ";
00100 
00101     //
00102     //  Print decoded instruction
00103     //
00104 
00105     outs << setw(26) << left;
00106     outs << inst->disassemble(cur_pc, debugSymbolTable);
00107 
00108     if (ran) {
00109         outs << " : ";
00110 
00111         if (Debug::ExecOpClass) {
00112             outs << Enums::OpClassStrings[inst->opClass()] << " : ";
00113         }
00114 
00115         if (Debug::ExecResult && !predicate) {
00116             outs << "Predicated False";
00117         }
00118 
00119         if (Debug::ExecResult && data_status != DataInvalid) {
00120             ccprintf(outs, " D=%#018x", data.as_int);
00121         }
00122 
00123         if (Debug::ExecEffAddr && getMemValid())
00124             outs << " A=0x" << hex << addr;
00125 
00126         if (Debug::ExecFetchSeq && fetch_seq_valid)
00127             outs << "  FetchSeq=" << dec << fetch_seq;
00128 
00129         if (Debug::ExecCPSeq && cp_seq_valid)
00130             outs << "  CPSeq=" << dec << cp_seq;
00131 
00132         if (Debug::ExecFlags) {
00133             outs << "  flags=(";
00134             inst->printFlags(outs, "|");
00135             outs << ")";
00136         }
00137     }
00138 
00139     //
00140     //  End of line...
00141     //
00142     outs << endl;
00143 }

00145 void
00146 Trace::ExeTracerRecord::dump()
00147 {
00148     /*
00149      * The behavior this check tries to achieve is that if ExecMacro is on,
00150      * the macroop will be printed. If it's on and microops are also on, it's
00151      * printed before the microops start printing to give context. If the
00152      * microops aren't printed, then it's printed only when the final microop
00153      * finishes. Macroops then behave like regular instructions and don't
00154      * complete/print when they fault.
00155      */
00156     if (Debug::ExecMacro && staticInst->isMicroop() &&
00157         ((Debug::ExecMicro &&
00158             macroStaticInst && staticInst->isFirstMicroop()) ||
00159             (!Debug::ExecMicro &&
00160              macroStaticInst && staticInst->isLastMicroop()))) {
00161         traceInst(macroStaticInst, false);
00162     }
00163     if (Debug::ExecMicro || !staticInst->isMicroop()) {
00164         traceInst(staticInst, true);
00165     }
00166 }

00295     void clearLoadLocks(Request *req = NULL)
00296     {
00297         if (!req) {
00298             // No request, invaldate all locks to this line
00299             lockList.clear();
00300         } else {
00301             // Only invalidate locks that overlap with this request
00302             std::list<Lock>::iterator lock_itr = lockList.begin();
00303             while (lock_itr != lockList.end()) {
00304                 if (lock_itr->overlapping(req)) {
00305                     lock_itr = lockList.erase(lock_itr);
00306                 } else {
00307                     ++lock_itr;
00308                 }
00309             }
00310         }

00354     bool checkWrite(PacketPtr pkt)
00355     {
00356         Request *req = pkt->req;
00357         if (pkt->isLLSC()) {
00358             // it's a store conditional... have to check for matching
00359             // load locked.
00360             bool success = false;
00361 
00362             for (std::list<Lock>::iterator i = lockList.begin();
00363                  i != lockList.end(); ++i)
00364             {
00365                 if (i->matchesContext(req)) {
00366                     // it's a store conditional, and as far as the memory
00367                     // system can tell, the requesting context's lock is
00368                     // still valid.
00369                     success = true;
00370                     break;
00371                 }
00372             }
00373 
00374             req->setExtraData(success ? 1 : 0);
00375             clearLoadLocks(req);
00376             return success;
00377         } else {
00378             // for *all* stores (conditional or otherwise) we have to
00379             // clear the list of load-locks as they're all invalid now.
00380             clearLoadLocks(req);
00381             return true;
00382         }

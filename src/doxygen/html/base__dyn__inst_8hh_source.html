00978                                 RequestPtr &sreqHigh)
00979 {
00980     // Check to see if the request crosses the next level block boundary.
00981     unsigned block_size = cpu->cacheLineSize();
00982     Addr addr = req->getVaddr();
00983     Addr split_addr = roundDown(addr + req->getSize() - 1, block_size);
00984     assert(split_addr <= addr || split_addr - addr < block_size);

00986     // Spans two blocks.
00987     if (split_addr > addr) {
00988         req->splitOnVaddr(split_addr, sreqLow, sreqHigh);

01020             savedSreqHigh = state->sreqHigh;
01021         }
01022     } else {

01024             new WholeTranslationState(req, sreqLow, sreqHigh, NULL, res, mode);
01025

01027         DataTranslation<BaseDynInstPtr> *stransLow =
01028             new DataTranslation<BaseDynInstPtr>(this, state, 0);
01029         DataTranslation<BaseDynInstPtr> *stransHigh =
01030             new DataTranslation<BaseDynInstPtr>(this, state, 1);
01031 
01032         cpu->dtb->translateTiming(sreqLow, thread->getTC(), stransLow, mode);
01033         cpu->dtb->translateTiming(sreqHigh, thread->getTC(), stransHigh, mode);
01034 
01035         if (!translationCompleted()) {

01037             // fault. Overwrite any existing fault we might have from a previous
01038             // execution of this instruction (e.g. an uncachable load that
01039             // couldn't execute because it wasn't at the head of the ROB).

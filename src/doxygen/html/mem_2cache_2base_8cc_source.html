00091 void
00092 BaseCache::CacheSlavePort::setBlocked()
00093 {
00094     assert(!blocked);
00095     DPRINTF(CachePort, "Port is blocking new requests\n");
00096     blocked = true;
00097     // if we already scheduled a retry in this cycle, but it has not yet
00098     // happened, cancel it
00099     if (sendRetryEvent.scheduled()) {
00100         owner.deschedule(sendRetryEvent);
00101         DPRINTF(CachePort, "Port descheduled retry\n");
00102         mustSendRetry = true;
00103     }
00104 }

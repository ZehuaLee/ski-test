00078 bool
00079 MessageBuffer::areNSlotsAvailable(unsigned int n)
00080 {
00081 
00082     // fast path when message buffers have infinite size
00083     if (m_max_size == 0) {
00084         return true;
00085     }
00086 
00087     // determine the correct size for the current cycle
00088     // pop operations shouldn't effect the network's visible size
00089     // until next cycle, but enqueue operations effect the visible
00090     // size immediately
00091     unsigned int current_size = 0;
00092 
00093     if (m_time_last_time_pop < m_sender->clockEdge()) {
00094         // no pops this cycle - heap size is correct
00095         current_size = m_prio_heap.size();
00096     } else {
00097         if (m_time_last_time_enqueue < m_sender->curCycle()) {
00098             // no enqueues this cycle - m_size_at_cycle_start is correct
00099             current_size = m_size_at_cycle_start;
00100         } else {
00101             // both pops and enqueues occured this cycle - add new
00102             // enqueued msgs to m_size_at_cycle_start
00103             current_size = m_size_at_cycle_start + m_msgs_this_cycle;
00104         }
00105     }
00106 
00107     // now compare the new size with our max size
00108     if (current_size + n <= m_max_size) {
00109         return true;
00110     } else {
00111         DPRINTF(RubyQueue, "n: %d, current_size: %d, heap size: %d, "
00112                 "m_max_size: %d\n",
00113                 n, current_size, m_prio_heap.size(), m_max_size);
00114         m_not_avail_count++;
00115         return false;
00116     }
00117 }

00144 void
00145 MessageBuffer::enqueue(MsgPtr message, Cycles delta)
00146 {
00147     assert(m_ordering_set);
00148 
00149     // record current time incase we have a pop that also adjusts my size
00150     if (m_time_last_time_enqueue < m_sender->curCycle()) {
00151         m_msgs_this_cycle = 0;  // first msg this cycle
00152         m_time_last_time_enqueue = m_sender->curCycle();
00153     }
00154 
00155     m_msg_counter++;
00156     m_msgs_this_cycle++;
00157 
00158     // Calculate the arrival time of the message, that is, the first
00159     // cycle the message can be dequeued.
00160     assert(delta > 0);
00161     Tick current_time = m_sender->clockEdge();
00162     Tick arrival_time = 0;
00163 
00164     if (!RubySystem::getRandomization() || !m_randomization) {
00165         // No randomization
00166         arrival_time = current_time + delta * m_sender->clockPeriod();
00167     } else {
00168         // Randomization - ignore delta
00169         if (m_strict_fifo) {
00170             if (m_last_arrival_time < current_time) {
00171                 m_last_arrival_time = current_time;
00172             }
00173             arrival_time = m_last_arrival_time +
00174                            random_time() * m_sender->clockPeriod();
00175         } else {
00176             arrival_time = current_time +
00177                            random_time() * m_sender->clockPeriod();
00178         }
00179     }
00180 
00181     // Check the arrival time
00182     assert(arrival_time > current_time);
00183     if (m_strict_fifo) {
00184         if (arrival_time < m_last_arrival_time) {
00185             panic("FIFO ordering violated: %s name: %s current time: %d "
00186                   "delta: %d arrival_time: %d last arrival_time: %d\n",
00187                   *this, m_name, current_time,
00188                   delta * m_sender->clockPeriod(),
00189                   arrival_time, m_last_arrival_time);
00190         }
00191     }
00192 
00193     // If running a cache trace, don't worry about the last arrival checks
00194     if (!RubySystem::getWarmupEnabled()) {
00195         m_last_arrival_time = arrival_time;
00196     }
00197 
00198     // compute the delay cycles and set enqueue time
00199     Message* msg_ptr = message.get();
00200     assert(msg_ptr != NULL);
00201 
00202     assert(m_sender->clockEdge() >= msg_ptr->getLastEnqueueTime() &&
00203            "ensure we aren't dequeued early");
00204 
00205     msg_ptr->updateDelayedTicks(m_sender->clockEdge());
00206     msg_ptr->setLastEnqueueTime(arrival_time);
00207 
00208     // Insert the message into the priority heap
00209     MessageBufferNode thisNode(arrival_time, m_msg_counter, message);
00210     m_prio_heap.push_back(thisNode);
00211     push_heap(m_prio_heap.begin(), m_prio_heap.end(),
00212         greater<MessageBufferNode>());
00213 
00214     DPRINTF(RubyQueue, "Enqueue arrival_time: %lld, Message: %s\n",
00215             arrival_time, *(message.get()));
00216 
00217     // Schedule the wakeup
00218     assert(m_consumer != NULL);
00219     m_consumer->scheduleEventAbsolute(arrival_time);
00220     m_consumer->storeEventInfo(m_vnet_id);
00221 }

00372         Message *msg = m_prio_heap[i].m_msgptr.get();

00379          map_iter != m_stall_msg_map.end();

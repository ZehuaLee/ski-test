00364     // Checks the cache and returns the id of the port that has the requested
00365     // address within its range
00366     inline PortID checkPortCache(Addr addr) const {
00367         if (portCache[0].valid && portCache[0].range.contains(addr)) {
00368             return portCache[0].id;
00369         }
00370         if (portCache[1].valid && portCache[1].range.contains(addr)) {
00371             return portCache[1].id;
00372         }
00373         if (portCache[2].valid && portCache[2].range.contains(addr)) {
00374             return portCache[2].id;
00375         }
00376 
00377         return InvalidPortID;
00378     }

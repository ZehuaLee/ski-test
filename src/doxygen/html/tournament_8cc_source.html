00068         fatal("Invalid global predictor size!\n");

00112     for (int i = 0; i < choicePredictorSize; ++i)

00115     //Set up historyRegisterMask
00116     historyRegisterMask = mask(globalHistoryBits);
00117 
00118     //Check that predictors don't use more bits than they have available
00119     if (globalHistoryMask > historyRegisterMask) {

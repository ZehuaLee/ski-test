00347         decodeStatus[tid] == Unblocking) {
00348         if (FullSystem) {
00349             toFetch->decodeUnblock[tid] = 1;
00350         } else {
00351             // In syscall emulation, we can have both a block and a squash due

00356                 toFetch->decodeBlock[tid] = 0;

00366     unsigned squash_count = 0;
00367

00370             fromFetch->insts[i]->setSquashed();

00373     }

00491 template <class Impl>
00492 bool
00493 DefaultDecode<Impl>::checkSignalsAndUpdate(ThreadID tid)
00494 {
00495     // Check if there's a squash signal, squash if there is.
00496     // Check stall signals, block if necessary.
00497     // If status was blocked
00498     //     Check if stall conditions have passed
00499     //         if so then go to unblocking
00500     // If status was Squashing
00501     //     check if squashing is not high.  Switch to running this cycle.
00502 
00503     // Update the per thread stall statuses.
00504     readStallSignals(tid);
00505 
00506     // Check squash signals from commit.
00507     if (fromCommit->commitInfo[tid].squash) {
00508 
00509         DPRINTF(Decode, "[tid:%u]: Squashing instructions due to squash "
00510                 "from commit.\n", tid);
00511 
00512         squash(tid);
00513 
00514         return true;
00515     }
00516 
00517     if (checkStall(tid)) {
00518         return block(tid);
00519     }
00520 
00521     if (decodeStatus[tid] == Blocked) {
00522         DPRINTF(Decode, "[tid:%u]: Done blocking, switching to unblocking.\n",
00523                 tid);
00524 
00525         decodeStatus[tid] = Unblocking;
00526 
00527         unblock(tid);
00528 
00529         return true;
00530     }
00531 
00532     if (decodeStatus[tid] == Squashing) {
00533         // Switch status to running if decode isn't being told to block or
00534         // squash this cycle.
00535         DPRINTF(Decode, "[tid:%u]: Done squashing, switching to running.\n",
00536                 tid);
00537 
00538         decodeStatus[tid] = Running;
00539 
00540         return false;
00541     }
00542 
00543     // If we've reached this point, we have not gotten any signals that
00544     // cause decode to change its status.  Decode remains the same as before.
00545     return false;
00546 }

00584 template<class Impl>
00585 void
00586 DefaultDecode<Impl>::decode(bool &status_change, ThreadID tid)
00587 {
00588     // If status is Running or idle,
00589     //     call decodeInsts()
00590     // If status is Unblocking,
00591     //     buffer any instructions coming from fetch
00592     //     continue trying to empty skid buffer
00593     //     check if stall conditions have passed
00594 
00595     if (decodeStatus[tid] == Blocked) {
00596         ++decodeBlockedCycles;
00597     } else if (decodeStatus[tid] == Squashing) {
00598         ++decodeSquashCycles;
00599     }
00600 
00601     // Decode should try to decode as many instructions as its bandwidth
00602     // will allow, as long as it is not currently blocked.
00603     if (decodeStatus[tid] == Running ||
00604         decodeStatus[tid] == Idle) {
00605         DPRINTF(Decode, "[tid:%u]: Not blocked, so attempting to run "
00606                 "stage.\n",tid);
00607 
00608         decodeInsts(tid);
00609     } else if (decodeStatus[tid] == Unblocking) {
00610         // Make sure that the skid buffer has something in it if the
00611         // status is unblocking.
00612         assert(!skidsEmpty());
00613 
00614         // If the status was unblocking, then instructions from the skid
00615         // buffer were used.  Remove those instructions and handle
00616         // the rest of unblocking.
00617         decodeInsts(tid);
00618 
00619         if (fetchInstsValid()) {
00620             // Add the current inputs to the skid buffer so they can be
00621             // reprocessed when this stage unblocks.
00622             skidInsert(tid);
00623         }
00624 
00625         status_change = unblock(tid) || status_change;
00626     }
00627 }

00634     // instructions coming from fetch, depending on decode's status.

00643         return;

00645         DPRINTF(Decode, "[tid:%u] Unblocking, removing insts from skid "
00646                 "buffer.\n",tid);
00647         ++decodeUnblockCycles;
00648     } else if (decodeStatus[tid] == Running) {
00649         ++decodeRunCycles;
00650     }
00651 
00652     DynInstPtr inst;
00653

00655         &insts_to_decode = decodeStatus[tid] == Unblocking ?
00656         skidBuffer[tid] : insts[tid];
00657

00659 
00660     while (insts_available > 0 && toRenameIndex < decodeWidth) {
00661         assert(!insts_to_decode.empty());
00662 
00663         inst = insts_to_decode.front();
00664 
00665         insts_to_decode.pop();
00666 
00667         DPRINTF(Decode, "[tid:%u]: Processing instruction [sn:%lli] with "
00668                 "PC %s\n", tid, inst->seqNum, inst->pcState());

00672                     "squashed, skipping.\n",
00673                     tid, inst->seqNum, inst->pcState());
00674 
00675             ++decodeSquashedInsts;
00676 
00677             --insts_available;
00678 
00679             continue;
00680         }
00681 
00682         // Also check if instructions have no source registers.  Mark
00683         // them as ready to issue at any time.  Not sure if this check
00684         // should exist here or at a later stage; however it doesn't matter
00685         // too much for function correctness.
00686         if (inst->numSrcRegs() == 0) {
00687             inst->setCanIssue();
00688         }
00689 
00690         // This current instruction is valid, so add it into the decode
00691         // queue.  The next instruction may not be valid, so check to
00692         // see if branches were predicted correctly.

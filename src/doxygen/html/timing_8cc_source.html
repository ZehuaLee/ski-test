00745 {
00746     // received a response from the dcache: complete the load or store
00747     // instruction
00748     assert(!pkt->isError());

00750            pkt->req->getFlags().isSet(Request::NO_ACCESS));
00751

00757         SplitFragmentSenderState * send_state =
00758             dynamic_cast<SplitFragmentSenderState *>(pkt->senderState);

00766             dynamic_cast<SplitMainSenderState *>(big_pkt->senderState);
00767         assert(main_send_state);

00769         assert(main_send_state->outstanding != 0);

00773             return;

00776             big_pkt->senderState = NULL;

00832 bool
00833 TimingSimpleCPU::DcachePort::recvTimingResp(PacketPtr pkt)
00834 {
00835     DPRINTF(SimpleCPU, "Received load/store response %#x\n", pkt->getAddr());
00836 
00837     // The timing CPU is not really ticked, instead it relies on the
00838     // memory system (fetch and load/store) to set the pace.
00839     if (!tickEvent.scheduled()) {
00840         // Delay processing of returned data until next CPU clock edge
00841         tickEvent.schedule(pkt, cpu->clockEdge());

00845         // faster than a CPU we could get two responses in the

00848             cpu->schedule(retryRespEvent, cpu->clockEdge(Cycles(1)));

00859 void
00860 TimingSimpleCPU::DcachePort::recvReqRetry()
00861 {
00862     // we shouldn't get a retry unless we have a packet that we're
00863     // waiting to transmit
00864     assert(cpu->dcache_pkt != NULL);
00865     assert(cpu->_status == DcacheRetry);
00866     PacketPtr tmp = cpu->dcache_pkt;
00867     if (tmp->senderState) {
00868         // This is a packet from a split access.
00869         SplitFragmentSenderState * send_state =
00870             dynamic_cast<SplitFragmentSenderState *>(tmp->senderState);
00871         assert(send_state);
00872         PacketPtr big_pkt = send_state->bigPkt;
00873         
00874         SplitMainSenderState * main_send_state =
00875             dynamic_cast<SplitMainSenderState *>(big_pkt->senderState);
00876         assert(main_send_state);
00877 
00878         if (sendTimingReq(tmp)) {
00879             // If we were able to send without retrying, record that fact
00880             // and try sending the other fragment.
00881             send_state->clearFromParent();
00882             int other_index = main_send_state->getPendingFragment();
00883             if (other_index > 0) {
00884                 tmp = main_send_state->fragments[other_index];
00885                 cpu->dcache_pkt = tmp;
00886                 if ((big_pkt->isRead() && cpu->handleReadPacket(tmp)) ||
00887                         (big_pkt->isWrite() && cpu->handleWritePacket())) {
00888                     main_send_state->fragments[other_index] = NULL;
00889                 }
00890             } else {
00891                 cpu->_status = DcacheWaitResponse;
00892                 // memory system takes ownership of packet
00893                 cpu->dcache_pkt = NULL;
00894             }
00895         }
00896     } else if (sendTimingReq(tmp)) {
00897         cpu->_status = DcacheWaitResponse;
00898         // memory system takes ownership of packet
00899         cpu->dcache_pkt = NULL;
00900     }
00901 }

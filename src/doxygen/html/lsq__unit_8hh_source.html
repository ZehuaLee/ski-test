00562     // Make sure this isn't a strictly ordered load

00564     // only if they're at the head of the LSQ and are ready to commit

00571                 load_inst->seqNum, load_inst->pcState());

00574         // memory.  This is quite ugly.  @todo: Figure out the proper
00575         // place to really handle request deletes.

00577         if (TheISA::HasUnalignedMemAcc && sreqLow) {
00578             delete sreqLow;
00579             delete sreqHigh;

00582             "Strictly ordered load [sn:%llx] PC %s\n",

00591     DPRINTF(LSQUnit, "Read called, load idx: %i, store idx: %i, "
00592             "storeHead: %i addr: %#x%s\n",
00593             load_idx, store_idx, storeHead, req->getPaddr(),
00594             sreqLow ? " split" : "");

00596     if (req->isLLSC()) {
00597         assert(!sreqLow);
00598         // Disable recording the result temporarily.  Writing to misc
00599         // regs normally updates the result, but this is not the
00600         // desired behavior when handling store conditionals.
00601         load_inst->recordResult(false);
00602         TheISA::handleLockedRead(load_inst.get(), req);
00603         load_inst->recordResult(true);
00604     }

00607         assert(!load_inst->memData);
00608         load_inst->memData = new uint8_t[64];

00846                 delete data_pkt;

00849                 delete sreqLow;

00851                 sreqLow = NULL;
00852                 sreqHigh = NULL;
00853             } else {

00869         return NoFault;

00871

00059 void
00060 PhysRegFile::initFreeList(UnifiedFreeList *freeList)
00061 {
00062     // Initialize the free lists.
00063     PhysRegIndex reg_idx = 0;
00064 
00065     // The initial batch of registers are the integer ones
00066     while (reg_idx < baseFloatRegIndex) {
00067         freeList->addIntReg(reg_idx++);
00068     }
00069 
00070     // The next batch of the registers are the floating-point physical
00071     // registers; put them onto the floating-point free list.
00072     while (reg_idx < baseCCRegIndex) {
00073         freeList->addFloatReg(reg_idx++);
00074     }
00075 
00076     // The rest of the registers are the condition-code physical
00077     // registers; put them onto the condition-code free list.
00078     while (reg_idx < totalNumRegs) {
00079         freeList->addCCReg(reg_idx++);
00080     }
00081 }

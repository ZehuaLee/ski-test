00065 DRAMSim2Wrapper::DRAMSim2Wrapper(const std::string& config_file,
00066                                  const std::string& system_file,
00067                                  const std::string& working_dir,
00068                                  const std::string& trace_file,
00069                                  unsigned int memory_size_mb,
00070                                  bool enable_debug,
00071                                  const std::string& name) :
00072     dramsim(new DRAMSim::MultiChannelMemorySystem(config_file, system_file,
00073                                                   working_dir, trace_file,
00074                                                   memory_size_mb, NULL, NULL)),
00075     _clockPeriod(0.0), _queueSize(0), _burstSize(0)
00076 {
00077     _name = name;
00078 
00079     // tell DRAMSim2 to ignore its internal notion of a CPU frequency
00080     dramsim->setCPUClockSpeed(0);
00081 
00082     // switch on debug output if requested
00083     if (enable_debug)
00084         SHOW_SIM_OUTPUT = 1;
00085 
00086     // there is no way of getting DRAMSim2 to tell us what frequency
00087     // it is assuming, so we have to extract it ourselves
00088     _clockPeriod = extractConfig<double>("tCK=",
00089                                          working_dir + '/' + config_file);
00090 
00091     if (!_clockPeriod)
00092         fatal("DRAMSim2 wrapper failed to get clock\n");
00093 
00094     // we also need to know what transaction queue size DRAMSim2 is
00095     // using so we can stall when responses are blocked
00096    _queueSize = extractConfig<unsigned int>("TRANS_QUEUE_DEPTH=",
00097                                             working_dir + '/' + system_file);
00098 
00099     if (!_queueSize)
00100         fatal("DRAMSim2 wrapper failed to get queue size\n");
00101 
00102 
00103    // finally, get the data bus bits and burst length so we can add a
00104    // sanity check for the burst size
00105     unsigned int dataBusBits =
00106         extractConfig<unsigned int>("JEDEC_DATA_BUS_BITS=",
00107                                     working_dir + '/' + system_file);
00108    unsigned int burstLength =
00109        extractConfig<unsigned int>("BL=", working_dir + '/' + config_file);
00110 
00111    if (!dataBusBits || !burstLength)
00112        fatal("DRAMSim22 wrapper failed to get burst size\n");
00113 
00114    _burstSize = dataBusBits * burstLength / 8;
00115 }

00122 template <typename T>
00123 T
00124 DRAMSim2Wrapper::extractConfig(const std::string& field_name,
00125                                const std::string& file_name) const
00126 {
00127     std::ifstream file_stream(file_name.c_str(), ios::in);
00128 
00129     if (!file_stream.good())
00130         fatal("DRAMSim2 wrapper could not open %s for reading\n", file_name);
00131 
00132     bool found = false;
00133     T res;
00134     std::string line;
00135     while (!found && file_stream) {
00136         getline(file_stream, line);
00137         if (line.substr(0, field_name.size()) == field_name) {
00138             found = true;
00139             istringstream iss(line.substr(field_name.size()));
00140             iss >> res;
00141         }
00142     }
00143 
00144     file_stream.close();
00145 
00146     if (!found)
00147         fatal("DRAMSim2 wrapper could not find %s in %s\n", field_name,
00148               file_name);
00149 
00150     return res;
00151 }

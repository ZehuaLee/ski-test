00336 {
00337     const bool global_enable(reg_pmcr.e);
00338 
00339     for (int i = 0; i < counters.size(); ++i) {
00340         CounterState &ctr(counters[i]);

00384 void
00385 PMU::handleEvent(CounterId id, uint64_t delta)
00386 {
00387     CounterState &ctr(getCounter(id));
00388     const bool overflowed(reg_pmovsr & (1 << id));
00389 
00390     if (isFiltered(ctr))
00391         return;
00392 
00393     // Handle the "count every 64 cycles" mode
00394     if (id == PMCCNTR && reg_pmcr.d) {
00395         clock_remainder += delta;
00396         delta = (clock_remainder >> 6);
00397         clock_remainder &= 63;
00398     }
00399 
00400     // Add delta and handle (new) overflows
00401     if (ctr.add(delta) && !overflowed) {
00402         DPRINTF(PMUVerbose, "PMU counter '%i' overflowed.\n", id);
00403         reg_pmovsr |= (1 << id);
00404         // Deliver a PMU interrupt if interrupt delivery is enabled
00405         // for this counter.
00406         if (reg_pminten  & (1 << id))
00407             raiseInterrupt();
00408     }
00409 }

00411 void
00412 PMU::updateCounter(CounterId id, CounterState &ctr)
00413 {
00414     if (!ctr.enabled) {
00415         if (!ctr.listeners.empty()) {
00416             DPRINTF(PMUVerbose, "updateCounter(%i): Disabling counter\n", id);
00417             ctr.listeners.clear();
00418         }
00419     } else {
00420         DPRINTF(PMUVerbose, "updateCounter(%i): Enable event id 0x%x\n",
00421                 id, ctr.eventId);
00422 
00423         // Attach all probes belonging to this event
00424         auto range(pmuEventTypes.equal_range(ctr.eventId));
00425         for (auto it = range.first; it != range.second; ++it) {
00426             const EventType &et(it->second);
00427 
00428             DPRINTF(PMUVerbose, "\tProbe: %s:%s\n", et.obj->name(), et.name);
00429             ctr.listeners.emplace_back(et.create(*this, id));
00430         }
00431 
00432         /* The SW_INCR event type is a special case which doesn't need
00433          * any probes since it is controlled by software and the PMU
00434          * itself.
00435          */
00436         if (ctr.listeners.empty() && ctr.eventId != ARCH_EVENT_SW_INCR) {
00437             warn("Can't enable PMU counter of type '0x%x': "
00438                  "No such event type.\n", ctr.eventId);
00439         }
00440     }

00574 bool
00575 PMU::CounterState::add(uint64_t delta)
00576 {
00577     const uint64_t msb(1ULL << (overflow64 ? 63 : 31));
00578     const uint64_t old_value(value);
00579 
00580     assert(delta > 0);
00581 
00582     value += delta;
00583 
00584     // Overflow if the msb goes from 1 to 0
00585     return (old_value & msb) && !(value & msb);

00372 void
00373 loadsymbol(ThreadContext *tc)
00374 {
00375     DPRINTF(PseudoInst, "PseudoInst::loadsymbol()\n");
00376     if (!FullSystem)
00377         panicFsOnlyPseudoInst("loadsymbol");
00378 
00379     const string &filename = tc->getCpuPtr()->system->params()->symbolfile;
00380     if (filename.empty()) {
00381         return;
00382     }
00383 
00384     std::string buffer;
00385     ifstream file(filename.c_str());
00386 
00387     if (!file)
00388         fatal("file error: Can't open symbol table file %s\n", filename);
00389 
00390     while (!file.eof()) {
00391         getline(file, buffer);
00392 
00393         if (buffer.empty())
00394             continue;
00395 
00396         string::size_type idx = buffer.find(' ');
00397         if (idx == string::npos)
00398             continue;
00399 
00400         string address = "0x" + buffer.substr(0, idx);
00401         eat_white(address);
00402         if (address.empty())
00403             continue;
00404 
00405         // Skip over letter and space
00406         string symbol = buffer.substr(idx + 3);
00407         eat_white(symbol);
00408         if (symbol.empty())
00409             continue;
00410 
00411         Addr addr;
00412         if (!to_number(address, addr))
00413             continue;
00414 
00415         if (!tc->getSystemPtr()->kernelSymtab->insert(addr, symbol))
00416             continue;
00417 
00418 
00419         DPRINTF(Loader, "Loaded symbol: %s @ %#llx\n", symbol, addr);
00420     }
00421     file.close();
00422 }

00536     while (len > 0) {
00537         int bytes = ::read(fd, p, len);
00538         if (bytes <= 0)
00539             break;
00540 
00541         p += bytes;
00542         result += bytes;
00543         len -= bytes;
00544     }
00545 
00546     close(fd);
00547     CopyIn(tc, vaddr, buf, result);
00548     delete [] buf;

00175     //Benign    Fault     5         Bound-Range                           #BR
00176     //Benign    Fault     6         Invalid-Opcode                        #UD
00177     //Benign    Fault     7         Device-Not-Available                  #NM
00178     //Benign    Abort     8         Double-Fault                          #DF
00179     //                    9         Coprocessor-Segment-Overrun
00180     //Contrib   Fault     10        Invalid-TSS                           #TS
00181     //Contrib   Fault     11        Segment-Not-Present                   #NP
00182     //Contrib   Fault     12        Stack                                 #SS
00183     //Contrib   Fault     13        General-Protection                    #GP
00184     //Either    Fault     14        Page-Fault                            #PF
00185     //                    15        Reserved
00186     //Benign    Fault     16        x87 Floating-Point Exception Pending  #MF
00187     //Benign    Fault     17        Alignment-Check                       #AC
00188     //Benign    Abort     18        Machine-Check                         #MC
00189     //Benign    Fault     19        SIMD Floating-Point                   #XF
00190     //                    20-29     Reserved
00191     //Contrib   ?         30        Security Exception                    #SX
00192     //                    31        Reserved
00193     //Benign    Interrupt 0-255     External Interrupts                   #INTR

00081 GlobalSimLoopExitEvent *
00082 simulate(Tick num_cycles)
00083 {
00084     // The first time simulate() is called from the Python code, we need to
00085     // create a thread for each of event queues referenced by the
00086     // instantiated sim objects.
00087     static bool threads_initialized = false;
00088     static std::vector<std::thread *> threads;
00089 
00090     if (!threads_initialized) {
00091         threadBarrier = new Barrier(numMainEventQueues);
00092 
00093         // the main thread (the one we're currently running on)
00094         // handles queue 0, so we only need to allocate new threads
00095         // for queues 1..N-1.  We'll call these the "subordinate" threads.
00096         for (uint32_t i = 1; i < numMainEventQueues; i++) {
00097             threads.push_back(new std::thread(thread_loop, mainEventQueue[i]));
00098         }
00099 
00100         threads_initialized = true;
00101         simulate_limit_event =
00102             new GlobalSimLoopExitEvent(mainEventQueue[0]->getCurTick(),
00103                                        "simulate() limit reached", 0);
00104     }
00105 
00106     inform("Entering event queue @ %d.  Starting simulation...\n", curTick());
00107 
00108     if (num_cycles < MaxTick - curTick())
00109         num_cycles = curTick() + num_cycles;
00110     else // counter would roll over or be set to MaxTick anyhow
00111         num_cycles = MaxTick;
00112 
00113     simulate_limit_event->reschedule(num_cycles);
00114 
00115     GlobalSyncEvent *quantum_event = NULL;
00116     if (numMainEventQueues > 1) {
00117         if (simQuantum == 0) {
00118             fatal("Quantum for multi-eventq simulation not specified");
00119         }
00120 
00121         quantum_event = new GlobalSyncEvent(curTick() + simQuantum, simQuantum,
00122                             EventBase::Progress_Event_Pri, 0);
00123 
00124         inParallelMode = true;
00125     }
00126 
00127     // all subordinate (created) threads should be waiting on the
00128     // barrier; the arrival of the main thread here will satisfy the
00129     // barrier, and all threads will enter doSimLoop in parallel
00130     threadBarrier->wait();
00131     Event *local_event = doSimLoop(mainEventQueue[0]);
00132     assert(local_event != NULL);
00133 
00134     inParallelMode = false;
00135 
00136     // locate the global exit event and return it to Python
00137     BaseGlobalEvent *global_event = local_event->globalEvent();
00138     assert(global_event != NULL);
00139 
00140     GlobalSimLoopExitEvent *global_exit_event =
00141         dynamic_cast<GlobalSimLoopExitEvent *>(global_event);
00142     assert(global_exit_event != NULL);
00143 
00145     if (quantum_event != NULL) {
00146         quantum_event->deschedule();
00147         delete quantum_event;
00148     }
00149 
00150     return global_exit_event;
00151 }

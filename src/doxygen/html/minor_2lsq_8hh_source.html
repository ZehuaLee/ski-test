00580     /* The store buffer contains committed cacheable stores on
00581      * their way to memory decoupled from subsequence instruction execution.
00582      * Before trying to issue a cacheable read from 'requests' to memory,
00583      * the store buffer is checked to see if a previous store contains the
00584      * needed data (StoreBuffer::canForwardDataToLoad) which can be
00585      * forwarded in lieu of a memory access.  If there are outstanding
00586      * stores in the transfers queue, they must be promoted to the store
00587      * buffer (and so be commited) before they can be correctly checked
00588      * for forwarding. */
00589     StoreBuffer storeBuffer;

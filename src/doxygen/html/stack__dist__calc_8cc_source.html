00126 // The updateSum method is a recursive function which updates
00127 // the node sums till the root. It also deletes the nodes that
00128 // are not used anymore.
00129 uint64_t
00130 StackDistCalc::updateSum(Node* node, bool from_left,
00131                          uint64_t sum_from_below, uint64_t level,
00132                          uint64_t stack_dist, bool discard_node)
00133 {
00134     ++level;
00135 
00136     // Make a copy of the node variables and work on them
00137     // as a node may be deleted by this function
00138     uint64_t node_sum_l = node->sumLeft;
00139     uint64_t node_sum_r = node->sumRight;
00140     bool node_left = node->isLeftNode;
00141     bool node_discard_left = node->discardLeft;
00142     bool node_discard_right = node->discardRight;
00143     uint64_t node_n_index = node->nodeIndex;
00144     Node* node_parent_ptr = node->parent;
00145 
00146     // For verification
00147     if (verifyStack) {
00148         // This sanity check makes sure that the left_sum and
00149         // right_sum of the node is not greater than the
00150         // maximum possible value given by the leaves below it
00151         // for example a node in layer 3 (tree[3]) can at most
00152         // have 8 leaves (4 to the left and 4 to the right)
00153         // thus left_sum and right_sum should be <= 4
00154         panic_if(node_sum_l > (1 << (level - 1)),
00155                  "Error in sum left of level %ul, node index %ull, "
00156                  "Sum =  %ull \n", level, node_n_index, node_sum_l);
00157 
00158         panic_if(node_sum_r > (1 << (level - 1)),
00159                  "Error in sum right of level %ul, node index %ull, "
00160                  "Sum =  %ull \n", level, node_n_index, node_sum_r);
00161     }
00162 
00163     // Update the left sum or the right sum depending on the
00164     // from_left flag. Variable stack_dist is updated only
00165     // when arriving from Left.
00166     if (from_left) {
00167         // update sumLeft
00168         node_sum_l = sum_from_below;
00169         stack_dist +=  node_sum_r;
00170     } else {
00171         // update sum_r
00172         node_sum_r = sum_from_below;
00173     }
00174 
00175     // sum_from_below == 0 can be a leaf discard operation
00176     if (discard_node && !sum_from_below) {
00177         if (from_left)
00178             node_discard_left = true;
00179         else
00180             node_discard_right = true;
00181     }
00182 
00183     // Update the node variables with new values
00184     node->nodeIndex = node_n_index;
00185     node->sumLeft = node_sum_l;
00186     node->sumRight = node_sum_r;
00187     node->isLeftNode = node_left;
00188     node->discardLeft = node_discard_left;
00189     node->discardRight = node_discard_right;
00190 
00191     // Delete the node if it is not required anymore
00192     if (node_discard_left && node_discard_right &&
00193         discard_node && node_parent_ptr && !sum_from_below) {
00194         delete node;
00195         tree[level].erase(node_n_index);
00196         discard_node = true;
00197     } else {
00198         // propogate discard_node as false upwards if the
00199         // above conditions are not met.
00200         discard_node = false;
00201     }
00202 
00203     // Recursively call the updateSum operation till the
00204     // root node is reached
00205     if (node_parent_ptr) {
00206         stack_dist = updateSum(node_parent_ptr, node_left,
00207                                node_sum_l + node_sum_r,
00208                                level, stack_dist, discard_node);
00209     }
00210 
00211     return stack_dist;
00212 }

00272 // Update tree is a tree balancing operation which maintains
00273 // the binary tree structure. This method is called whenever
00274 // index%2 == 0 (i.e. every alternate cycle)
00275 // The two main operation are :
00276 // OP1. moving the root node one layer up if index counter
00277 //    crosses power of 2
00278 // OP2. Addition of intermediate nodes as and when required
00279 //      and linking them to their parents in the layer above.
00280 void
00281 StackDistCalc::updateTree()
00282 {
00283     uint64_t i;
00284 
00285     if (isPowerOf2(index)) {
00286         // OP1. moving the root node one layer up if index counter
00287         //    crosses power of 2
00288         // If index counter crosses a power of 2, then add a
00289         // new tree layer above and create a new Root node in it.
00290         // After the root is created the old node
00291         // in the layer below is updated to point to this
00292         // newly created root node. The sum_l of this new root node
00293         // becomes the sum_l + sum_r of the old node.
00294         //
00295         // After this root update operation a chain of intermediate
00296         // nodes is created from root layer to tree[1](one layer
00297         // above the leaf layer)
00298 
00299         // Create a new root node
00300         Node* newRootNode = new Node();
00301 
00302         // Update its sum_l as the sum_l+sum_r from below
00303         newRootNode->sumLeft = tree[getTreeDepth()][0]->sumRight +
00304             tree[getTreeDepth()][0]->sumLeft;
00305         // Update its discard left flag if the node below has
00306         // has both discardLeft and discardRight set.
00307         newRootNode->discardLeft = tree[getTreeDepth()][0]->discardLeft &&
00308             tree[getTreeDepth()][0]->discardRight;
00309 
00310         // Map type variable, representing a layer in the tree
00311         IndexNodeMap treeLevel;
00312         // Add a new layer to the tree
00313         tree.push_back(treeLevel);
00314         nextIndex.push_back(1);
00315         tree[getTreeDepth()][newRootNode->nodeIndex] = newRootNode;
00316 
00317         // Update the parent pointer at lower layer to
00318         // point to newly created root node
00319         tree[getTreeDepth() - 1][0]->parent = tree[getTreeDepth()][0];
00320 
00321         // Add intermediate nodes from root till bottom (one layer above the
00322         // leaf layer)
00323         for (i = getTreeDepth() - 1; i >= 1; --i) {
00324             Node* newINode = new Node();
00325             // newNode is left or right child depending on the number of nodes
00326             // in that layer
00327             if (nextIndex[i] % 2 == 0) {
00328                 newINode->isLeftNode = true;
00329             } else {
00330                 newINode->isLeftNode = false;
00331             }
00332 
00333             newINode->parent = tree[i + 1][nextIndex[i + 1] - 1];
00334             newINode->nodeIndex = ++nextIndex[i] - 1;
00335             tree[i][newINode->nodeIndex] = newINode;
00336         }
00337     } else {
00338         // OP2. Addition of intermediate nodes as and when required
00339         //      and linking them to their parents in the layer above.
00340         //
00341         // At layer 1 a new INode is added whenever index%(2^1)==0
00342         // (multiples of 2)
00343         //
00344         // At layer 2 a new INode is added whenever index%(2^2)==0
00345         // (multiples of 4)
00346         //
00347         // At layer 3 a new INode is added whenever index%(2^3)==0
00348         // (multiples of 8)
00349         //...
00350         //
00351         // At layer N a new INode is added whenever index%(2^N)==0
00352         // (multiples of 2^N)
00353         for (i = getTreeDepth() - 1; i >= 1; --i) {
00354             // Traverse each layer from root to leaves and add a new
00355             // intermediate node if required. Link the parent_ptr of
00356             // the new node to the parent in the above layer.
00357 
00358             if ((index % (1 << i)) == 0) {
00359                 // Checks if current (index % 2^treeDepth) == 0 if true
00360                 // a new node at that layer is created
00361                 Node* newINode = new Node();
00362 
00363                 // newNode is left or right child depending on the
00364                 // number of nodes in that layer.
00365                 if (nextIndex[i] % 2 == 0) {
00366                     newINode->isLeftNode = true;
00367                 } else {
00368                     newINode->isLeftNode = false;
00369                 }
00370 
00371                 // Pointing to its parent in the upper layer
00372                 newINode->parent = tree[i + 1][nextIndex[i + 1] - 1];
00373                 newINode->nodeIndex = ++nextIndex[i] - 1;
00374                 tree[i][newINode->nodeIndex] = newINode;
00375             }
00376         }
00377     }
00378 }

00380 // This function is called everytime to get the stack distance and add
00381 // a new node. A feature to mark an old node in the tree is
00382 // added. This is useful if it is required to see the reuse
00383 // pattern. For example, BackInvalidates from the lower level (Membus)
00384 // to L2, can be marked (isMarked flag of Node set to True). And then
00385 // later if this same address is accessed by L1, the value of the
00386 // isMarked flag would be True. This would give some insight on how
00387 // the BackInvalidates policy of the lower level affect the read/write
00388 // accesses in an application.
00389 std::pair< uint64_t, bool>
00390 StackDistCalc::calcStackDistAndUpdate(const Addr r_address, bool addNewNode)
00391 {
00392     Node* newLeafNode;
00393     // Return index if the address was already present in stack
00394     uint64_t r_index = index;
00395 
00396     auto ai = aiMap.lower_bound(r_address);
00397 
00398     // Default value of flag indicating as the left or right leaf
00399     bool isLeft     = true;
00400     // Default value of isMarked flag for each node.
00401     bool _mark = false;
00402     // By default stackDistacne is treated as infinity
00403     uint64_t stack_dist;
00404 
00405     // Lookup aiMap by giving address as the key:
00406     // If found take address and Lookup in tree
00407     // Update tree from leaves by making B(found index) = 0
00408     // Add sums to right till root while Updating them
00409     // Stack Distance of that address sums to right
00410     if (ai != aiMap.end() && !(aiMap.key_comp()(r_address, ai->first))) {
00411         // key already exists
00412         // save the index counter value when this address was
00413         // encountered before and update it to the current index value
00414         r_index = ai->second;
00415 
00416         if (addNewNode) {
00417             // Update aiMap aiMap(Address) = current index
00418             ai->second   = index;
00419         } else {
00420             aiMap.erase(r_address);
00421         }
00422 
00423         // Call update tree operation on the tree starting with
00424         // the r_index value found above. This function would return
00425         // the value of the stack distcance.
00426         stack_dist = updateSumsLeavesToRoot(tree[0][r_index], false);
00427         newLeafNode = tree[0][r_index];
00428         // determine if this node was marked earlier
00429         _mark = newLeafNode->isMarked;
00430         delete newLeafNode;
00431         tree[0].erase(r_index);
00432     } else {
00433         if (addNewNode) {
00434             // Update aiMap aiMap(Address) = current index
00435             aiMap[r_address] = index;
00436         }
00437         // Update infinity bin count
00438         // By default stackDistacne is treated as infinity
00439         stack_dist = Infinity;
00440     }
00441 
00442     if (addNewNode) {
00443         // If index%2 == 0 then update tree
00444         if (index % 2 == 0) {
00445             updateTree();
00446         } else {
00447             // At odd values of index counter, a new right-type node is
00448             // added to the leaf layer, else a left-type node is added
00449             isLeft = false;
00450         }
00451 
00452         // Add new leaf node in the leaf layer (tree[0])
00453         // set n_index = current index
00454         newLeafNode = new Node();
00455         ++nextIndex[0];
00456         newLeafNode->nodeIndex=index;
00457         newLeafNode->isLeftNode=isLeft;
00458         // Point the parent pointer to the intermediate node above
00459         newLeafNode->parent = tree[1][nextIndex[1] - 1];
00460         tree[0][index] = newLeafNode;
00461         // call an update operation which would update the tree after
00462         // addition of this new leaf node.
00463         updateSumsLeavesToRoot(tree[0][index], true);
00464 
00465         // For verification
00466         if (verifyStack) {
00467             // This function checks the sanity of the tree to make sure the
00468             // last node in the link of parent pointers is the root node.
00469             // It takes a leaf node as an argument and traveses upwards till
00470             // the root layer to check if the last parent is null
00471             sanityCheckTree(tree[0][index]);
00472 
00473             // Push the same element in debug stack, and check
00474             uint64_t verify_stack_dist = verifyStackDist(r_address, true);
00475             panic_if(verify_stack_dist != stack_dist,
00476                      "Expected stack-distance for address \
00477                              %#lx is %#lx but found %#lx",
00478                      r_address, verify_stack_dist, stack_dist);
00479             printStack();
00480         }
00481 
00482         // The index counter is updated at the end of each transaction
00483         // (unique or non-unique)
00484         ++index;
00485     }
00486 
00487     return (std::make_pair(stack_dist, _mark));
00488 }

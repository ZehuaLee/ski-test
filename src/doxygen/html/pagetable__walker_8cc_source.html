00544 void
00545 Walker::WalkerState::setupWalk(Addr vaddr)
00546 {
00547     VAddr addr = vaddr;
00548     CR3 cr3 = tc->readMiscRegNoEffect(MISCREG_CR3);
00549     // Check if we're in long mode or not
00550     Efer efer = tc->readMiscRegNoEffect(MISCREG_EFER);
00551     dataSize = 8;
00552     Addr topAddr;
00553     if (efer.lma) {
00554         // Do long mode.
00555         state = LongPML4;
00556         topAddr = (cr3.longPdtb << 12) + addr.longl4 * dataSize;
00557         enableNX = efer.nxe;
00558     } else {
00559         // We're in some flavor of legacy mode.
00560         CR4 cr4 = tc->readMiscRegNoEffect(MISCREG_CR4);
00561         if (cr4.pae) {
00562             // Do legacy PAE.
00563             state = PAEPDP;
00564             topAddr = (cr3.paePdtb << 5) + addr.pael3 * dataSize;
00565             enableNX = efer.nxe;
00566         } else {
00567             dataSize = 4;
00568             topAddr = (cr3.pdtb << 12) + addr.norml2 * dataSize;
00569             if (cr4.pse) {
00570                 // Do legacy PSE.
00571                 state = PSEPD;
00572             } else {
00573                 // Do legacy non PSE.
00574                 state = PD;
00575             }
00576             enableNX = false;
00577         }
00578     }
00579 
00580     nextState = Ready;
00581     entry.vaddr = vaddr;
00582 
00583     Request::Flags flags = Request::PHYSICAL;
00584     if (cr3.pcd)
00585         flags.set(Request::UNCACHEABLE);
00586     RequestPtr request = new Request(topAddr, dataSize, flags,
00587                                      walker->masterId);
00588     read = new Packet(request, MemCmd::ReadReq);
00589     read->allocate();

00592 bool
00593 Walker::WalkerState::recvPacket(PacketPtr pkt)
00594 {
00595     assert(pkt->isResponse());
00596     assert(inflight);
00597     assert(state == Waiting);
00598     inflight--;
00599     if (pkt->isRead()) {
00600         // should not have a pending read it we also had one outstanding
00601         assert(!read);
00602 
00603         // @todo someone should pay for this
00604         pkt->headerDelay = pkt->payloadDelay = 0;
00605 
00606         state = nextState;
00607         nextState = Ready;
00608         PacketPtr write = NULL;
00609         read = pkt;
00610         timingFault = stepWalk(write);
00611         state = Waiting;
00612         assert(timingFault == NoFault || read == NULL);
00613         if (write) {
00614             writes.push_back(write);
00615         }
00616         sendPackets();
00617     } else {
00618         sendPackets();
00619     }
00620     if (inflight == 0 && read == NULL && writes.size() == 0) {
00621         state = Ready;
00622         nextState = Waiting;
00623         if (timingFault == NoFault) {
00624             /*
00625              * Finish the translation. Now that we now the right entry is
00626              * in the TLB, this should work with no memory accesses.
00627              * There could be new faults unrelated to the table walk like
00628              * permissions violations, so we'll need the return value as
00629              * well.
00630              */
00631             bool delayedResponse;
00632             Fault fault = walker->tlb->translate(req, tc, NULL, mode,
00633                                                  delayedResponse, true);
00634             assert(!delayedResponse);
00635             // Let the CPU continue.
00636             translation->finish(fault, req, tc, mode);
00637         } else {

00648 Walker::WalkerState::sendPackets()
00649 {
00650     //If we're already waiting for the port to become available, just return.
00651     if (retrying)
00652         return;

00654     //Reads always have priority
00655     if (read) {
00656         PacketPtr pkt = read;
00657         read = NULL;
00658         inflight++;
00659         if (!walker->sendTiming(this, pkt)) {
00660             retrying = true;
00661             read = pkt;
00662             inflight--;
00663             return;
00664         }
00665     }
00666     //Send off as many of the writes as we can.
00667     while (writes.size()) {
00668         PacketPtr write = writes.back();
00669         writes.pop_back();
00670         inflight++;
00671         if (!walker->sendTiming(this, write)) {
00672             retrying = true;
00673             writes.push_back(write);
00674             inflight--;
00675             return;
00676         }
00677     }

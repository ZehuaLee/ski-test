00129      * Read the request structure and do endian conversion if

00136 
00137     Status status;

00140 
00141     switch (req.type) {
00142       case T_IN:
00143         status = parent.read(req, desc, sizeof(BlkRequest), data_size);
00144         break;
00145 
00146       case T_OUT:
00147         status = parent.write(req, desc, sizeof(BlkRequest), data_size);
00148         break;
00149 
00150       case T_FLUSH:
00151         status = S_OK;
00152         break;
00153 
00154       default:
00155         warn("Unsupported IO request: %i\n", req.type);
00156         status = S_UNSUPP;
00157         break;
00158     }
00159 
00160     desc->chainWrite(sizeof(BlkRequest) + data_size,
00161                      &status, sizeof(status));
00162 
00163     // Tell the guest that we are done with this descriptor.
00164     produceDescriptor(desc, sizeof(BlkRequest) + data_size + sizeof(Status));

00076 MsrBase::printMsrBase(std::ostream &os) const
00077 {
00078     printMnemonic(os);
00079     bool apsr = false;
00080     bool foundPsr = false;
00081     for (unsigned i = 0; i < numDestRegs(); i++) {
00082         int idx = destRegIdx(i);
00083         if (idx < Misc_Reg_Base) {
00084             continue;
00085         }
00086         idx -= Misc_Reg_Base;
00087         if (idx == MISCREG_CPSR) {
00088             os << "cpsr_";
00089             foundPsr = true;
00090             break;
00091         }
00092         if (idx == MISCREG_SPSR) {
00093             if (bits(byteMask, 1, 0)) {
00094                 os << "spsr_";
00095             } else {
00096                 os << "apsr_";
00097                 apsr = true;
00098             }
00099             foundPsr = true;
00100             break;
00101         }
00102     }
00103     if (!foundPsr) {
00104         os << "????";
00105         return;
00106     }
00107     if (bits(byteMask, 3)) {
00108         if (apsr) {
00109             os << "nzcvq";
00110         } else {
00111             os << "f";
00112         }
00113     }
00114     if (bits(byteMask, 2)) {
00115         if (apsr) {
00116             os << "g";
00117         } else {
00118             os << "s";
00119         }
00120     }
00121     if (bits(byteMask, 1)) {
00122         os << "x";
00123     }
00124     if (bits(byteMask, 0)) {
00125         os << "c";
00126     }

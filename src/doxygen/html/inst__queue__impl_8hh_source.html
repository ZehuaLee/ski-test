00095 {
00096     assert(fuPool);
00097 
00098     numThreads = params->numThreads;
00099 
00100     // Set the number of total physical registers
00101     numPhysRegs = params->numPhysIntRegs + params->numPhysFloatRegs +
00102         params->numPhysCCRegs;
00103 
00104     //Create an entry for each physical register within the
00105     //dependency graph.
00106     dependGraph.resize(numPhysRegs);
00107 
00108     // Resize the register scoreboard.
00109     regScoreboard.resize(numPhysRegs);
00110 
00111     //Initialize Mem Dependence Units
00112     for (ThreadID tid = 0; tid < numThreads; tid++) {
00113         memDepUnit[tid].init(params, tid);
00114         memDepUnit[tid].setIQ(this);
00115     }
00116 
00117     resetState();
00118 
00119     std::string policy = params->smtIQPolicy;
00120 
00121     //Convert string to lowercase
00122     std::transform(policy.begin(), policy.end(), policy.begin(),
00123                    (int(*)(int)) tolower);
00124 
00125     //Figure out resource sharing policy
00126     if (policy == "dynamic") {
00127         iqPolicy = Dynamic;
00128 
00129         //Set Max Entries to Total ROB Capacity
00130         for (ThreadID tid = 0; tid < numThreads; tid++) {
00131             maxEntries[tid] = numEntries;
00132         }
00133 
00134     } else if (policy == "partitioned") {
00135         iqPolicy = Partitioned;
00136 
00137         //@todo:make work if part_amt doesnt divide evenly.
00138         int part_amt = numEntries / numThreads;
00139 
00140         //Divide ROB up evenly
00141         for (ThreadID tid = 0; tid < numThreads; tid++) {
00142             maxEntries[tid] = part_amt;
00143         }
00144 
00145         DPRINTF(IQ, "IQ sharing policy set to Partitioned:"

00151 
00152         int thresholdIQ = (int)((double)threshold * numEntries);
00153

00381 template <class Impl>
00382 void
00383 InstructionQueue<Impl>::resetState()
00384 {
00385     //Initialize thread IQ counts
00386     for (ThreadID tid = 0; tid <numThreads; tid++) {
00387         count[tid] = 0;
00388         instList[tid].clear();
00389     }
00390 
00391     // Initialize the number of free IQ entries.
00392     freeEntries = numEntries;
00393 
00394     // Note that in actuality, the registers corresponding to the logical
00395     // registers start off as ready.  However this doesn't matter for the
00396     // IQ as the instruction should have been correctly told if those
00397     // registers are ready in rename.  Thus it can all be initialized as
00398     // unready.
00399     for (int i = 0; i < numPhysRegs; ++i) {
00400         regScoreboard[i] = false;
00401     }
00402 
00403     for (ThreadID tid = 0; tid < numThreads; ++tid) {
00404         squashedSeqNum[tid] = 0;
00405     }
00406 
00407     for (int i = 0; i < Num_OpClasses; ++i) {
00408         while (!readyInsts[i].empty())
00409             readyInsts[i].pop();
00410         queueOnList[i] = false;
00411         readyIt[i] = listOrder.end();
00412     }
00413     nonSpecInsts.clear();
00414     listOrder.clear();
00415     deferredMemInsts.clear();
00416     blockedMemInsts.clear();
00417     retryMemInsts.clear();
00418     wbOutstanding = 0;
00419 }

00566 {
00567     new_inst->isFloating() ? fpInstQueueWrites++ : intInstQueueWrites++;
00568     // Make sure the instruction is valid
00569     assert(new_inst);
00570 
00571     DPRINTF(IQ, "Adding instruction [sn:%lli] PC %s to the IQ.\n",
00572             new_inst->seqNum, new_inst->pcState());
00573 
00574     assert(freeEntries != 0);
00575 
00576     instList[new_inst->threadNumber].push_back(new_inst);

00583     // dependencies.
00584     addToDependents(new_inst);
00585 
00586     // Have this instruction set itself as the producer of its destination
00587     // register(s).
00588     addToProducers(new_inst);
00589 
00590     if (new_inst->isMemRef()) {
00591         memDepUnit[new_inst->threadNumber].insert(new_inst);
00592     } else {
00593         addIfReady(new_inst);
00594     }
00595 
00596     ++iqInstsAdded;
00597 
00598     count[new_inst->threadNumber]++;
00599 
00600     assert(freeEntries == (numEntries - countInsts()));

00704     ListOrderEntry queue_entry;
00705     OpClass op_class = (*list_order_it).queueType;
00706     ListOrderIt next_it = list_order_it;
00707 
00708     ++next_it;
00709 
00710     queue_entry.queueType = op_class;
00711     queue_entry.oldestInst = readyInsts[op_class].top()->seqNum;
00712 
00713     while (next_it != listOrder.end() &&
00714            (*next_it).oldestInst < queue_entry.oldestInst) {
00715         ++next_it;
00716     }
00717 
00718     readyIt[op_class] = listOrder.insert(next_it, queue_entry);
00719 }

00721 template <class Impl>
00722 void
00723 InstructionQueue<Impl>::processFUCompletion(DynInstPtr &inst, int fu_idx)
00724 {
00725     DPRINTF(IQ, "Processing FU completion [sn:%lli]\n", inst->seqNum);
00726     assert(!cpu->switchedOut());
00727     // The CPU could have been sleeping until this op completed (*extremely*
00728     // long latency op).  Wake it if it was.  This may be overkill.
00729    --wbOutstanding;
00730     iewStage->wakeCPU();
00731 
00732     if (fu_idx > -1)
00733         fuPool->freeUnitNextCycle(fu_idx);
00734 
00735     // @todo: Ensure that these FU Completions happen at the beginning
00736     // of a cycle, otherwise they could add too many instructions to
00737     // the queue.
00738     issueToExecuteQueue->access(-1)->size++;
00739     instsToExecute.push_back(inst);
00740 }

00742 // @todo: Figure out a better way to remove the squashed items from the
00743 // lists.  Checking the top item of each list to see if it's squashed
00744 // wastes time and forces jumps.
00745 template <class Impl>
00746 void
00747 InstructionQueue<Impl>::scheduleReadyInsts()
00748 {
00749     DPRINTF(IQ, "Attempting to schedule ready instructions from "
00750             "the IQ.\n");
00751 
00752     IssueStruct *i2e_info = issueToExecuteQueue->access(0);
00753 
00754     DynInstPtr mem_inst;
00755     while (mem_inst = getDeferredMemInstToExecute()) {
00756         addReadyMemInst(mem_inst);
00757     }
00758 
00759     // See if any cache blocked instructions are able to be executed
00760     while (mem_inst = getBlockedMemInstToExecute()) {
00761         addReadyMemInst(mem_inst);
00762     }
00763 
00764     // Have iterator to head of the list
00765     // While I haven't exceeded bandwidth or reached the end of the list,
00766     // Try to get a FU that can do what this op needs.
00767     // If successful, change the oldestInst to the new top of the list, put
00768     // the queue in the proper place in the list.
00769     // Increment the iterator.
00770     // This will avoid trying to schedule a certain op class if there are no
00771     // FUs that handle it.
00772     int total_issued = 0;
00773     ListOrderIt order_it = listOrder.begin();
00774     ListOrderIt order_end_it = listOrder.end();
00775 
00776     while (total_issued < totalWidth && order_it != order_end_it) {
00777         OpClass op_class = (*order_it).queueType;
00778 
00779         assert(!readyInsts[op_class].empty());
00780 
00781         DynInstPtr issuing_inst = readyInsts[op_class].top();
00782 
00783         issuing_inst->isFloating() ? fpInstQueueReads++ : intInstQueueReads++;
00784 
00785         assert(issuing_inst->seqNum == (*order_it).oldestInst);
00786 
00787         if (issuing_inst->isSquashed()) {
00788             readyInsts[op_class].pop();
00789 
00790             if (!readyInsts[op_class].empty()) {
00791                 moveToYoungerInst(order_it);
00792             } else {
00793                 readyIt[op_class] = listOrder.end();
00794                 queueOnList[op_class] = false;
00795             }
00796 
00797             listOrder.erase(order_it++);
00798 
00799             ++iqSquashedInstsIssued;
00800 
00801             continue;
00802         }
00803 
00804         int idx = -2;
00805         Cycles op_latency = Cycles(1);
00806         ThreadID tid = issuing_inst->threadNumber;
00807 
00808         if (op_class != No_OpClass) {
00809             idx = fuPool->getUnit(op_class);
00810             issuing_inst->isFloating() ? fpAluAccesses++ : intAluAccesses++;
00811             if (idx > -1) {
00812                 op_latency = fuPool->getOpLatency(op_class);
00813             }
00814         }
00815 
00816         // If we have an instruction that doesn't require a FU, or a
00817         // valid FU, then schedule for execution.
00818         if (idx == -2 || idx != -1) {
00819             if (op_latency == Cycles(1)) {
00820                 i2e_info->size++;
00821                 instsToExecute.push_back(issuing_inst);
00822 
00823                 // Add the FU onto the list of FU's to be freed next
00824                 // cycle if we used one.
00825                 if (idx >= 0)
00826                     fuPool->freeUnitNextCycle(idx);
00827             } else {
00828                 bool pipelined = fuPool->isPipelined(op_class);
00829                 // Generate completion event for the FU
00830                 ++wbOutstanding;
00831                 FUCompletion *execution = new FUCompletion(issuing_inst,
00832                                                            idx, this);
00833 
00834                 cpu->schedule(execution,
00835                               cpu->clockEdge(Cycles(op_latency - 1)));
00836 
00837                 if (!pipelined) {
00838                     // If FU isn't pipelined, then it must be freed
00839                     // upon the execution completing.
00840                     execution->setFreeFU();
00841                 } else {
00842                     // Add the FU onto the list of FU's to be freed next cycle.
00843                     fuPool->freeUnitNextCycle(idx);
00844                 }
00845             }
00846 
00847             DPRINTF(IQ, "Thread %i: Issuing instruction PC %s "
00848                     "[sn:%lli]\n",
00849                     tid, issuing_inst->pcState(),
00850                     issuing_inst->seqNum);
00851 
00852             readyInsts[op_class].pop();
00853 
00854             if (!readyInsts[op_class].empty()) {
00855                 moveToYoungerInst(order_it);
00856             } else {
00857                 readyIt[op_class] = listOrder.end();
00858                 queueOnList[op_class] = false;
00859             }
00860 
00861             issuing_inst->setIssued();
00862             ++total_issued;
00863 
00864 #if TRACING_ON
00865             issuing_inst->issueTick = curTick() - issuing_inst->fetchTick;
00866 #endif
00867 
00868             if (!issuing_inst->isMemRef()) {
00869                 // Memory instructions can not be freed from the IQ until they
00870                 // complete.
00871                 ++freeEntries;
00872                 count[tid]--;
00873                 issuing_inst->clearInIQ();
00874             } else {
00875                 memDepUnit[tid].issue(issuing_inst);
00876             }
00877 
00878             listOrder.erase(order_it++);
00879             statIssuedInstType[tid][op_class]++;
00880         } else {
00881             statFuBusy[op_class]++;
00882             fuBusy[tid]++;
00883             ++order_it;
00884         }
00885     }
00886 
00887     numIssuedDist.sample(total_issued);
00888     iqInstsIssued+= total_issued;
00889 
00890     // If we issued any instructions, tell the CPU we had activity.
00891     // @todo If the way deferred memory instructions are handeled due to
00892     // translation changes then the deferredMemInsts condition should be removed
00893     // from the code below.
00894     if (total_issued || !retryMemInsts.empty() || !deferredMemInsts.empty()) {
00895         cpu->activityThisCycle();
00896     } else {
00897         DPRINTF(IQ, "Not able to schedule any instructions.\n");
00898     }
00899 }

00947 template <class Impl>
00948 int
00949 InstructionQueue<Impl>::wakeDependents(DynInstPtr &completed_inst)
00950 {
00951     int dependents = 0;
00952 
00953     // The instruction queue here takes care of both floating and int ops
00954     if (completed_inst->isFloating()) {
00955         fpInstQueueWakeupQccesses++;
00956     } else {
00957         intInstQueueWakeupAccesses++;
00958     }
00959 
00960     DPRINTF(IQ, "Waking dependents of completed instruction.\n");
00961 
00962     assert(!completed_inst->isSquashed());
00963 
00964     // Tell the memory dependence unit to wake any dependents on this
00965     // instruction if it is a memory instruction.  Also complete the memory
00966     // instruction at this point since we know it executed without issues.
00967     // @todo: Might want to rename "completeMemInst" to something that
00968     // indicates that it won't need to be replayed, and call this
00969     // earlier.  Might not be a big deal.
00970     if (completed_inst->isMemRef()) {
00971         memDepUnit[completed_inst->threadNumber].wakeDependents(completed_inst);
00972         completeMemInst(completed_inst);
00973     } else if (completed_inst->isMemBarrier() ||
00974                completed_inst->isWriteBarrier()) {
00975         memDepUnit[completed_inst->threadNumber].completeBarrier(completed_inst);
00976     }
00977 
00978     for (int dest_reg_idx = 0;
00979          dest_reg_idx < completed_inst->numDestRegs();
00980          dest_reg_idx++)
00981     {
00982         PhysRegIndex dest_reg =
00983             completed_inst->renamedDestRegIdx(dest_reg_idx);
00984 
00985         // Special case of uniq or control registers.  They are not
00986         // handled by the IQ and thus have no dependency graph entry.
00987         // @todo Figure out a cleaner way to handle this.
00988         if (dest_reg >= numPhysRegs) {
00989             DPRINTF(IQ, "dest_reg :%d, numPhysRegs: %d\n", dest_reg,
00990                     numPhysRegs);
00991             continue;
00992         }
00993 
00994         DPRINTF(IQ, "Waking any dependents on register %i.\n",
00995                 (int) dest_reg);
00996 
00997         //Go through the dependency chain, marking the registers as
00998         //ready within the waiting instructions.
00999         DynInstPtr dep_inst = dependGraph.pop(dest_reg);
01000 
01001         while (dep_inst) {
01002             DPRINTF(IQ, "Waking up a dependent instruction, [sn:%lli] "
01003                     "PC %s.\n", dep_inst->seqNum, dep_inst->pcState());
01004 
01005             // Might want to give more information to the instruction
01006             // so that it knows which of its source registers is
01007             // ready.  However that would mean that the dependency
01008             // graph entries would need to hold the src_reg_idx.
01009             dep_inst->markSrcRegReady();
01010 
01011             addIfReady(dep_inst);
01012 
01013             dep_inst = dependGraph.pop(dest_reg);
01014 
01015             ++dependents;
01016         }
01017 
01018         // Reset the head node now that all of its dependents have
01019         // been woken up.
01020         assert(dependGraph.empty(dest_reg));
01021         dependGraph.clearInst(dest_reg);
01022 
01023         // Mark the scoreboard as having that register ready.
01024         regScoreboard[dest_reg] = true;
01025     }
01026     return dependents;
01027 }

01172 template <class Impl>
01173 void
01174 InstructionQueue<Impl>::doSquash(ThreadID tid)
01175 {
01176     // Start at the tail.
01177     ListIt squash_it = instList[tid].end();
01178     --squash_it;
01179 
01180     DPRINTF(IQ, "[tid:%i]: Squashing until sequence number %i!\n",
01181             tid, squashedSeqNum[tid]);
01182 
01183     // Squash any instructions younger than the squashed sequence number
01184     // given.
01185     while (squash_it != instList[tid].end() &&
01186            (*squash_it)->seqNum > squashedSeqNum[tid]) {
01187 
01188         DynInstPtr squashed_inst = (*squash_it);
01189         squashed_inst->isFloating() ? fpInstQueueWrites++ : intInstQueueWrites++;
01190 
01191         // Only handle the instruction if it actually is in the IQ and
01192         // hasn't already been squashed in the IQ.
01193         if (squashed_inst->threadNumber != tid ||
01194             squashed_inst->isSquashedInIQ()) {
01195             --squash_it;
01196             continue;
01197         }
01198 
01199         if (!squashed_inst->isIssued() ||
01200             (squashed_inst->isMemRef() &&
01201              !squashed_inst->memOpDone())) {
01202 
01203             DPRINTF(IQ, "[tid:%i]: Instruction [sn:%lli] PC %s squashed.\n",
01204                     tid, squashed_inst->seqNum, squashed_inst->pcState());
01205 
01206             bool is_acq_rel = squashed_inst->isMemBarrier() &&
01207                          (squashed_inst->isLoad() ||
01208                            (squashed_inst->isStore() &&
01209                              !squashed_inst->isStoreConditional()));
01210 
01211             // Remove the instruction from the dependency list.
01212             if (is_acq_rel ||
01213                 (!squashed_inst->isNonSpeculative() &&
01214                  !squashed_inst->isStoreConditional() &&
01215                  !squashed_inst->isMemBarrier() &&
01216                  !squashed_inst->isWriteBarrier())) {
01217 
01218                 for (int src_reg_idx = 0;
01219                      src_reg_idx < squashed_inst->numSrcRegs();
01220                      src_reg_idx++)
01221                 {
01222                     PhysRegIndex src_reg =
01223                         squashed_inst->renamedSrcRegIdx(src_reg_idx);
01224 
01225                     // Only remove it from the dependency graph if it
01226                     // was placed there in the first place.
01227 
01228                     // Instead of doing a linked list traversal, we
01229                     // can just remove these squashed instructions
01230                     // either at issue time, or when the register is
01231                     // overwritten.  The only downside to this is it
01232                     // leaves more room for error.
01233 
01234                     if (!squashed_inst->isReadySrcRegIdx(src_reg_idx) &&
01235                         src_reg < numPhysRegs) {
01236                         dependGraph.remove(src_reg, squashed_inst);
01237                     }
01238 
01239 
01240                     ++iqSquashedOperandsExamined;
01241                 }
01242             } else if (!squashed_inst->isStoreConditional() ||
01243                        !squashed_inst->isCompleted()) {
01244                 NonSpecMapIt ns_inst_it =
01245                     nonSpecInsts.find(squashed_inst->seqNum);
01246 
01247                 // we remove non-speculative instructions from
01248                 // nonSpecInsts already when they are ready, and so we
01249                 // cannot always expect to find them
01250                 if (ns_inst_it == nonSpecInsts.end()) {
01251                     // loads that became ready but stalled on a
01252                     // blocked cache are alreayd removed from
01253                     // nonSpecInsts, and have not faulted
01254                     assert(squashed_inst->getFault() != NoFault ||
01255                            squashed_inst->isMemRef());
01256                 } else {
01257 
01258                     (*ns_inst_it).second = NULL;
01259 
01260                     nonSpecInsts.erase(ns_inst_it);
01261 
01262                     ++iqSquashedNonSpecRemoved;
01263                 }
01264             }
01265 
01266             // Might want to also clear out the head of the dependency graph.
01267 
01268             // Mark it as squashed within the IQ.
01269             squashed_inst->setSquashedInIQ();
01270 
01271             // @todo: Remove this hack where several statuses are set so the
01272             // inst will flow through the rest of the pipeline.
01273             squashed_inst->setIssued();
01274             squashed_inst->setCanCommit();
01275             squashed_inst->clearInIQ();
01276 
01277             //Update Thread IQ Count
01278             count[squashed_inst->threadNumber]--;
01279 
01280             ++freeEntries;
01281         }
01282 
01283         instList[tid].erase(squash_it--);
01284         ++iqSquashedInstsExamined;
01285     }
01286 }

01288 template <class Impl>
01289 bool
01290 InstructionQueue<Impl>::addToDependents(DynInstPtr &new_inst)
01291 {
01292     // Loop through the instruction's source registers, adding
01293     // them to the dependency list if they are not ready.
01294     int8_t total_src_regs = new_inst->numSrcRegs();
01295     bool return_val = false;
01296 
01297     for (int src_reg_idx = 0;
01298          src_reg_idx < total_src_regs;
01299          src_reg_idx++)
01300     {
01301         // Only add it to the dependency graph if it's not ready.
01302         if (!new_inst->isReadySrcRegIdx(src_reg_idx)) {
01303             PhysRegIndex src_reg = new_inst->renamedSrcRegIdx(src_reg_idx);
01304 
01305             // Check the IQ's scoreboard to make sure the register
01306             // hasn't become ready while the instruction was in flight
01307             // between stages.  Only if it really isn't ready should
01308             // it be added to the dependency graph.
01309             if (src_reg >= numPhysRegs) {
01310                 continue;
01311             } else if (!regScoreboard[src_reg]) {
01312                 DPRINTF(IQ, "Instruction PC %s has src reg %i that "
01313                         "is being added to the dependency chain.\n",
01314                         new_inst->pcState(), src_reg);
01315 
01316                 dependGraph.insert(src_reg, new_inst);
01317 
01318                 // Change the return value to indicate that something
01319                 // was added to the dependency graph.
01320                 return_val = true;
01321             } else {
01322                 DPRINTF(IQ, "Instruction PC %s has src reg %i that "
01323                         "became ready before it reached the IQ.\n",
01324                         new_inst->pcState(), src_reg);
01325                 // Mark a register ready within the instruction.
01326                 new_inst->markSrcRegReady(src_reg_idx);
01327             }
01328         }
01329     }
01330 
01331     return return_val;
01332 }

01334 template <class Impl>
01335 void
01336 InstructionQueue<Impl>::addToProducers(DynInstPtr &new_inst)
01337 {
01338     // Nothing really needs to be marked when an instruction becomes
01339     // the producer of a register's value, but for convenience a ptr
01340     // to the producing instruction will be placed in the head node of
01341     // the dependency links.
01342     int8_t total_dest_regs = new_inst->numDestRegs();
01343 
01344     for (int dest_reg_idx = 0;
01345          dest_reg_idx < total_dest_regs;
01346          dest_reg_idx++)
01347     {
01348         PhysRegIndex dest_reg = new_inst->renamedDestRegIdx(dest_reg_idx);
01349 
01350         // Instructions that use the misc regs will have a reg number
01351         // higher than the normal physical registers.  In this case these
01352         // registers are not renamed, and there is no need to track
01353         // dependencies as these instructions must be executed at commit.
01354         if (dest_reg >= numPhysRegs) {
01355             continue;
01356         }
01357 
01358         if (!dependGraph.empty(dest_reg)) {

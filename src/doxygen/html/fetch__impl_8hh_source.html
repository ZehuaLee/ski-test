00445 bool
00446 DefaultFetch<Impl>::isDrained() const
00447 {
00448     /* Make sure that threads are either idle of that the commit stage
00449      * has signaled that draining has completed by setting the drain
00450      * stall flag. This effectively forces the pipeline to be disabled
00451      * until the whole system is drained (simulation may continue to
00452      * drain other components).
00453      */
00454     for (ThreadID i = 0; i < numThreads; ++i) {
00455         // Verify fetch queues are drained
00456         if (!fetchQueue[i].empty())
00457             return false;
00458 
00459         // Return false if not idle or drain stalled
00460         if (fetchStatus[i] != Idle) {
00461             if (fetchStatus[i] == Blocked && stalls[i].drain)
00462                 continue;
00463             else
00464                 return false;
00465         }
00466     }
00467 
00468     /* The pipeline might start up again in the middle of the drain
00469      * cycle if the finish translation event is scheduled, so make
00470      * sure that's not the case.
00471      */
00472     return !finishTranslationEvent.scheduled();

00663                     mem_req->getPaddr());
00664             fetchStatus[tid] = NoGoodAddr;
00665             delete mem_req;
00666             memReq[tid] = NULL;
00667             return;
00668         }
00669 
00670         // Build packet here.
00671         PacketPtr data_pkt = new Packet(mem_req, MemCmd::ReadReq);
00672         data_pkt->dataDynamic(new uint8_t[fetchBufferSize]);
00673 
00674         fetchBufferPC[tid] = fetchBufferBlockPC;
00675         fetchBufferValid[tid] = false;
00676         DPRINTF(Fetch, "Fetch: Doing instruction read.\n");
00677 
00678         fetchedCacheLines++;
00679 
00680         // Access the cache.
00681         if (!cpu->getInstPort().sendTimingReq(data_pkt)) {
00682             assert(retryPkt == NULL);
00683             assert(retryTid == InvalidThreadID);
00684             DPRINTF(Fetch, "[tid:%i] Out of MSHRs!\n", tid);
00685 
00686             fetchStatus[tid] = IcacheWaitRetry;
00687             retryPkt = data_pkt;
00688             retryTid = tid;
00689             cacheBlocked = true;
00690         } else {
00691             DPRINTF(Fetch, "[tid:%i]: Doing Icache access.\n", tid);
00692             DPRINTF(Activity, "[tid:%i]: Activity: Waiting on I-cache "
00693                     "response.\n", tid);
00694             lastIcacheStall[tid] = curTick();
00695             fetchStatus[tid] = IcacheWaitResponse;
00696         }
00697     } else {
00698         // Don't send an instruction to decode if we can't handle it.
00699         if (!(numInst < fetchWidth) || !(fetchQueue[tid].size() < fetchQueueSize)) {
00700             assert(!finishTranslationEvent.scheduled());
00701             finishTranslationEvent.setFault(fault);
00702             finishTranslationEvent.setReq(mem_req);
00703             cpu->schedule(finishTranslationEvent,
00704                           cpu->clockEdge(Cycles(1)));
00705             return;
00706         }
00707         DPRINTF(Fetch, "[tid:%i] Got back req with addr %#x but expected %#x\n",
00708                 tid, mem_req->getVaddr(), memReq[tid]->getVaddr());
00709         // Translation faulted, icache request won't be sent.
00710         delete mem_req;
00711         memReq[tid] = NULL;
00712 
00713         // Send the fault to commit.  This thread will not do anything
00714         // until commit handles the fault.  The only other way it can
00715         // wake up is if a squash comes along and changes the PC.
00716         TheISA::PCState fetchPC = pc[tid];
00717 
00718         DPRINTF(Fetch, "[tid:%i]: Translation faulted, building noop.\n", tid);
00719         // We will use a nop in ordier to carry the fault.
00720         DynInstPtr instruction = buildInst(tid,
00721                 decoder[tid]->decode(TheISA::NoopMachInst, fetchPC.instAddr()),
00722                 NULL, fetchPC, fetchPC, false);
00723 
00724         instruction->setPredTarg(fetchPC);
00725         instruction->fault = fault;
00726         wroteToTimeBuffer = true;
00727 
00728         DPRINTF(Activity, "Activity this cycle.\n");
00729         cpu->activityThisCycle();
00730 
00731         fetchStatus[tid] = TrapPending;
00732 
00733         DPRINTF(Fetch, "[tid:%i]: Blocked, need to handle the trap.\n", tid);
00734         DPRINTF(Fetch, "[tid:%i]: fault (%s) detected @ PC %s.\n",
00735                 tid, fault->name(), pc[tid]);
00736     }
00737     _status = updateFetchStatus();
00738 }

00740 template <class Impl>
00741 inline void
00742 DefaultFetch<Impl>::doSquash(const TheISA::PCState &newPC,
00743                              const DynInstPtr squashInst, ThreadID tid)
00744 {
00745     DPRINTF(Fetch, "[tid:%i]: Squashing, setting PC to: %s.\n",
00746             tid, newPC);
00747 
00748     pc[tid] = newPC;
00749     fetchOffset[tid] = 0;
00750     if (squashInst && squashInst->pcState().instAddr() == newPC.instAddr())
00751         macroop[tid] = squashInst->macroop;
00752     else
00753         macroop[tid] = NULL;
00754     decoder[tid]->reset();
00755 
00756     // Clear the icache miss if it's outstanding.
00757     if (fetchStatus[tid] == IcacheWaitResponse) {
00758         DPRINTF(Fetch, "[tid:%i]: Squashing outstanding Icache miss.\n",
00759                 tid);
00760         memReq[tid] = NULL;
00761     } else if (fetchStatus[tid] == ItlbWait) {
00762         DPRINTF(Fetch, "[tid:%i]: Squashing outstanding ITLB miss.\n",
00763                 tid);
00764         memReq[tid] = NULL;
00765     }
00766 
00767     // Get rid of the retrying packet if it was from this thread.
00768     if (retryTid == tid) {
00769         assert(cacheBlocked);
00770         if (retryPkt) {
00771             delete retryPkt->req;
00772             delete retryPkt;
00773         }
00774         retryPkt = NULL;
00775         retryTid = InvalidThreadID;
00776     }
00777 
00778     fetchStatus[tid] = Squashing;
00779 
00780     // Empty fetch queue
00781     fetchQueue[tid].clear();
00782 
00783     // microops are being squashed, it is not known wheather the
00784     // youngest non-squashed microop was  marked delayed commit
00785     // or not. Setting the flag to true ensures that the
00786     // interrupts are not handled when they cannot be, though
00787     // some opportunities to handle interrupts may be missed.
00788     delayedCommit[tid] = true;
00789 
00790     ++fetchSquashCycles;
00791 }

00877 template <class Impl>
00878 void
00879 DefaultFetch<Impl>::tick()
00880 {
00881     list<ThreadID>::iterator threads = activeThreads->begin();
00882     list<ThreadID>::iterator end = activeThreads->end();
00883     bool status_change = false;
00884 
00885     wroteToTimeBuffer = false;
00886 
00887     for (ThreadID i = 0; i < numThreads; ++i) {
00888         issuePipelinedIfetch[i] = false;
00889     }
00890 
00891     while (threads != end) {
00892         ThreadID tid = *threads++;
00893 
00894         // Check the signals for each thread to determine the proper status
00895         // for each thread.
00896         bool updated_status = checkSignalsAndUpdate(tid);
00897         status_change =  status_change || updated_status;
00898     }
00899 
00900     DPRINTF(Fetch, "Running stage.\n");
00901 
00902     if (FullSystem) {
00903         if (fromCommit->commitInfo[0].interruptPending) {
00904             interruptPending = true;
00905         }
00906 
00907         if (fromCommit->commitInfo[0].clearInterrupt) {
00908             interruptPending = false;
00909         }
00910     }
00911 
00912     for (threadFetched = 0; threadFetched < numFetchingThreads;
00913          threadFetched++) {
00914         // Fetch each of the actively fetching threads.
00915         fetch(status_change);
00916     }
00917 
00918     // Record number of instructions fetched this cycle for distribution.
00919     fetchNisnDist.sample(numInst);
00920 
00921     if (status_change) {
00922         // Change the fetch stage status if there was a status change.
00923         _status = updateFetchStatus();
00924     }
00925 
00926     // Issue the next I-cache request if possible.
00927     for (ThreadID i = 0; i < numThreads; ++i) {
00928         if (issuePipelinedIfetch[i]) {
00929             pipelineIcacheAccesses(i);
00930         }
00931     }
00932 
00933     // Send instructions enqueued into the fetch queue to decode.
00934     // Limit rate by fetchWidth.  Stall if decode is stalled.
00935     unsigned insts_to_decode = 0;
00936     unsigned available_insts = 0;
00937 
00938     for (auto tid : *activeThreads) {
00939         if (!stalls[tid].decode) {
00940             available_insts += fetchQueue[tid].size();
00941         }
00942     }
00943 
00944     // Pick a random thread to start trying to grab instructions from
00945     auto tid_itr = activeThreads->begin();
00946     std::advance(tid_itr, random_mt.random<uint8_t>(0, activeThreads->size() - 1));
00947 
00948     while (available_insts != 0 && insts_to_decode < decodeWidth) {
00949         ThreadID tid = *tid_itr;
00950         if (!stalls[tid].decode && !fetchQueue[tid].empty()) {
00951             auto inst = fetchQueue[tid].front();
00952             toDecode->insts[toDecode->size++] = inst;
00953             DPRINTF(Fetch, "[tid:%i][sn:%i]: Sending instruction to decode from "
00954                     "fetch queue. Fetch queue size: %i.\n",
00955                     tid, inst->seqNum, fetchQueue[tid].size());
00956 
00957             wroteToTimeBuffer = true;
00958             fetchQueue[tid].pop_front();
00959             insts_to_decode++;
00960             available_insts--;
00961         }
00962 
00963         tid_itr++;
00964         // Wrap around if at end of active threads list
00965         if (tid_itr == activeThreads->end())
00966             tid_itr = activeThreads->begin();
00967     }
00968 
00969     // If there was activity this cycle, inform the CPU of it.
00970     if (wroteToTimeBuffer) {
00971         DPRINTF(Activity, "Activity this cycle.\n");
00972         cpu->activityThisCycle();
00973     }
00974 
00975     // Reset the number of the instruction we've fetched.
00976     numInst = 0;
00977 }

00979 template <class Impl>
00980 bool
00981 DefaultFetch<Impl>::checkSignalsAndUpdate(ThreadID tid)
00982 {
00983     // Update the per thread stall statuses.
00984     if (fromDecode->decodeBlock[tid]) {
00985         stalls[tid].decode = true;
00986     }
00987 
00988     if (fromDecode->decodeUnblock[tid]) {
00989         assert(stalls[tid].decode);
00990         assert(!fromDecode->decodeBlock[tid]);
00991         stalls[tid].decode = false;
00992     }
00993 
00994     // Check squash signals from commit.
00995     if (fromCommit->commitInfo[tid].squash) {
00996 
00997         DPRINTF(Fetch, "[tid:%u]: Squashing instructions due to squash "
00998                 "from commit.\n",tid);
00999         // In any case, squash.
01000         squash(fromCommit->commitInfo[tid].pc,
01001                fromCommit->commitInfo[tid].doneSeqNum,
01002                fromCommit->commitInfo[tid].squashInst, tid);
01003 
01004         // If it was a branch mispredict on a control instruction, update the
01005         // branch predictor with that instruction, otherwise just kill the
01006         // invalid state we generated in after sequence number
01007         if (fromCommit->commitInfo[tid].mispredictInst &&
01008             fromCommit->commitInfo[tid].mispredictInst->isControl()) {
01009             branchPred->squash(fromCommit->commitInfo[tid].doneSeqNum,
01010                               fromCommit->commitInfo[tid].pc,
01011                               fromCommit->commitInfo[tid].branchTaken,
01012                               tid);
01013         } else {
01014             branchPred->squash(fromCommit->commitInfo[tid].doneSeqNum,
01015                               tid);
01016         }
01017 
01018         return true;
01019     } else if (fromCommit->commitInfo[tid].doneSeqNum) {
01020         // Update the branch predictor if it wasn't a squashed instruction
01021         // that was broadcasted.
01022         branchPred->update(fromCommit->commitInfo[tid].doneSeqNum, tid);
01023     }
01024 
01025     // Check squash signals from decode.
01026     if (fromDecode->decodeInfo[tid].squash) {
01027         DPRINTF(Fetch, "[tid:%u]: Squashing instructions due to squash "
01028                 "from decode.\n",tid);
01029 
01030         // Update the branch predictor.
01031         if (fromDecode->decodeInfo[tid].branchMispredict) {
01032             branchPred->squash(fromDecode->decodeInfo[tid].doneSeqNum,
01033                               fromDecode->decodeInfo[tid].nextPC,
01034                               fromDecode->decodeInfo[tid].branchTaken,
01035                               tid);
01036         } else {
01037             branchPred->squash(fromDecode->decodeInfo[tid].doneSeqNum,
01038                               tid);
01039         }
01040 
01041         if (fetchStatus[tid] != Squashing) {
01042 
01043             DPRINTF(Fetch, "Squashing from decode with PC = %s\n",
01044                 fromDecode->decodeInfo[tid].nextPC);
01045             // Squash unless we're already squashing
01046             squashFromDecode(fromDecode->decodeInfo[tid].nextPC,
01047                              fromDecode->decodeInfo[tid].squashInst,
01048                              fromDecode->decodeInfo[tid].doneSeqNum,
01049                              tid);
01050 
01051             return true;
01052         }
01053     }
01054 
01055     if (checkStall(tid) &&
01056         fetchStatus[tid] != IcacheWaitResponse &&
01057         fetchStatus[tid] != IcacheWaitRetry &&
01058         fetchStatus[tid] != ItlbWait &&
01059         fetchStatus[tid] != QuiescePending) {
01060         DPRINTF(Fetch, "[tid:%i]: Setting to blocked\n",tid);
01061 
01062         fetchStatus[tid] = Blocked;
01063 
01064         return true;
01065     }
01066 
01067     if (fetchStatus[tid] == Blocked ||
01068         fetchStatus[tid] == Squashing) {
01069         // Switch status to running if fetch isn't being told to block or
01070         // squash this cycle.
01071         DPRINTF(Fetch, "[tid:%i]: Done squashing, switching to running.\n",
01072                 tid);
01073 
01074         fetchStatus[tid] = Running;
01075 
01076         return true;
01077     }
01078 
01079     // If we've reached this point, we have not gotten any signals that
01080     // cause fetch to change its status.  Fetch remains the same as before.
01081     return false;
01082 }

01138 template<class Impl>
01139 void
01140 DefaultFetch<Impl>::fetch(bool &status_change)
01141 {
01143     // Start actual fetch
01145     ThreadID tid = getFetchingThread(fetchPolicy);
01146 
01147     assert(!cpu->switchedOut());
01148 
01149     if (tid == InvalidThreadID) {
01150         // Breaks looping condition in tick()
01151         threadFetched = numFetchingThreads;
01152 
01153         if (numThreads == 1) {  // @todo Per-thread stats
01154             profileStall(0);
01155         }
01156 
01157         return;
01158     }
01159 
01160     DPRINTF(Fetch, "Attempting to fetch from [tid:%i]\n", tid);
01161 
01162     // The current PC.
01163     TheISA::PCState thisPC = pc[tid];
01164 
01165     Addr pcOffset = fetchOffset[tid];
01166     Addr fetchAddr = (thisPC.instAddr() + pcOffset) & BaseCPU::PCMask;
01167 
01168     bool inRom = isRomMicroPC(thisPC.microPC());
01169 
01170     // If returning from the delay of a cache miss, then update the status
01171     // to running, otherwise do the cache access.  Possibly move this up
01172     // to tick() function.
01173     if (fetchStatus[tid] == IcacheAccessComplete) {
01174         DPRINTF(Fetch, "[tid:%i]: Icache miss is complete.\n", tid);
01175 
01176         fetchStatus[tid] = Running;
01177         status_change = true;
01178     } else if (fetchStatus[tid] == Running) {
01179         // Align the fetch PC so its at the start of a fetch buffer segment.
01180         Addr fetchBufferBlockPC = fetchBufferAlignPC(fetchAddr);
01181 
01182         // If buffer is no longer valid or fetchAddr has moved to point
01183         // to the next cache block, AND we have no remaining ucode
01184         // from a macro-op, then start fetch from icache.
01185         if (!(fetchBufferValid[tid] && fetchBufferBlockPC == fetchBufferPC[tid])
01186             && !inRom && !macroop[tid]) {
01187             DPRINTF(Fetch, "[tid:%i]: Attempting to translate and read "
01188                     "instruction, starting at PC %s.\n", tid, thisPC);
01189 
01190             fetchCacheLine(fetchAddr, tid, thisPC.instAddr());
01191 
01192             if (fetchStatus[tid] == IcacheWaitResponse)
01193                 ++icacheStallCycles;
01194             else if (fetchStatus[tid] == ItlbWait)
01195                 ++fetchTlbCycles;
01196             else
01197                 ++fetchMiscStallCycles;
01198             return;
01199         } else if ((checkInterrupt(thisPC.instAddr()) && !delayedCommit[tid])) {
01200             // Stall CPU if an interrupt is posted and we're not issuing
01201             // an delayed commit micro-op currently (delayed commit instructions
01202             // are not interruptable by interrupts, only faults)
01203             ++fetchMiscStallCycles;
01204             DPRINTF(Fetch, "[tid:%i]: Fetch is stalled!\n", tid);
01205             return;
01206         }
01207     } else {
01208         if (fetchStatus[tid] == Idle) {
01209             ++fetchIdleCycles;
01210             DPRINTF(Fetch, "[tid:%i]: Fetch is idle!\n", tid);
01211         }
01212 
01213         // Status is Idle, so fetch should do nothing.
01214         return;
01215     }
01216 
01217     ++fetchCycles;
01218 
01219     TheISA::PCState nextPC = thisPC;
01220 
01221     StaticInstPtr staticInst = NULL;
01222     StaticInstPtr curMacroop = macroop[tid];
01223 
01224     // If the read of the first instruction was successful, then grab the
01225     // instructions from the rest of the cache line and put them into the
01226     // queue heading to decode.
01227 
01228     DPRINTF(Fetch, "[tid:%i]: Adding instructions to queue to "
01229             "decode.\n", tid);
01230 
01231     // Need to keep track of whether or not a predicted branch
01232     // ended this fetch block.
01233     bool predictedBranch = false;
01234 
01235     // Need to halt fetch if quiesce instruction detected
01236     bool quiesce = false;
01237 
01238     TheISA::MachInst *cacheInsts =
01239         reinterpret_cast<TheISA::MachInst *>(fetchBuffer[tid]);
01240 
01241     const unsigned numInsts = fetchBufferSize / instSize;
01242     unsigned blkOffset = (fetchAddr - fetchBufferPC[tid]) / instSize;
01243 
01244     // Loop through instruction memory from the cache.
01245     // Keep issuing while fetchWidth is available and branch is not
01246     // predicted taken
01247     while (numInst < fetchWidth && fetchQueue[tid].size() < fetchQueueSize
01248            && !predictedBranch && !quiesce) {
01249         // We need to process more memory if we aren't going to get a
01250         // StaticInst from the rom, the current macroop, or what's already
01251         // in the decoder.
01252         bool needMem = !inRom && !curMacroop &&
01253             !decoder[tid]->instReady();
01254         fetchAddr = (thisPC.instAddr() + pcOffset) & BaseCPU::PCMask;
01255         Addr fetchBufferBlockPC = fetchBufferAlignPC(fetchAddr);
01256 
01257         if (needMem) {
01258             // If buffer is no longer valid or fetchAddr has moved to point
01259             // to the next cache block then start fetch from icache.
01260             if (!fetchBufferValid[tid] ||
01261                 fetchBufferBlockPC != fetchBufferPC[tid])
01262                 break;
01263 
01264             if (blkOffset >= numInsts) {
01265                 // We need to process more memory, but we've run out of the
01266                 // current block.
01267                 break;
01268             }
01269 
01270             if (ISA_HAS_DELAY_SLOT && pcOffset == 0) {
01271                 // Walk past any annulled delay slot instructions.
01272                 Addr pcAddr = thisPC.instAddr() & BaseCPU::PCMask;
01273                 while (fetchAddr != pcAddr && blkOffset < numInsts) {
01274                     blkOffset++;
01275                     fetchAddr += instSize;
01276                 }
01277                 if (blkOffset >= numInsts)
01278                     break;
01279             }
01280 
01281             MachInst inst = TheISA::gtoh(cacheInsts[blkOffset]);
01282             decoder[tid]->moreBytes(thisPC, fetchAddr, inst);
01283 
01284             if (decoder[tid]->needMoreBytes()) {
01285                 blkOffset++;
01286                 fetchAddr += instSize;
01287                 pcOffset += instSize;
01288             }
01289         }
01290 
01291         // Extract as many instructions and/or microops as we can from
01292         // the memory we've processed so far.
01293         do {
01294             if (!(curMacroop || inRom)) {
01295                 if (decoder[tid]->instReady()) {
01296                     staticInst = decoder[tid]->decode(thisPC);
01297 
01298                     // Increment stat of fetched instructions.
01299                     ++fetchedInsts;
01300 
01301                     if (staticInst->isMacroop()) {
01302                         curMacroop = staticInst;
01303                     } else {
01304                         pcOffset = 0;
01305                     }
01306                 } else {
01307                     // We need more bytes for this instruction so blkOffset and
01308                     // pcOffset will be updated
01309                     break;
01310                 }
01311             }
01312             // Whether we're moving to a new macroop because we're at the

01381                 "for this cycle.\n", tid);

01384                 "fetch buffer.\n", tid);
01385     }

01392     }
01393

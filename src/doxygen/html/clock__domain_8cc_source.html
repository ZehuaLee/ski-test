00076 SrcClockDomain::SrcClockDomain(const Params *p) :
00077     ClockDomain(p, p->voltage_domain),
00078     freqOpPoints(p->clock),
00079     _domainID(p->domain_id),
00080     _perfLevel(p->init_perf_level)
00081 {
00082     VoltageDomain *vdom = p->voltage_domain;
00083 
00084     fatal_if(freqOpPoints.empty(), "DVFS: Empty set of frequencies for "\
00085              "domain %d %s\n", _domainID, name());
00086 
00087     fatal_if(!vdom, "DVFS: Empty voltage domain specified for "\
00088              "domain %d %s\n", _domainID, name());
00089 
00090     fatal_if((vdom->numVoltages() > 1) &&
00091              (vdom->numVoltages() != freqOpPoints.size()),
00092              "DVFS: Number of frequency and voltage scaling points do "\
00093              "not match: %d:%d ID: %d %s.\n", vdom->numVoltages(),
00094              freqOpPoints.size(), _domainID, name());
00095 
00096     // Frequency (& voltage) points should be declared in descending order,
00097     // NOTE: Frequency is inverted to ticks, so checking for ascending ticks
00098     fatal_if(!std::is_sorted(freqOpPoints.begin(), freqOpPoints.end()),
00099              "DVFS: Frequency operation points not in descending order for "\
00100              "domain with ID %d\n", _domainID);
00101 
00102     fatal_if(_perfLevel >= freqOpPoints.size(), "DVFS: Initial DVFS point %d "\
00103              "is outside of list for Domain ID: %d\n", _perfLevel, _domainID);
00104 
00105     clockPeriod(freqOpPoints[_perfLevel]);
00106 
00107     vdom->registerSrcClockDom(this);
00108 }

00110 void
00111 SrcClockDomain::clockPeriod(Tick clock_period)
00112 {
00113     if (clock_period == 0) {
00114         fatal("%s has a clock period of zero\n", name());
00115     }
00116 
00117     // Align all members to the current tick
00118     for (auto m = members.begin(); m != members.end(); ++m) {
00119         (*m)->updateClockPeriod();
00120     }
00121 
00122     _clockPeriod = clock_period;
00123 
00124     DPRINTF(ClockDomain,
00125             "Setting clock period to %d ticks for source clock %s\n",
00126             _clockPeriod, name());
00127 
00128     // inform any derived clocks they need to updated their period
00129     for (auto c = children.begin(); c != children.end(); ++c) {
00130         (*c)->updateClockPeriod();
00131     }
00132 }

00134 void
00135 SrcClockDomain::perfLevel(PerfLevel perf_level)
00136 {
00137     assert(validPerfLevel(perf_level));
00138 
00139     if (perf_level == _perfLevel) {
00140         // Silently ignore identical overwrites
00141         return;
00142     }
00143 
00144     DPRINTF(ClockDomain, "DVFS: Switching performance level of domain %s "\
00145             "(id: %d) from  %d to %d\n", name(), domainID(), _perfLevel,
00146             perf_level);
00147 
00148     _perfLevel = perf_level;
00149 
00150     // Signal the voltage domain that we have changed our perf level so that the
00151     // voltage domain can recompute its performance level
00152     voltageDomain()->sanitiseVoltages();
00153 
00154     // Integrated switching of the actual clock value, too
00155     clockPeriod(clkPeriodAtPerfLevel());
00156 }

00186 DerivedClockDomain::DerivedClockDomain(const Params *p) :
00187     ClockDomain(p, p->clk_domain->voltageDomain()),
00188     parent(*p->clk_domain),
00189     clockDivider(p->clk_divider)
00190 {
00191     // Ensure that clock divider setting works as frequency divider and never
00192     // work as frequency multiplier
00193     if (clockDivider < 1) {
00194        fatal("Clock divider param cannot be less than 1");
00195     }
00196 
00197     // let the parent keep track of this derived domain so that it can
00198     // propagate changes
00199     parent.addDerivedDomain(this);
00200 
00201     // update our clock period based on the parents clock
00202     updateClockPeriod();
00203 }

00205 void
00206 DerivedClockDomain::updateClockPeriod()
00207 {
00208     // Align all members to the current tick
00209     for (auto m = members.begin(); m != members.end(); ++m) {
00210         (*m)->updateClockPeriod();
00211     }
00212 
00213     // recalculate the clock period, relying on the fact that changes
00214     // propagate downwards in the tree
00215     _clockPeriod = parent.clockPeriod() * clockDivider;
00216 
00217     DPRINTF(ClockDomain,
00218             "Setting clock period to %d ticks for derived clock %s\n",
00219             _clockPeriod, name());
00220 
00221     // inform any derived clocks
00222     for (auto c = children.begin(); c != children.end(); ++c) {
00223         (*c)->updateClockPeriod();
00224     }
00225 }

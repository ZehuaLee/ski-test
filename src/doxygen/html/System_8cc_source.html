00278 void
00279 RubySystem::startup()
00280 {
00281 
00282     // Ruby restores state from a checkpoint by resetting the clock to 0 and
00283     // playing the requests that can possibly re-generate the cache state.
00284     // The clock value is set to the actual checkpointed value once all the
00285     // requests have been executed.
00286     //
00287     // This way of restoring state is pretty finicky. For example, if a
00288     // Ruby component reads time before the state has been restored, it would
00289     // cache this value and hence its clock would not be reset to 0, when
00290     // Ruby resets the global clock. This can potentially result in a
00291     // deadlock.
00292     //
00293     // The solution is that no Ruby component should read time before the
00294     // simulation starts. And then one also needs to hope that the time
00295     // Ruby finishes restoring the state is less than the time when the
00296     // state was checkpointed.
00297 
00298     if (m_warmup_enabled) {
00299         // save the current tick value
00300         Tick curtick_original = curTick();
00301         // save the event queue head
00302         Event* eventq_head = eventq->replaceHead(NULL);
00303         // set curTick to 0 and reset Ruby System's clock
00304         setCurTick(0);
00305         resetClock();
00306 
00307         // Schedule an event to start cache warmup
00308         enqueueRubyEvent(curTick());
00309         simulate();
00310 
00311         delete m_cache_recorder;
00312         m_cache_recorder = NULL;
00313         m_systems_to_warmup--;
00314         if (m_systems_to_warmup == 0) {
00315             m_warmup_enabled = false;
00316         }
00317 
00318         // Restore eventq head
00319         eventq_head = eventq->replaceHead(eventq_head);
00320         // Restore curTick and Ruby System's clock
00321         setCurTick(curtick_original);
00322         resetClock();
00323     }
00324 
00325     resetStats();

00344 bool
00345 RubySystem::functionalRead(PacketPtr pkt)
00346 {
00347     Address address(pkt->getAddr());
00348     Address line_address(address);
00349     line_address.makeLineAddress();
00350 
00351     AccessPermission access_perm = AccessPermission_NotPresent;
00352     int num_controllers = m_abs_cntrl_vec.size();
00353 
00354     DPRINTF(RubySystem, "Functional Read request for %s\n",address);
00355 
00356     unsigned int num_ro = 0;
00357     unsigned int num_rw = 0;
00358     unsigned int num_busy = 0;
00359     unsigned int num_backing_store = 0;
00360     unsigned int num_invalid = 0;
00361 
00362     // In this loop we count the number of controllers that have the given
00363     // address in read only, read write and busy states.
00364     for (unsigned int i = 0; i < num_controllers; ++i) {
00365         access_perm = m_abs_cntrl_vec[i]-> getAccessPermission(line_address);
00366         if (access_perm == AccessPermission_Read_Only)
00367             num_ro++;
00368         else if (access_perm == AccessPermission_Read_Write)
00369             num_rw++;
00370         else if (access_perm == AccessPermission_Busy)
00371             num_busy++;
00372         else if (access_perm == AccessPermission_Backing_Store)
00373             // See RubySlicc_Exports.sm for details, but Backing_Store is meant
00374             // to represent blocks in memory *for Broadcast/Snooping protocols*,
00375             // where memory has no idea whether it has an exclusive copy of data
00376             // or not.
00377             num_backing_store++;
00378         else if (access_perm == AccessPermission_Invalid ||
00379                  access_perm == AccessPermission_NotPresent)
00380             num_invalid++;
00381     }
00382     assert(num_rw <= 1);
00383 
00384     // This if case is meant to capture what happens in a Broadcast/Snoop
00385     // protocol where the block does not exist in the cache hierarchy. You
00386     // only want to read from the Backing_Store memory if there is no copy in
00387     // the cache hierarchy, otherwise you want to try to read the RO or RW
00388     // copies existing in the cache hierarchy (covered by the else statement).
00389     // The reason is because the Backing_Store memory could easily be stale, if
00390     // there are copies floating around the cache hierarchy, so you want to read
00391     // it only if it's not in the cache hierarchy at all.
00392     if (num_invalid == (num_controllers - 1) && num_backing_store == 1) {
00393         DPRINTF(RubySystem, "only copy in Backing_Store memory, read from it\n");
00394         for (unsigned int i = 0; i < num_controllers; ++i) {
00395             access_perm = m_abs_cntrl_vec[i]->getAccessPermission(line_address);
00396             if (access_perm == AccessPermission_Backing_Store) {
00397                 m_abs_cntrl_vec[i]->functionalRead(line_address, pkt);
00398                 return true;
00399             }
00400         }
00401     } else if (num_ro > 0 || num_rw == 1) {
00402         // In Broadcast/Snoop protocols, this covers if you know the block
00403         // exists somewhere in the caching hierarchy, then you want to read any
00404         // valid RO or RW block.  In directory protocols, same thing, you want
00405         // to read any valid readable copy of the block.
00406         DPRINTF(RubySystem, "num_busy = %d, num_ro = %d, num_rw = %d\n",
00407                 num_busy, num_ro, num_rw);
00408         // In this loop, we try to figure which controller has a read only or
00409         // a read write copy of the given address. Any valid copy would suffice
00410         // for a functional read.
00411         for (unsigned int i = 0;i < num_controllers;++i) {
00412             access_perm = m_abs_cntrl_vec[i]->getAccessPermission(line_address);
00413             if (access_perm == AccessPermission_Read_Only ||
00414                 access_perm == AccessPermission_Read_Write) {
00415                 m_abs_cntrl_vec[i]->functionalRead(line_address, pkt);
00416                 return true;
00417             }
00418         }
00419     }
00420 
00421     return false;
00422 }

00061     /*
00062      * If an access needs to be broken into fragments, currently at most two,
00063      * the the following two classes are used as the sender state of the
00064      * packets so the CPU can keep track of everything. In the main packet
00065      * sender state, there's an array with a spot for each fragment. If a
00066      * fragment has already been accepted by the CPU, aka isn't waiting for
00067      * a retry, it's pointer is NULL. After each fragment has successfully
00068      * been processed, the "outstanding" counter is decremented. Once the
00069      * count is zero, the entire larger access is complete.
00070      */
00071     class SplitMainSenderState : public Packet::SenderState
00072     {
00073       public:
00074         int outstanding;
00075         PacketPtr fragments[2];
00076 
00077         int
00078         getPendingFragment()
00079         {
00080             if (fragments[0]) {
00081                 return 0;
00082             } else if (fragments[1]) {
00083                 return 1;
00084             } else {
00085                 return -1;
00086             }
00087         }
00088     };

00128 void
00129 X86ISA::I82094AA::writeReg(uint8_t offset, uint32_t value)
00130 {
00131     if (offset == 0x0) {
00132         id = bits(value, 31, 24);
00133     } else if (offset == 0x1) {
00134         // The IOAPICVER register is read only.
00135     } else if (offset == 0x2) {
00136         arbId = bits(value, 31, 24);
00137     } else if (offset >= 0x10 && offset <= (0x10 + TableSize * 2)) {
00138         int index = (offset - 0x10) / 2;
00139         if (offset % 2) {
00140             redirTable[index].topDW = value;
00141             redirTable[index].topReserved = 0;
00142         } else {
00143             redirTable[index].bottomDW = value;
00144             redirTable[index].bottomReserved = 0;
00145         }
00146     } else {
00147         warn("Access to undefined I/O APIC register %#x.\n", offset);
00148     }
00149     DPRINTF(I82094AA,
00150             "Wrote %#x to I/O APIC register %#x .\n", value, offset);

00153 uint32_t
00154 X86ISA::I82094AA::readReg(uint8_t offset)
00155 {
00156     uint32_t result = 0;
00157     if (offset == 0x0) {
00158         result = id << 24;
00159     } else if (offset == 0x1) {
00160         result = ((TableSize - 1) << 16) | APICVersion;
00161     } else if (offset == 0x2) {
00162         result = arbId << 24;
00163     } else if (offset >= 0x10 && offset <= (0x10 + TableSize * 2)) {
00164         int index = (offset - 0x10) / 2;
00165         if (offset % 2) {
00166             result = redirTable[index].topDW;
00167         } else {
00168             result = redirTable[index].bottomDW;
00169         }
00170     } else {
00171         warn("Access to undefined I/O APIC register %#x.\n", offset);
00172     }
00173     DPRINTF(I82094AA,
00174             "Read %#x from I/O APIC register %#x.\n", result, offset);
00175     return result;

00178 void
00179 X86ISA::I82094AA::signalInterrupt(int line)
00180 {
00181     DPRINTF(I82094AA, "Received interrupt %d.\n", line);
00182     assert(line < TableSize);
00183     RedirTableEntry entry = redirTable[line];
00184     if (entry.mask) {
00185         DPRINTF(I82094AA, "Entry was masked.\n");
00186         return;
00187     } else {
00188         TriggerIntMessage message = 0;
00189         message.destination = entry.dest;
00190         if (entry.deliveryMode == DeliveryMode::ExtInt) {
00191             assert(extIntPic);
00192             message.vector = extIntPic->getVector();
00193         } else {
00194             message.vector = entry.vector;
00195         }
00196         message.deliveryMode = entry.deliveryMode;
00197         message.destMode = entry.destMode;
00198         message.level = entry.polarity;
00199         message.trigger = entry.trigger;
00200         ApicList apics;
00201         int numContexts = sys->numContexts();
00202         if (message.destMode == 0) {
00203             if (message.deliveryMode == DeliveryMode::LowestPriority) {
00204                 panic("Lowest priority delivery mode from the "
00205                         "IO APIC aren't supported in physical "
00206                         "destination mode.\n");
00207             }
00208             if (message.destination == 0xFF) {
00209                 for (int i = 0; i < numContexts; i++) {
00210                     apics.push_back(i);
00211                 }
00212             } else {
00213                 apics.push_back(message.destination);
00214             }
00215         } else {
00216             for (int i = 0; i < numContexts; i++) {
00217                 Interrupts *localApic = sys->getThreadContext(i)->
00218                     getCpuPtr()->getInterruptController();
00219                 if ((localApic->readReg(APIC_LOGICAL_DESTINATION) >> 24) &
00220                         message.destination) {
00221                     apics.push_back(localApic->getInitialApicId());
00222                 }
00223             }
00224             if (message.deliveryMode == DeliveryMode::LowestPriority &&
00225                     apics.size()) {
00226                 // The manual seems to suggest that the chipset just does
00227                 // something reasonable for these instead of actually using
00228                 // state from the local APIC. We'll just rotate an offset
00229                 // through the set of APICs selected above.
00230                 uint64_t modOffset = lowestPriorityOffset % apics.size();
00231                 lowestPriorityOffset++;
00232                 ApicList::iterator apicIt = apics.begin();
00233                 while (modOffset--) {
00234                     apicIt++;
00235                     assert(apicIt != apics.end());
00236                 }
00237                 int selected = *apicIt;
00238                 apics.clear();
00239                 apics.push_back(selected);
00240             }
00241         }
00242         intMasterPort.sendMessage(apics, message, sys->isTimingMode());
00243     }
00244 }

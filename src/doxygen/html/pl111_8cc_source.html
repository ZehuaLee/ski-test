00475 void
00476 Pl111::fillFifo()
00477 {
00478     while ((dmaPendingNum < maxOutstandingDma) && (maxAddr >= curAddr + dmaSize )) {
00479         // concurrent dma reads need different dma done events
00480         // due to assertion in scheduling state
00481         ++dmaPendingNum;
00482 
00483         assert(!dmaDoneEventFree.empty());
00484         DmaDoneEvent *event(dmaDoneEventFree.back());
00485         dmaDoneEventFree.pop_back();
00486         assert(!event->scheduled());
00487 
00488         // We use a uncachable request here because the requests from the CPU
00489         // will be uncacheable as well. If we have uncacheable and cacheable
00490         // requests in the memory system for the same address it won't be
00491         // pleased
00492         dmaPort.dmaAction(MemCmd::ReadReq, curAddr + startAddr, dmaSize,
00493                           event, curAddr + dmaBuffer,
00494                           0, Request::UNCACHEABLE);
00495         curAddr += dmaSize;
00496     }
00497 }

00499 void
00500 Pl111::dmaDone()
00501 {
00502     DPRINTF(PL111, "DMA Done\n");
00503 
00504     Tick maxFrameTime = lcdTiming2.cpl * height * pixelClock;
00505 
00506     --dmaPendingNum;
00507 
00508     if (maxAddr == curAddr && !dmaPendingNum) {
00509         if ((curTick() - startTime) > maxFrameTime) {
00510             warn("CLCD controller buffer underrun, took %d ticks when should"
00511                  " have taken %d\n", curTick() - startTime, maxFrameTime);
00512             lcdRis.underflow = 1;
00513             if (!intEvent.scheduled())
00514                 schedule(intEvent, clockEdge());
00515         }
00516 
00517         assert(!readEvent.scheduled());
00518         fb.copyIn(dmaBuffer, converter);
00519         if (vnc)
00520             vnc->setDirty();
00521 
00522         if (enableCapture) {
00523             DPRINTF(PL111, "-- write out frame buffer into bmp\n");
00524 
00525             if (!pic)
00526                 pic = simout.create(csprintf("%s.framebuffer.bmp", sys->name()), true);
00527 
00528             assert(pic);
00529             pic->seekp(0);
00530             bmp.write(*pic);
00531         }
00532 
00533         // schedule the next read based on when the last frame started
00534         // and the desired fps (i.e. maxFrameTime), we turn the
00535         // argument into a relative number of cycles in the future
00536         if (lcdControl.lcden)
00537             schedule(readEvent, clockEdge(ticksToCycles(startTime -
00538                                                         curTick() +
00539                                                         maxFrameTime)));
00540     }
00541 
00542     if (dmaPendingNum > (maxOutstandingDma - waterMark))
00543         return;
00544 
00545     if (!fillFifoEvent.scheduled())
00546         schedule(fillFifoEvent, clockEdge());
00547 }

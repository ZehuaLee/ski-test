00129 void
00130 KvmGic::setIntState(uint8_t type, uint8_t vcpu, uint16_t irq, bool high)
00131 {
00132     assert(type < KVM_ARM_IRQ_TYPE_MASK);
00133     assert(vcpu < KVM_ARM_IRQ_VCPU_MASK);
00134     assert(irq < KVM_ARM_IRQ_NUM_MASK);
00135     const uint32_t line(
00136         (type << KVM_ARM_IRQ_TYPE_SHIFT) |
00137         (vcpu << KVM_ARM_IRQ_VCPU_SHIFT) |
00138         (irq << KVM_ARM_IRQ_NUM_SHIFT));
00139 
00140     vm.setIRQLine(line, high);

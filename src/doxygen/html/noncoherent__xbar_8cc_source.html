00056 NoncoherentXBar::NoncoherentXBar(const NoncoherentXBarParams *p)
00057     : BaseXBar(p)
00058 {
00059     // create the ports based on the size of the master and slave
00060     // vector ports, and the presence of the default port, the ports
00061     // are enumerated starting from zero
00062     for (int i = 0; i < p->port_master_connection_count; ++i) {
00063         std::string portName = csprintf("%s.master[%d]", name(), i);
00064         MasterPort* bp = new NoncoherentXBarMasterPort(portName, *this, i);
00065         masterPorts.push_back(bp);
00066         reqLayers.push_back(new ReqLayer(*bp, *this,
00067                                          csprintf(".reqLayer%d", i)));
00068     }
00069 
00070     // see if we have a default slave device connected and if so add
00071     // our corresponding master port
00072     if (p->port_default_connection_count) {
00073         defaultPortID = masterPorts.size();
00074         std::string portName = name() + ".default";
00075         MasterPort* bp = new NoncoherentXBarMasterPort(portName, *this,
00076                                                       defaultPortID);
00077         masterPorts.push_back(bp);
00078         reqLayers.push_back(new ReqLayer(*bp, *this, csprintf(".reqLayer%d",
00079                                                               defaultPortID)));
00080     }
00081 
00082     // create the slave ports, once again starting at zero
00083     for (int i = 0; i < p->port_slave_connection_count; ++i) {
00084         std::string portName = csprintf("%s.slave[%d]", name(), i);
00085         SlavePort* bp = new NoncoherentXBarSlavePort(portName, *this, i);
00086         slavePorts.push_back(bp);
00087         respLayers.push_back(new RespLayer(*bp, *this,
00088                                            csprintf(".respLayer%d", i)));
00089     }

00126     // forwarding the packet
00127     unsigned int pkt_size = pkt->hasData() ? pkt->getSize() : 0;
00128     unsigned int pkt_cmd = pkt->cmdToIndex();

00131     Tick old_header_delay = pkt->headerDelay;
00132 
00133     // a request sees the frontend and forward latency

00138

00140     Tick packetFinishTime = clockEdge(Cycles(1)) + pkt->payloadDelay;
00141

00143     // remember if we are expecting a response

00146

00149

00157         // restore the header delay as it is additive
00158         pkt->headerDelay = old_header_delay;
00159 
00160         // occupy until the header is sent

00167     // remember where to route the response to
00168     if (expect_response) {

00093 /*
00094  * Uncompress and unmarshal the code object stored in the
00095  * EmbeddedPython
00096  */
00097 PyObject *
00098 EmbeddedPython::getCode() const
00099 {
00100     Bytef marshalled[len];
00101     uLongf unzlen = len;
00102     int ret = uncompress(marshalled, &unzlen, (const Bytef *)code, zlen);
00103     if (ret != Z_OK)
00104         panic("Could not uncompress code: %s\n", zError(ret));
00105     assert(unzlen == (uLongf)len);
00106 
00107     return PyMarshal_ReadObjectFromString((char *)marshalled, len);
00108 }

00192 /*
00193  * Start up the M5 simulator.  This mostly vectors into the python
00194  * main function.
00195  */
00196 int
00197 m5Main(int argc, char **argv)
00198 {
00199 #if HAVE_PROTOBUF
00200     // Verify that the version of the protobuf library that we linked
00201     // against is compatible with the version of the headers we
00202     // compiled against.
00203     GOOGLE_PROTOBUF_VERIFY_VERSION;
00204 #endif
00205 
00206     PySys_SetArgv(argc, argv);
00207 
00208     // We have to set things up in the special __main__ module
00209     PyObject *module = PyImport_AddModule(PyCC("__main__"));
00210     if (module == NULL)
00211         panic("Could not import __main__");
00212     PyObject *dict = PyModule_GetDict(module);
00213 
00214     // import the main m5 module
00215     PyObject *result;
00216     const char **command = m5MainCommands;
00217 
00218     // evaluate each command in the m5MainCommands array (basically a
00219     // bunch of python statements.
00220     while (*command) {
00221         result = PyRun_String(*command, Py_file_input, dict, dict);
00222         if (!result) {
00223             PyErr_Print();
00224             return 1;
00225         }
00226         Py_DECREF(result);
00227 
00228         command++;
00229     }
00230 
00231 #if HAVE_PROTOBUF
00232     google::protobuf::ShutdownProtobufLibrary();
00233 #endif
00234 
00235     return 0;

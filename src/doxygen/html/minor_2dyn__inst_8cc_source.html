00171         unsigned int num_dest_regs = staticInst->numDestRegs();
00172 
00173         std::ostringstream regs_str;
00174 
00175         /* Format lists of src and dest registers for microops and
00176          *  'full' instructions */
00177         if (!staticInst->isMacroop()) {
00178             regs_str << " srcRegs=";
00179 
00180             unsigned int src_reg = 0;
00181             while (src_reg < num_src_regs) {
00182                 printRegName(regs_str, staticInst->srcRegIdx(src_reg));
00183 
00184                 src_reg++;
00185                 if (src_reg != num_src_regs)
00186                     regs_str << ',';
00187             }
00188 
00189             regs_str << " destRegs=";
00190 
00191             unsigned int dest_reg = 0;
00192             while (dest_reg < num_dest_regs) {
00193                 printRegName(regs_str, staticInst->destRegIdx(dest_reg));
00194 
00195                 dest_reg++;
00196                 if (dest_reg != num_dest_regs)
00197                     regs_str << ',';
00198             }
00199 
00200 #if THE_ISA == ARM_ISA
00201             regs_str << " extMachInst=" << std::hex << std::setw(16)
00202                 << std::setfill('0') << staticInst->machInst << std::dec;
00203 #endif
00204         }
00205 
00206         std::ostringstream flags;
00207         staticInst->printFlags(flags, " ");
00208 
00209         MINORINST(&named_object, "id=%s addr=0x%x inst=\"%s\" class=%s"
00210             " flags=\"%s\"%s%s\n",
00211             id, pc.instAddr(),

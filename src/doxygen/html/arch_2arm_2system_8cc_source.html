00056 ArmSystem::ArmSystem(Params *p)
00057     : System(p), bootldr(NULL), _haveSecurity(p->have_security),
00058       _haveLPAE(p->have_lpae),
00059       _haveVirtualization(p->have_virtualization),
00060       _genericTimer(nullptr),
00061       _highestELIs64(p->highest_el_is_64),
00062       _resetAddr64(p->reset_addr_64),
00063       _physAddrRange64(p->phys_addr_range_64),
00064       _haveLargeAsid64(p->have_large_asid_64),
00065       multiProc(p->multi_proc)
00066 {
00067     // Check if the physical address range is valid
00068     if (_highestELIs64 && (
00069             _physAddrRange64 < 32 ||
00070             _physAddrRange64 > 48 ||
00071             (_physAddrRange64 % 4 != 0 && _physAddrRange64 != 42))) {
00072         fatal("Invalid physical address range (%d)\n", _physAddrRange64);
00073     }
00074 
00075     if (p->boot_loader != "") {
00076         bootldr = createObjectFile(p->boot_loader);
00077 
00078         if (!bootldr)
00079             fatal("Could not read bootloader: %s\n", p->boot_loader);
00080 
00081         if ((bootldr->getArch() == ObjectFile::Arm64) && !_highestELIs64) {
00082             warn("Highest ARM exception-level set to AArch32 but bootloader "
00083                   "is for AArch64. Assuming you wanted these to match.\n");
00084             _highestELIs64 = true;
00085         } else if ((bootldr->getArch() == ObjectFile::Arm) && _highestELIs64) {
00086             warn("Highest ARM exception-level set to AArch64 but bootloader "
00087                   "is for AArch32. Assuming you wanted these to match.\n");
00088             _highestELIs64 = false;
00089         }
00090 
00091         bootldr->loadGlobalSymbols(debugSymbolTable);
00092 
00093     }
00094     debugPrintkEvent = addKernelFuncEvent<DebugPrintkEvent>("dprintk");
00095 }

00097 void
00098 ArmSystem::initState()
00099 {
00100     // Moved from the constructor to here since it relies on the
00101     // address map being resolved in the interconnect
00102 
00103     // Call the initialisation of the super class
00104     System::initState();
00105 
00106     const Params* p = params();
00107 
00108     if (bootldr) {
00109         bootldr->loadSections(physProxy);
00110 
00111         uint8_t jump_to_bl_32[] =
00112         {
00113             0x07, 0xf0, 0xa0, 0xe1  // branch to r7 in aarch32
00114         };
00115 
00116         uint8_t jump_to_bl_64[] =
00117         {
00118             0xe0, 0x00, 0x1f, 0xd6  // instruction "br x7" in aarch64
00119         };
00120 
00121         // write the jump to branch table into address 0
00122         if (!_highestELIs64)
00123             physProxy.writeBlob(0x0, jump_to_bl_32, sizeof(jump_to_bl_32));
00124         else
00125             physProxy.writeBlob(0x0, jump_to_bl_64, sizeof(jump_to_bl_64));
00126 
00127         inform("Using bootloader at address %#x\n", bootldr->entryPoint());
00128 
00129         // Put the address of the boot loader into r7 so we know
00130         // where to branch to after the reset fault
00131         // All other values needed by the boot loader to know what to do
00132         if (!p->gic_cpu_addr || !p->flags_addr)
00133             fatal("gic_cpu_addr && flags_addr must be set with bootloader\n");
00134 
00135         for (int i = 0; i < threadContexts.size(); i++) {
00136             if (!_highestELIs64)
00137                 threadContexts[i]->setIntReg(3, (kernelEntry & loadAddrMask) +
00138                         loadAddrOffset);
00139             threadContexts[i]->setIntReg(4, params()->gic_cpu_addr);
00140             threadContexts[i]->setIntReg(5, params()->flags_addr);
00141             threadContexts[i]->setIntReg(7, bootldr->entryPoint());
00142         }
00143         inform("Using kernel entry physical address at %#x\n",
00144                (kernelEntry & loadAddrMask) + loadAddrOffset);
00145     } else {
00146         // Set the initial PC to be at start of the kernel code
00147         if (!_highestELIs64)
00148             threadContexts[0]->pcState((kernelEntry & loadAddrMask) +
00149                     loadAddrOffset);
00150     }
00151 }

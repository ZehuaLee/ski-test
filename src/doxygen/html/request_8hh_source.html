00405     void splitOnVaddr(Addr split_addr, RequestPtr &req1, RequestPtr &req2)
00406     {
00407         assert(privateFlags.isSet(VALID_VADDR));
00408         assert(privateFlags.noneSet(VALID_PADDR));
00409         assert(split_addr > _vaddr && split_addr < _vaddr + _size);
00410         req1 = new Request(*this);
00411         req2 = new Request(*this);
00412         req1->_size = split_addr - _vaddr;
00413         req2->_vaddr = split_addr;
00414         req2->_size = _size - req1->_size;
00415     }

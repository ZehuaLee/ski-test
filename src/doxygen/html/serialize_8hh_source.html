00209 // A SerializableBuilder serves as an evaluation context for a set of
00210 // parameters that describe a specific instance of a Serializable.  This
00211 // evaluation context corresponds to a section in the .ini file (as
00212 // with the base ParamContext) plus an optional node in the
00213 // configuration hierarchy (the configNode member) for resolving
00214 // Serializable references.  SerializableBuilder is an abstract superclass;
00215 // derived classes specialize the class for particular subclasses of
00216 // Serializable (e.g., BaseCache).
00217 //
00218 // For typical usage, see the definition of
00219 // SerializableClass::createObject().
00220 //
00221 class SerializableBuilder
00222 {
00223   public:
00224

00227     virtual ~SerializableBuilder() {}

00229     // Create the actual Serializable corresponding to the parameter
00230     // values in this context.  This function is overridden in derived
00231     // classes to call a specific constructor for a particular
00232     // subclass of Serializable.
00233     virtual Serializable *create() = 0;
00234 };

00236 //
00237 // An instance of SerializableClass corresponds to a class derived from
00238 // Serializable.  The SerializableClass instance serves to bind the string
00239 // name (found in the config file) to a function that creates an
00240 // instance of the appropriate derived class.
00241 //
00242 // This would be much cleaner in Smalltalk or Objective-C, where types
00243 // are first-class objects themselves.
00244 //
00245 class SerializableClass
00246 {
00247   public:
00248 
00249     // Type CreateFunc is a pointer to a function that creates a new
00250     // simulation object builder based on a .ini-file parameter
00251     // section (specified by the first string argument), a unique name
00252     // for the object (specified by the second string argument), and
00253     // an optional config hierarchy node (specified by the third
00254     // argument).  A pointer to the new SerializableBuilder is returned.
00255     typedef Serializable *(*CreateFunc)(Checkpoint *cp,
00256                                         const std::string &section);
00257 
00258     static std::map<std::string,CreateFunc> *classMap;
00259 
00260     // Constructor.  For example:
00261     //
00262     // SerializableClass baseCacheSerializableClass("BaseCacheSerializable",
00263     //                         newBaseCacheSerializableBuilder);
00264     //
00265     SerializableClass(const std::string &className, CreateFunc createFunc);
00266 
00267     // create Serializable given name of class and pointer to
00268     // configuration hierarchy node
00269     static Serializable *createObject(Checkpoint *cp,
00270                                       const std::string &section);
00271 };

00076 void
00077 SimPoint::profile(const std::pair<SimpleThread*, StaticInstPtr>& p)
00078 {
00079     SimpleThread* thread = p.first;
00080     const StaticInstPtr &inst = p.second;
00081 
00082     if (inst->isMicroop() && !inst->isLastMicroop())
00083         return;
00084 
00085     if (!currentBBVInstCount)
00086         currentBBV.first = thread->pcState().instAddr();
00087 
00088     ++intervalCount;
00089     ++currentBBVInstCount;
00090 
00091     // If inst is control inst, assume end of basic block.
00092     if (inst->isControl()) {
00093         currentBBV.second = thread->pcState().instAddr();
00094 
00095         auto map_itr = bbMap.find(currentBBV);
00096         if (map_itr == bbMap.end()){
00097             // If a new (previously unseen) basic block is found,
00098             // add a new unique id, record num of insts and insert into bbMap.
00099             BBInfo info;
00100             info.id = bbMap.size() + 1;
00101             info.insts = currentBBVInstCount;
00102             info.count = currentBBVInstCount;
00103             bbMap.insert(std::make_pair(currentBBV, info));
00104         } else {
00105             // If basic block is seen before, just increment the count by the
00106             // number of insts in basic block.
00107             BBInfo& info = map_itr->second;
00108             info.count += currentBBVInstCount;
00109         }
00110         currentBBVInstCount = 0;
00111 
00112         // Reached end of interval if the sum of the current inst count
00113         // (intervalCount) and the excessive inst count from the previous
00114         // interval (intervalDrift) is greater than/equal to the interval size.
00115         if (intervalCount + intervalDrift >= intervalSize) {
00116             // summarize interval and display BBV info
00117             std::vector<std::pair<uint64_t, uint64_t> > counts;
00118             for (auto map_itr = bbMap.begin(); map_itr != bbMap.end();
00119                     ++map_itr) {
00120                 BBInfo& info = map_itr->second;
00121                 if (info.count != 0) {
00122                     counts.push_back(std::make_pair(info.id, info.count));
00123                     info.count = 0;
00124                 }
00125             }
00126             std::sort(counts.begin(), counts.end());
00127 
00128             // Print output BBV info
00129             *simpointStream << "T";
00130             for (auto cnt_itr = counts.begin(); cnt_itr != counts.end();
00131                     ++cnt_itr) {
00132                 *simpointStream << ":" << cnt_itr->first
00133                                 << ":" << cnt_itr->second << " ";
00134             }
00135             *simpointStream << "\n";
00136 
00137             intervalDrift = (intervalCount + intervalDrift) - intervalSize;
00138             intervalCount = 0;
00139         }
00140     }
00141 }

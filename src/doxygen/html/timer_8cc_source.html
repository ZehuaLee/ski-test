00131 Tick
00132 PosixKvmTimer::calcResolution()
00133 {
00134     struct timespec ts;
00135 
00136     if (clock_getres(clockID, &ts) == -1)
00137         panic("PosixKvmTimer: Failed to get timer resolution\n");
00138 
00139     const uint64_t res_ns(ts.tv_sec * 1000000000ULL + ts.tv_nsec);
00140     // We preferrably want ticksFromHostNs() to calculate the the
00141     // ceiling rather than truncating the value. However, there are
00142     // other cases where truncating is fine, so we just add 1 here to
00143     // make sure that the actual resolution is strictly less than what
00144     // we return. We could get all kinds of nasty behavior if
00145     // arm(resolution) is called and the resulting time is 0 (which
00146     // could happen if we truncate the results and the resolution is
00147     // 1ns).
00148     const Tick resolution(ticksFromHostNs(res_ns) + 1);
00149     // It might not make sense to enter into KVM for less than a
00150     // certain number of host cycles. In some systems (e.g., Linux)
00151     // the resolution of the timer we use is 1ns (a few cycles on most
00152     // CPUs), which isn't very useful.
00153     const Tick min_cycles(ticksFromHostCycles(MIN_HOST_CYCLES));
00154 
00155     return std::max(resolution, min_cycles);
00156 }

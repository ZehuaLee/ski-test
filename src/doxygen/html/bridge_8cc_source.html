00147 bool
00148 Bridge::BridgeSlavePort::recvTimingReq(PacketPtr pkt)
00149 {
00150     DPRINTF(Bridge, "recvTimingReq: %s addr 0x%x\n",
00151             pkt->cmdString(), pkt->getAddr());
00152 
00153     // we should not see a timing request if we are already in a retry
00154     assert(!retryReq);
00155 
00156     DPRINTF(Bridge, "Response queue size: %d outresp: %d\n",
00157             transmitList.size(), outstandingResponses);
00158 
00159     // if the request queue is full then there is no hope
00160     if (masterPort.reqQueueFull()) {
00161         DPRINTF(Bridge, "Request queue full\n");
00162         retryReq = true;
00163     } else {
00164         // look at the response queue if we expect to see a response
00165         bool expects_response = pkt->needsResponse() &&
00166             !pkt->memInhibitAsserted();
00167         if (expects_response) {
00168             if (respQueueFull()) {
00169                 DPRINTF(Bridge, "Response queue full\n");
00170                 retryReq = true;
00171             } else {
00172                 // ok to send the request with space for the response
00173                 DPRINTF(Bridge, "Reserving space for response\n");
00174                 assert(outstandingResponses != respQueueLimit);
00175                 ++outstandingResponses;
00176 
00177                 // no need to set retryReq to false as this is already the
00178                 // case
00179             }
00180         }
00181 
00182         if (!retryReq) {
00183             // @todo: We need to pay for this and not just zero it out
00184             pkt->headerDelay = pkt->payloadDelay = 0;
00185 
00186             masterPort.schedTimingReq(pkt, bridge.clockEdge(delay));
00187         }
00188     }
00189 
00190     // remember that we are now stalling a packet and that we have to
00191     // tell the sending master to retry once space becomes available,
00192     // we make no distinction whether the stalling is due to the
00193     // request queue or response queue being full
00194     return !retryReq;

00242

00244 
00245     assert(req.tick <= curTick());
00246 
00247     PacketPtr pkt = req.pkt;
00248 
00249     DPRINTF(Bridge, "trySend request addr 0x%x, queue size %d\n",

00253         // send successful
00254         transmitList.pop_front();
00255         DPRINTF(Bridge, "trySend request successful\n");
00256 
00257         // If there are more packets to send, schedule event to try again.
00258         if (!transmitList.empty()) {
00259             DeferredPacket next_req = transmitList.front();
00260             DPRINTF(Bridge, "Scheduling next send\n");
00261             bridge.schedule(sendEvent, std::max(next_req.tick,
00262                                                 bridge.clockEdge()));
00263         }
00264 
00265         // if we have stalled a request due to a full request queue,
00266         // then send a retry at this point, also note that if the
00267         // request we stalled was waiting for the response queue
00268         // rather than the request queue we might stall it again
00269         slavePort.retryStalledReq();
00270     }
00271 
00272     // if the send failed, then we try again once we receive a retry,
00273     // and therefore there is no need to take any action
00274 }

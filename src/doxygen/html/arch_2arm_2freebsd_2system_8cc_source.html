00076 void
00077 FreebsdArmSystem::initState()
00078 {
00079     // Moved from the constructor to here since it relies on the
00080     // address map being resolved in the interconnect
00081 
00082     // Call the initialisation of the super class
00083     GenericArmSystem::initState();
00084 
00085     // Load symbols at physical address, we might not want
00086     // to do this permanently, for but early bootup work
00087     // it is helpful.
00088     if (params()->early_kernel_symbols) {
00089         kernel->loadGlobalSymbols(kernelSymtab, loadAddrMask);
00090         kernel->loadGlobalSymbols(debugSymbolTable, loadAddrMask);
00091     }
00092 
00093     // Setup boot data structure
00094     Addr addr = 0;
00095 
00096     // Check if the kernel image has a symbol that tells us it supports
00097     // device trees.
00098     bool kernel_has_fdt_support =
00099         kernelSymtab->findAddress("fdt_get_range", addr);
00100     bool dtb_file_specified = params()->dtb_filename != "";
00101 
00102     if (!dtb_file_specified)
00103         fatal("dtb file is not specified\n");
00104 
00105     if (!kernel_has_fdt_support)
00106         fatal("kernel must have fdt support\n");
00107 
00108     // Kernel supports flattened device tree and dtb file specified.
00109     // Using Device Tree Blob to describe system configuration.
00110     inform("Loading DTB file: %s at address %#x\n", params()->dtb_filename,
00111             params()->atags_addr + loadAddrOffset);
00112 
00113     ObjectFile *dtb_file = createObjectFile(params()->dtb_filename, true);
00114     if (!dtb_file) {
00115         fatal("couldn't load DTB file: %s\n", params()->dtb_filename);
00116     }
00117 
00118     DtbObject *_dtb_file = dynamic_cast<DtbObject*>(dtb_file);
00119 
00120     if (_dtb_file) {
00121         if (!_dtb_file->addBootCmdLine(params()->boot_osflags.c_str(),
00122                                        params()->boot_osflags.size())) {
00123             warn("couldn't append bootargs to DTB file: %s\n",
00124                  params()->dtb_filename);
00125         }
00126     } else {
00127         warn("dtb_file cast failed; couldn't append bootargs "
00128              "to DTB file: %s\n", params()->dtb_filename);
00129     }
00130 
00131     Addr ra = _dtb_file->findReleaseAddr();
00132     if (ra)
00133         bootReleaseAddr = ra & ~ULL(0x7F);
00134 
00135     dtb_file->setTextBase(params()->atags_addr + loadAddrOffset);
00136     dtb_file->loadSections(physProxy);
00137     delete dtb_file;
00138 
00139     // Kernel boot requirements to set up r0, r1 and r2 in ARMv7
00140     for (int i = 0; i < threadContexts.size(); i++) {
00141         threadContexts[i]->setIntReg(0, 0);
00142         threadContexts[i]->setIntReg(1, params()->machine_type);
00143         threadContexts[i]->setIntReg(2, params()->atags_addr + loadAddrOffset);
00144     }
00145 }

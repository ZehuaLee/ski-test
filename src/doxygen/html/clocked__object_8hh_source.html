00085     void update() const
00086     {
00087         // both tick and cycle are up-to-date and we are done, note
00088         // that the >= is important as it captures cases where tick
00089         // has already passed curTick()
00090         if (tick >= curTick())
00091             return;
00092 
00093         // optimise for the common case and see if the tick should be
00094         // advanced by a single clock period
00095         tick += clockPeriod();
00096         ++cycle;
00097 
00098         // see if we are done at this point
00099         if (tick >= curTick())
00100             return;
00101 
00102         // if not, we have to recalculate the cycle and tick, we
00103         // perform the calculations in terms of relative cycles to
00104         // allow changes to the clock period in the future
00105         Cycles elapsedCycles(divCeil(curTick() - tick, clockPeriod()));
00106         cycle += elapsedCycles;
00107         tick += elapsedCycles * clockPeriod();
00108     }

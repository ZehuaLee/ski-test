00082 void
00083 Ticked::unserialize(Checkpoint *cp, const std::string &section)
00084 {
00085     uint64_t lastStoppedUint = 0;
00086 
00087     /* lastStopped is optional on checkpoint restore as this object may be
00088      *  being restored from one which has a common base (and so possibly
00089      *  many common checkpointed values) but where Ticked is used in the
00090      *  checkpointed object but not this one.
00091      *  An example would be a CPU model using Ticked restores from a
00092      *  simple CPU without without Ticked */
00093     optParamIn(cp, section, "lastStopped", lastStoppedUint);
00094 
00095     lastStopped = Cycles(lastStoppedUint);

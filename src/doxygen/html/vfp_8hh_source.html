00263 template <typename T>
00264 uint64_t
00265 vfpFpToFixed(T val, bool isSigned, uint8_t width, uint8_t imm, bool
00266              useRmode = true, VfpRoundingMode roundMode = VfpRoundZero,
00267              bool aarch64 = false)
00268 {
00269     int  rmode;
00270     bool roundAwayFix = false;
00271 
00272     if (!useRmode) {
00273         rmode = fegetround();
00274     } else {
00275         switch (roundMode)
00276         {
00277           case VfpRoundNearest:
00278             rmode = FeRoundNearest;
00279             break;
00280           case VfpRoundUpward:
00281             rmode = FeRoundUpward;
00282             break;
00283           case VfpRoundDown:
00284             rmode = FeRoundDown;
00285             break;
00286           case VfpRoundZero:
00287             rmode = FeRoundZero;
00288             break;
00289           case VfpRoundAway:
00290             // There is no equivalent rounding mode, use round down and we'll
00291             // fix it later
00292             rmode        = FeRoundDown;
00293             roundAwayFix = true;
00294             break;
00295           default:
00296             panic("Unsupported roundMode %d\n", roundMode);
00297         }
00298     }
00299     __asm__ __volatile__("" : "=m" (rmode) : "m" (rmode));
00300     fesetround(FeRoundNearest);
00301     val = val * pow(2.0, imm);
00302     __asm__ __volatile__("" : "=m" (val) : "m" (val));
00303     fesetround(rmode);
00304     feclearexcept(FeAllExceptions);
00305     __asm__ __volatile__("" : "=m" (val) : "m" (val));
00306     T origVal = val;
00307     val = rint(val);
00308     __asm__ __volatile__("" : "=m" (val) : "m" (val));
00309 
00310     int exceptions = fetestexcept(FeAllExceptions);
00311 
00312     int fpType = std::fpclassify(val);
00313     if (fpType == FP_SUBNORMAL || fpType == FP_NAN) {
00314         if (fpType == FP_NAN) {
00315             exceptions |= FeInvalid;
00316         }
00317         val = 0.0;
00318     } else if (origVal != val) {
00319         switch (rmode) {
00320           case FeRoundNearest:
00321             if (origVal - val > 0.5)
00322                 val += 1.0;
00323             else if (val - origVal > 0.5)
00324                 val -= 1.0;
00325             break;
00326           case FeRoundDown:
00327             if (roundAwayFix) {
00328                 // The ordering on the subtraction looks a bit odd in that we
00329                 // don't do the obvious origVal - val, instead we do
00330                 // -(val - origVal). This is required to get the corruct bit
00331                 // exact behaviour when very close to the 0.5 threshold.
00332                 volatile T error = val;
00333                 error -= origVal;
00334                 error = -error;
00335                 if ( (error >  0.5) ||
00336                     ((error == 0.5) && (val >= 0)) )
00337                     val += 1.0;
00338             } else {
00339                 if (origVal < val)
00340                     val -= 1.0;
00341             }
00342             break;
00343           case FeRoundUpward:
00344             if (origVal > val)
00345                 val += 1.0;
00346             break;
00347         }
00348         exceptions |= FeInexact;
00349     }
00350 
00351     __asm__ __volatile__("" : "=m" (val) : "m" (val));
00352 
00353     if (isSigned) {
00354         bool     outOfRange = false;
00355         int64_t  result     = (int64_t) val;
00356         uint64_t finalVal;
00357 
00358         if (!aarch64) {
00359             if (width == 16) {
00360                 finalVal = (int16_t)val;
00361             } else if (width == 32) {
00362                 finalVal =(int32_t)val;
00363             } else if (width == 64) {
00364                 finalVal = result;
00365             } else {
00366                 panic("Unsupported width %d\n", width);
00367             }
00368 
00369             // check if value is in range
00370             int64_t minVal = ~mask(width-1);
00371             if ((double)val < minVal) {
00372                 outOfRange = true;
00373                 finalVal = minVal;
00374             }
00375             int64_t maxVal = mask(width-1);
00376             if ((double)val > maxVal) {
00377                 outOfRange = true;
00378                 finalVal = maxVal;
00379             }
00380         } else {
00381             bool isNeg = val < 0;
00382             finalVal = result & mask(width);
00383             // If the result is supposed to be less than 64 bits check that the
00384             // upper bits that got thrown away are just sign extension bits
00385             if (width != 64) {
00386                 outOfRange = ((uint64_t) result >> (width - 1)) !=
00387                              (isNeg ? mask(64-width+1) : 0);
00388             }
00389             // Check if the original floating point value doesn't matches the
00390             // integer version we are also out of range. So create a saturated
00391             // result.
00392             if (isNeg) {
00393                 outOfRange |= val < result;
00394                 if (outOfRange) {
00395                     finalVal = 1LL << (width-1);
00396                 }
00397             } else {
00398                 outOfRange |= val > result;
00399                 if (outOfRange) {
00400                     finalVal = mask(width-1);
00401                 }
00402             }
00403         }
00404 
00405         // Raise an exception if the value was out of range
00406         if (outOfRange) {
00407             exceptions |= FeInvalid;
00408             exceptions &= ~FeInexact;
00409         }
00410         setFPExceptions(exceptions);
00411         return finalVal;
00412     } else {
00413         if ((double)val < 0) {
00414             exceptions |= FeInvalid;
00415             exceptions &= ~FeInexact;
00416             setFPExceptions(exceptions);
00417             return 0;
00418         }
00419 
00420         uint64_t result = ((uint64_t) val) & mask(width);
00421         if (val > result) {
00422             exceptions |= FeInvalid;
00423             exceptions &= ~FeInexact;
00424             setFPExceptions(exceptions);
00425             return mask(width);
00426         }
00427 
00428         setFPExceptions(exceptions);
00429         return result;
00430     }
00431 };

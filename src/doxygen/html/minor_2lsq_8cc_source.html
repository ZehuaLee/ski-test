00350 void
00351 LSQ::SplitDataRequest::makeFragmentRequests()
00352 {
00353     Addr base_addr = request.getVaddr();
00354     unsigned int whole_size = request.getSize();
00355     unsigned int line_width = port.lineWidth;
00356 
00357     unsigned int fragment_size;
00358     Addr fragment_addr;
00359 
00360     /* Assume that this transfer is across potentially many block snap
00361      * boundaries:
00362      *
00363      * |      _|________|________|________|___     |
00364      * |     |0| 1      | 2      | 3      | 4 |    |
00365      * |     |_|________|________|________|___|    |
00366      * |       |        |        |        |        |
00367      *
00368      *  The first transfer (0) can be up to lineWidth in size.
00369      *  All the middle transfers (1-3) are lineWidth in size
00370      *  The last transfer (4) can be from zero to lineWidth - 1 in size
00371      */
00372     unsigned int first_fragment_offset =
00373         addrBlockOffset(base_addr, line_width);
00374     unsigned int last_fragment_size =
00375         addrBlockOffset(base_addr + whole_size, line_width);
00376     unsigned int first_fragment_size =
00377         line_width - first_fragment_offset;
00378 
00379     unsigned int middle_fragments_total_size =
00380         whole_size - (first_fragment_size + last_fragment_size);
00381 
00382     assert(addrBlockOffset(middle_fragments_total_size, line_width) == 0);
00383 
00384     unsigned int middle_fragment_count =
00385         middle_fragments_total_size / line_width;
00386 
00387     numFragments = 1 /* first */ + middle_fragment_count +
00388         (last_fragment_size == 0 ? 0 : 1);
00389 
00390     DPRINTFS(MinorMem, (&port), "Dividing transfer into %d fragmentRequests."
00391         " First fragment size: %d Last fragment size: %d\n",
00392         numFragments, first_fragment_size,
00393         (last_fragment_size == 0 ? line_width : last_fragment_size));
00394 
00395     assert(((middle_fragment_count * line_width) +
00396         first_fragment_size + last_fragment_size) == whole_size);
00397 
00398     fragment_addr = base_addr;
00399     fragment_size = first_fragment_size;
00400 
00401     /* Just past the last address in the request */
00402     Addr end_addr = base_addr + whole_size;
00403 
00404     for (unsigned int fragment_index = 0; fragment_index < numFragments;
00405          fragment_index++)
00406     {
00407         bool M5_VAR_USED is_last_fragment = false;
00408 
00409         if (fragment_addr == base_addr) {
00410             /* First fragment */
00411             fragment_size = first_fragment_size;
00412         } else {
00413             if ((fragment_addr + line_width) > end_addr) {
00414                 /* Adjust size of last fragment */
00415                 fragment_size = end_addr - fragment_addr;
00416                 is_last_fragment = true;
00417             } else {
00418                 /* Middle fragments */
00419                 fragment_size = line_width;
00420             }
00421         }
00422 
00423         Request *fragment = new Request();
00424 
00425         fragment->setThreadContext(request.contextId(), /* thread id */ 0);
00426         fragment->setVirt(0 /* asid */,
00427             fragment_addr, fragment_size, request.getFlags(),
00428             request.masterId(),
00429             request.getPC());
00430 
00431         DPRINTFS(MinorMem, (&port), "Generating fragment addr: 0x%x size: %d"
00432             " (whole request addr: 0x%x size: %d) %s\n",
00433             fragment_addr, fragment_size, base_addr, whole_size,
00434             (is_last_fragment ? "last fragment" : ""));
00435 
00436         fragment_addr += fragment_size;
00437 
00438         fragmentRequests.push_back(fragment);
00439     }
00440 }

00527 void
00528 LSQ::SplitDataRequest::retireResponse(PacketPtr response)
00529 {
00530     assert(numRetiredFragments < numFragments);
00531 
00532     DPRINTFS(MinorMem, (&port), "Retiring fragment addr: 0x%x size: %d"
00533         " offset: 0x%x (retired fragment num: %d) %s\n",
00534         response->req->getVaddr(), response->req->getSize(),
00535         request.getVaddr() - response->req->getVaddr(),
00536         numRetiredFragments,
00537         (fault == NoFault ? "" : fault->name()));
00538 
00539     numRetiredFragments++;
00540 
00541     if (skipped) {
00542         /* Skip because we already knew the request had faulted or been
00543          *  skipped */
00544         DPRINTFS(MinorMem, (&port), "Skipping this fragment\n");
00545     } else if (response->isError()) {
00546         /* Mark up the error and leave to execute to handle it */
00547         DPRINTFS(MinorMem, (&port), "Fragment has an error, skipping\n");
00548         setSkipped();
00549         packet->copyError(response);
00550     } else {
00551         if (isLoad) {
00552             if (!data) {
00553                 /* For a split transfer, a Packet must be constructed
00554                  *  to contain all returning data.  This is that packet's
00555                  *  data */
00556                 data = new uint8_t[request.getSize()];
00557             }
00558 
00559             /* Populate the portion of the overall response data represented
00560              *  by the response fragment */
00561             std::memcpy(
00562                 data + (response->req->getVaddr() - request.getVaddr()),
00563                 response->getConstPtr<uint8_t>(),
00564                 response->req->getSize());
00565         }
00566     }
00567 
00568     /* Complete early if we're skipping are no more in-flight accesses */
00569     if (skipped && !hasPacketsInMemSystem()) {
00570         DPRINTFS(MinorMem, (&port), "Completed skipped burst\n");
00571         setState(Complete);
00572         if (packet->needsResponse())
00573             packet->makeResponse();
00574     }
00575 
00576     if (numRetiredFragments == numFragments)
00577         setState(Complete);
00578 
00579     if (!skipped && isComplete()) {
00580         DPRINTFS(MinorMem, (&port), "Completed burst %d\n", packet != NULL);
00581 
00582         DPRINTFS(MinorMem, (&port), "Retired packet isRead: %d isWrite: %d"

00584              " %s\n", packet->isRead(), packet->isWrite(),

00586              response->getSize());
00587 
00588         /* A request can become complete by several paths, this is a sanity
00589          *  check to make sure the packet's data is created */
00590         if (!data) {
00591             data = new uint8_t[request.getSize()];
00592         }
00593 
00594         if (isLoad) {
00595             DPRINTFS(MinorMem, (&port), "Copying read data\n");
00596             std::memcpy(packet->getPtr<uint8_t>(), data, request.getSize());

00602 }

00672     AddrRangeCoverage ret = NoAddrRangeCoverage;
00673 
00674     /* Traverse the store buffer in reverse order (most to least recent)
00675      *  and try to find a slot whose address range overlaps this request */
00676     while (ret == NoAddrRangeCoverage && i != slots.rend()) {
00677         LSQRequestPtr slot = *i;
00678 
00679         if (slot->packet) {
00680             AddrRangeCoverage coverage = slot->containsAddrRangeOf(request);
00681 
00682             if (coverage != NoAddrRangeCoverage) {
00683                 DPRINTF(MinorMem, "Forwarding: slot: %d result: %s thisAddr:"
00684                     " 0x%x thisSize: %d slotAddr: 0x%x slotSize: %d\n",
00685                     slot_index, coverage,
00686                     request->request.getPaddr(), request->request.getSize(),
00687                     slot->request.getPaddr(), slot->request.getSize());
00688 
00689                 found_slot = slot_index;
00690                 ret = coverage;
00691             }
00692         }
00693 
00694         i++;
00695         slot_index--;
00696     }
00697 
00698     return ret;
00699 }

00741 void
00742 LSQ::StoreBuffer::step()
00743 {
00744     DPRINTF(MinorMem, "StoreBuffer step numUnissuedAccesses: %d\n",
00745         numUnissuedAccesses);
00746 
00747     if (numUnissuedAccesses != 0 && lsq.state == LSQ::MemoryRunning) {
00748         /* Clear all the leading barriers */
00749         while (!slots.empty() &&
00750             slots.front()->isComplete() && slots.front()->isBarrier())
00751         {
00752             LSQRequestPtr barrier = slots.front();
00753 
00754             DPRINTF(MinorMem, "Clearing barrier for inst: %s\n",
00755                 *(barrier->inst));
00756 
00757             numUnissuedAccesses--;
00758             lsq.clearMemBarrier(barrier->inst);
00759             slots.pop_front();
00760 
00761             delete barrier;
00762         }
00763 
00764         auto i = slots.begin();
00765         bool issued = true;
00766         unsigned int issue_count = 0;
00767 
00768         /* Skip trying if the memory system is busy */
00769         if (lsq.state == LSQ::MemoryNeedsRetry)
00770             issued = false;
00771 
00772         /* Try to issue all stores in order starting from the head
00773          *  of the queue.  Responses are allowed to be retired
00774          *  out of order */
00775         while (issued &&
00776             issue_count < storeLimitPerCycle &&
00777             lsq.canSendToMemorySystem() &&
00778             i != slots.end())
00779         {
00780             LSQRequestPtr request = *i;
00781 
00782             DPRINTF(MinorMem, "Considering request: %s, sentAllPackets: %d"
00783                 " state: %s\n",
00784                 *(request->inst), request->sentAllPackets(),
00785                 request->state);
00786 
00787             if (request->isBarrier() && request->isComplete()) {
00788                 /* Give up at barriers */
00789                 issued = false;
00790             } else if (!(request->state == LSQRequest::StoreBufferIssuing &&
00791                 request->sentAllPackets()))
00792             {
00793                 DPRINTF(MinorMem, "Trying to send request: %s to memory"
00794                     " system\n", *(request->inst));
00795 
00796                 if (lsq.tryToSend(request)) {
00797                     countIssuedStore(request);
00798                     issue_count++;
00799                 } else {
00800                     /* Don't step on to the next store buffer entry if this
00801                      *  one hasn't issued all its packets as the store
00802                      *  buffer must still enforce ordering */
00803                     issued = false;
00804                 }
00805             }
00806             i++;
00807         }
00808     }
00809 }

00812 LSQ::completeMemBarrierInst(MinorDynInstPtr inst,
00813     bool committed)
00814 {
00815     if (committed) {
00816         /* Not already sent to the store buffer as a store request? */
00817         if (!inst->inStoreBuffer) {

00819              *  until there are none in flight */
00820             storeBuffer.insert(new BarrierDataRequest(*this, inst));
00821         }
00822     } else {
00823         /* Clear the barrier anyway if it wasn't actually committed */
00824         clearMemBarrier(inst);

00857 void
00858 LSQ::tryToSendToTransfers(LSQRequestPtr request)
00859 {
00860     if (state == MemoryNeedsRetry) {
00861         DPRINTF(MinorMem, "Request needs retry, not issuing to"
00862             " memory until retry arrives\n");
00863         return;
00864     }
00865 
00866     if (request->state == LSQRequest::InTranslation) {
00867         DPRINTF(MinorMem, "Request still in translation, not issuing to"
00868             " memory\n");
00869         return;
00870     }
00871 
00872     assert(request->state == LSQRequest::Translated ||
00873         request->state == LSQRequest::RequestIssuing ||
00874         request->state == LSQRequest::Failed ||
00875         request->state == LSQRequest::Complete);
00876 
00877     if (requests.empty() || requests.front() != request) {
00878         DPRINTF(MinorMem, "Request not at front of requests queue, can't"
00879             " issue to memory\n");
00880         return;
00881     }
00882 
00883     if (transfers.unreservedRemainingSpace() == 0) {
00884         DPRINTF(MinorMem, "No space to insert request into transfers"
00885             " queue\n");
00886         return;
00887     }
00888 
00889     if (request->isComplete() || request->state == LSQRequest::Failed) {
00890         DPRINTF(MinorMem, "Passing a %s transfer on to transfers"
00891             " queue\n", (request->isComplete() ? "completed" : "failed"));
00892         request->setState(LSQRequest::Complete);
00893         request->setSkipped();
00894         moveFromRequestsToTransfers(request);
00895         return;
00896     }
00897 
00898     if (!execute.instIsRightStream(request->inst)) {
00899         /* Wrong stream, try to abort the transfer but only do so if
00900          *  there are no packets in flight */
00901         if (request->hasPacketsInMemSystem()) {
00902             DPRINTF(MinorMem, "Request's inst. is from the wrong stream,"
00903                 " waiting for responses before aborting request\n");
00904         } else {
00905             DPRINTF(MinorMem, "Request's inst. is from the wrong stream,"
00906                 " aborting request\n");
00907             request->setState(LSQRequest::Complete);
00908             request->setSkipped();
00909             moveFromRequestsToTransfers(request);
00910         }
00911         return;
00912     }
00913 
00914     if (request->fault != NoFault) {
00915         if (request->inst->staticInst->isPrefetch()) {
00916             DPRINTF(MinorMem, "Not signalling fault for faulting prefetch\n");
00917         }
00918         DPRINTF(MinorMem, "Moving faulting request into the transfers"
00919             " queue\n");
00920         request->setState(LSQRequest::Complete);
00921         request->setSkipped();
00922         moveFromRequestsToTransfers(request);
00923         return;
00924     }
00925 
00926     bool is_load = request->isLoad;
00927     bool is_llsc = request->request.isLLSC();
00928     bool is_swap = request->request.isSwap();
00929     bool bufferable = !(request->request.isStrictlyOrdered() ||
00930         is_llsc || is_swap);
00931 
00932     if (is_load) {
00933         if (numStoresInTransfers != 0) {
00934             DPRINTF(MinorMem, "Load request with stores still in transfers"
00935                 " queue, stalling\n");
00936             return;
00937         }
00938     } else {
00939         /* Store.  Can it be sent to the store buffer? */
00940         if (bufferable && !request->request.isMmappedIpr()) {
00941             request->setState(LSQRequest::StoreToStoreBuffer);
00942             moveFromRequestsToTransfers(request);
00943             DPRINTF(MinorMem, "Moving store into transfers queue\n");
00944             return;
00945         }
00946     }
00947 
00948     /* Check if this is the head instruction (and so must be executable as
00949      *  its stream sequence number was checked above) for loads which must
00950      *  not be speculatively issued and stores which must be issued here */
00951     if (!bufferable) {
00952         if (!execute.instIsHeadInst(request->inst)) {
00953             DPRINTF(MinorMem, "Memory access not the head inst., can't be"
00954                 " sure it can be performed, not issuing\n");
00955             return;
00956         }
00957 
00958         unsigned int forwarding_slot = 0;
00959 
00960         if (storeBuffer.canForwardDataToLoad(request, forwarding_slot) !=
00961             NoAddrRangeCoverage)
00962         {
00963             DPRINTF(MinorMem, "Memory access can receive forwarded data"
00964                 " from the store buffer, need to wait for store buffer to"
00965                 " drain\n");
00966             return;
00967         }
00968     }
00969 
00970     /* True: submit this packet to the transfers queue to be sent to the
00971      * memory system.
00972      * False: skip the memory and push a packet for this request onto
00973      * requests */
00974     bool do_access = true;
00975 
00976     if (!is_llsc) {
00977         /* Check for match in the store buffer */
00978         if (is_load) {
00979             unsigned int forwarding_slot = 0;
00980             AddrRangeCoverage forwarding_result =
00981                 storeBuffer.canForwardDataToLoad(request,
00982                 forwarding_slot);
00983 
00984             switch (forwarding_result) {
00985               case FullAddrRangeCoverage:
00986                 /* Forward data from the store buffer into this request and
00987                  *  repurpose this request's packet into a response packet */
00988                 storeBuffer.forwardStoreData(request, forwarding_slot);
00989                 request->packet->makeResponse();
00990 
00991                 /* Just move between queues, no access */
00992                 do_access = false;
00993                 break;
00994               case PartialAddrRangeCoverage:
00995                 DPRINTF(MinorMem, "Load partly satisfied by store buffer"
00996                     " data. Must wait for the store to complete\n");
00997                 return;
00998                 break;
00999               case NoAddrRangeCoverage:
01000                 DPRINTF(MinorMem, "No forwardable data from store buffer\n");
01001                 /* Fall through to try access */
01002                 break;
01003             }
01004         }
01005     } else {
01006         if (!canSendToMemorySystem()) {
01007             DPRINTF(MinorMem, "Can't send request to memory system yet\n");
01008             return;
01009         }
01010 
01011         SimpleThread &thread = *cpu.threads[request->inst->id.threadId];
01012 
01013         TheISA::PCState old_pc = thread.pcState();
01014         ExecContext context(cpu, thread, execute, request->inst);
01015 
01016         /* Handle LLSC requests and tests */
01017         if (is_load) {
01018             TheISA::handleLockedRead(&context, &request->request);
01019         } else {
01020             do_access = TheISA::handleLockedWrite(&context,
01021                 &request->request, cacheBlockMask);
01022 
01023             if (!do_access) {
01024                 DPRINTF(MinorMem, "Not perfoming a memory "
01025                     "access for store conditional\n");
01026             }
01027         }
01028         thread.pcState(old_pc);
01029     }
01030 
01031     /* See the do_access comment above */
01032     if (do_access) {
01033         if (!canSendToMemorySystem()) {
01034             DPRINTF(MinorMem, "Can't send request to memory system yet\n");
01035             return;
01036         }
01037 
01038         /* Remember if this is an access which can't be idly
01039          *  discarded by an interrupt */
01040         if (!bufferable && !request->issuedToMemory) {
01041             numAccessesIssuedToMemory++;
01042             request->issuedToMemory = true;
01043         }
01044 
01045         if (tryToSend(request))
01046             moveFromRequestsToTransfers(request);
01047     } else {
01048         request->setState(LSQRequest::Complete);
01049         moveFromRequestsToTransfers(request);
01050     }
01051 }

01053 bool
01054 LSQ::tryToSend(LSQRequestPtr request)
01055 {
01056     bool ret = false;
01057 
01058     if (!canSendToMemorySystem()) {
01059         DPRINTF(MinorMem, "Can't send request: %s yet, no space in memory\n",
01060             *(request->inst));
01061     } else {
01062         PacketPtr packet = request->getHeadPacket();
01063 
01064         DPRINTF(MinorMem, "Trying to send request: %s addr: 0x%x\n",
01065             *(request->inst), packet->req->getVaddr());
01066 
01067         /* The sender state of the packet *must* be an LSQRequest
01068          *  so the response can be correctly handled */
01069         assert(packet->findNextSenderState<LSQRequest>());
01070 
01071         if (request->request.isMmappedIpr()) {
01072             ThreadContext *thread =
01073                 cpu.getContext(request->request.threadId());
01074 
01075             if (request->isLoad) {
01076                 DPRINTF(MinorMem, "IPR read inst: %s\n", *(request->inst));
01077                 TheISA::handleIprRead(thread, packet);
01078             } else {
01079                 DPRINTF(MinorMem, "IPR write inst: %s\n", *(request->inst));
01080                 TheISA::handleIprWrite(thread, packet);
01081             }
01082 
01083             request->stepToNextPacket();
01084             ret = request->sentAllPackets();
01085 
01086             if (!ret) {
01087                 DPRINTF(MinorMem, "IPR access has another packet: %s\n",
01088                     *(request->inst));
01089             }
01090 
01091             if (ret)
01092                 request->setState(LSQRequest::Complete);
01093             else
01094                 request->setState(LSQRequest::RequestIssuing);
01095         } else if (dcachePort.sendTimingReq(packet)) {
01096             DPRINTF(MinorMem, "Sent data memory request\n");
01097 
01098             numAccessesInMemorySystem++;
01099 
01100             request->stepToNextPacket();
01101 
01102             ret = request->sentAllPackets();
01103 
01104             switch (request->state) {
01105               case LSQRequest::Translated:
01106               case LSQRequest::RequestIssuing:
01107                 /* Fully or partially issued a request in the transfers
01108                  *  queue */
01109                 request->setState(LSQRequest::RequestIssuing);
01110                 break;
01111               case LSQRequest::StoreInStoreBuffer:
01112               case LSQRequest::StoreBufferIssuing:
01113                 /* Fully or partially issued a request in the store
01114                  *  buffer */
01115                 request->setState(LSQRequest::StoreBufferIssuing);
01116                 break;
01117               default:
01118                 assert(false);
01119                 break;
01120             }
01121 
01122             state = MemoryRunning;
01123         } else {
01124             DPRINTF(MinorMem,
01125                 "Sending data memory request - needs retry\n");
01126 
01127             /* Needs to be resent, wait for that */
01128             state = MemoryNeedsRetry;
01129             retryRequest = request;
01130 
01131             switch (request->state) {
01132               case LSQRequest::Translated:
01133               case LSQRequest::RequestIssuing:
01134                 request->setState(LSQRequest::RequestNeedsRetry);
01135                 break;
01136               case LSQRequest::StoreInStoreBuffer:
01137               case LSQRequest::StoreBufferIssuing:
01138                 request->setState(LSQRequest::StoreBufferNeedsRetry);
01139                 break;
01140               default:
01141                 assert(false);
01142                 break;
01143             }
01144         }
01145     }
01146 
01147     return ret;
01148 }

01150 void
01151 LSQ::moveFromRequestsToTransfers(LSQRequestPtr request)
01152 {
01153     assert(!requests.empty() && requests.front() == request);
01154     assert(transfers.unreservedRemainingSpace() != 0);
01155 
01156     /* Need to count the number of stores in the transfers
01157      *  queue so that loads know when their store buffer forwarding
01158      *  results will be correct (only when all those stores
01159      *  have reached the store buffer) */
01160     if (!request->isLoad)
01161         numStoresInTransfers++;
01162 
01163     requests.pop();
01164     transfers.push(request);
01165 }

01174 bool
01175 LSQ::recvTimingResp(PacketPtr response)
01176 {
01177     LSQRequestPtr request =
01178         safe_cast<LSQRequestPtr>(response->popSenderState());
01179 
01180     DPRINTF(MinorMem, "Received response packet inst: %s"
01181         " addr: 0x%x cmd: %s\n",
01182         *(request->inst), response->getAddr(),
01183         response->cmd.toString());
01184 
01185     numAccessesInMemorySystem--;
01186 
01187     if (response->isError()) {
01188         DPRINTF(MinorMem, "Received error response packet: %s\n",
01189             *request->inst);
01190     }
01191 
01192     switch (request->state) {
01193       case LSQRequest::RequestIssuing:
01194       case LSQRequest::RequestNeedsRetry:
01195         /* Response to a request from the transfers queue */
01196         request->retireResponse(response);
01197 
01198         DPRINTF(MinorMem, "Has outstanding packets?: %d %d\n",
01199             request->hasPacketsInMemSystem(), request->isComplete());
01200 
01201         break;
01202       case LSQRequest::StoreBufferIssuing:
01203       case LSQRequest::StoreBufferNeedsRetry:
01204         /* Response to a request from the store buffer */
01205         request->retireResponse(response);
01206 
01207         /* Remove completed requests unless they are barriers (which will
01208          *  need to be removed in order */
01209         if (request->isComplete()) {
01210             if (!request->isBarrier()) {
01211                 storeBuffer.deleteRequest(request);
01212             } else {
01213                 DPRINTF(MinorMem, "Completed transfer for barrier: %s"
01214                     " leaving the request as it is also a barrier\n",
01215                     *(request->inst));
01216             }
01217         }
01218         break;
01219       default:
01220         /* Shouldn't be allowed to receive a response from another
01221          *  state */
01222         assert(false);
01223         break;
01224     }
01225 
01226     /* We go to idle even if there are more things in the requests queue
01227      * as it's the job of step to actually step us on to the next
01228      * transaction */
01229 
01230     /* Let's try and wake up the processor for the next cycle */
01231     cpu.wakeupOnEvent(Pipeline::ExecuteStageId);
01232 
01233     /* Never busy */
01234     return true;
01235 }

01237 void
01238 LSQ::recvReqRetry()
01239 {
01240     DPRINTF(MinorMem, "Received retry request\n");
01241 
01242     assert(state == MemoryNeedsRetry);
01243 
01244     switch (retryRequest->state) {
01245       case LSQRequest::RequestNeedsRetry:
01246         /* Retry in the requests queue */
01247         retryRequest->setState(LSQRequest::Translated);
01248         break;
01249       case LSQRequest::StoreBufferNeedsRetry:
01250         /* Retry in the store buffer */
01251         retryRequest->setState(LSQRequest::StoreInStoreBuffer);
01252         break;
01253       default:
01254         assert(false);
01255     }
01256 
01257     /* Set state back to MemoryRunning so that the following
01258      *  tryToSend can actually send.  Note that this won't
01259      *  allow another transfer in as tryToSend should
01260      *  issue a memory request and either succeed for this
01261      *  request or return the LSQ back to MemoryNeedsRetry */
01262     state = MemoryRunning;
01263 
01264     /* Try to resend the request */
01265     if (tryToSend(retryRequest)) {
01266         /* Successfully sent, need to move the request */
01267         switch (retryRequest->state) {
01268           case LSQRequest::RequestIssuing:
01269             /* In the requests queue */
01270             moveFromRequestsToTransfers(retryRequest);
01271             break;
01272           case LSQRequest::StoreBufferIssuing:
01273             /* In the store buffer */
01274             storeBuffer.countIssuedStore(retryRequest);
01275             break;
01276           default:
01277             assert(false);
01278             break;
01279         }
01280 
01281         retryRequest = NULL;
01282     }
01283 }

01357 void
01358 LSQ::step()
01359 {
01360     /* Try to move address-translated requests between queues and issue
01361      *  them */

01377          *  capable of being moved to the store buffer */

01465 void
01466 LSQ::pushRequest(MinorDynInstPtr inst, bool isLoad, uint8_t *data,
01467     unsigned int size, Addr addr, unsigned int flags, uint64_t *res)
01468 {
01469     bool needs_burst = transferNeedsBurst(addr, size, lineWidth);
01470     LSQRequestPtr request;
01471 
01472     /* Copy given data into the request.  The request will pass this to the
01473      *  packet and then it will own the data */
01474     uint8_t *request_data = NULL;
01475 
01476     DPRINTF(MinorMem, "Pushing request (%s) addr: 0x%x size: %d flags:"
01477         " 0x%x%s lineWidth : 0x%x\n",
01478         (isLoad ? "load" : "store"), addr, size, flags,
01479             (needs_burst ? " (needs burst)" : ""), lineWidth);
01480 
01481     if (!isLoad) {
01482         /* request_data becomes the property of a ...DataRequest (see below)
01483          *  and destroyed by its destructor */
01484         request_data = new uint8_t[size];
01485         if (flags & Request::CACHE_BLOCK_ZERO) {
01486             /* For cache zeroing, just use zeroed data */
01487             std::memset(request_data, 0, size);
01488         } else {
01489             std::memcpy(request_data, data, size);
01490         }
01491     }
01492 
01493     if (needs_burst) {
01494         request = new SplitDataRequest(
01495             *this, inst, isLoad, request_data, res);
01496     } else {
01497         request = new SingleDataRequest(
01498             *this, inst, isLoad, request_data, res);
01499     }
01500 
01501     if (inst->traceData)
01502         inst->traceData->setMem(addr, size, flags);
01503 
01504     request->request.setThreadContext(cpu.cpuId(), /* thread id */ 0);
01505     request->request.setVirt(0 /* asid */,
01506         addr, size, flags, cpu.dataMasterId(),
01507         /* I've no idea why we need the PC, but give it */
01508         inst->pc.instAddr());
01509 
01510     requests.push(request);
01511     request->startAddrTranslation();
01512 }

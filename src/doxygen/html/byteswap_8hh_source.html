00061 //These functions actually perform the swapping for parameters
00062 //of various bit lengths
00063 inline uint64_t
00064 swap_byte64(uint64_t x)
00065 {
00066 #if defined(__linux__)
00067     return bswap_64(x);
00068 #elif defined(__APPLE__)
00069     return OSSwapInt64(x);
00070 #else
00071     return  (uint64_t)((((uint64_t)(x) & 0xff) << 56) |
00072             ((uint64_t)(x) & 0xff00ULL) << 40 |
00073             ((uint64_t)(x) & 0xff0000ULL) << 24 |
00074             ((uint64_t)(x) & 0xff000000ULL) << 8 |
00075             ((uint64_t)(x) & 0xff00000000ULL) >> 8 |
00076             ((uint64_t)(x) & 0xff0000000000ULL) >> 24 |
00077             ((uint64_t)(x) & 0xff000000000000ULL) >> 40 |
00078             ((uint64_t)(x) & 0xff00000000000000ULL) >> 56) ;
00079 #endif
00080 }

00109 // This function lets the compiler figure out how to call the
00110 // swap_byte functions above for different data types.  Since the
00111 // sizeof() values are known at compile time, it should inline to a
00112 // direct call to the right swap_byteNN() function.
00113 template <typename T>
00114 inline T swap_byte(T x) {
00115     if (sizeof(T) == 8)
00116         return swap_byte64((uint64_t)x);
00117     else if (sizeof(T) == 4)
00118         return swap_byte32((uint32_t)x);
00119     else if (sizeof(T) == 2)
00120         return swap_byte16((uint16_t)x);
00121     else if (sizeof(T) == 1)
00122         return x;
00123     else
00124         panic("Can't byte-swap values larger than 64 bits");
00125 }

00065 void
00066 ThreadState::serialize(std::ostream &os)
00067 {
00068     SERIALIZE_ENUM(_status);
00069     // thread_num and cpu_id are deterministic from the config
00070     SERIALIZE_SCALAR(funcExeInst);
00071 
00072     if (!FullSystem)
00073         return;
00074 
00075     Tick quiesceEndTick = 0;
00076     if (quiesceEvent->scheduled())
00077         quiesceEndTick = quiesceEvent->when();
00078     SERIALIZE_SCALAR(quiesceEndTick);
00079     if (kernelStats)
00080         kernelStats->serialize(os);
00081 }

00086 
00087     UNSERIALIZE_ENUM(_status);

00089     UNSERIALIZE_SCALAR(funcExeInst);

00094     Tick quiesceEndTick;

00096     if (quiesceEndTick)
00097         baseCpu->schedule(quiesceEvent, quiesceEndTick);
00098     if (kernelStats)
00099         kernelStats->unserialize(cp, section);

00102 void
00103 ThreadState::initMemProxies(ThreadContext *tc)
00104 {
00105     // The port proxies only refer to the data port on the CPU side
00106     // and can safely be done at init() time even if the CPU is not
00107     // connected, i.e. when restoring from a checkpoint and later
00108     // switching the CPU in.
00109     if (FullSystem) {
00110         assert(physProxy == NULL);
00111         // This cannot be done in the constructor as the thread state
00112         // itself is created in the base cpu constructor and the
00113         // getDataPort is a virtual function
00114         physProxy = new PortProxy(baseCpu->getDataPort(),
00115                                   baseCpu->cacheLineSize());
00116 
00117         assert(virtProxy == NULL);
00118         virtProxy = new FSTranslatingPortProxy(tc);
00119     } else {
00120         assert(proxy == NULL);
00121         proxy = new SETranslatingPortProxy(baseCpu->getDataPort(),
00122                                            process,
00123                                            SETranslatingPortProxy::NextPage);
00124     }
00125 }

00078 void
00079 Switch::addOutPort(const vector<MessageBuffer*>& out,
00080                    const NetDest& routing_table_entry,
00081                    Cycles link_latency, int bw_multiplier)
00082 {
00083     // Create a throttle
00084     Throttle* throttle_ptr = new Throttle(m_id, m_throttles.size(),
00085                                           link_latency, bw_multiplier,
00086                                           m_network_ptr->getEndpointBandwidth(),
00087                                           this);
00088 
00089     m_throttles.push_back(throttle_ptr);
00090 
00091     // Create one buffer per vnet (these are intermediaryQueues)
00092     vector<MessageBuffer*> intermediateBuffers;
00093 
00094     for (int i = 0; i < out.size(); ++i) {
00095         if (out[i] != nullptr) {
00096             out[i]->setSender(this);
00097         }
00098 
00099         MessageBuffer* buffer_ptr = new MessageBuffer;
00100         // Make these queues ordered
00101         buffer_ptr->setOrdering(true);
00102         if (m_network_ptr->getBufferSize() > 0) {
00103             buffer_ptr->resize(m_network_ptr->getBufferSize());
00104         }
00105 
00106         intermediateBuffers.push_back(buffer_ptr);
00107         m_buffers_to_free.push_back(buffer_ptr);
00108 
00109         buffer_ptr->setSender(this);
00110         buffer_ptr->setReceiver(this);
00111     }
00112 
00113     // Hook the queues to the PerfectSwitch
00114     m_perfect_switch->addOutPort(intermediateBuffers, routing_table_entry);
00115 
00116     // Hook the queues to the Throttle
00117     throttle_ptr->addLinks(intermediateBuffers, out);
00118 }

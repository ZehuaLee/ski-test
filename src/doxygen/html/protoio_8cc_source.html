00046 ProtoOutputStream::ProtoOutputStream(const string& filename) :
00047     fileStream(filename.c_str(), ios::out | ios::binary | ios::trunc),
00048     wrappedFileStream(NULL), gzipStream(NULL), zeroCopyStream(NULL)
00049 {
00050     if (!fileStream.good())
00051         panic("Could not open %s for writing\n", filename);
00052 
00053     // Wrap the output file in a zero copy stream, that in turn is
00054     // wrapped in a gzip stream if the filename ends with .gz. The
00055     // latter stream is in turn wrapped in a coded stream
00056     wrappedFileStream = new io::OstreamOutputStream(&fileStream);
00057     if (filename.find_last_of('.') != string::npos &&
00058         filename.substr(filename.find_last_of('.') + 1) == "gz") {
00059         gzipStream = new io::GzipOutputStream(wrappedFileStream);
00060         zeroCopyStream = gzipStream;
00061     } else {
00062         zeroCopyStream = wrappedFileStream;
00063     }
00064 
00065     // Write the magic number to the file
00066     io::CodedOutputStream codedStream(zeroCopyStream);
00067     codedStream.WriteLittleEndian32(magicNumber);
00068 
00069     // Note that each type of stream (packet, instruction etc) should
00070     // add its own header and perform the appropriate checks
00071 }

00083 ProtoOutputStream::write(const Message& msg)
00084 {
00085     // Due to the byte limit of the coded stream we create it for
00086     // every single mesage (based on forum discussions around the size
00087     // limitation)
00088     io::CodedOutputStream codedStream(zeroCopyStream);
00089 
00090     // Write the size of the message to the stream
00091     codedStream.WriteVarint32(msg.ByteSize());
00092 
00093     // Write the message itself to the stream
00094     msg.SerializeWithCachedSizes(&codedStream);

00097 ProtoInputStream::ProtoInputStream(const string& filename) :
00098     fileStream(filename.c_str(), ios::in | ios::binary), fileName(filename),
00099     useGzip(false),
00100     wrappedFileStream(NULL), gzipStream(NULL), zeroCopyStream(NULL)
00101 {
00102     if (!fileStream.good())
00103         panic("Could not open %s for reading\n", filename);
00104 
00105     // check the magic number to see if this is a gzip stream
00106     unsigned char bytes[2];
00107     fileStream.read((char*) bytes, 2);
00108     useGzip = fileStream.good() && bytes[0] == 0x1f && bytes[1] == 0x8b;
00109 
00110     // seek to the start of the input file and clear any flags
00111     fileStream.clear();
00112     fileStream.seekg(0, ifstream::beg);
00113 
00114     createStreams();

00117 void
00118 ProtoInputStream::createStreams()
00119 {
00120     // All streams should be NULL at this point
00121     assert(wrappedFileStream == NULL && gzipStream == NULL &&
00122            zeroCopyStream == NULL);
00123 
00124     // Wrap the input file in a zero copy stream, that in turn is
00125     // wrapped in a gzip stream if the filename ends with .gz. The
00126     // latter stream is in turn wrapped in a coded stream
00127     wrappedFileStream = new io::IstreamInputStream(&fileStream);
00128     if (useGzip) {
00129         gzipStream = new io::GzipInputStream(wrappedFileStream);
00130         zeroCopyStream = gzipStream;
00131     } else {
00132         zeroCopyStream = wrappedFileStream;
00133     }
00134 
00135     uint32_t magic_check;
00136     io::CodedInputStream codedStream(zeroCopyStream);
00137     if (!codedStream.ReadLittleEndian32(&magic_check) ||
00138         magic_check != magicNumber)
00139         panic("Input file %s is not a valid gem5 proto format.\n",
00140               fileName);

00175 bool
00176 ProtoInputStream::read(Message& msg)
00177 {
00178     // Read a message from the stream by getting the size, using it as
00179     // a limit when parsing the message, then popping the limit again
00180     uint32_t size;
00181 
00182     // Due to the byte limit of the coded stream we create it for
00183     // every single mesage (based on forum discussions around the size
00184     // limitation)
00185     io::CodedInputStream codedStream(zeroCopyStream);
00186     if (codedStream.ReadVarint32(&size)) {
00187         io::CodedInputStream::Limit limit = codedStream.PushLimit(size);
00188         if (msg.ParseFromCodedStream(&codedStream)) {
00189             codedStream.PopLimit(limit);
00190             // All went well, the message is parsed and the limit is
00191             // popped again
00192             return true;
00193         } else {
00194             panic("Unable to read message from coded stream %s\n",
00195                   fileName);
00196         }
00197     }
00198 
00199     return false;
00200 }

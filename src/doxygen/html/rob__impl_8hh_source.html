00057 ROB<Impl>::ROB(O3CPU *_cpu, DerivO3CPUParams *params)
00058     : cpu(_cpu),
00059       numEntries(params->numROBEntries),
00060       squashWidth(params->squashWidth),
00061       numInstsInROB(0),
00062       numThreads(params->numThreads)
00063 {
00064     std::string policy = params->smtROBPolicy;
00065 
00066     //Convert string to lowercase
00067     std::transform(policy.begin(), policy.end(), policy.begin(),
00068                    (int(*)(int)) tolower);
00069 
00070     //Figure out rob policy
00071     if (policy == "dynamic") {
00072         robPolicy = Dynamic;
00073 
00074         //Set Max Entries to Total ROB Capacity
00075         for (ThreadID tid = 0; tid < numThreads; tid++) {
00076             maxEntries[tid] = numEntries;
00077         }
00078 
00079     } else if (policy == "partitioned") {
00080         robPolicy = Partitioned;
00081         DPRINTF(Fetch, "ROB sharing policy set to Partitioned\n");
00082 
00083         //@todo:make work if part_amt doesnt divide evenly.
00084         int part_amt = numEntries / numThreads;
00085 
00086         //Divide ROB up evenly
00087         for (ThreadID tid = 0; tid < numThreads; tid++) {
00088             maxEntries[tid] = part_amt;
00089         }
00090 
00091     } else if (policy == "threshold") {
00092         robPolicy = Threshold;
00093         DPRINTF(Fetch, "ROB sharing policy set to Threshold\n");
00094 
00095         int threshold =  params->smtROBThreshold;;
00096 
00097         //Divide up by threshold amount
00098         for (ThreadID tid = 0; tid < numThreads; tid++) {
00099             maxEntries[tid] = threshold;
00100         }
00101     } else {
00102         assert(0 && "Invalid ROB Sharing Policy.Options Are:{Dynamic,"
00103                     "Partitioned, Threshold}");
00104     }
00105 
00106     resetState();
00107 }

00329 void
00330 ROB<Impl>::doSquash(ThreadID tid)
00331 {
00332     robWrites++;
00333     DPRINTF(ROB, "[tid:%u]: Squashing instructions until [sn:%i].\n",
00334             tid, squashedSeqNum[tid]);
00335 
00336     assert(squashIt[tid] != instList[tid].end());
00337 
00338     if ((*squashIt[tid])->seqNum < squashedSeqNum[tid]) {
00339         DPRINTF(ROB, "[tid:%u]: Done squashing instructions.\n",
00340                 tid);
00341 
00342         squashIt[tid] = instList[tid].end();
00343 
00344         doneSquashing[tid] = true;
00345         return;
00346     }
00347 
00348     bool robTailUpdate = false;
00349 
00350     for (int numSquashed = 0;
00351          numSquashed < squashWidth &&
00352          squashIt[tid] != instList[tid].end() &&
00353          (*squashIt[tid])->seqNum > squashedSeqNum[tid];
00354          ++numSquashed)
00355     {
00356         DPRINTF(ROB, "[tid:%u]: Squashing instruction PC %s, seq num %i.\n",
00357                 (*squashIt[tid])->threadNumber,
00358                 (*squashIt[tid])->pcState(),
00359                 (*squashIt[tid])->seqNum);
00360 
00361         // Mark the instruction as squashed, and ready to commit so that
00362         // it can drain out of the pipeline.
00363         (*squashIt[tid])->setSquashed();
00364 
00365         (*squashIt[tid])->setCanCommit();
00366 
00367 
00368         if (squashIt[tid] == instList[tid].begin()) {
00369             DPRINTF(ROB, "Reached head of instruction list while "
00370                     "squashing.\n");
00371 
00372             squashIt[tid] = instList[tid].end();
00373 
00374             doneSquashing[tid] = true;
00375 
00376             return;
00377         }
00378 
00379         InstIt tail_thread = instList[tid].end();
00380         tail_thread--;
00381 
00382         if ((*squashIt[tid]) == (*tail_thread))
00383             robTailUpdate = true;
00384 
00385         squashIt[tid]--;
00386     }
00387 
00388 
00389     // Check if ROB is done squashing.
00390     if ((*squashIt[tid])->seqNum <= squashedSeqNum[tid]) {
00391         DPRINTF(ROB, "[tid:%u]: Done squashing instructions.\n",
00392                 tid);
00393 
00394         squashIt[tid] = instList[tid].end();
00395 
00396         doneSquashing[tid] = true;
00397     }
00398 
00399     if (robTailUpdate) {
00400         updateTail();
00401     }
00402 }

00411 
00412     // @todo: set ActiveThreads through ROB or CPU

00414     list<ThreadID>::iterator end = activeThreads->end();
00415

00420             continue;
00421 
00422         if (first_valid) {
00423             head = instList[tid].begin();
00424             lowest_num = (*head)->seqNum;
00425             first_valid = false;
00426             continue;

00428 
00429         InstIt head_thread = instList[tid].begin();
00430 
00431         DynInstPtr head_inst = (*head_thread);
00432 
00433         assert(head_inst != 0);
00434 
00435         if (head_inst->seqNum < lowest_num) {
00436             head = head_thread;
00437             lowest_num = head_inst->seqNum;

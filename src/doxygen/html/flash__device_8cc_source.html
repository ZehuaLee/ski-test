00078     pageSize(p->page_size),
00079     GCActivePercentage(p->GC_active),

00083     dataDistribution(p->data_distribution),
00084     numPlanes(p->num_planes),

00087     blocksPerDisk(0),
00088     planeMask(numPlanes - 1),
00089     drainManager(NULL),
00090     planeEventQueue(numPlanes),
00091     planeEvent(this)
00092 {
00093 
00094     /*
00095      * Let 'a' be a power of two of n bits, written such that a-n is the msb
00096      * and a-0 is the lsb. Since it is a power of two, only one bit (a-x,
00097      * with 0 <= x <= n) is set. If we subtract one from this number the bits
00098      * a-(x-1) to a-0 are set and all the other bits are cleared. Hence a
00099      * bitwise AND with those two numbers results in an integer with all bits
00100      * cleared.
00101      */
00102     if(numPlanes & planeMask)
00103         fatal("Number of planes is not a power of 2 in flash device.\n");
00104 }

00353 FlashDevice::remap(uint64_t logic_page_addr)
00354 {
00358     if (blockEmptyEntries[locationTable[logic_page_addr].block] > 0) {
00359         //just a remap
00360         //update tables
00361         --blockEmptyEntries[locationTable[logic_page_addr].block];
00362         //access to this table won't be sequential anymore
00363         locationTable[logic_page_addr].page = pagesPerBlock + 2;
00364         //access new block
00365         Tick time = accessTimes(locationTable[logic_page_addr].block,
00366                                 ActionWrite);
00367 
00368         DPRINTF(FlashDevice, "Remap returns %d ticks\n", time);
00369         return time;
00370 
00371     } else {
00372         //calculate how much time GC would have taken
00373         uint32_t block = locationTable[logic_page_addr].block;
00374         Tick time = ((GCActivePercentage *
00375                        (accessTimes(block, ActionCopy) +
00376                         accessTimes(block, ActionErase)))
00377                      / 100);
00378 
00379         //use block as the logical start address of the block
00380         block = locationTable[logic_page_addr].block * pagesPerBlock;
00381 
00382         //assumption: clean will improve locality
00383         for (uint32_t count = 0; count < pageSize; count++) {
00384             locationTable[block + count].page = (block + count) %
00385                 pagesPerBlock;
00386             ++count;
00387         }
00388 
00389         blockEmptyEntries[locationTable[logic_page_addr].block] =
00390             pagesPerBlock;
00391         /*stats*/
00392         ++stats.totalGCActivations;

00394         DPRINTF(FlashDevice, "Remap with erase action returns %d ticks\n",

00405 Tick
00406 FlashDevice::accessTimes(uint64_t block, Actions action)
00407 {
00408     Tick time = 0;
00409 
00410     switch(action) {
00411       case ActionRead: {
00413           time = readLatency;
00414       } break;
00415 
00416       case ActionWrite: {
00418           time = writeLatency + readLatency;
00419       } break;
00420 
00421       case ActionErase: {
00423           time = eraseLatency + readLatency;
00424       } break;
00425 
00426       case ActionCopy: {
00428           uint32_t validpages = blockValidEntries[block];
00429           time = validpages * (readLatency + writeLatency);
00430       } break;
00431 
00432       default: break;
00433     }
00434 
00435     //Used to determine sequential action.
00436     DPRINTF(FlashDevice, "Access returns %d ticks\n", time);

00075 void
00076 VirtIOConsole::TermRecvQueue::trySend()
00077 {
00078     DPRINTF(VIOConsole, "trySend\n");
00079 
00080     // Send data as long as the terminal has outgoing data and we can
00081     // get free descriptors (i.e., there are buffers available to
00082     // send) from the guest.
00083     VirtDescriptor *d;
00084     while (parent.term.dataAvailable() && (d = consumeDescriptor())) {
00085         DPRINTF(VIOConsole, "Got descriptor (len: %i)\n", d->size());
00086         size_t len(0);
00087         while (parent.term.dataAvailable() && len < d->size()) {
00088             uint8_t in(parent.term.in());
00089             d->chainWrite(len, &in, sizeof(uint8_t));
00090             ++len;
00091         }
00092 
00093         // Tell the guest that we are done with this descriptor.
00094         produceDescriptor(d, len);
00095         parent.kick();
00096     }
00097 }

00099 void
00100 VirtIOConsole::TermTransQueue::onNotifyDescriptor(VirtDescriptor *desc)
00101 {
00102     DPRINTF(VIOConsole, "Got input data descriptor (len: %i)\n",
00103             desc->size());
00104 
00105     // Copy the data from the guest and forward it to the
00106     // terminal.
00107     const size_t size(desc->chainSize());
00108     uint8_t data[size];
00109     desc->chainRead(0, data, size);
00110     for (int i = 0; i < desc->size(); ++i)
00111         parent.term.out(data[i]);
00112 
00113     // Tell the guest that we are done with this descriptor.
00114     produceDescriptor(desc, 0);
00115     parent.kick();
00116 }

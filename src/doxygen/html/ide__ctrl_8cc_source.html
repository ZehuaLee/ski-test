00392 void
00393 IdeController::Channel::accessBMI(Addr offset,
00394         int size, uint8_t *data, bool read)
00395 {
00396     assert(offset + size <= sizeof(BMIRegs));
00397     if (read) {
00398         memcpy(data, (uint8_t *)&bmiRegs + offset, size);
00399     } else {
00400         switch (offset) {
00401           case BMICommand:
00402             {
00403                 if (size != sizeof(uint8_t))
00404                     panic("Invalid BMIC write size: %x\n", size);
00405 
00406                 BMICommandReg oldVal = bmiRegs.command;
00407                 BMICommandReg newVal = *data;
00408 
00409                 // if a DMA transfer is in progress, R/W control cannot change
00410                 if (oldVal.startStop && oldVal.rw != newVal.rw)
00411                     oldVal.rw = newVal.rw;
00412 
00413                 if (oldVal.startStop != newVal.startStop) {
00414                     if (selected == NULL)
00415                         panic("DMA start for disk which does not exist\n");
00416 
00417                     if (oldVal.startStop) {
00418                         DPRINTF(IdeCtrl, "Stopping DMA transfer\n");
00419                         bmiRegs.status.active = 0;
00420 
00421                         selected->abortDma();
00422                     } else {
00423                         DPRINTF(IdeCtrl, "Starting DMA transfer\n");
00424                         bmiRegs.status.active = 1;
00425 
00426                         selected->startDma(letoh(bmiRegs.bmidtp));
00427                     }
00428                 }
00429 
00430                 bmiRegs.command = newVal;
00431             }
00432             break;
00433           case BMIStatus:
00434             {
00435                 if (size != sizeof(uint8_t))
00436                     panic("Invalid BMIS write size: %x\n", size);
00437 
00438                 BMIStatusReg oldVal = bmiRegs.status;
00439                 BMIStatusReg newVal = *data;
00440 
00441                 // the BMIDEA bit is read only
00442                 newVal.active = oldVal.active;
00443 
00444                 // to reset (set 0) IDEINTS and IDEDMAE, write 1 to each
00445                 if ((oldVal.intStatus == 1) && (newVal.intStatus == 1)) {
00446                     newVal.intStatus = 0; // clear the interrupt?
00447                 } else {
00448                     // Assigning two bitunion fields to each other does not
00449                     // work as intended, so we need to use this temporary variable
00450                     // to get around the bug.
00451                     uint8_t tmp = oldVal.intStatus;
00452                     newVal.intStatus = tmp;
00453                 }
00454                 if ((oldVal.dmaError == 1) && (newVal.dmaError == 1)) {
00455                     newVal.dmaError = 0;
00456                 } else {
00457                     uint8_t tmp = oldVal.dmaError;
00458                     newVal.dmaError = tmp;
00459                 }
00460 
00461                 bmiRegs.status = newVal;
00462             }
00463             break;
00464           case BMIDescTablePtr:
00465             if (size != sizeof(uint32_t))
00466                 panic("Invalid BMIDTP write size: %x\n", size);
00467             bmiRegs.bmidtp = htole(*(uint32_t *)data & ~0x3);
00468             break;
00469           default:
00470             if (size != sizeof(uint8_t) && size != sizeof(uint16_t) &&
00471                     size != sizeof(uint32_t))
00472                 panic("IDE controller write of invalid write size: %x\n", size);
00473             memcpy((uint8_t *)&bmiRegs + offset, data, size);
00474         }

00478 void
00479 IdeController::dispatchAccess(PacketPtr pkt, bool read)
00480 {
00481     if (pkt->getSize() != 1 && pkt->getSize() != 2 && pkt->getSize() !=4)
00482          panic("Bad IDE read size: %d\n", pkt->getSize());
00483 
00484     if (!ioEnabled) {
00485         pkt->makeAtomicResponse();
00486         DPRINTF(IdeCtrl, "io not enabled\n");
00487         return;
00488     }
00489 
00490     Addr addr = pkt->getAddr();
00491     int size = pkt->getSize();
00492     uint8_t *dataPtr = pkt->getPtr<uint8_t>();
00493 
00494     if (addr >= primary.cmdAddr &&
00495             addr < (primary.cmdAddr + primary.cmdSize)) {
00496         addr -= primary.cmdAddr;
00497         // linux may have shifted the address by ioShift,
00498         // here we shift it back, similarly for ctrlOffset.
00499         addr >>= ioShift;
00500         primary.accessCommand(addr, size, dataPtr, read);
00501     } else if (addr >= primary.ctrlAddr &&
00502                addr < (primary.ctrlAddr + primary.ctrlSize)) {
00503         addr -= primary.ctrlAddr;
00504         addr += ctrlOffset;
00505         primary.accessControl(addr, size, dataPtr, read);
00506     } else if (addr >= secondary.cmdAddr &&
00507                addr < (secondary.cmdAddr + secondary.cmdSize)) {
00508         addr -= secondary.cmdAddr;
00509         secondary.accessCommand(addr, size, dataPtr, read);
00510     } else if (addr >= secondary.ctrlAddr &&
00511                addr < (secondary.ctrlAddr + secondary.ctrlSize)) {
00512         addr -= secondary.ctrlAddr;
00513         secondary.accessControl(addr, size, dataPtr, read);
00514     } else if (addr >= bmiAddr && addr < (bmiAddr + bmiSize)) {
00515         if (!read && !bmEnabled)
00516             return;
00517         addr -= bmiAddr;
00518         if (addr < sizeof(Channel::BMIRegs)) {
00519             primary.accessBMI(addr, size, dataPtr, read);
00520         } else {
00521             addr -= sizeof(Channel::BMIRegs);
00522             secondary.accessBMI(addr, size, dataPtr, read);
00523         }
00524     } else {
00525         panic("IDE controller access to invalid address: %#x\n", addr);
00526     }
00527 
00528 #ifndef NDEBUG
00529     uint32_t data;
00530     if (pkt->getSize() == 1)
00531         data = pkt->get<uint8_t>();
00532     else if (pkt->getSize() == 2)
00533         data = pkt->get<uint16_t>();
00534     else
00535         data = pkt->get<uint32_t>();
00536     DPRINTF(IdeCtrl, "%s from offset: %#x size: %#x data: %#x\n",
00537             read ? "Read" : "Write", pkt->getAddr(), pkt->getSize(), data);
00538 #endif
00539 
00540     pkt->makeAtomicResponse();
00541 }

00144     Fault fault = NoFault;
00145     int fullSize = size;
00146     Addr secondAddr = roundDown(addr + size - 1, cacheLineSize());
00147     bool checked_flags = false;
00148     bool flags_match = true;
00149     Addr pAddr = 0x0;
00150 
00151 
00152     if (secondAddr > addr)
00153        size = secondAddr - addr;
00154 
00155     // Need to account for multiple accesses like the Atomic and TimingSimple
00156     while (1) {
00157         memReq = new Request(0, addr, size, flags, masterId,
00158                              thread->pcState().instAddr(), tc->contextId(), 0);
00159 
00160         // translate to physical address
00161         fault = dtb->translateFunctional(memReq, tc, BaseTLB::Read);
00162 
00163         if (!checked_flags && fault == NoFault && unverifiedReq) {
00164             flags_match = checkFlags(unverifiedReq, memReq->getVaddr(),
00165                                      memReq->getPaddr(), memReq->getFlags());
00166             pAddr = memReq->getPaddr();
00167             checked_flags = true;
00168         }
00169 
00170         // Now do the access
00171         if (fault == NoFault &&
00172             !memReq->getFlags().isSet(Request::NO_ACCESS)) {
00173             PacketPtr pkt = Packet::createRead(memReq);
00174 
00175             pkt->dataStatic(data);
00176 
00177             if (!(memReq->isUncacheable() || memReq->isMmappedIpr())) {
00178                 // Access memory to see if we have the same data
00179                 dcachePort->sendFunctional(pkt);
00180             } else {
00181                 // Assume the data is correct if it's an uncached access
00182                 memcpy(data, unverifiedMemData, size);
00183             }
00184 
00185             delete memReq;
00186             memReq = NULL;
00187             delete pkt;
00188         }
00189 
00190         if (fault != NoFault) {
00191             if (memReq->isPrefetch()) {
00192                 fault = NoFault;
00193             }
00194             delete memReq;
00195             memReq = NULL;
00196             break;
00197         }
00198 
00199         if (memReq != NULL) {
00200             delete memReq;
00201         }
00202 
00203         //If we don't need to access a second cache line, stop now.
00204         if (secondAddr <= addr)
00205         {
00206             break;
00207         }
00208 
00209         // Setup for accessing next cache line
00210         data += size;
00211         unverifiedMemData += size;
00212         size = addr + fullSize - secondAddr;
00213         addr = secondAddr;
00214     }
00215 
00216     if (!flags_match) {
00217         warn("%lli: Flags do not match CPU:%#x %#x %#x Checker:%#x %#x %#x\n",
00218              curTick(), unverifiedReq->getVaddr(), unverifiedReq->getPaddr(),

00227 CheckerCPU::writeMem(uint8_t *data, unsigned size,
00228                      Addr addr, unsigned flags, uint64_t *res)

00230     Fault fault = NoFault;

00232     bool flags_match = true;
00233     Addr pAddr = 0x0;
00234     static uint8_t zero_data[64] = {};
00235

00237 
00238     Addr secondAddr = roundDown(addr + size - 1, cacheLineSize());
00239 
00240     if (secondAddr > addr)
00241         size = secondAddr - addr;
00242 
00243     // Need to account for a multiple access like Atomic and Timing CPUs
00244     while (1) {
00245         memReq = new Request(0, addr, size, flags, masterId,
00246                              thread->pcState().instAddr(), tc->contextId(), 0);
00247 
00248         // translate to physical address
00249         fault = dtb->translateFunctional(memReq, tc, BaseTLB::Write);
00250 
00251         if (!checked_flags && fault == NoFault && unverifiedReq) {
00252            flags_match = checkFlags(unverifiedReq, memReq->getVaddr(),
00253                                     memReq->getPaddr(), memReq->getFlags());
00254            pAddr = memReq->getPaddr();
00255            checked_flags = true;
00256         }
00257 
00258         /*
00259          * We don't actually check memory for the store because there
00260          * is no guarantee it has left the lsq yet, and therefore we
00261          * can't verify the memory on stores without lsq snooping
00262          * enabled.  This is left as future work for the Checker: LSQ snooping
00263          * and memory validation after stores have committed.
00264          */
00265         bool was_prefetch = memReq->isPrefetch();
00266 
00267         delete memReq;
00268 
00269         //If we don't need to access a second cache line, stop now.
00270         if (fault != NoFault || secondAddr <= addr)
00271         {
00272             if (fault != NoFault && was_prefetch) {
00273               fault = NoFault;
00274             }
00275             break;
00276         }
00277 
00278         //Update size and access address
00279         size = addr + fullSize - secondAddr;
00280         //And access the right address.
00281         addr = secondAddr;
00282    }
00283 
00284    if (!flags_match) {
00285        warn("%lli: Flags do not match CPU:%#x %#x Checker:%#x %#x %#x\n",
00286             curTick(), unverifiedReq->getVaddr(), unverifiedReq->getPaddr(),
00287             unverifiedReq->getFlags(), addr, pAddr, flags);
00288        handleError();
00289    }
00290 
00291    // Assume the result was the same as the one passed in.  This checker
00292    // doesn't check if the SC should succeed or fail, it just checks the
00293    // value.
00294    if (unverifiedReq && res && unverifiedReq->extraDataValid())
00295        *res = unverifiedReq->getExtraData();
00296 
00297    // Entire purpose here is to make sure we are getting the
00298    // same data to send to the mem system as the CPU did.
00299    // Cannot check this is actually what went to memory because
00300    // there stores can be in ld/st queue or coherent operations
00301    // overwriting values.
00302    bool extraData = false;
00303    if (unverifiedReq) {
00304        extraData = unverifiedReq->extraDataValid() ?
00305                         unverifiedReq->getExtraData() : true;
00306    }
00307 
00308    // If the request is to ZERO a cache block, there is no data to check
00309    // against, but it's all zero. We need something to compare to, so use a
00310    // const set of zeros.
00311    if (flags & Request::CACHE_BLOCK_ZERO) {
00312        assert(!data);
00313        assert(sizeof(zero_data) <= fullSize);
00314        data = zero_data;
00315    }
00316 
00317    if (unverifiedReq && unverifiedMemData &&
00318        memcmp(data, unverifiedMemData, fullSize) && extraData) {
00319            warn("%lli: Store value does not match value sent to memory! "
00320                   "data: %#x inst_data: %#x", curTick(), data,
00321                   unverifiedMemData);
00322        handleError();
00323    }
00324 
00325    return fault;
00326 }

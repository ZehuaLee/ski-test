00067 bool
00068 RemoteGDB::acc(Addr va, size_t len)
00069 {
00070     if (FullSystem) {
00071         Walker *walker = context->getDTBPtr()->getWalker();
00072         unsigned logBytes;
00073         Fault fault = walker->startFunctional(context, va, logBytes,
00074                                               BaseTLB::Read);
00075         if (fault != NoFault)
00076             return false;
00077 
00078         Addr endVa = va + len - 1;
00079         if ((va & ~mask(logBytes)) == (endVa & ~mask(logBytes)))
00080             return true;
00081 
00082         fault = walker->startFunctional(context, endVa, logBytes,
00083                                         BaseTLB::Read);
00084         return fault == NoFault;
00085     } else {
00086         TlbEntry entry;
00087         return context->getProcessPtr()->pTable->lookup(va, entry);
00088     }

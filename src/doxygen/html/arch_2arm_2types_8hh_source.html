00368         void
00369         advance()
00370         {
00371             Base::advance();
00372             flags = nextFlags;
00373             npc(pc() + (thumb() ? 2 : 4));
00374 
00375             if (_nextItstate) {
00376                 _itstate = _nextItstate;
00377                 _nextItstate = 0;
00378             } else if (_itstate) {
00379                 ITSTATE it = _itstate;
00380                 uint8_t cond_mask = it.mask;
00381                 uint8_t thumb_cond = it.cond;
00382                 DPRINTF(Decoder, "Advancing ITSTATE from %#x,%#x.\n",
00383                         thumb_cond, cond_mask);
00384                 cond_mask <<= 1;
00385                 uint8_t new_bit = bits(cond_mask, 4);
00386                 cond_mask &= mask(4);
00387                 if (cond_mask == 0)
00388                     thumb_cond = 0;
00389                 else
00390                     replaceBits(thumb_cond, 0, new_bit);
00391                 DPRINTF(Decoder, "Advancing ITSTATE to %#x,%#x.\n",
00392                         thumb_cond, cond_mask);
00393                 it.mask = cond_mask;
00394                 it.cond = thumb_cond;
00395                 _itstate = it;
00396             }
00397         }

00431         // Perform an interworking branch.
00432         void
00433         instIWNPC(Addr val)
00434         {
00435             bool thumbEE = (thumb() && jazelle());
00436 
00437             Addr newPC = val;
00438             if (thumbEE) {
00439                 if (bits(newPC, 0)) {
00440                     newPC = newPC & ~mask(1);
00441                 }  // else we have a bad interworking address; do not call
00442                    // panic() since the instruction could be executed
00443                    // speculatively
00444             } else {
00445                 if (bits(newPC, 0)) {
00446                     nextThumb(true);
00447                     newPC = newPC & ~mask(1);
00448                 } else if (!bits(newPC, 1)) {
00449                     nextThumb(false);
00450                 } else {
00451                     // This state is UNPREDICTABLE in the ARM architecture
00452                     // The easy thing to do is just mask off the bit and
00453                     // stay in the current mode, so we'll do that.
00454                     newPC &= ~mask(2);
00455                 }
00456             }
00457             npc(newPC);
00458         }

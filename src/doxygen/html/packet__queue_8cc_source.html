00102 void
00103 PacketQueue::schedSendTiming(PacketPtr pkt, Tick when, bool force_order)
00104 {
00105     DPRINTF(PacketQueue, "%s for %s address %x size %d when %lu ord: %i\n",
00106             __func__, pkt->cmdString(), pkt->getAddr(), pkt->getSize(), when,
00107             force_order);
00108 
00109     // we can still send a packet before the end of this tick
00110     assert(when >= curTick());
00111 
00112     // express snoops should never be queued
00113     assert(!pkt->isExpressSnoop());
00114 
00115     // add a very basic sanity check on the port to ensure the
00116     // invisible buffer is not growing beyond reasonable limits
00117     if (transmitList.size() > 100) {
00118         panic("Packet queue %s has grown beyond 100 packets\n",
00119               name());
00120     }
00121 
00122     // if requested, force the timing to be in-order by changing the when
00123     // parameter
00124     if (force_order && !transmitList.empty()) {
00125         Tick back = transmitList.back().tick;
00126 
00127         // fudge timing if required; relies on the code below to do the right
00128         // thing (push_back) with the updated time-stamp
00129         if (when < back) {
00130             DPRINTF(PacketQueue, "%s force_order shifted packet %s address "\
00131                     "%x from %lu to %lu\n", __func__, pkt->cmdString(),
00132                     pkt->getAddr(), when, back);
00133             when = back;
00134         }
00135     }
00136 
00137     // nothing on the list, or earlier than current front element,
00138     // schedule an event
00139     if (transmitList.empty() || when < transmitList.front().tick) {
00140         // force_order-ed in here only when list is empty
00141         assert(!force_order || transmitList.empty());
00142         // note that currently we ignore a potentially outstanding retry
00143         // and could in theory put a new packet at the head of the
00144         // transmit list before retrying the existing packet
00145         transmitList.emplace_front(DeferredPacket(when, pkt));
00146         schedSendEvent(when);
00147         return;
00148     }
00149 
00150     // we should either have an outstanding retry, or a send event
00151     // scheduled, but there is an unfortunate corner case where the
00152     // x86 page-table walker and timing CPU send out a new request as
00153     // part of the receiving of a response (called by
00154     // PacketQueue::sendDeferredPacket), in which we end up calling
00155     // ourselves again before we had a chance to update waitingOnRetry
00156     // assert(waitingOnRetry || sendEvent.scheduled());
00157 
00158     // list is non-empty and this belongs at the end
00159     if (when >= transmitList.back().tick) {
00160         transmitList.emplace_back(DeferredPacket(when, pkt));
00161         return;
00162     }
00163 
00164     // forced orders never need insertion in the middle
00165     assert(!force_order);
00166 
00167     // this belongs in the middle somewhere, insertion sort
00168     auto i = transmitList.begin();
00169     ++i; // already checked for insertion at front
00170     while (i != transmitList.end() && when >= i->tick)
00171         ++i;
00172     transmitList.emplace(i, DeferredPacket(when, pkt));
00173 }

00175 void
00176 PacketQueue::schedSendEvent(Tick when)
00177 {
00178     // if we are waiting on a retry just hold off
00179     if (waitingOnRetry) {
00180         DPRINTF(PacketQueue, "Not scheduling send as waiting for retry\n");
00181         assert(!sendEvent.scheduled());
00182         return;
00183     }
00184 
00185     if (when != MaxTick) {
00186         // we cannot go back in time, and to be consistent we stick to
00187         // one tick in the future
00188         when = std::max(when, curTick() + 1);
00189         // @todo Revisit the +1
00190 
00191         if (!sendEvent.scheduled()) {
00192             em.schedule(&sendEvent, when);
00193         } else if (when < sendEvent.when()) {
00194             // if the new time is earlier than when the event
00195             // currently is scheduled, move it forward
00196             em.reschedule(&sendEvent, when);
00197         }
00198     } else {
00199         // we get a MaxTick when there is no more to send, so if we're
00200         // draining, we may be done at this point
00201         if (drainManager && transmitList.empty() && !sendEvent.scheduled()) {
00202             DPRINTF(Drain, "PacketQueue done draining,"
00203                     "processing drain event\n");
00204             drainManager->signalDrainDone();
00205             drainManager = NULL;
00206         }
00207     }
00208 }

00210 void
00211 PacketQueue::sendDeferredPacket()
00212 {
00213     // sanity checks
00214     assert(!waitingOnRetry);
00215     assert(deferredPacketReady());
00216 
00217     DeferredPacket dp = transmitList.front();
00218 
00219     // take the packet of the list before sending it, as sending of
00220     // the packet in some cases causes a new packet to be enqueued
00221     // (most notaly when responding to the timing CPU, leading to a
00222     // new request hitting in the L1 icache, leading to a new
00223     // response)
00224     transmitList.pop_front();
00225 
00226     // use the appropriate implementation of sendTiming based on the
00227     // type of queue
00228     waitingOnRetry = !sendTiming(dp.pkt);
00229 
00230     // if we succeeded and are not waiting for a retry, schedule the
00231     // next send
00232     if (!waitingOnRetry) {
00233         schedSendEvent(deferredPacketReadyTime());
00234     } else {
00235         // put the packet back at the front of the list
00236         transmitList.emplace_front(dp);
00237     }
00238 }

00183 void
00184 FUPipeline::advance()
00185 {
00186     bool was_stalled = stalled;
00187 
00188     /* If an instruction was pushed into the pipeline, set the delay before
00189      *  the next instruction can follow */
00190     if (alreadyPushed()) {
00191         if (nextInsertCycle <= timeSource.curCycle()) {
00192             nextInsertCycle = timeSource.curCycle() + description.issueLat;
00193         }
00194     } else if (was_stalled && nextInsertCycle != 0) {
00195         /* Don't count stalled cycles as part of the issue latency */
00196         ++nextInsertCycle;
00197     }
00198     FUPipelineBase::advance();

00052 bool
00053 InvalidateGenerator::initiate()
00054 {
00055     MasterPort* port;
00056     Request::Flags flags;
00057     PacketPtr pkt;
00058     Packet::Command cmd;
00059 
00060     // For simplicity, requests are assumed to be 1 byte-sized
00061     Request *req = new Request(m_address, 1, flags, masterId);
00062 
00063     //
00064     // Based on the current state, issue a load or a store
00065     //
00066     if (m_status == InvalidateGeneratorStatus_Load_Waiting) {
00067         DPRINTF(DirectedTest, "initiating read\n");
00068         cmd = MemCmd::ReadReq;
00069         port = m_directed_tester->getCpuPort(m_active_read_node);
00070         pkt = new Packet(req, cmd);
00071     } else if (m_status == InvalidateGeneratorStatus_Inv_Waiting) {
00072         DPRINTF(DirectedTest, "initiating invalidating write\n");
00073         cmd = MemCmd::WriteReq;
00074         port = m_directed_tester->getCpuPort(m_active_inv_node);
00075         pkt = new Packet(req, cmd);
00076     } else {
00077         panic("initiate was unexpectedly called\n");
00078     }
00079     pkt->allocate();
00080 
00081     if (port->sendTimingReq(pkt)) {
00082         DPRINTF(DirectedTest, "initiating request - successful\n");
00083         if (m_status == InvalidateGeneratorStatus_Load_Waiting) {
00084             m_status = InvalidateGeneratorStatus_Load_Pending;
00085         } else {
00086             m_status = InvalidateGeneratorStatus_Inv_Pending;
00087         }
00088         return true;
00089     } else {
00090         // If the packet did not issue, must delete
00091         // Note: No need to delete the data, the packet destructor
00092         // will delete it
00093         delete pkt->req;
00094         delete pkt;
00095 
00096         DPRINTF(DirectedTest, "failed to issue request - sequencer not ready\n");
00097         return false;
00098     }
00099 }

00101 void 
00102 InvalidateGenerator::performCallback(uint32_t proc, Addr address)
00103 {
00104     assert(m_address == address);  
00105 
00106     if (m_status == InvalidateGeneratorStatus_Load_Pending) {
00107         assert(m_active_read_node == proc);
00108         m_active_read_node++;
00109         //
00110         // Once all cpus have the block in S state, issue the invalidate
00111         //
00112         if (m_active_read_node == m_num_cpus) {
00113             m_status = InvalidateGeneratorStatus_Inv_Waiting;
00114             m_active_read_node = 0;
00115         } else {
00116             m_status = InvalidateGeneratorStatus_Load_Waiting;
00117         }
00118     } else if (m_status == InvalidateGeneratorStatus_Inv_Pending) {
00119         assert(m_active_inv_node == proc);
00120         m_active_inv_node++;
00121         if (m_active_inv_node == m_num_cpus) {
00122             m_address += m_addr_increment_size;
00123             m_active_inv_node = 0;
00124         }
00125         //
00126         // Invalidate completed, send that info to the tester and restart
00127         // the cycle
00128         //
00129         m_directed_tester->incrementCycleCompletions();
00130         m_status = InvalidateGeneratorStatus_Load_Waiting;
00131     } 
00132     
00133 }

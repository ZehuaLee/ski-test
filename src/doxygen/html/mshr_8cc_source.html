00079 inline void
00080 MSHR::TargetList::add(PacketPtr pkt, Tick readyTime,
00081                       Counter order, Target::Source source, bool markPending)
00082 {
00083     if (source != Target::FromSnoop) {
00084         if (pkt->needsExclusive()) {
00085             needsExclusive = true;
00086         }
00087 
00088         // StoreCondReq is effectively an upgrade if it's in an MSHR
00089         // since it would have been failed already if we didn't have a
00090         // read-only copy
00091         if (pkt->isUpgrade() || pkt->cmd == MemCmd::StoreCondReq) {
00092             hasUpgrade = true;
00093         }
00094     }
00095 
00096     if (markPending) {
00097         // Iterate over the SenderState stack and see if we find
00098         // an MSHR entry. If we do, set the downstreamPending
00099         // flag. Otherwise, do nothing.
00100         MSHR *mshr = pkt->findNextSenderState<MSHR>();
00101         if (mshr != NULL) {
00102             assert(!mshr->downstreamPending);
00103             mshr->downstreamPending = true;
00104         }
00105     }
00106 
00107     emplace_back(Target(pkt, readyTime, order, source, markPending));
00108 }

00237 MSHR::markInService(bool pending_dirty_resp)
00238 {
00239     assert(!inService);
00240     if (isForwardNoResponse()) {
00241         // we just forwarded the request packet & don't expect a
00242         // response, so get rid of it
00243         assert(getNumTargets() == 1);
00244         popTarget();
00245         return true;

00250     postInvalidate = postDowngrade = false;

00271  * Adds a target to an MSHR
00272  */
00273 void
00274 MSHR::allocateTarget(PacketPtr pkt, Tick whenReady, Counter _order)
00275 {
00276     // assume we'd never issue a prefetch when we've got an
00277     // outstanding miss
00278     assert(pkt->cmd != MemCmd::HardPFReq);
00279 
00280     // uncacheable accesses always allocate a new MSHR, and cacheable
00281     // accesses ignore any uncacheable MSHRs, thus we should never
00282     // have targets addded if originally allocated uncacheable
00283     assert(!_isUncacheable);
00284 
00285     // if there's a request already in service for this MSHR, we will
00286     // have to defer the new target until after the response if any of
00287     // the following are true:
00288     // - there are other targets already deferred
00289     // - there's a pending invalidate to be applied after the response
00290     //   comes back (but before this target is processed)
00291     // - this target requires an exclusive block and either we're not
00292     //   getting an exclusive block back or we have already snooped
00293     //   another read request that will downgrade our exclusive block
00294     //   to shared
00295     if (inService &&
00296         (!deferredTargets.empty() || hasPostInvalidate() ||
00297          (pkt->needsExclusive() &&
00298           (!isPendingDirty() || hasPostDowngrade() || isForward)))) {
00299         // need to put on deferred list
00300         if (hasPostInvalidate())
00301             replaceUpgrade(pkt);
00302         deferredTargets.add(pkt, whenReady, _order, Target::FromCPU, true);
00303     } else {
00304         // No request outstanding, or still OK to append to
00305         // outstanding request: append to regular target list.  Only
00306         // mark pending if current request hasn't been issued yet
00307         // (isn't in service).
00308         targets.add(pkt, whenReady, _order, Target::FromCPU, !inService);
00309     }
00310 }

00312 bool
00313 MSHR::handleSnoop(PacketPtr pkt, Counter _order)
00314 {
00315     DPRINTF(Cache, "%s for %s addr %#llx size %d\n", __func__,
00316             pkt->cmdString(), pkt->getAddr(), pkt->getSize());
00317     if (!inService || (pkt->isExpressSnoop() && downstreamPending)) {
00318         // Request has not been issued yet, or it's been issued
00319         // locally but is buffered unissued at some downstream cache
00320         // which is forwarding us this snoop.  Either way, the packet
00321         // we're snooping logically precedes this MSHR's request, so
00322         // the snoop has no impact on the MSHR, but must be processed
00323         // in the standard way by the cache.  The only exception is
00324         // that if we're an L2+ cache buffering an UpgradeReq from a
00325         // higher-level cache, and the snoop is invalidating, then our
00326         // buffered upgrades must be converted to read exclusives,
00327         // since the upper-level cache no longer has a valid copy.
00328         // That is, even though the upper-level cache got out on its
00329         // local bus first, some other invalidating transaction
00330         // reached the global bus before the upgrade did.
00331         if (pkt->needsExclusive()) {
00332             targets.replaceUpgrades();
00333             deferredTargets.replaceUpgrades();
00334         }
00335 
00336         return false;
00337     }
00338 
00339     // From here on down, the request issued by this MSHR logically
00340     // precedes the request we're snooping.
00341     if (pkt->needsExclusive()) {
00342         // snooped request still precedes the re-request we'll have to
00343         // issue for deferred targets, if any...
00344         deferredTargets.replaceUpgrades();
00345     }
00346 
00347     if (hasPostInvalidate()) {
00348         // a prior snoop has already appended an invalidation, so
00349         // logically we don't have the block anymore; no need for
00350         // further snooping.
00351         return true;
00352     }
00353 
00354     if (isPendingDirty() || pkt->isInvalidate()) {
00355         // We need to save and replay the packet in two cases:
00356         // 1. We're awaiting an exclusive copy, so ownership is pending,
00357         //    and we need to respond after we receive data.
00358         // 2. It's an invalidation (e.g., UpgradeReq), and we need
00359         //    to forward the snoop up the hierarchy after the current
00360         //    transaction completes.
00361         
00362         // Actual target device (typ. a memory) will delete the
00363         // packet on reception, so we need to save a copy here.
00364 
00365         // Clear flags and also allocate new data as the original
00366         // packet data storage may have been deleted by the time we
00367         // get to send this packet.
00368         PacketPtr cp_pkt = nullptr;
00369 
00370         if (isPendingDirty()) {
00371             // Case 1: The new packet will need to get the response from the
00372             // MSHR already queued up here
00373             cp_pkt = new Packet(pkt, true, true);
00374             pkt->assertMemInhibit();
00375             // in the case of an uncacheable request there is no need
00376             // to set the exclusive flag, but since the recipient does
00377             // not care there is no harm in doing so
00378             pkt->setSupplyExclusive();
00379         } else {
00380             // Case 2: We only need to buffer the packet for information
00381             // purposes; the original request can proceed without waiting
00382             // => Create a copy of the request, as that may get deallocated as
00383             // well
00384             cp_pkt = new Packet(new Request(*pkt->req), pkt->cmd);
00385             DPRINTF(Cache, "Copying packet %p -> %p and request %p -> %p\n",
00386                     pkt, cp_pkt, pkt->req, cp_pkt->req);
00387         }
00388         targets.add(cp_pkt, curTick(), _order, Target::FromSnoop,
00389                     downstreamPending && targets.needsExclusive);
00390 
00391         if (pkt->needsExclusive()) {
00392             // This transaction will take away our pending copy
00393             postInvalidate = true;
00394         }
00395     }
00396 
00397     if (!pkt->needsExclusive() && !pkt->req->isUncacheable()) {
00398         // This transaction will get a read-shared copy, downgrading
00399         // our copy if we had an exclusive one
00400         postDowngrade = true;
00401         pkt->assertShared();
00402     }
00403 
00404     return true;
00405 }

00429 void
00430 MSHR::handleFill(PacketPtr pkt, CacheBlk *blk)
00431 {
00432     if (!pkt->sharedAsserted()
00433         && !(hasPostInvalidate() || hasPostDowngrade())
00434         && deferredTargets.needsExclusive) {
00435         // We got an exclusive response, but we have deferred targets
00436         // which are waiting to request an exclusive copy (not because
00437         // of a pending invalidate).  This can happen if the original
00438         // request was for a read-only (non-exclusive) block, but we
00439         // got an exclusive copy anyway because of the E part of the
00440         // MOESI/MESI protocol.  Since we got the exclusive copy
00441         // there's no need to defer the targets, so move them up to
00442         // the regular target list.
00443         assert(!targets.needsExclusive);
00444         targets.needsExclusive = true;
00445         // if any of the deferred targets were upper-level cache
00446         // requests marked downstreamPending, need to clear that
00447         assert(!downstreamPending);  // not pending here anymore
00448         deferredTargets.clearDownstreamPending();
00449         // this clears out deferredTargets too
00450         targets.splice(targets.end(), deferredTargets);
00451         deferredTargets.resetFlags();
00452     }
00453 }

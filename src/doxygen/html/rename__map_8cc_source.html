00059 SimpleRenameMap::RenameInfo
00060 SimpleRenameMap::rename(RegIndex arch_reg)
00061 {
00062     PhysRegIndex renamed_reg;
00063 
00064     // Record the current physical register that is renamed to the
00065     // requested architected register.
00066     PhysRegIndex prev_reg = map[arch_reg];
00067 
00068     // If it's not referencing the zero register, then rename the
00069     // register.
00070     if (arch_reg != zeroReg) {
00071         renamed_reg = freeList->getReg();
00072 
00073         map[arch_reg] = renamed_reg;
00074     } else {
00075         // Otherwise return the zero register so nothing bad happens.
00076         assert(prev_reg == zeroReg);
00077         renamed_reg = zeroReg;
00078     }
00079 
00080     DPRINTF(Rename, "Renamed reg %d to physical reg %d old mapping was %d\n",
00081             arch_reg, renamed_reg, prev_reg);
00082 
00083     return RenameInfo(renamed_reg, prev_reg);
00084 }

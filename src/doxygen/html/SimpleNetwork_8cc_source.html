00226  * The simple network has an array of switches. These switches have buffers

00230 bool
00231 SimpleNetwork::functionalRead(Packet *pkt)
00232 {
00233     for (unsigned int i = 0; i < m_switches.size(); i++) {
00234         if (m_switches[i]->functionalRead(pkt)) {
00235             return true;
00236         }
00237     }
00238 
00239     for (unsigned int i = 0; i < m_buffers_to_free.size(); ++i) {
00240         if (m_buffers_to_free[i]->functionalRead(pkt)) {
00241             return true;
00242         }
00243     }
00244 
00245     return false;
00246 }

00122 TrafficGen::drain(DrainManager *dm)
00123 {
00124     if (!updateEvent.scheduled()) {
00125         // no event has been scheduled yet (e.g. switched from atomic mode)
00126         return 0;
00127     }
00128 
00129     if (retryPkt == NULL) {
00130         // shut things down
00131         nextPacketTick = MaxTick;
00132         nextTransitionTick = MaxTick;
00133         deschedule(updateEvent);
00134         return 0;
00135     } else {
00136         drainManager = dm;

00180 void
00181 TrafficGen::update()
00182 {
00183     // if we have reached the time for the next state transition, then
00184     // perform the transition
00185     if (curTick() >= nextTransitionTick) {
00186         transition();
00187     } else {
00188         assert(curTick() >= nextPacketTick);
00189         // get the next packet and try to send it
00190         PacketPtr pkt = states[currState]->getNextPacket();
00191 
00192         // suppress packets that are not destined for a memory, such as
00193         // device accesses that could be part of a trace
00194         if (system->isMemAddr(pkt->getAddr())) {
00195             numPackets++;
00196             if (!port.sendTimingReq(pkt)) {
00197                 retryPkt = pkt;
00198                 retryPktTick = curTick();
00199             }
00200         } else {
00201             DPRINTF(TrafficGen, "Suppressed packet %s 0x%x\n",
00202                     pkt->cmdString(), pkt->getAddr());
00203         }
00204     }
00205 
00206     // if we are waiting for a retry, do not schedule any further
00207     // events, in the case of a transition or a successful send, go
00208     // ahead and determine when the next update should take place
00209     if (retryPkt == NULL) {
00210         // schedule next update event based on either the next execute
00211         // tick or the next transition, which ever comes first
00212         nextPacketTick = states[currState]->nextPacketTick(elasticReq, 0);
00213         Tick nextEventTick = std::min(nextPacketTick, nextTransitionTick);
00214         DPRINTF(TrafficGen, "Next event scheduled at %lld\n", nextEventTick);
00215         schedule(updateEvent, nextEventTick);
00216     }
00217 }

00441 void
00442 TrafficGen::transition()
00443 {
00444     // exit the current state
00445     states[currState]->exit();
00446 
00447     // determine next state
00448     double p = random_mt.random<double>();
00449     assert(currState < transitionMatrix.size());
00450     double cumulative = 0.0;
00451     size_t i = 0;
00452     do {
00453         cumulative += transitionMatrix[currState][i];
00454         ++i;
00455     } while (cumulative < p && i < transitionMatrix[currState].size());
00456 
00457     enterState(i - 1);
00458 }

00473 void
00474 TrafficGen::recvReqRetry()
00475 {
00476     assert(retryPkt != NULL);
00477 
00478     DPRINTF(TrafficGen, "Received retry\n");
00479     numRetries++;
00480     // attempt to send the packet, and if we are successful start up
00481     // the machinery again
00482     if (port.sendTimingReq(retryPkt)) {
00483         retryPkt = NULL;
00484         // remember how much delay was incurred due to back-pressure
00485         // when sending the request, we also use this to derive
00486         // the tick for the next packet
00487         Tick delay = curTick() - retryPktTick;
00488         retryPktTick = 0;
00489         retryTicks += delay;
00490 
00491         if (drainManager == NULL) {
00492             // packet is sent, so find out when the next one is due
00493             nextPacketTick = states[currState]->nextPacketTick(elasticReq,
00494                                                                delay);
00495             Tick nextEventTick = std::min(nextPacketTick, nextTransitionTick);
00496             schedule(updateEvent, std::max(curTick(), nextEventTick));
00497         } else {
00498             // shut things down
00499             nextPacketTick = MaxTick;
00500             nextTransitionTick = MaxTick;
00501             drainManager->signalDrainDone();
00502             // Clear the drain event once we're done with it.
00503             drainManager = NULL;
00504         }
00505     }
00506 }

00286 unsigned int
00287 BaseKvmCPU::drain(DrainManager *dm)
00288 {
00289     if (switchedOut())
00290         return 0;
00291 
00292     DPRINTF(Drain, "BaseKvmCPU::drain\n");
00293     switch (_status) {
00294       case Running:
00295         // The base KVM code is normally ready when it is in the
00296         // Running state, but the architecture specific code might be
00297         // of a different opinion. This may happen when the CPU been
00298         // notified of an event that hasn't been accepted by the vCPU
00299         // yet.
00300         if (!archIsDrained()) {
00301             drainManager = dm;
00302             return 1;
00303         }
00304 
00305         // The state of the CPU is consistent, so we don't need to do
00306         // anything special to drain it. We simply de-schedule the
00307         // tick event and enter the Idle state to prevent nasty things
00308         // like MMIOs from happening.
00309         if (tickEvent.scheduled())
00310             deschedule(tickEvent);
00311         _status = Idle;
00312 
00314       case Idle:
00315         // Idle, no need to drain
00316         assert(!tickEvent.scheduled());
00317 
00318         // Sync the thread context here since we'll need it when we
00319         // switch CPUs or checkpoint the CPU.
00320         syncThreadContext();
00321 
00322         return 0;
00323 
00324       case RunningServiceCompletion:
00325         // The CPU has just requested a service that was handled in
00326         // the RunningService state, but the results have still not
00327         // been reported to the CPU. Now, we /could/ probably just
00328         // update the register state ourselves instead of letting KVM
00329         // handle it, but that would be tricky. Instead, we enter KVM
00330         // and let it do its stuff.
00331         drainManager = dm;
00332 
00333         DPRINTF(Drain, "KVM CPU is waiting for service completion, "
00334                 "requesting drain.\n");
00335         return 1;
00336 
00337       case RunningService:
00338         // We need to drain since the CPU is waiting for service (e.g., MMIOs)
00339         drainManager = dm;
00340 
00341         DPRINTF(Drain, "KVM CPU is waiting for service, requesting drain.\n");
00342         return 1;
00343 
00344       default:
00345         panic("KVM: Unhandled CPU state in drain()\n");
00346         return 0;
00347     }
00348 }

00519 void
00520 BaseKvmCPU::tick()
00521 {
00522     Tick delay(0);
00523     assert(_status != Idle);
00524 
00525     switch (_status) {
00526       case RunningService:
00527         // handleKvmExit() will determine the next state of the CPU
00528         delay = handleKvmExit();
00529 
00530         if (tryDrain())
00531             _status = Idle;
00532         break;
00533 
00534       case RunningServiceCompletion:
00535       case Running: {
00536           const uint64_t nextInstEvent(
00537               !comInstEventQueue[0]->empty() ?
00538               comInstEventQueue[0]->nextTick() : UINT64_MAX);
00539           // Enter into KVM and complete pending IO instructions if we
00540           // have an instruction event pending.
00541           const Tick ticksToExecute(
00542               nextInstEvent > ctrInsts ?
00543               curEventQueue()->nextTick() - curTick() : 0);
00544 
00545           // We might need to update the KVM state.
00546           syncKvmState();
00547 
00548           // Setup any pending instruction count breakpoints using
00549           // PerfEvent if we are going to execute more than just an IO
00550           // completion.
00551           if (ticksToExecute > 0)
00552               setupInstStop();
00553 
00554           DPRINTF(KvmRun, "Entering KVM...\n");
00555           if (drainManager) {
00556               // Force an immediate exit from KVM after completing
00557               // pending operations. The architecture-specific code
00558               // takes care to run until it is in a state where it can
00559               // safely be drained.
00560               delay = kvmRunDrain();
00561           } else {
00562               delay = kvmRun(ticksToExecute);
00563           }
00564 
00565           // The CPU might have been suspended before entering into
00566           // KVM. Assume that the CPU was suspended /before/ entering
00567           // into KVM and skip the exit handling.
00568           if (_status == Idle)
00569               break;
00570 
00571           // Entering into KVM implies that we'll have to reload the thread
00572           // context from KVM if we want to access it. Flag the KVM state as
00573           // dirty with respect to the cached thread context.
00574           kvmStateDirty = true;
00575 
00576           // Enter into the RunningService state unless the
00577           // simulation was stopped by a timer.
00578           if (_kvmRun->exit_reason !=  KVM_EXIT_INTR) {
00579               _status = RunningService;
00580           } else {
00581               ++numExitSignal;
00582               _status = Running;
00583           }
00584 
00585           // Service any pending instruction events. The vCPU should
00586           // have exited in time for the event using the instruction
00587           // counter configured by setupInstStop().
00588           comInstEventQueue[0]->serviceEvents(ctrInsts);
00589           system->instEventQueue.serviceEvents(system->totalNumInsts);
00590 
00591           if (tryDrain())
00592               _status = Idle;
00593       } break;
00594 
00595       default:
00596         panic("BaseKvmCPU entered tick() in an illegal state (%i)\n",
00597               _status);
00598     }
00599 
00600     // Schedule a new tick if we are still running
00601     if (_status != Idle)
00602         schedule(tickEvent, clockEdge(ticksToCycles(delay)));
00603 }

00624 Tick
00625 BaseKvmCPU::kvmRun(Tick ticks)
00626 {
00627     Tick ticksExecuted;
00628     DPRINTF(KvmRun, "KVM: Executing for %i ticks\n", ticks);
00629 
00630     if (ticks == 0) {
00631         // Settings ticks == 0 is a special case which causes an entry
00632         // into KVM that finishes pending operations (e.g., IO) and
00633         // then immediately exits.
00634         DPRINTF(KvmRun, "KVM: Delivering IO without full guest entry\n");
00635 
00636         ++numVMHalfEntries;
00637 
00638         // Send a KVM_KICK_SIGNAL to the vCPU thread (i.e., this
00639         // thread). The KVM control signal is masked while executing
00640         // in gem5 and gets unmasked temporarily as when entering
00641         // KVM. See setSignalMask() and setupSignalHandler().
00642         kick();
00643 
00644         // Start the vCPU. KVM will check for signals after completing
00645         // pending operations (IO). Since the KVM_KICK_SIGNAL is
00646         // pending, this forces an immediate exit to gem5 again. We
00647         // don't bother to setup timers since this shouldn't actually
00648         // execute any code (other than completing half-executed IO
00649         // instructions) in the guest.
00650         ioctlRun();
00651 
00652         // We always execute at least one cycle to prevent the
00653         // BaseKvmCPU::tick() to be rescheduled on the same tick
00654         // twice.
00655         ticksExecuted = clockPeriod();
00656     } else {
00657         // This method is executed as a result of a tick event. That
00658         // means that the event queue will be locked when entering the
00659         // method. We temporarily unlock the event queue to allow
00660         // other threads to steal control of this thread to inject
00661         // interrupts. They will typically lock the queue and then
00662         // force an exit from KVM by kicking the vCPU.
00663         EventQueue::ScopedRelease release(curEventQueue());
00664 
00665         if (ticks < runTimer->resolution()) {
00666             DPRINTF(KvmRun, "KVM: Adjusting tick count (%i -> %i)\n",
00667                     ticks, runTimer->resolution());
00668             ticks = runTimer->resolution();
00669         }
00670 
00671         // Get hardware statistics after synchronizing contexts. The KVM
00672         // state update might affect guest cycle counters.
00673         uint64_t baseCycles(getHostCycles());
00674         uint64_t baseInstrs(hwInstructions.read());
00675 
00676         // Arm the run timer and start the cycle timer if it isn't
00677         // controlled by the overflow timer. Starting/stopping the cycle
00678         // timer automatically starts the other perf timers as they are in
00679         // the same counter group.
00680         runTimer->arm(ticks);
00681         if (!perfControlledByTimer)
00682             hwCycles.start();
00683 
00684         ioctlRun();
00685 
00686         runTimer->disarm();
00687         if (!perfControlledByTimer)
00688             hwCycles.stop();
00689 
00690         // The control signal may have been delivered after we exited
00691         // from KVM. It will be pending in that case since it is
00692         // masked when we aren't executing in KVM. Discard it to make
00693         // sure we don't deliver it immediately next time we try to
00694         // enter into KVM.
00695         discardPendingSignal(KVM_KICK_SIGNAL);
00696 
00697         const uint64_t hostCyclesExecuted(getHostCycles() - baseCycles);

00998 Tick
00999 BaseKvmCPU::doMMIOAccess(Addr paddr, void *data, int size, bool write)
01000 {
01001     ThreadContext *tc(thread->getTC());
01002     syncThreadContext();
01003 
01004     Request mmio_req(paddr, size, Request::UNCACHEABLE, dataMasterId());
01005     mmio_req.setThreadContext(tc->contextId(), 0);
01006     // Some architectures do need to massage physical addresses a bit
01007     // before they are inserted into the memory system. This enables
01008     // APIC accesses on x86 and m5ops where supported through a MMIO
01009     // interface.
01010     BaseTLB::Mode tlb_mode(write ? BaseTLB::Write : BaseTLB::Read);
01011     Fault fault(tc->getDTBPtr()->finalizePhysical(&mmio_req, tc, tlb_mode));
01012     if (fault != NoFault)
01013         warn("Finalization of MMIO address failed: %s\n", fault->name());
01014 
01015 
01016     const MemCmd cmd(write ? MemCmd::WriteReq : MemCmd::ReadReq);
01017     Packet pkt(&mmio_req, cmd);
01018     pkt.dataStatic(data);
01019 
01020     if (mmio_req.isMmappedIpr()) {
01021         // We currently assume that there is no need to migrate to a
01022         // different event queue when doing IPRs. Currently, IPRs are
01023         // only used for m5ops, so it should be a valid assumption.
01024         const Cycles ipr_delay(write ?
01025                              TheISA::handleIprWrite(tc, &pkt) :
01026                              TheISA::handleIprRead(tc, &pkt));
01027         threadContextDirty = true;
01028         return clockPeriod() * ipr_delay;
01029     } else {
01030         // Temporarily lock and migrate to the event queue of the
01031         // VM. This queue is assumed to "own" all devices we need to
01032         // access if running in multi-core mode.
01033         EventQueue::ScopedMigration migrate(vm.eventQueue());
01034 
01035         return dataPort.sendAtomic(&pkt);
01036     }
01037 }

01142 {
01143     int discardedSignal;
01144 
01145     // Setting the timeout to zero causes sigtimedwait to return
01146     // immediately.
01147     struct timespec timeout;
01148     timeout.tv_sec = 0;

01246 BaseKvmCPU::setupInstCounter(uint64_t period)
01247 {
01248     // No need to do anything if we aren't attaching for the first
01249     // time or the period isn't changing.
01250     if (period == activeInstPeriod && hwInstructions.attached())
01251         return;

01253     PerfKvmCounterConfig cfgInstructions(PERF_TYPE_HARDWARE,
01254                                          PERF_COUNT_HW_INSTRUCTIONS);

01257     // exclude_host since different architectures use slightly
01258     // different APIs in the kernel.

01260         .exclude_host(true);

01264         cfgInstructions.wakeupEvents(1)
01265             .samplePeriod(period);

01269     // sampling settings. See PerfKvmCounter::period() for details.

01274                           0, // TID (0 => currentThread)
01275                           hwCycles);

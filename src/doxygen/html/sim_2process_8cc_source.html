00348 bool
00349 Process::fixupStackFault(Addr vaddr)
00350 {
00351     // Check if this is already on the stack and there's just no page there
00352     // yet.
00353     if (vaddr >= stack_min && vaddr < stack_base) {
00354         allocateMem(roundDown(vaddr, PageBytes), PageBytes);
00355         return true;
00356     }
00357 
00358     // We've accessed the next page of the stack, so extend it to include
00359     // this address.
00360     if (vaddr < stack_min && vaddr >= stack_base - max_stack_size) {
00361         while (vaddr < stack_min) {
00362             stack_min -= TheISA::PageBytes;
00363             if (stack_base - stack_min > max_stack_size)
00364                 fatal("Maximum stack size exceeded\n");
00365             allocateMem(stack_min, TheISA::PageBytes);
00366             inform("Increasing stack size by one page.");
00367         };
00368         return true;
00369     }
00370     return false;
00371 }

00409

00411         stderr_fd = STDOUT_FILENO;

00421             panic("Unable to seek to correct location in file: %s", err);

00423 
00424     fdo_stdin->fd = stdin_fd;

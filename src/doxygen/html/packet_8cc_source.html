00176 Packet::checkFunctional(Printable *obj, Addr addr, bool is_secure, int size,
00177                         uint8_t *_data)
00178 {
00179     Addr func_start = getAddr();
00180     Addr func_end   = getAddr() + getSize() - 1;
00181     Addr val_start  = addr;
00182     Addr val_end    = val_start + size - 1;
00183 
00184     if (is_secure != _isSecure || func_start > val_end ||
00185         val_start > func_end) {
00186         // no intersection
00187         return false;
00188     }
00189 
00190     // check print first since it doesn't require data
00191     if (isPrint()) {
00192         assert(!_data);
00193         safe_cast<PrintReqState*>(senderState)->printObj(obj);
00194         return false;
00195     }
00196 
00197     // we allow the caller to pass NULL to signify the other packet
00198     // has no data
00199     if (!_data) {
00200         return false;
00201     }
00202 
00203     // offset of functional request into supplied value (could be
00204     // negative if partial overlap)
00205     int offset = func_start - val_start;
00206 
00207     if (isRead()) {
00208         if (func_start >= val_start && func_end <= val_end) {
00209             memcpy(getPtr<uint8_t>(), _data + offset, getSize());
00210             if (bytesValid.empty())
00211                 bytesValid.resize(getSize(), true);
00212             // complete overlap, and as the current packet is a read
00213             // we are done
00214             return true;
00215         } else {
00216             // Offsets and sizes to copy in case of partial overlap
00217             int func_offset;
00218             int val_offset;
00219             int overlap_size;
00220 
00221             // calculate offsets and copy sizes for the two byte arrays
00222             if (val_start < func_start && val_end <= func_end) {
00223                 // the one we are checking against starts before and
00224                 // ends before or the same
00225                 val_offset = func_start - val_start;
00226                 func_offset = 0;
00227                 overlap_size = val_end - func_start;
00228             } else if (val_start >= func_start && val_end > func_end) {
00229                 // the one we are checking against starts after or the
00230                 // same, and ends after
00231                 val_offset = 0;
00232                 func_offset = val_start - func_start;
00233                 overlap_size = func_end - val_start;
00234             } else if (val_start >= func_start && val_end <= func_end) {
00235                 // the one we are checking against is completely
00236                 // subsumed in the current packet, possibly starting
00237                 // and ending at the same address
00238                 val_offset = 0;
00239                 func_offset = val_start - func_start;
00240                 overlap_size = size;
00241             } else if (val_start < func_start && val_end > func_end) {
00242                 // the current packet is completely subsumed in the
00243                 // one we are checking against
00244                 val_offset = func_start - val_start;
00245                 func_offset = 0;
00246                 overlap_size = func_end - func_start;
00247             } else {
00248                 panic("Missed a case for checkFunctional with "
00249                       " %s 0x%x size %d, against 0x%x size %d\n",
00250                       cmdString(), getAddr(), getSize(), addr, size);
00251             }
00252 
00253             // copy partial data into the packet's data array
00254             uint8_t *dest = getPtr<uint8_t>() + func_offset;
00255             uint8_t *src = _data + val_offset;
00256             memcpy(dest, src, overlap_size);
00257 
00258             // initialise the tracking of valid bytes if we have not
00259             // used it already
00260             if (bytesValid.empty())
00261                 bytesValid.resize(getSize(), false);
00262 
00263             // track if we are done filling the functional access
00264             bool all_bytes_valid = true;
00265 
00266             int i = 0;
00267 
00268             // check up to func_offset
00269             for (; all_bytes_valid && i < func_offset; ++i)
00270                 all_bytes_valid &= bytesValid[i];
00271 
00272             // update the valid bytes
00273             for (i = func_offset; i < func_offset + overlap_size; ++i)
00274                 bytesValid[i] = true;
00275 
00276             // check the bit after the update we just made
00277             for (; all_bytes_valid && i < getSize(); ++i)
00278                 all_bytes_valid &= bytesValid[i];
00279 
00280             return all_bytes_valid;
00281         }
00282     } else if (isWrite()) {
00283         if (offset >= 0) {
00284             memcpy(_data + offset, getConstPtr<uint8_t>(),
00285                    (min(func_end, val_end) - func_start) + 1);
00286         } else {
00287             // val_start > func_start
00288             memcpy(_data, getConstPtr<uint8_t>() - offset,
00289                    (min(func_end, val_end) - val_start) + 1);
00290         }
00291     } else {
00292         panic("Don't know how to handle command %s\n", cmdString());
00293     }
00294 
00295     // keep going with request by default
00296     return false;
00297 }

00117 void
00118 NetworkLink::wakeup()
00119 {
00120     if (!link_srcQueue->isReady(curCycle()))
00121         return;
00122 
00123     flit *t_flit = link_srcQueue->getTopFlit();
00124     t_flit->set_time(curCycle() + m_latency);
00125     linkBuffer->insert(t_flit);
00126     link_consumer->scheduleEventAbsolute(clockEdge(m_latency));
00127 
00128     m_link_utilized++;
00129     m_vc_load[t_flit->get_vc()]++;
00130 }

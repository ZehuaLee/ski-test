00037 LocalBP::LocalBP(const LocalBPParams *params)
00038     : BPredUnit(params),
00039       localPredictorSize(params->localPredictorSize),
00040       localCtrBits(params->localCtrBits)
00041 {
00042     if (!isPowerOf2(localPredictorSize)) {
00043         fatal("Invalid local predictor size!\n");
00044     }
00045 
00046     localPredictorSets = localPredictorSize / localCtrBits;
00047 
00048     if (!isPowerOf2(localPredictorSets)) {
00049         fatal("Invalid number of local predictor sets! Check localCtrBits.\n");
00050     }
00051 
00052     // Setup the index mask.
00053     indexMask = localPredictorSets - 1;
00054 
00055     DPRINTF(Fetch, "index mask: %#x\n", indexMask);
00056 
00057     // Setup the array of counters for the local predictor.
00058     localCtrs.resize(localPredictorSets);
00059 
00060     for (unsigned i = 0; i < localPredictorSets; ++i)
00061         localCtrs[i].setBits(localCtrBits);
00062 
00063     DPRINTF(Fetch, "local predictor size: %i\n",
00064             localPredictorSize);
00065 
00066     DPRINTF(Fetch, "local counter bits: %i\n", localCtrBits);
00067 
00068     DPRINTF(Fetch, "instruction shift amount: %i\n",
00069             instShiftAmt);

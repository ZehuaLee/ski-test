00047 DecoderFaultInst::DecoderFaultInst(ExtMachInst _machInst)
00048     : ArmStaticInst("gem5decoderFault", _machInst, No_OpClass),
00049       faultId(static_cast<DecoderFault>(
00050                   static_cast<uint8_t>(_machInst.decoderFault)))
00051 {
00052     // Don't call execute() if we're on a speculative path and the

00057 Fault
00058 DecoderFaultInst::execute(ExecContext *xc, Trace::InstRecord *traceData) const
00059 {
00060     const PCState pc_state(xc->pcState());
00061     const Addr pc(pc_state.instAddr());
00062 
00063     switch (faultId) {
00064       case DecoderFault::UNALIGNED:
00065         if (machInst.aarch64) {
00066             return std::make_shared<PCAlignmentFault>(pc);
00067         } else {
00068             // TODO: We should check if we the receiving end is in
00069             // aarch64 mode and raise a PCAlignment fault instead.
00070             return std::make_shared<PrefetchAbort>(
00071                 pc, ArmFault::AlignmentFault);
00072         }
00073 
00074       case DecoderFault::PANIC:
00075         panic("Internal error in instruction decoder\n");
00076 
00077       case DecoderFault::OK:
00078         panic("Decoder fault instruction without decoder fault.\n");
00079     }
00080 
00081     panic("Unhandled fault type");
00082 }

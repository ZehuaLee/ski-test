00433 template <class Impl>
00434 void
00435 LSQUnit<Impl>::checkSnoop(PacketPtr pkt)
00436 {
00437     int load_idx = loadHead;
00438     DPRINTF(LSQUnit, "Got snoop for address %#x\n", pkt->getAddr());
00439 
00440     // Unlock the cpu-local monitor when the CPU sees a snoop to a locked
00441     // address. The CPU can speculatively execute a LL operation after a pending
00442     // SC operation in the pipeline and that can make the cache monitor the CPU
00443     // is connected to valid while it really shouldn't be.
00444     for (int x = 0; x < cpu->numContexts(); x++) {
00445         ThreadContext *tc = cpu->getContext(x);
00446         bool no_squash = cpu->thread[x]->noSquashFromTC;
00447         cpu->thread[x]->noSquashFromTC = true;
00448         TheISA::handleLockedSnoop(tc, pkt, cacheBlockMask);
00449         cpu->thread[x]->noSquashFromTC = no_squash;
00450     }
00451 
00452     Addr invalidate_addr = pkt->getAddr() & cacheBlockMask;
00453 
00454     DynInstPtr ld_inst = loadQueue[load_idx];
00455     if (ld_inst) {
00456         Addr load_addr = ld_inst->physEffAddr & cacheBlockMask;
00457         // Check that this snoop didn't just invalidate our lock flag
00458         if (ld_inst->effAddrValid() && load_addr == invalidate_addr &&
00459             ld_inst->memReqFlags & Request::LLSC)
00460             TheISA::handleLockedSnoopHit(ld_inst.get());
00461     }
00462 
00463     // If this is the only load in the LSQ we don't care
00464     if (load_idx == loadTail)
00465         return;
00466 
00467     incrLdIdx(load_idx);
00468 
00469     bool force_squash = false;
00470 
00471     while (load_idx != loadTail) {
00472         DynInstPtr ld_inst = loadQueue[load_idx];
00473 
00474         if (!ld_inst->effAddrValid() || ld_inst->strictlyOrdered()) {
00475             incrLdIdx(load_idx);
00476             continue;
00477         }
00478 
00479         Addr load_addr = ld_inst->physEffAddr & cacheBlockMask;
00480         DPRINTF(LSQUnit, "-- inst [sn:%lli] load_addr: %#x to pktAddr:%#x\n",
00481                     ld_inst->seqNum, load_addr, invalidate_addr);
00482 
00483         if (load_addr == invalidate_addr || force_squash) {
00484             if (needsTSO) {
00485                 // If we have a TSO system, as all loads must be ordered with
00486                 // all other loads, this load as well as *all* subsequent loads
00487                 // need to be squashed to prevent possible load reordering.
00488                 force_squash = true;
00489             }
00490             if (ld_inst->possibleLoadViolation() || force_squash) {
00491                 DPRINTF(LSQUnit, "Conflicting load at addr %#x [sn:%lli]\n",
00492                         pkt->getAddr(), ld_inst->seqNum);
00493 
00494                 // Mark the load for re-execution
00495                 ld_inst->fault = std::make_shared<ReExec>();
00496             } else {
00497                 DPRINTF(LSQUnit, "HitExternal Snoop for addr %#x [sn:%lli]\n",
00498                         pkt->getAddr(), ld_inst->seqNum);
00499 
00500                 // Make sure that we don't lose a snoop hitting a LOCKED
00501                 // address since the LOCK* flags don't get updated until
00502                 // commit.
00503                 if (ld_inst->memReqFlags & Request::LLSC)
00504                     TheISA::handleLockedSnoopHit(ld_inst.get());
00505 
00506                 // If a older load checks this and it's true
00507                 // then we might have missed the snoop
00508                 // in which case we need to invalidate to be sure
00509                 ld_inst->hitExternalSnoop(true);
00510             }
00511         }
00512         incrLdIdx(load_idx);
00513     }
00514     return;
00515 }

00517 template <class Impl>
00518 Fault
00519 LSQUnit<Impl>::checkViolations(int load_idx, DynInstPtr &inst)
00520 {
00521     Addr inst_eff_addr1 = inst->effAddr >> depCheckShift;
00522     Addr inst_eff_addr2 = (inst->effAddr + inst->effSize - 1) >> depCheckShift;
00523 
00529     while (load_idx != loadTail) {
00530         DynInstPtr ld_inst = loadQueue[load_idx];
00531         if (!ld_inst->effAddrValid() || ld_inst->strictlyOrdered()) {
00532             incrLdIdx(load_idx);
00533             continue;
00534         }
00535 
00536         Addr ld_eff_addr1 = ld_inst->effAddr >> depCheckShift;
00537         Addr ld_eff_addr2 =
00538             (ld_inst->effAddr + ld_inst->effSize - 1) >> depCheckShift;
00539 
00540         if (inst_eff_addr2 >= ld_eff_addr1 && inst_eff_addr1 <= ld_eff_addr2) {
00541             if (inst->isLoad()) {
00542                 // If this load is to the same block as an external snoop
00543                 // invalidate that we've observed then the load needs to be
00544                 // squashed as it could have newer data
00545                 if (ld_inst->hitExternalSnoop()) {
00546                     if (!memDepViolator ||
00547                             ld_inst->seqNum < memDepViolator->seqNum) {
00548                         DPRINTF(LSQUnit, "Detected fault with inst [sn:%lli] "
00549                                 "and [sn:%lli] at address %#x\n",
00550                                 inst->seqNum, ld_inst->seqNum, ld_eff_addr1);
00551                         memDepViolator = ld_inst;
00552 
00553                         ++lsqMemOrderViolation;
00554 
00555                         return std::make_shared<GenericISA::M5PanicFault>(
00556                             "Detected fault with inst [sn:%lli] and "
00557                             "[sn:%lli] at address %#x\n",
00558                             inst->seqNum, ld_inst->seqNum, ld_eff_addr1);
00559                     }
00560                 }
00561 
00562                 // Otherwise, mark the load has a possible load violation
00563                 // and if we see a snoop before it's commited, we need to squash
00564                 ld_inst->possibleLoadViolation(true);
00565                 DPRINTF(LSQUnit, "Found possible load violation at addr: %#x"
00566                         " between instructions [sn:%lli] and [sn:%lli]\n",
00567                         inst_eff_addr1, inst->seqNum, ld_inst->seqNum);
00568             } else {
00569                 // A load/store incorrectly passed this store.
00570                 // Check if we already have a violator, or if it's newer
00571                 // squash and refetch.
00572                 if (memDepViolator && ld_inst->seqNum > memDepViolator->seqNum)
00573                     break;
00574 
00575                 DPRINTF(LSQUnit, "Detected fault with inst [sn:%lli] and "
00576                         "[sn:%lli] at address %#x\n",
00577                         inst->seqNum, ld_inst->seqNum, ld_eff_addr1);
00578                 memDepViolator = ld_inst;
00579 
00580                 ++lsqMemOrderViolation;
00581 
00582                 return std::make_shared<GenericISA::M5PanicFault>(
00583                     "Detected fault with "
00584                     "inst [sn:%lli] and [sn:%lli] at address %#x\n",
00585                     inst->seqNum, ld_inst->seqNum, ld_eff_addr1);
00586             }
00587         }
00588 
00589         incrLdIdx(load_idx);
00590     }
00591     return NoFault;
00592 }

00598 Fault
00599 LSQUnit<Impl>::executeLoad(DynInstPtr &inst)
00600 {
00601     using namespace TheISA;
00602     // Execute a specific load.
00603     Fault load_fault = NoFault;
00604 
00605     DPRINTF(LSQUnit, "Executing load PC %s, [sn:%lli]\n",
00606             inst->pcState(), inst->seqNum);
00607 
00608     assert(!inst->isSquashed());
00609 
00610     load_fault = inst->initiateAcc();
00611 
00612     if (inst->isTranslationDelayed() &&
00613         load_fault == NoFault)
00614         return load_fault;
00615 
00616     // If the instruction faulted or predicated false, then we need to send it
00617     // along to commit without the instruction completing.
00618     if (load_fault != NoFault || !inst->readPredicate()) {
00619         // Send this instruction to commit, also make sure iew stage
00620         // realizes there is activity.  Mark it as executed unless it
00621         // is a strictly ordered load that needs to hit the head of
00622         // commit.
00623         if (!inst->readPredicate())
00624             inst->forwardOldRegs();
00625         DPRINTF(LSQUnit, "Load [sn:%lli] not executed from %s\n",
00626                 inst->seqNum,
00627                 (load_fault != NoFault ? "fault" : "predication"));
00628         if (!(inst->hasRequest() && inst->strictlyOrdered()) ||
00629             inst->isAtCommit()) {
00630             inst->setExecuted();
00631         }
00632         iewStage->instToCommit(inst);
00633         iewStage->activityThisCycle();
00634     } else {
00635         assert(inst->effAddrValid());
00636         int load_idx = inst->lqIdx;
00637         incrLdIdx(load_idx);
00638 
00639         if (checkLoads)
00640             return checkViolations(load_idx, inst);
00641     }
00642 
00643     return load_fault;
00644 }

00665     Fault store_fault = store_inst->initiateAcc();
00666 
00667     if (store_inst->isTranslationDelayed() &&
00668         store_fault == NoFault)
00669         return store_fault;
00670 
00671     if (!store_inst->readPredicate())

00677 
00678         return store_fault;
00679     } else if (!store_inst->readPredicate()) {

00777     // complete last cycle because there weren't enough cache ports available.
00778     if (TheISA::HasUnalignedMemAcc) {
00779         writebackPendingStore();

00782     while (storesToWB > 0 &&
00783            storeWBIdx != storeTail &&
00784            storeQueue[storeWBIdx].inst &&

00791                     " is blocked!\n");

00795         // Store didn't write any data so no need to write it back to
00796         // memory.

00798             completeStore(storeWBIdx);

00818             assert(storeQueue[storeWBIdx].sreqHigh);
00819         }

00822 
00823         Request *req = storeQueue[storeWBIdx].req;
00824         RequestPtr sreqLow = storeQueue[storeWBIdx].sreqLow;
00825         RequestPtr sreqHigh = storeQueue[storeWBIdx].sreqHigh;
00826 
00827         storeQueue[storeWBIdx].committed = true;
00828 
00829         assert(!inst->memData);
00830         inst->memData = new uint8_t[req->getSize()];
00831 
00832         if (storeQueue[storeWBIdx].isAllZeros)
00833             memset(inst->memData, 0, req->getSize());
00834         else
00835             memcpy(inst->memData, storeQueue[storeWBIdx].data, req->getSize());
00836

00838         PacketPtr snd_data_pkt = NULL;
00839 
00840         LSQSenderState *state = new LSQSenderState;
00841         state->isLoad = false;

00843         state->inst = inst;
00844 
00845         if (!TheISA::HasUnalignedMemAcc || !storeQueue[storeWBIdx].isSplit) {
00846 
00847             // Build a single data packet if the store isn't split.
00848             data_pkt = Packet::createWrite(req);

00853             data_pkt = Packet::createWrite(sreqLow);
00854             snd_data_pkt = Packet::createWrite(sreqHigh);
00855 
00856             data_pkt->dataStatic(inst->memData);
00857             snd_data_pkt->dataStatic(inst->memData + sreqLow->getSize());
00858 
00859             data_pkt->senderState = state;
00860             snd_data_pkt->senderState = state;
00861 
00862             state->isSplit = true;
00863             state->outstanding = 2;
00864 
00865             // Can delete the main request now.
00866             delete req;
00867             req = sreqLow;
00868         }
00869 
00870         DPRINTF(LSQUnit, "D-Cache: Writing back store idx:%i PC:%s "
00871                 "to Addr:%#x, data:%#x [sn:%lli]\n",

00873                 req->getPaddr(), (int)*(inst->memData),
00874                 inst->seqNum);
00875 
00876         // @todo: Remove this SC hack once the memory system handles it.
00877         if (inst->isStoreConditional()) {
00878             assert(!storeQueue[storeWBIdx].isSplit);
00879             // Disable recording the result temporarily.  Writing to
00880             // misc regs normally updates the result, but this is not
00881             // the desired behavior when handling store conditionals.
00882             inst->recordResult(false);
00883             bool success = TheISA::handleLockedWrite(inst.get(), req, cacheBlockMask);
00884             inst->recordResult(true);
00885 
00886             if (!success) {

00888                 DPRINTF(LSQUnit, "Store conditional [sn:%lli] failed.  "
00889                         "Instantly completing it.\n",
00890                         inst->seqNum);

00892                 cpu->schedule(wb, curTick() + 1);
00893                 if (cpu->checker) {
00894                     // Make sure to set the LLSC data for verification
00895                     // if checker is loaded
00896                     inst->reqToVerify->setExtraData(0);
00897                     inst->completeAcc(data_pkt);
00898                 }
00899                 completeStore(storeWBIdx);
00900                 incrStIdx(storeWBIdx);
00901                 continue;
00902             }
00903         } else {
00904             // Non-store conditionals do not need a writeback.
00905             state->noWB = true;
00906         }
00907 
00908         bool split =
00909             TheISA::HasUnalignedMemAcc && storeQueue[storeWBIdx].isSplit;
00910 
00911         ThreadContext *thread = cpu->tcBase(lsqID);
00912 
00913         if (req->isMmappedIpr()) {
00914             assert(!inst->isStoreConditional());
00915             TheISA::handleIprWrite(thread, data_pkt);
00916             delete data_pkt;
00917             if (split) {
00918                 assert(snd_data_pkt->req->isMmappedIpr());
00919                 TheISA::handleIprWrite(thread, snd_data_pkt);
00920                 delete snd_data_pkt;
00921                 delete sreqLow;
00922                 delete sreqHigh;
00923             }
00924             delete state;
00925             delete req;
00926             completeStore(storeWBIdx);
00927             incrStIdx(storeWBIdx);
00928         } else if (!sendStore(data_pkt)) {
00929             DPRINTF(IEW, "D-Cache became blocked when writing [sn:%lli], will"
00930                     "retry later\n",
00931                     inst->seqNum);
00932 
00933             // Need to store the second packet, if split.
00934             if (split) {
00935                 state->pktToSend = true;
00936                 state->pendingPacket = snd_data_pkt;
00937             }
00938         } else {
00939 
00940             // If split, try to send the second packet too
00941             if (split) {
00942                 assert(snd_data_pkt);
00943 
00944                 // Ensure there are enough ports to use.
00945                 if (usedPorts < cachePorts) {
00946                     ++usedPorts;
00947                     if (sendStore(snd_data_pkt)) {
00948                         storePostSend(snd_data_pkt);
00949                     } else {
00950                         DPRINTF(IEW, "D-Cache became blocked when writing"
00951                                 " [sn:%lli] second packet, will retry later\n",
00952                                 inst->seqNum);
00953                     }
00954                 } else {
00955 
00956                     // Store the packet for when there's free ports.
00957                     assert(pendingPkt == NULL);
00958                     pendingPkt = snd_data_pkt;
00959                     hasPendingPkt = true;
00960                 }
00961             } else {
00962 
00963                 // Not a split store.
00964                 storePostSend(data_pkt);
00965             }
00966         }
00967     }
00968 
00969     // Not sure this should set it to 0.
00970     usedPorts = 0;
00971 
00972     assert(stores >= 0 && storesToWB >= 0);

01111 template <class Impl>
01112 void
01113 LSQUnit<Impl>::writeback(DynInstPtr &inst, PacketPtr pkt)
01114 {
01115     iewStage->wakeCPU();
01116 
01117     // Squashed instructions do not need to complete their access.
01118     if (inst->isSquashed()) {
01119         assert(!inst->isStore());
01120         ++lsqIgnoredResponses;
01121         return;
01122     }
01123 
01124     if (!inst->isExecuted()) {
01125         inst->setExecuted();
01126 
01127         if (inst->fault == NoFault) {
01128             // Complete access to copy data to proper place.
01129             inst->completeAcc(pkt);
01130         } else {
01131             // If the instruction has an outstanding fault, we cannot complete
01132             // the access as this discards the current fault.
01133 
01134             // If we have an outstanding fault, the fault should only be of
01135             // type ReExec.
01136             assert(dynamic_cast<ReExec*>(inst->fault.get()) != nullptr);
01137 
01138             DPRINTF(LSQUnit, "Not completing instruction [sn:%lli] access "
01139                     "due to pending fault.\n", inst->seqNum);
01140         }
01141     }
01142 
01143     // Need to insert instruction into queue to commit
01144     iewStage->instToCommit(inst);
01145 
01146     iewStage->activityThisCycle();

01148     // see if this load changed the PC
01149     iewStage->checkMisprediction(inst);

00107 void
00108 Throttle::operateVnet(int vnet, int &bw_remaining, bool &schedule_wakeup,
00109                       MessageBuffer *in, MessageBuffer *out)
00110 {
00111     if (out == nullptr || in == nullptr) {
00112         return;
00113     }
00114     assert(m_units_remaining[vnet] >= 0);
00115 
00116     while (bw_remaining > 0 && (in->isReady() || m_units_remaining[vnet] > 0) &&
00117                                 out->areNSlotsAvailable(1)) {
00118 
00119         // See if we are done transferring the previous message on
00120         // this virtual network
00121         if (m_units_remaining[vnet] == 0 && in->isReady()) {
00122             // Find the size of the message we are moving
00123             MsgPtr msg_ptr = in->peekMsgPtr();
00124             NetworkMessage* net_msg_ptr =
00125                 safe_cast<NetworkMessage*>(msg_ptr.get());
00126             m_units_remaining[vnet] +=
00127                 network_message_to_size(net_msg_ptr);
00128 
00129             DPRINTF(RubyNetwork, "throttle: %d my bw %d bw spent "
00130                     "enqueueing net msg %d time: %lld.\n",
00131                     m_node, getLinkBandwidth(), m_units_remaining[vnet],
00132                     g_system_ptr->curCycle());
00133 
00134             // Move the message
00135             in->dequeue();
00136             out->enqueue(msg_ptr, m_link_latency);
00137 
00138             // Count the message
00139             m_msg_counts[net_msg_ptr->getMessageSize()][vnet]++;
00140             DPRINTF(RubyNetwork, "%s\n", *out);
00141         }
00142 
00143         // Calculate the amount of bandwidth we spent on this message
00144         int diff = m_units_remaining[vnet] - bw_remaining;
00145         m_units_remaining[vnet] = max(0, diff);
00146         bw_remaining = max(0, -diff);
00147     }
00148 
00149     if (bw_remaining > 0 && (in->isReady() || m_units_remaining[vnet] > 0) &&
00150                              !out->areNSlotsAvailable(1)) {
00151         DPRINTF(RubyNetwork, "vnet: %d", vnet);
00152 
00153         // schedule me to wakeup again because I'm waiting for my
00154         // output queue to become available
00155         schedule_wakeup = true;
00156     }
00157 }

00159 void
00160 Throttle::wakeup()
00161 {
00162     // Limits the number of message sent to a limited number of bytes/cycle.
00163     assert(getLinkBandwidth() > 0);
00164     int bw_remaining = getLinkBandwidth();
00165 
00166     m_wakeups_wo_switch++;
00167     bool schedule_wakeup = false;
00168 
00169     // variable for deciding the direction in which to iterate
00170     bool iteration_direction = false;
00171 
00172 
00173     // invert priorities to avoid starvation seen in the component network
00174     if (m_wakeups_wo_switch > PRIORITY_SWITCH_LIMIT) {
00175         m_wakeups_wo_switch = 0;
00176         iteration_direction = true;
00177     }
00178 
00179     if (iteration_direction) {
00180         for (int vnet = 0; vnet < m_vnets; ++vnet) {
00181             operateVnet(vnet, bw_remaining, schedule_wakeup,
00182                         m_in[vnet], m_out[vnet]);
00183         }
00184     } else {
00185         for (int vnet = m_vnets-1; vnet >= 0; --vnet) {
00186             operateVnet(vnet, bw_remaining, schedule_wakeup,
00187                         m_in[vnet], m_out[vnet]);
00188         }
00189     }
00190 
00191     // We should only wake up when we use the bandwidth
00192     // This is only mostly true
00193     // assert(bw_remaining != getLinkBandwidth());
00194 
00195     // Record that we used some or all of the link bandwidth this cycle
00196     double ratio = 1.0 - (double(bw_remaining) / double(getLinkBandwidth()));
00197 
00198     // If ratio = 0, we used no bandwidth, if ratio = 1, we used all
00199     m_link_utilization_proxy += ratio;
00200 
00201     if (bw_remaining > 0 && !schedule_wakeup) {
00202         // We have extra bandwidth and our output buffer was
00203         // available, so we must not have anything else to do until
00204         // another message arrives.
00205         DPRINTF(RubyNetwork, "%s not scheduled again\n", *this);
00206     } else {
00207         DPRINTF(RubyNetwork, "%s scheduled again\n", *this);
00208 
00209         // We are out of bandwidth for this cycle, so wakeup next
00210         // cycle and continue
00211         scheduleEvent(Cycles(1));
00212     }
00213 }

00260 network_message_to_size(NetworkMessage* net_msg_ptr)
00261 {
00262     assert(net_msg_ptr != NULL);
00263 
00264     int size = Network::MessageSizeType_to_int(net_msg_ptr->getMessageSize());
00265     size *=  MESSAGE_SIZE_MULTIPLIER;

00267     // Artificially increase the size of broadcast messages

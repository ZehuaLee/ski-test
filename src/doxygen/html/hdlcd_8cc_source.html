00315 void
00316 HDLcd::updateVideoParams(bool unserializing = false)
00317 {
00318     const uint16_t bpp M5_VAR_USED = bytesPerPixel() << 3;
00319 
00320     // Workaround configuration bugs where multiple display
00321     // controllers are attached to the same VNC server by reattaching
00322     // enabled devices. This isn't ideal, but works as long as only
00323     // one display controller is active at a time.
00324     if (command.enable && vnc)
00325         vnc->setFrameBuffer(&fb);
00326 
00327     // updating these parameters while LCD is enabled is not supported
00328     if (frameUnderway && !unserializing)
00329         panic("Attempting to change some HDLCD parameters while the controller"
00330                 " is active is not allowed");
00331 
00332     // resize the virtualDisplayBuffer unless we are unserializing - it may
00333     //   have changed size
00334     // there must be no outstanding DMA transactions for this to work
00335     if (!unserializing) {
00336         assert(dmaPendingNum == 0);
00337 
00338         virtualDisplayBuffer.resize(bytesPerPixel() * area());
00339         fb.resize(width(), height());
00340         fb.clear();
00341 
00342         std::fill(virtualDisplayBuffer.begin(), virtualDisplayBuffer.end(),
00343                   0);
00344     }
00345 
00346     DPRINTF(HDLcd, "bpp = %d\n", bpp);
00347     DPRINTF(HDLcd, "display size = %d x %d\n", width(), height());
00348 #if TRACING_ON
00349     const size_t totalLinesPerFrame = v_back_porch.val + 1 +
00350                                       v_data.val + 1 +
00351                                       v_front_porch.val + 1 +
00352                                       v_sync.val + 1;
00353     const double fps = (double)SimClock::Frequency /
00354             (double)(PClksPerLine() * totalLinesPerFrame * pixelClock);
00355 #endif
00356     DPRINTF(HDLcd, "simulated refresh rate ~ %.1ffps generating ~ %.1fMB/s "
00357             "traffic ([%.1fMHz, T=%d sim clocks] pclk, %d bpp => %.1fMB/s peak requirement)\n",
00358             fps,
00359             fps * virtualDisplayBuffer.size() / 1024 / 1024,
00360             (double)SimClock::Frequency / pixelClock / 1000000.0,
00361             pixelClock,
00362             bpp,
00363             (double)(SimClock::Frequency / pixelClock * (bpp / 8)) / 1024 / 1024);

00366 void
00367 HDLcd::startFrame()
00368 {
00369     // 0. Check that we are in the appropriate state
00370     assert(!frameUnderway);
00371     if (!command.enable)
00372         return;
00373     DPRINTF(HDLcd, "Frame read started\n");
00374     if (doUpdateParams) {
00375         updateVideoParams();
00376         doUpdateParams = false;
00377     }
00378     frameUnderway = true;
00379     assert(!virtualDisplayBuffer.empty());
00380     assert(pixelBufferSize == 0);
00381     assert(dmaBytesInFlight == 0);
00382     assert(dmaPendingNum == 0);
00383     assert(dmaDoneEventFree.size() == dmaDoneEventAll.size());
00384     assert(!renderPixelEvent.scheduled());
00385     // currently only support positive line pitches equal to the line length
00386     assert(width() * bytesPerPixel() == fb_line_pitch);
00387 
00388     // 1. Start DMA'ing the frame; subsequent transactions created as we go
00389     dmaCurAddr = dmaStartAddr = fb_base;
00390     dmaMaxAddr = static_cast<Addr>(width() * height() * bytesPerPixel()) +
00391                     dmaCurAddr;
00392     frameReadStartTime = curTick();
00393     pixelIndex = 0;
00394     frameUnderrun = false;
00395     fillPixelBuffer();
00396 
00397     // 2. Schedule first pixelclock read; subsequent reads generated as we go
00398     Tick firstPixelReadTick = curTick() + pixelClock * (
00399                                   PClksPerLine() * (v_sync.val + 1 +
00400                                                     v_back_porch.val + 1) +
00401                                   h_sync.val + 1 +
00402                                   h_back_porch.val + 1);
00403     schedule(renderPixelEvent, firstPixelReadTick);
00404 }

00406 void
00407 HDLcd::fillPixelBuffer()
00408 {
00409     // - am I under the LCD dma transaction total?
00410     // - do I have more data to transfer?
00411     // - have I not yet underrun for this frame?
00412     // - is there room to put the data in the pixel buffer including any
00413     //   outstanding dma transfers in flight?
00414     while ((dmaPendingNum < maxOutstandingDma()) &&
00415            (dmaMaxAddr > dmaCurAddr) &&
00416            !frameUnderrun &&
00417            bytesFreeInPixelBuffer() > dmaBurstLength() * AXI_PORT_WIDTH) {
00418         // try largest transaction size allowed first but switch to smaller
00419         // sizes for trailing bytes
00420         size_t transaction_size = dmaBurstLength() * AXI_PORT_WIDTH;
00421         while (transaction_size > (dmaMaxAddr - dmaCurAddr))
00422             transaction_size >>= 1;
00423         assert(transaction_size > 0);
00424 
00425         // concurrent dma reads need different dma done events
00426         // due to assertion in scheduling state
00427         ++dmaPendingNum;
00428 
00429         assert(!dmaDoneEventFree.empty());
00430         DmaDoneEvent *event(dmaDoneEventFree.back());
00431         dmaDoneEventFree.pop_back();
00432         assert(event);
00433         assert(!event->scheduled());
00434 
00435         // We use a uncachable request here because the requests from the CPU
00436         // will be uncacheable as well. If we have uncacheable and cacheable
00437         // requests in the memory system for the same address it won't be
00438         // pleased
00439         uint8_t *const dma_dst(
00440             virtualDisplayBuffer.data() + dmaCurAddr - dmaStartAddr);
00441         event->setTransactionSize(transaction_size);
00442         dmaPort.dmaAction(MemCmd::ReadReq, dmaCurAddr, transaction_size, event,
00443                           dma_dst, 0, Request::UNCACHEABLE);
00444         dmaCurAddr += transaction_size;
00445         dmaBytesInFlight += transaction_size;
00446     }
00447 }

00449 void
00450 HDLcd::renderPixel()
00451 {
00452     // try to handle multiple pixels at a time; doing so reduces the accuracy
00453     //   of the underrun detection but lowers simulation overhead
00454     const size_t count = 32;
00455     assert(width() % count == 0); // not set up to handle trailing pixels
00456 
00457     // have we underrun on this frame anytime before?
00458     if (frameUnderrun) {
00459         // the LCD controller gives up on a frame if an underrun occurs and
00460         //   resumes regular operation on the next frame
00461         pixelBufferSize = 0;
00462     } else {
00463         // did we underrun on this set of pixels?
00464         if (pixelBufferSize < bytesPerPixel() * count) {
00465             warn("HDLcd controller buffer underrun\n");
00466             frameUnderrun = true;
00467             int_rawstat.underrun = 1;
00468             if (!intEvent.scheduled())
00469                 schedule(intEvent, clockEdge());
00470         } else {
00471             // emulate the pixel read from the internal buffer
00472             pixelBufferSize -= bytesPerPixel() * count;
00473         }
00474     }
00475 
00476     // the DMA may have previously stalled due to the buffer being full;
00477     //   give it a kick; it knows not to fill if at end of frame, underrun, etc
00478     if (!fillPixelBufferEvent.scheduled())
00479         schedule(fillPixelBufferEvent, clockEdge());
00480 
00481     // schedule the next pixel read according to where it is in the frame
00482     pixelIndex += count;
00483     assert(pixelIndex <= width() * height());
00484     size_t x = pixelIndex % width();
00485     Tick nextEventTick = curTick();
00486     if (x == 0) {
00487         // start of new line
00488         nextEventTick += pixelClock * ((h_front_porch.val + 1) +
00489                                        (h_back_porch.val + 1) +
00490                                        (h_sync.val + 1));
00491         if (pixelIndex == width() * height()) {
00492             // end of frame
00493             nextEventTick += PClksPerLine() * (v_front_porch.val + 1) *
00494                              pixelClock;
00495             schedule(endFrameEvent, nextEventTick);
00496             return;
00497         }
00498     } else {
00499         nextEventTick += pixelClock * count;
00500     }
00501 
00502     schedule(renderPixelEvent, nextEventTick);
00503 }

00554 void
00555 HDLcd::dmaDone(DmaDoneEvent *event)
00556 {
00557     const size_t transactionLength = event->getTransactionSize();
00558     assert(pixelBufferSize + transactionLength < PIXEL_BUFFER_CAPACITY);
00559     assert(dmaCurAddr <= dmaMaxAddr);
00560 
00561     dmaDoneEventFree.push_back(event);
00562     --dmaPendingNum;
00563     assert(MAX_OUTSTANDING_DMA_REQ_CAPACITY - dmaDoneEventFree.size() ==
00564             dmaPendingNum);
00565 
00566     // add the data to the pixel buffer
00567     dmaBytesInFlight -= transactionLength;
00568     pixelBufferSize += transactionLength;
00569 
00570     // schedule another dma transaction if:
00571     // - we're not done reading the frame
00572     // - there is sufficient room in the pixel buffer for another transaction
00573     // - another fillPixelBufferEvent is not already scheduled
00574     const size_t targetTransSize = dmaBurstLength() * AXI_PORT_WIDTH;
00575     if ((dmaCurAddr < dmaMaxAddr) &&
00576         (bytesFreeInPixelBuffer() + targetTransSize < PIXEL_BUFFER_CAPACITY) &&
00577         !fillPixelBufferEvent.scheduled()) {
00578         schedule(fillPixelBufferEvent, clockEdge());
00579     }
00580 }

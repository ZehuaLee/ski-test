00050 bool
00051 SeriesRequestGenerator::initiate()
00052 {
00053     DPRINTF(DirectedTest, "initiating request\n");
00054     assert(m_status == SeriesRequestGeneratorStatus_Thinking);
00055 
00056     MasterPort* port = m_directed_tester->getCpuPort(m_active_node);
00057 
00058     Request::Flags flags;
00059 
00060     // For simplicity, requests are assumed to be 1 byte-sized
00061     Request *req = new Request(m_address, 1, flags, masterId);
00062 
00063     Packet::Command cmd;
00064     bool do_write = (random_mt.random(0, 100) < m_percent_writes);
00065     if (do_write) {
00066         cmd = MemCmd::WriteReq;
00067     } else {
00068         cmd = MemCmd::ReadReq;
00069     }
00070 
00071     PacketPtr pkt = new Packet(req, cmd);
00072     pkt->allocate();
00073 
00074     if (port->sendTimingReq(pkt)) {
00075         DPRINTF(DirectedTest, "initiating request - successful\n");
00076         m_status = SeriesRequestGeneratorStatus_Request_Pending;
00077         return true;
00078     } else {
00079         // If the packet did not issue, must delete
00080         // Note: No need to delete the data, the packet destructor
00081         // will delete it
00082         delete pkt->req;
00083         delete pkt;
00084 
00085         DPRINTF(DirectedTest, "failed to initiate request - sequencer not ready\n");
00086         return false;
00087     }
00088 }

00090 void 
00091 SeriesRequestGenerator::performCallback(uint32_t proc, Addr address)
00092 {
00093     assert(m_active_node == proc);
00094     assert(m_address == address);  
00095     assert(m_status == SeriesRequestGeneratorStatus_Request_Pending);
00096 
00097     m_status = SeriesRequestGeneratorStatus_Thinking;
00098     m_active_node++;
00099     if (m_active_node == m_num_cpus) {
00100         //
00101         // Cycle of requests completed, increment cycle completions and restart

00104         m_directed_tester->incrementCycleCompletions();

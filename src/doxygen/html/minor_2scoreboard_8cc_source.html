00051 {
00052     RegClass reg_class = regIdxToClass(reg);
00053     bool ret = false;
00054 
00055     if (reg == TheISA::ZeroReg) {
00056         /* Don't bother with the zero register */
00057         ret = false;
00058     } else {
00059         switch (reg_class)
00060         {
00061           case IntRegClass:
00062             scoreboard_index = reg;
00063             ret = true;
00064             break;

00066             scoreboard_index = TheISA::NumIntRegs + TheISA::NumCCRegs +
00067                 reg - TheISA::FP_Reg_Base;

00070           case CCRegClass:
00071             scoreboard_index = TheISA::NumIntRegs + reg - TheISA::FP_Reg_Base;
00072             ret = true;

00074           case MiscRegClass:
00075               /* Don't bother with Misc registers */
00076             ret = false;

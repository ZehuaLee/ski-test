00088 Tick
00089 I2CBus::write(PacketPtr pkt)
00090 {
00091     assert(pkt->getAddr() == pioAddr + SB_CONTROLS ||
00092            pkt->getAddr() == pioAddr + SB_CONTROLC);
00093 
00094     updateSignals(pkt);
00095 
00096     // Check if the bus master is starting a new transmission.
00097     if (isStart(pkt)) {
00098         state = RECEIVING_ADDR;
00099         message = 0x00;
00100         currBit = 7;
00101         /* Most i2c devices expect something special (e.g., command,
00102          * register address) in the first byte they receive so they
00103          * must be notified somehow that this is a new transmission.
00104          */
00105         for (auto& d : devices) {
00106             d.second->i2cStart();
00107         }
00108         return pioDelay;
00109     }
00110 
00111     // Check if the bus master is ending a transmission.
00112     if (isEnd(pkt)) {
00113         state = IDLE;
00114         return pioDelay;
00115     }
00116 
00117     // Only change state when the clock is transitioning from low to high.
00118     // This may not perfectly mimic physical i2c devices but the important
00119     // part is to only do the following once per clock cycle.
00120     if (isClockSet(pkt)) {
00121         switch (state) {
00122           case RECEIVING_ADDR:
00123             if (currBit >= 0) {
00124                 message |= sda << currBit;
00125                 currBit--;
00126             } else {
00127                 i2cAddr = message >> 1;
00128                 assert(devices.find(i2cAddr) != devices.end());
00129                 if (message & 0x01) {
00130                     state = SENDING_DATA;
00131                     message = devices[i2cAddr]->read();
00132                 } else {
00133                     state = RECEIVING_DATA;
00134                     message = 0x00;
00135                 }
00136                 currBit = 7;
00137                 sda = 0; /* Ack */
00138             }
00139             break;
00140           case RECEIVING_DATA:
00141             if (currBit >= 0) {
00142                 message |= sda << currBit;
00143                 currBit--;
00144             } else {
00145                 devices[i2cAddr]->write(message);
00146                 message = 0x00;
00147                 currBit = 7;
00148                 sda = 0; /* Ack */
00149             }
00150             break;
00151           case SENDING_DATA:
00152             if (currBit >= 0) {
00153                 sda = (message >> currBit) & 0x01;
00154                 currBit--;
00155             } else {
00156                 if (!sda) /* Check for ack from the bus master. */
00157                     message = devices[i2cAddr]->read();
00158                 currBit = 7;
00159             }
00160             break;
00161           case IDLE:
00162           default:
00163             panic("Invalid state on posedge of clock in I2CBus::write.\n");
00164             break;
00165         }
00166     }
00167 
00168     return pioDelay;
00169 }

00171 void
00172 I2CBus::updateSignals(PacketPtr pkt)
00173 {
00174     uint8_t msg = pkt->get<uint8_t>();
00175     Addr daddr = pkt->getAddr() - pioAddr;
00176 
00177     switch (daddr) {
00178       case SB_CONTROLS:
00179         scl = (msg & 1) ? 1 : scl;
00180         sda = (msg & 2) ? 1 : sda;
00181         break;
00182       case SB_CONTROLC:
00183         scl = (msg & 1) ? 0 : scl;
00184         sda = (msg & 2) ? 0 : sda;
00185         break;
00186       default:
00187         break;
00188     }
00189 }

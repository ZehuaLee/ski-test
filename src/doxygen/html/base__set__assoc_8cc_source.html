00056 BaseSetAssoc::BaseSetAssoc(const Params *p)
00057     :BaseTags(p), assoc(p->assoc),
00058      numSets(p->size / (p->block_size * p->assoc)),
00059      sequentialAccess(p->sequential_access)
00060 {
00061     // Check parameters
00062     if (blkSize < 4 || !isPowerOf2(blkSize)) {
00063         fatal("Block size must be at least 4 and a power of 2");
00064     }
00065     if (numSets <= 0 || !isPowerOf2(numSets)) {
00066         fatal("# of sets must be non-zero and a power of 2");
00067     }
00068     if (assoc <= 0) {
00069         fatal("associativity must be greater than zero");
00070     }
00071 
00072     blkMask = blkSize - 1;
00073     setShift = floorLog2(blkSize);
00074     setMask = numSets - 1;
00075     tagShift = setShift + floorLog2(numSets);
00077     warmupBound = numSets * assoc;
00078 
00079     sets = new SetType[numSets];
00080     blks = new BlkType[numSets * assoc];
00081     // allocate data storage in one big chunk
00082     numBlocks = numSets * assoc;
00083     dataBlks = new uint8_t[numBlocks * blkSize];
00084 
00085     unsigned blkIndex = 0;       // index into blks array
00086     for (unsigned i = 0; i < numSets; ++i) {
00087         sets[i].assoc = assoc;
00088 
00089         sets[i].blks = new BlkType*[assoc];
00090 
00091         // link in the data blocks
00092         for (unsigned j = 0; j < assoc; ++j) {
00093             // locate next cache block
00094             BlkType *blk = &blks[blkIndex];
00095             blk->data = &dataBlks[blkSize*blkIndex];
00096             ++blkIndex;
00097 
00098             // invalidate new cache block
00099             blk->invalidate();
00100 
00101             //EGH Fix Me : do we need to initialize blk?
00102 
00103             // Setting the tag to j is just to prevent long chains in the hash
00104             // table; won't matter because the block is invalid
00105             blk->tag = j;
00106             blk->whenReady = 0;
00107             blk->isTouched = false;
00108             blk->size = blkSize;
00109             sets[i].blks[j]=blk;
00110             blk->set = i;
00111         }
00112     }
00113 }

00190                 age_index = 2;

00192                 age_index = 3;

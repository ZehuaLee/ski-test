00081 bool
00082 BasePrefetcher::observeAccess(const PacketPtr &pkt) const
00083 {
00084     Addr addr = pkt->getAddr();
00085     bool fetch = pkt->req->isInstFetch();
00086     bool read = pkt->isRead();
00087     bool inv = pkt->isInvalidate();
00088     bool is_secure = pkt->isSecure();
00089 
00090     if (pkt->req->isUncacheable()) return false;
00091     if (fetch && !onInst) return false;
00092     if (!fetch && !onData) return false;
00093     if (!fetch && read && !onRead) return false;
00094     if (!fetch && !read && !onWrite) return false;
00095     if (!fetch && !read && inv) return false;
00096 
00097     if (onMiss) {
00098         return !inCache(addr, is_secure) &&
00099                !inMissQueue(addr, is_secure);
00100     }
00101 
00102     return true;
00103 }

00119 LinearGen::nextPacketTick(bool elastic, Tick delay) const
00120 {
00121     // Check to see if we have reached the data limit. If dataLimit is
00122     // zero we do not have a data limit and therefore we will keep
00123     // generating requests for the entire residency in this state.
00124     if (dataLimit && dataManipulated >= dataLimit) {
00125         DPRINTF(TrafficGen, "Data limit for LinearGen reached.\n");
00126         // there are no more requests, therefore return MaxTick
00127         return MaxTick;
00128     } else {
00129         // return the time when the next request should take place
00130         Tick wait = random_mt.random(minPeriod, maxPeriod);
00131 
00132         // compensate for the delay experienced to not be elastic, by
00133         // default the value we generate is from the time we are
00134         // asked, so the elasticity happens automatically
00135         if (!elastic) {
00136             if (wait < delay)
00137                 wait = 0;
00138             else
00139                 wait -= delay;
00140         }
00141 
00142         return curTick() + wait;

00331 void
00332 DramGen::genStartAddr(unsigned int new_bank, unsigned int new_rank)
00333 {
00334     // start by picking a random address in the range
00335     addr = random_mt.random<Addr>(startAddr, endAddr - 1);
00336 
00337     // round down to start address of a block, i.e. a DRAM burst
00338     addr -= addr % blocksize;
00339 
00340     // insert the bank bits at the right spot, and align the
00341     // address to achieve the required hit length, this involves
00342     // finding the appropriate start address such that all
00343     // sequential packets target successive columns in the same
00344     // page
00345 
00346     // for example, if we have a stride size of 192B, which means
00347     // for LPDDR3 where burstsize = 32B we have numSeqPkts = 6,
00348     // the address generated previously can be such that these
00349     // 192B cross the page boundary, hence it needs to be aligned
00350     // so that they all belong to the same page for page hit
00351     unsigned int columns_per_page = pageSize / blocksize;
00352 
00353     // pick a random column, but ensure that there is room for
00354     // numSeqPkts sequential columns in the same page
00355     unsigned int new_col =
00356         random_mt.random<unsigned int>(0, columns_per_page - numSeqPkts);
00357 
00358     if (addrMapping == 1) {
00359         // addrMapping=1: RoRaBaCoCh/RoRaBaChCo
00360         // Block bits, then page bits, then bank bits, then rank bits
00361         replaceBits(addr, blockBits + pageBits + bankBits - 1,
00362                     blockBits + pageBits, new_bank);
00363         replaceBits(addr, blockBits + pageBits - 1, blockBits, new_col);
00364         if (rankBits != 0) {
00365             replaceBits(addr, blockBits + pageBits + bankBits +rankBits - 1,
00366                         blockBits + pageBits + bankBits, new_rank);
00367         }
00368     } else if (addrMapping == 0) {
00369         // addrMapping=0: RoCoRaBaCh
00370         // Block bits, then bank bits, then rank bits, then page bits
00371         replaceBits(addr, blockBits + bankBits - 1, blockBits, new_bank);
00372         replaceBits(addr, blockBits + bankBits + rankBits + pageBits - 1,
00373                     blockBits + bankBits + rankBits, new_col);
00374         if (rankBits != 0) {
00375             replaceBits(addr, blockBits + bankBits + rankBits - 1,
00376                         blockBits + bankBits, new_rank);
00377         }
00378     }
00379 }

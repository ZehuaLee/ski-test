00222     getInterrupt(ThreadContext *tc)
00223     {
00224         HCR  hcr  = tc->readMiscReg(MISCREG_HCR);

00226         SCR  scr  = tc->readMiscReg(MISCREG_SCR);
00227 
00228         // Calculate a few temp vars so we can work out if there's a pending
00229         // virtual interrupt, and if its allowed to happen
00230         // ARM ARM Issue C section B1.9.9, B1.9.11, and B1.9.13
00231         bool isHypMode   = cpsr.mode == MODE_HYP;
00232         bool isSecure    = inSecureState(scr, cpsr);
00233         bool allowVIrq   = !cpsr.i && hcr.imo && !isSecure && !isHypMode;
00234         bool allowVFiq   = !cpsr.f && hcr.fmo && !isSecure && !isHypMode;
00235         bool allowVAbort = !cpsr.a && hcr.amo && !isSecure && !isHypMode;
00236 
00237         if ( !(intStatus || (hcr.vi && allowVIrq) || (hcr.vf && allowVFiq) ||
00238                (hcr.va && allowVAbort)) )
00239             return NoFault;
00240 
00241         bool take_irq = takeInt(tc, INT_IRQ);
00242         bool take_fiq = takeInt(tc, INT_FIQ);
00243         bool take_ea =  takeInt(tc, INT_ABT);
00244 
00245 
00246         if (interrupts[INT_IRQ] && take_irq)
00247             return std::make_shared<Interrupt>();
00248         if ((interrupts[INT_VIRT_IRQ] || hcr.vi) && allowVIrq)
00249             return std::make_shared<VirtualInterrupt>();
00250         if (interrupts[INT_FIQ] && take_fiq)
00251             return std::make_shared<FastInterrupt>();
00252         if ((interrupts[INT_VIRT_FIQ] || hcr.vf) && allowVFiq)
00253             return std::make_shared<VirtualFastInterrupt>();
00254         if (interrupts[INT_ABT] && take_ea)
00255             return std::make_shared<SystemError>();
00256         if (hcr.va && allowVAbort)
00257             return std::make_shared<VirtualDataAbort>(
00258                 0, TlbEntry::DomainType::NoAccess, false,
00259                 ArmFault::AsynchronousExternalAbort);
00260         if (interrupts[INT_RST])
00261             return std::make_shared<Reset>();
00262         if (interrupts[INT_SEV])
00263             return std::make_shared<ArmSev>();
00264 
00265         panic("intStatus and interrupts not in sync\n");
00266     }

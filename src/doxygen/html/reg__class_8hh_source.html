00066 RegClass regIdxToClass(TheISA::RegIndex reg_idx,
00067                        TheISA::RegIndex *rel_reg_idx = NULL)
00068 {
00069     assert(reg_idx < TheISA::Max_Reg_Index);
00070     RegClass cl;
00071     int offset;
00072 
00073     if (reg_idx < TheISA::FP_Reg_Base) {
00074         cl = IntRegClass;
00075         offset = 0;
00076     } else if (reg_idx < TheISA::CC_Reg_Base) {
00077         cl = FloatRegClass;
00078         offset = TheISA::FP_Reg_Base;
00079     } else if (reg_idx < TheISA::Misc_Reg_Base) {
00080         // if there are no CC regs, the ISA should set
00081         // CC_Reg_Base == Misc_Reg_Base so the if above
00082         // never succeeds
00083         cl = CCRegClass;
00084         offset = TheISA::CC_Reg_Base;
00085     } else {
00086         cl = MiscRegClass;
00087         offset = TheISA::Misc_Reg_Base;
00088     }
00089 
00090     if (rel_reg_idx)
00091         *rel_reg_idx = reg_idx - offset;
00092     return cl;
00093 }

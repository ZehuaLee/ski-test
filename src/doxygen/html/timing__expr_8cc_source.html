00088 uint64_t TimingExprUn::eval(TimingExprEvalContext &context)
00089 {
00090     uint64_t arg_value = arg->eval(context);
00091     uint64_t ret = 0;
00092 
00093     switch (op) {
00094       case Enums::timingExprSizeInBits:
00095         if (arg_value == 0)
00096             ret = 0;
00097         else
00098             ret = ceilLog2(arg_value);
00099         break;
00100       case Enums::timingExprNot:
00101         ret = arg_value != 0;
00102         break;
00103       case Enums::timingExprInvert:
00104         ret = ~arg_value;
00105         break;
00106       case Enums::timingExprSignExtend32To64:
00107         ret = static_cast<int64_t>(
00108             static_cast<int32_t>(arg_value));
00109         break;
00110       case Enums::timingExprAbs:
00111         if (static_cast<int64_t>(arg_value) < 0)
00112             ret = -arg_value;
00113         else
00114             ret = arg_value;
00115         break;
00116       default:
00117         break;
00118     }
00119 
00120     return ret;
00121 }

00123 uint64_t TimingExprBin::eval(TimingExprEvalContext &context)
00124 {
00125     uint64_t left_value = left->eval(context);
00126     uint64_t right_value = right->eval(context);
00127     uint64_t ret = 0;
00128 
00129     switch (op) {
00130       case Enums::timingExprAdd:
00131           ret = left_value + right_value;
00132           break;
00133       case Enums::timingExprSub:
00134           ret = left_value - right_value;
00135           break;
00136       case Enums::timingExprUMul:
00137           ret = left_value * right_value;
00138           break;
00139       case Enums::timingExprUDiv:
00140           if (right_value != 0) {
00141               ret = left_value / right_value;
00142           }
00143           break;
00144       case Enums::timingExprUCeilDiv:
00145           if (right_value != 0) {
00146               ret = (left_value + (right_value - 1)) / right_value;
00147           }
00148           break;
00149       case Enums::timingExprSMul:
00150           ret = static_cast<int64_t>(left_value) *
00151               static_cast<int64_t>(right_value);
00152           break;
00153       case Enums::timingExprSDiv:
00154           if (right_value != 0) {
00155               ret = static_cast<int64_t>(left_value) /
00156                   static_cast<int64_t>(right_value);
00157           }
00158           break;
00159       case Enums::timingExprEqual:
00160           ret = left_value == right_value;
00161           break;
00162       case Enums::timingExprNotEqual:
00163           ret = left_value != right_value;
00164           break;
00165       case Enums::timingExprULessThan:
00166           ret = left_value < right_value;
00167           break;
00168       case Enums::timingExprUGreaterThan:
00169           ret = left_value > right_value;
00170           break;
00171       case Enums::timingExprSLessThan:
00172           ret = static_cast<int64_t>(left_value) <
00173               static_cast<int64_t>(right_value);
00174           break;
00175       case Enums::timingExprSGreaterThan:
00176           ret = static_cast<int64_t>(left_value) >
00177               static_cast<int64_t>(right_value);
00178           break;
00179       case Enums::timingExprAnd:
00180           ret = (left_value != 0) && (right_value != 0);
00181           break;
00182       case Enums::timingExprOr:
00183           ret = (left_value != 0) || (right_value != 0);
00184           break;
00185       default:
00186           break;
00187     }
00188 
00189     return ret;

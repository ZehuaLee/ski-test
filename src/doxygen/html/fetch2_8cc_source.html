00123 void
00124 Fetch2::updateBranchPrediction(const BranchData &branch)
00125 {
00126     MinorDynInstPtr inst = branch.inst;
00127 
00128     /* Don't even consider instructions we didn't try to predict or faults */
00129     if (inst->isFault() || !inst->triedToPredict)
00130         return;
00131 
00132     switch (branch.reason) {
00133       case BranchData::NoBranch:
00134         /* No data to update */
00135         break;
00136       case BranchData::Interrupt:
00137         /* Never try to predict interrupts */
00138         break;
00139       case BranchData::SuspendThread:
00140         /* Don't need to act on suspends */
00141         break;
00142       case BranchData::WakeupFetch:
00143         /* Don't need to act on wakeups, no instruction tied to action. */
00144         break;
00145       case BranchData::HaltFetch:
00146         /* Don't need to act on fetch wakeup */
00147         break;
00148       case BranchData::BranchPrediction:
00149         /* Shouldn't happen.  Fetch2 is the only source of
00150          *  BranchPredictions */
00151         break;
00152       case BranchData::UnpredictedBranch:
00153         /* Unpredicted branch or barrier */
00154         DPRINTF(Branch, "Unpredicted branch seen inst: %s\n", *inst);
00155         branchPredictor.squash(inst->id.fetchSeqNum,
00156             branch.target, true, inst->id.threadId);
00157         break;
00158       case BranchData::CorrectlyPredictedBranch:
00159         /* Predicted taken, was taken */
00160         DPRINTF(Branch, "Branch predicted correctly inst: %s\n", *inst);
00161         branchPredictor.update(inst->id.fetchSeqNum,
00162             inst->id.threadId);
00163         break;
00164       case BranchData::BadlyPredictedBranch:
00165         /* Predicted taken, not taken */
00166         DPRINTF(Branch, "Branch mis-predicted inst: %s\n", *inst);
00167         branchPredictor.squash(inst->id.fetchSeqNum,
00168             branch.target /* Not used */, false, inst->id.threadId);
00169         break;
00170       case BranchData::BadlyPredictedBranchTarget:
00171         /* Predicted taken, was taken but to a different target */
00172         DPRINTF(Branch, "Branch mis-predicted target inst: %s target: %s\n",
00173             *inst, branch.target);
00174         branchPredictor.squash(inst->id.fetchSeqNum,
00175             branch.target, true, inst->id.threadId);
00176         break;
00177     }
00178 }

00182 {

00184 
00185     assert(!inst->predictedTaken);
00186 
00187     /* Skip non-control/sys call instructions */
00188     if (inst->staticInst->isControl() ||
00189         inst->staticInst->isSyscall())
00190     {
00191         /* Tried to predict */
00192         inst->triedToPredict = true;
00193 
00194         DPRINTF(Branch, "Trying to predict for inst: %s\n", *inst);
00195 
00196         if (branchPredictor.predict(inst->staticInst,
00197             inst->id.fetchSeqNum, inst_pc,
00198             inst->id.threadId))
00199         {
00200             inst->predictedTaken = true;
00201             inst->predictedTarget = inst_pc;
00202             branch.target = inst_pc;
00203         }
00204     } else {
00205         DPRINTF(Branch, "Not attempting prediction for inst: %s\n", *inst);
00206     }
00207 
00208     /* If we predict taken, set branch and update sequence numbers */
00209     if (inst->predictedTaken) {
00210         /* Update the predictionSeqNum and remember the streamSeqNum that it
00211          *  was associated with */
00212         expectedStreamSeqNum = inst->id.streamSeqNum;
00213 
00214         BranchData new_branch = BranchData(BranchData::BranchPrediction,
00215             inst->id.streamSeqNum, predictionSeqNum + 1,
00216             inst->predictedTarget, inst);
00217 
00218         /* Mark with a new prediction number by the stream number of the
00219          *  instruction causing the prediction */
00220         predictionSeqNum++;
00221         branch = new_branch;
00222 
00223         DPRINTF(Branch, "Branch predicted taken inst: %s target: %s"
00224             " new predictionSeqNum: %d\n",
00225             *inst, inst->predictedTarget, predictionSeqNum);

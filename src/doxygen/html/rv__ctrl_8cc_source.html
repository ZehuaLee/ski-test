00128 RealViewCtrl::write(PacketPtr pkt)
00129 {
00130     assert(pkt->getAddr() >= pioAddr && pkt->getAddr() < pioAddr + pioSize);
00131 
00132     Addr daddr = pkt->getAddr() - pioAddr;
00133     switch (daddr) {
00134       case Flash:
00135       case Clcd:
00136       case Osc0:
00137       case Osc1:
00138       case Osc2:
00139       case Osc3:
00140       case Osc4:
00141         break;
00142       case Lock:
00143         sysLock.lockVal = pkt->get<uint16_t>();
00144         break;
00145       case Flags:
00146         flags = pkt->get<uint32_t>();
00147         break;
00148       case FlagsClr:
00149         flags = 0;
00150         break;
00151       case CfgData:
00152         scData = pkt->get<uint32_t>();
00153         break;
00154       case CfgCtrl: {
00155           // A request is being submitted to read/write the system control
00156           // registers.  See
00157           // http://infocenter.arm.com/help/topic/com.arm.doc.dui0447h/CACDEFGH.html
00158           // For now, model as much of the OSC regs (can't find docs) as Linux
00159           // seems to require (can't find docs); some clocks are deemed to be 0,
00160           // giving all kinds of /0 problems booting Linux 3.9.  Return a
00161           // vaguely plausible number within the range the device trees state:
00162           uint32_t data = pkt->get<uint32_t>();
00163           uint16_t dev = bits(data, 11, 0);
00164           uint8_t pos = bits(data, 15, 12);
00165           uint8_t site = bits(data, 17, 16);
00166           uint8_t func = bits(data, 25, 20);
00167           uint8_t dcc = bits(data, 29, 26);
00168           bool wr = bits(data, 30);
00169           bool start = bits(data, 31);
00170 
00171           if (start) {
00172               if (wr) {
00173                   warn_once("SCReg: Writing %#x to dcc%d:site%d:pos%d:fn%d:dev%d\n",
00174                           scData, dcc, site, pos, func, dev);
00175                   // Only really support reading, for now!
00176               } else {
00177                   // Only deal with function 1 (oscillators) so far!
00178                   if (dcc != 0 || pos != 0 || func != 1) {
00179                       warn("SCReg: read from unknown area "
00180                            "(dcc %d:site%d:pos%d:fn%d:dev%d)\n",
00181                            dcc, site, pos, func, dev);
00182                   } else {
00183                       switch (site) {
00184                         case 0: { // Motherboard regs
00185                             switch(dev) {
00186                               case 0: // MCC clk
00187                                 scData = 25000000;
00188                                 break;
00189                               case 1: // CLCD clk
00190                                 scData = 25000000;
00191                                 break;
00192                               case 2: // PeriphClk 24MHz
00193                                 scData = 24000000;
00194                                 break;
00195                               default:
00196                                 scData = 0;
00197                                 warn("SCReg: read from unknown dev %d "
00198                                      "(site%d:pos%d:fn%d)\n",
00199                                      dev, site, pos, func);
00200                             }
00201                         } break;
00202                         case 1: { // Coretile 1 regs
00203                             switch(dev) {
00204                               case 0: // CPU PLL ref
00205                                 scData = 50000000;
00206                                 break;
00207                               case 4: // Muxed AXI master clock
00208                                 scData = 40000000;
00209                                 break;
00210                               case 5: // HDLCD clk
00211                                 scData = 50000000;
00212                                 break;
00213                               case 6: // SMB clock
00214                                 scData = 35000000;
00215                                 break;
00216                               case 7: // SYS PLL (also used for pl011 UART!)
00217                                 scData = 40000000;
00218                                 break;
00219                               case 8: // DDR PLL 40MHz fixed
00220                                 scData = 40000000;
00221                                 break;
00222                               default:
00223                                 scData = 0;
00224                                 warn("SCReg: read from unknown dev %d "
00225                                      "(site%d:pos%d:fn%d)\n",
00226                                      dev, site, pos, func);
00227                             }
00228                         } break;
00229                         default:
00230                           warn("SCReg: Read from unknown site %d (pos%d:fn%d:dev%d)\n",
00231                                site, pos, func, dev);
00232                       }
00233                       DPRINTF(RVCTRL, "SCReg: Will read %#x (ctrlWr %#x)\n", scData, data);
00234                   }
00235               }
00236           } else {
00237               DPRINTF(RVCTRL, "SCReg: write %#x to ctrl but not starting\n", data);
00238           }
00239       } break;
00240       case CfgStat:     // Weird to write this
00241       default:
00242         warn("Tried to write RVIO at offset %#x (data %#x) that doesn't exist\n",
00243              daddr, pkt->get<uint32_t>());
00244         break;
00245     }
00246     pkt->makeAtomicResponse();
00247     return pioDelay;
00248 }

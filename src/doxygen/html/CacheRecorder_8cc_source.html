00106                                                                 m_bytes_read);

00109 
00110         for (int rec_bytes_read = 0; rec_bytes_read < m_block_size_bytes;
00111                 rec_bytes_read += RubySystem::getBlockSizeBytes()) {
00112             Request* req = nullptr;
00113             MemCmd::Command requestType;
00114 
00115             if (traceRecord->m_type == RubyRequestType_LD) {
00116                 requestType = MemCmd::ReadReq;
00117                 req = new Request(traceRecord->m_data_address + rec_bytes_read,
00118                     RubySystem::getBlockSizeBytes(), 0, Request::funcMasterId);
00119             }   else if (traceRecord->m_type == RubyRequestType_IFETCH) {
00120                 requestType = MemCmd::ReadReq;
00121                 req = new Request(traceRecord->m_data_address + rec_bytes_read,
00122                         RubySystem::getBlockSizeBytes(),
00123                         Request::INST_FETCH, Request::funcMasterId);
00124             }   else {
00125                 requestType = MemCmd::WriteReq;
00126                 req = new Request(traceRecord->m_data_address + rec_bytes_read,
00127                     RubySystem::getBlockSizeBytes(), 0, Request::funcMasterId);
00128             }
00129 
00130             Packet *pkt = new Packet(req, requestType);
00131             pkt->dataStatic(traceRecord->m_data + rec_bytes_read);
00132 
00133             Sequencer* m_sequencer_ptr = m_seq_map[traceRecord->m_cntrl_id];
00134             assert(m_sequencer_ptr != NULL);
00135             m_sequencer_ptr->makeRequest(pkt);
00136         }
00137 
00138         m_bytes_read += (sizeof(TraceRecord) + m_block_size_bytes);

00161 uint64
00162 CacheRecorder::aggregateRecords(uint8_t** buf, uint64 total_size)
00163 {
00164     std::sort(m_records.begin(), m_records.end(), compareTraceRecords);
00165 
00166     int size = m_records.size();
00167     uint64 current_size = 0;
00168     int record_size = sizeof(TraceRecord) + m_block_size_bytes;
00169 
00170     for (int i = 0; i < size; ++i) {
00171         // Determine if we need to expand the buffer size
00172         if (current_size + record_size > total_size) {
00173             uint8_t* new_buf = new (nothrow) uint8_t[total_size * 2];
00174             if (new_buf == NULL) {
00175                 fatal("Unable to allocate buffer of size %s\n",
00176                       total_size * 2);
00177             }
00178             total_size = total_size * 2;
00179             uint8_t* old_buf = *buf;
00180             memcpy(new_buf, old_buf, current_size);
00181             *buf = new_buf;
00182             delete [] old_buf;
00183         }
00184 
00185         // Copy the current record into the buffer
00186         memcpy(&((*buf)[current_size]), m_records[i], record_size);
00187         current_size += record_size;
00188 
00189         free(m_records[i]);
00190         m_records[i] = NULL;
00191     }
00192 
00193     m_records.clear();
00194     return current_size;

00270 Tick
00271 PciDevice::readConfig(PacketPtr pkt)
00272 {
00273     int offset = pkt->getAddr() & PCI_CONFIG_SIZE;
00274 
00275     /* Return 0 for accesses to unimplemented PCI configspace areas */
00276     if (offset >= PCI_DEVICE_SPECIFIC &&
00277         offset < PCI_CONFIG_SIZE) {
00278         warn_once("Device specific PCI config space "
00279                   "not implemented for %s!\n", this->name());
00280         switch (pkt->getSize()) {
00281             case sizeof(uint8_t):
00282                 pkt->set<uint8_t>(0);
00283                 break;
00284             case sizeof(uint16_t):
00285                 pkt->set<uint16_t>(0);
00286                 break;
00287             case sizeof(uint32_t):
00288                 pkt->set<uint32_t>(0);
00289                 break;
00290             default:
00291                 panic("invalid access size(?) for PCI configspace!\n");
00292         }
00293     } else if (offset > PCI_CONFIG_SIZE) {
00294         panic("Out-of-range access to PCI config space!\n");
00295     }
00296 
00297     switch (pkt->getSize()) {
00298       case sizeof(uint8_t):
00299         pkt->set<uint8_t>(config.data[offset]);
00300         DPRINTF(PCIDEV,
00301             "readConfig:  dev %#x func %#x reg %#x 1 bytes: data = %#x\n",
00302             params()->pci_dev, params()->pci_func, offset,
00303             (uint32_t)pkt->get<uint8_t>());
00304         break;
00305       case sizeof(uint16_t):
00306         pkt->set<uint16_t>(*(uint16_t*)&config.data[offset]);
00307         DPRINTF(PCIDEV,
00308             "readConfig:  dev %#x func %#x reg %#x 2 bytes: data = %#x\n",
00309             params()->pci_dev, params()->pci_func, offset,
00310             (uint32_t)pkt->get<uint16_t>());
00311         break;
00312       case sizeof(uint32_t):
00313         pkt->set<uint32_t>(*(uint32_t*)&config.data[offset]);
00314         DPRINTF(PCIDEV,
00315             "readConfig:  dev %#x func %#x reg %#x 4 bytes: data = %#x\n",
00316             params()->pci_dev, params()->pci_func, offset,
00317             (uint32_t)pkt->get<uint32_t>());
00318         break;
00319       default:
00320         panic("invalid access size(?) for PCI configspace!\n");
00321     }
00322     pkt->makeAtomicResponse();
00323     return configDelay;
00324

00338 Tick
00339 PciDevice::writeConfig(PacketPtr pkt)
00340 {
00341     int offset = pkt->getAddr() & PCI_CONFIG_SIZE;
00342 
00343     /* No effect if we write to config space that is not implemented*/
00344     if (offset >= PCI_DEVICE_SPECIFIC &&
00345         offset < PCI_CONFIG_SIZE) {
00346         warn_once("Device specific PCI config space "
00347                   "not implemented for %s!\n", this->name());
00348         switch (pkt->getSize()) {
00349             case sizeof(uint8_t):
00350             case sizeof(uint16_t):
00351             case sizeof(uint32_t):
00352                 break;
00353             default:
00354                 panic("invalid access size(?) for PCI configspace!\n");
00355         }
00356     } else if (offset > PCI_CONFIG_SIZE) {
00357         panic("Out-of-range access to PCI config space!\n");
00358     }
00359 
00360     switch (pkt->getSize()) {
00361       case sizeof(uint8_t):
00362         switch (offset) {
00363           case PCI0_INTERRUPT_LINE:
00364             config.interruptLine = pkt->get<uint8_t>();
00365             break;
00366           case PCI_CACHE_LINE_SIZE:
00367             config.cacheLineSize = pkt->get<uint8_t>();
00368             break;
00369           case PCI_LATENCY_TIMER:
00370             config.latencyTimer = pkt->get<uint8_t>();
00371             break;
00372           /* Do nothing for these read-only registers */
00373           case PCI0_INTERRUPT_PIN:
00374           case PCI0_MINIMUM_GRANT:
00375           case PCI0_MAXIMUM_LATENCY:
00376           case PCI_CLASS_CODE:
00377           case PCI_REVISION_ID:
00378             break;
00379           default:
00380             panic("writing to a read only register");
00381         }
00382         DPRINTF(PCIDEV,
00383             "writeConfig: dev %#x func %#x reg %#x 1 bytes: data = %#x\n",
00384             params()->pci_dev, params()->pci_func, offset,
00385             (uint32_t)pkt->get<uint8_t>());
00386         break;
00387       case sizeof(uint16_t):
00388         switch (offset) {
00389           case PCI_COMMAND:
00390             config.command = pkt->get<uint8_t>();
00391             break;
00392           case PCI_STATUS:
00393             config.status = pkt->get<uint8_t>();
00394             break;
00395           case PCI_CACHE_LINE_SIZE:
00396             config.cacheLineSize = pkt->get<uint8_t>();
00397             break;
00398           default:
00399             panic("writing to a read only register");
00400         }
00401         DPRINTF(PCIDEV,
00402             "writeConfig: dev %#x func %#x reg %#x 2 bytes: data = %#x\n",
00403             params()->pci_dev, params()->pci_func, offset,
00404             (uint32_t)pkt->get<uint16_t>());
00405         break;
00406       case sizeof(uint32_t):
00407         switch (offset) {
00408           case PCI0_BASE_ADDR0:
00409           case PCI0_BASE_ADDR1:
00410           case PCI0_BASE_ADDR2:
00411           case PCI0_BASE_ADDR3:
00412           case PCI0_BASE_ADDR4:
00413           case PCI0_BASE_ADDR5:
00414             {
00415                 int barnum = BAR_NUMBER(offset);
00416 
00417                 if (!legacyIO[barnum]) {
00418                     // convert BAR values to host endianness
00419                     uint32_t he_old_bar = letoh(config.baseAddr[barnum]);
00420                     uint32_t he_new_bar = letoh(pkt->get<uint32_t>());
00421 
00422                     uint32_t bar_mask =
00423                         BAR_IO_SPACE(he_old_bar) ? BAR_IO_MASK : BAR_MEM_MASK;
00424 
00425                     // Writing 0xffffffff to a BAR tells the card to set the
00426                     // value of the bar to a bitmask indicating the size of
00427                     // memory it needs
00428                     if (he_new_bar == 0xffffffff) {
00429                         he_new_bar = ~(BARSize[barnum] - 1);
00430                     } else {
00431                         // does it mean something special to write 0 to a BAR?
00432                         he_new_bar &= ~bar_mask;
00433                         if (he_new_bar) {
00434                             BARAddrs[barnum] = BAR_IO_SPACE(he_old_bar) ?
00435                                 platform->calcPciIOAddr(he_new_bar) :
00436                                 platform->calcPciMemAddr(he_new_bar);
00437                             pioPort.sendRangeChange();
00438                         }
00439                     }
00440                     config.baseAddr[barnum] = htole((he_new_bar & ~bar_mask) |
00441                                                     (he_old_bar & bar_mask));
00442                 }
00443             }
00444             break;
00445 
00446           case PCI0_ROM_BASE_ADDR:
00447             if (letoh(pkt->get<uint32_t>()) == 0xfffffffe)
00448                 config.expansionROM = htole((uint32_t)0xffffffff);
00449             else
00450                 config.expansionROM = pkt->get<uint32_t>();
00451             break;
00452 
00453           case PCI_COMMAND:
00454             // This could also clear some of the error bits in the Status
00455             // register. However they should never get set, so lets ignore
00456             // it for now
00457             config.command = pkt->get<uint32_t>();
00458             break;
00459 
00460           default:
00461             DPRINTF(PCIDEV, "Writing to a read only register");
00462         }
00463         DPRINTF(PCIDEV,
00464             "writeConfig: dev %#x func %#x reg %#x 4 bytes: data = %#x\n",
00465             params()->pci_dev, params()->pci_func, offset,
00466             (uint32_t)pkt->get<uint32_t>());
00467         break;
00468       default:
00469         panic("invalid access size(?) for PCI configspace!\n");
00470     }
00471     pkt->makeAtomicResponse();
00472     return configDelay;
00473 }

00503         int msix_array_size = msixcap_mxc_ts + 1;

00506             pba_array_size++;

00511

00514                      msix_table[i].fields.addr_lo);
00515             paramOut(os, csprintf("msix_table[%d].addr_hi", i),
00516                      msix_table[i].fields.addr_hi);
00517             paramOut(os, csprintf("msix_table[%d].msg_data", i),
00518                      msix_table[i].fields.msg_data);
00519             paramOut(os, csprintf("msix_table[%d].vec_ctrl", i),

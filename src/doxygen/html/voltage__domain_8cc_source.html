00063 VoltageDomain::perfLevel(PerfLevel perf_level)
00064 {
00065     chatty_assert(perf_level < voltageOpPoints.size(),
00066                   "DVFS: Requested voltage ID %d is outside the known "\
00067                   "range for domain %s.\n", perf_level, name());
00068 
00069     if (perf_level == _perfLevel) {
00070         // Silently ignore identical overwrites
00071         return;
00072     }
00073 
00074     _perfLevel = perf_level;
00075 
00076     DPRINTF(VoltageDomain, "Setting voltage to %.3fV idx: %d for domain %s\n",
00077             voltage(), perf_level, name());

00080 bool
00081 VoltageDomain::sanitiseVoltages()
00082 {
00083     if (numVoltages() == 1)
00084         return false;
00085 
00086     // Find the highest requested performance level and update the voltage
00087     // domain with it
00088     PerfLevel perf_max = (PerfLevel)-1;
00089     for (auto dit = srcClockChildren.begin(); dit != srcClockChildren.end(); ++dit) {
00090         SrcClockDomain* d = *dit;
00091         chatty_assert(d->voltageDomain() == this, "DVFS: Clock domain %s "\
00092                       "(id: %d) should not be registered with voltage domain "\
00093                       "%s\n", d->name(), d->domainID(), name());
00094 
00095         PerfLevel perf = d->perfLevel();
00096 
00097         DPRINTF(VoltageDomain, "DVFS: Clock domain %s (id: %d) requests perf "\
00098                 "level %d\n", d->name(), d->domainID(), perf);
00099 
00100         // NOTE: Descending sort of performance levels: 0 - fast, 5 - slow
00101         if (perf < perf_max) {
00102             DPRINTF(VoltageDomain, "DVFS: Updating max perf level %d -> %d\n",
00103                     perf_max, perf);
00104             perf_max = perf;
00105         }
00106     }
00107     DPRINTF(VoltageDomain, "DVFS: Setting perf level of voltage domain %s "\
00108             "from %d to %d.\n", name(), perfLevel(), perf_max);
00109 
00110     // Set the performance level
00111     if (perf_max != perfLevel()) {
00112         perfLevel(perf_max);
00113         return true;
00114     } else {
00115         return false;
00116     }
00117 }

00283     bool intersects(const AddrRange& r) const
00284     {
00285         if (_start > r._end || _end < r._start)
00286             // start with the simple case of no overlap at all,
00287             // applicable even if we have interleaved ranges
00288             return false;

00290             // if neither range is interleaved, we are done
00291             return true;

00293         // now it gets complicated, focus on the cases we care about

00269             i->second = new SequencerRequest(pkt, request_type, curCycle());

00271         } else {
00272             // There is an outstanding read request for the cache line
00273             m_load_waiting_on_load++;
00274             return RequestStatus_Aliased;
00275         }
00276     }

00330 bool
00331 Sequencer::handleLlsc(const Address& address, SequencerRequest* request)
00332 {
00333     //
00334     // The success flag indicates whether the LLSC operation was successful.
00335     // LL ops will always succeed, but SC may fail if the cache line is no
00336     // longer locked.
00337     //
00338     bool success = true;
00339     if (request->m_type == RubyRequestType_Store_Conditional) {
00340         if (!m_dataCache_ptr->isLocked(address, m_version)) {
00341             //
00342             // For failed SC requests, indicate the failure to the cpu by
00343             // setting the extra data to zero.
00344             //
00345             request->pkt->req->setExtraData(0);
00346             success = false;
00347         } else {
00348             //
00349             // For successful SC requests, indicate the success to the cpu by
00350             // setting the extra data to one.  
00351             //
00352             request->pkt->req->setExtraData(1);
00353         }
00354         //
00355         // Independent of success, all SC operations must clear the lock
00356         //
00357         m_dataCache_ptr->clearLocked(address);
00358     } else if (request->m_type == RubyRequestType_Load_Linked) {
00359         //
00360         // Note: To fully follow Alpha LLSC semantics, should the LL clear any
00361         // previously locked cache lines?
00362         //
00363         m_dataCache_ptr->setLocked(address, m_version);
00364     } else if ((m_dataCache_ptr->isTagPresent(address)) &&
00365                (m_dataCache_ptr->isLocked(address, m_version))) {
00366         //
00367         // Normal writes should clear the locked address
00368         //
00369         m_dataCache_ptr->clearLocked(address);
00370     }
00371     return success;
00372 }

00401                     forwardRequestTime - initialRequestTime);

00403                     firstResponseTime - forwardRequestTime);

00407                 m_IncompleteTimes[respondingMach]++;

00579 RequestStatus

00581 {

00583         return RequestStatus_BufferFull;
00584     }
00585 
00586     RubyRequestType primary_type = RubyRequestType_NULL;
00587     RubyRequestType secondary_type = RubyRequestType_NULL;
00588 
00589     if (pkt->isLLSC()) {
00590         //
00591         // Alpha LL/SC instructions need to be handled carefully by the cache
00592         // coherence protocol to ensure they follow the proper semantics. In
00593         // particular, by identifying the operations as atomic, the protocol
00594         // should understand that migratory sharing optimizations should not
00595         // be performed (i.e. a load between the LL and SC should not steal
00596         // away exclusive permission).

00598         if (pkt->isWrite()) {
00599             DPRINTF(RubySequencer, "Issuing SC\n");
00600             primary_type = RubyRequestType_Store_Conditional;
00601         } else {
00602             DPRINTF(RubySequencer, "Issuing LL\n");
00603             assert(pkt->isRead());
00604             primary_type = RubyRequestType_Load_Linked;
00605         }
00606         secondary_type = RubyRequestType_ATOMIC;
00607     } else if (pkt->req->isLockedRMW()) {
00608         //
00609         // x86 locked instructions are translated to store cache coherence
00610         // requests because these requests should always be treated as read
00611         // exclusive operations and should leverage any migratory sharing
00612         // optimization built into the protocol.
00613         //
00614         if (pkt->isWrite()) {
00615             DPRINTF(RubySequencer, "Issuing Locked RMW Write\n");
00616             primary_type = RubyRequestType_Locked_RMW_Write;
00617         } else {
00618             DPRINTF(RubySequencer, "Issuing Locked RMW Read\n");
00619             assert(pkt->isRead());
00620             primary_type = RubyRequestType_Locked_RMW_Read;
00621         }
00622         secondary_type = RubyRequestType_ST;
00623     } else {
00624         if (pkt->isRead()) {
00625             if (pkt->req->isInstFetch()) {
00626                 primary_type = secondary_type = RubyRequestType_IFETCH;
00627             } else {
00628                 bool storeCheck = false;
00629                 // only X86 need the store check
00630                 if (system->getArch() == Arch::X86ISA) {
00631                     uint32_t flags = pkt->req->getFlags();
00632                     storeCheck = flags &
00633                         (X86ISA::StoreCheck << X86ISA::FlagShift);
00634                 }
00635                 if (storeCheck) {
00636                     primary_type = RubyRequestType_RMW_Read;
00637                     secondary_type = RubyRequestType_ST;
00638                 } else {
00639                     primary_type = secondary_type = RubyRequestType_LD;
00640                 }
00641             }
00642         } else if (pkt->isWrite()) {
00643             //
00644             // Note: M5 packets do not differentiate ST from RMW_Write
00645             //
00646             primary_type = secondary_type = RubyRequestType_ST;
00647         } else if (pkt->isFlush()) {
00648           primary_type = secondary_type = RubyRequestType_FLUSH;
00649         } else {
00650             panic("Unsupported ruby packet type\n");
00651         }
00652     }
00653 
00654     RequestStatus status = insertRequest(pkt, primary_type);
00655     if (status != RequestStatus_Ready)
00656         return status;
00657 
00658     issueRequest(pkt, secondary_type);
00659 
00660     // TODO: issue hardware prefetches here
00661     return RequestStatus_Issued;
00662 }

00664 void
00665 Sequencer::issueRequest(PacketPtr pkt, RubyRequestType secondary_type)
00666 {
00667     assert(pkt != NULL);
00668     int proc_id = -1;
00669     if (pkt->req->hasContextId()) {
00670         proc_id = pkt->req->contextId();
00671     }
00672 
00673     // If valid, copy the pc to the ruby request
00674     Addr pc = 0;
00675     if (pkt->req->hasPC()) {
00676         pc = pkt->req->getPC();
00677     }
00678 
00679     // check if the packet has data as for example prefetch and flush
00680     // requests do not
00681     std::shared_ptr<RubyRequest> msg =
00682         std::make_shared<RubyRequest>(clockEdge(), pkt->getAddr(),
00683                                       pkt->isFlush() ?
00684                                       nullptr : pkt->getPtr<uint8_t>(),
00685                                       pkt->getSize(), pc, secondary_type,
00686                                       RubyAccessMode_Supervisor, pkt,
00687                                       PrefetchBit_No, proc_id);
00688 
00689     DPRINTFR(ProtocolTrace, "%15s %3s %10s%20s %6s>%-6s %s %s\n",
00690             curTick(), m_version, "Seq", "Begin", "", "",
00691             msg->getPhysicalAddress(),
00692             RubyRequestType_to_string(secondary_type));
00693 
00694     Cycles latency(0);  // initialzed to an null value
00695 
00696     if (secondary_type == RubyRequestType_IFETCH)
00697         latency = m_instCache_ptr->getLatency();
00698     else
00699         latency = m_dataCache_ptr->getLatency();
00700 
00701     // Send the message to the cache controller
00702     assert(latency > 0);
00703 
00704     assert(m_mandatory_q_ptr != NULL);
00705     m_mandatory_q_ptr->enqueue(msg, latency);
00706 }

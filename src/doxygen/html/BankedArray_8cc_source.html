00050 bool
00051 BankedArray::tryAccess(int64 idx)
00052 {
00053     if (accessLatency == 0)
00054         return true;
00055 
00056     unsigned int bank = mapIndexToBank(idx);
00057     assert(bank < banks);
00058 
00059     if (busyBanks[bank].endAccess >= curTick()) {
00060         if (!(busyBanks[bank].startAccess == curTick() &&
00061             busyBanks[bank].idx == idx)) {
00062             return false;
00063         } else {
00064             // We tried to allocate resources twice
00065             // in the same cycle for the same addr
00066             return true;
00067         }
00068     }
00069 
00070     busyBanks[bank].idx = idx;
00071     busyBanks[bank].startAccess = curTick();
00072     busyBanks[bank].endAccess = curTick() +
00073         (accessLatency-1) * g_system_ptr->clockPeriod();
00074 
00075     return true;
00076 }

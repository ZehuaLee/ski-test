00737 static void
00738 forceSegAccessed(struct kvm_segment &seg)
00739 {

00743     // gem5 did set the access bits correctly, so we force it to one

00751 {
00752     struct kvm_sregs sregs;
00753 
00754 #define APPLY_SREG(kreg, mreg) sregs.kreg = tc->readMiscRegNoEffect(mreg)
00755 #define APPLY_SEGMENT(kreg, idx) setKvmSegmentReg(tc, sregs.kreg, idx)
00756 #define APPLY_DTABLE(kreg, idx) setKvmDTableReg(tc, sregs.kreg, idx)

00758     FOREACH_SREG();

00760     FOREACH_DTABLE();
00761 
00762 #undef APPLY_SREG
00763 #undef APPLY_SEGMENT
00764 #undef APPLY_DTABLE
00765 
00766     // Clear the interrupt bitmap
00767     memset(&sregs.interrupt_bitmap, 0, sizeof(sregs.interrupt_bitmap));
00768 
00769     // VMX requires CS, SS, DS, ES, FS, and GS to have the accessed
00770     // bit in the type field set.
00771     forceSegAccessed(sregs.cs);
00772     forceSegAccessed(sregs.ss);
00773     forceSegAccessed(sregs.ds);
00774     forceSegAccessed(sregs.es);
00775     forceSegAccessed(sregs.fs);
00776     forceSegAccessed(sregs.gs);
00777 
00778     // There are currently some cases where the active task isn't
00779     // marked as busy. This is illegal in VMX, so we force it to busy.
00780     if (sregs.tr.type == SEG_SYS_TYPE_TSS_AVAILABLE) {
00781         hack("tr.type (%i) is not busy. Forcing the busy bit.\n",
00782              sregs.tr.type);
00783         sregs.tr.type = SEG_SYS_TYPE_TSS_BUSY;
00784     }
00785 
00786     // VMX requires the DPL of SS and CS to be the same for
00787     // non-conforming code segments. It seems like m5 doesn't set the
00788     // DPL of SS correctly when taking interrupts, so we need to fix
00789     // that here.
00790     if ((sregs.cs.type == SEG_CS_TYPE_ACCESSED ||
00791          sregs.cs.type == SEG_CS_TYPE_READ_ACCESSED) &&
00792         sregs.cs.dpl != sregs.ss.dpl) {
00793 
00794         hack("CS.DPL (%i) != SS.DPL (%i): Forcing SS.DPL to %i\n",
00795              sregs.cs.dpl, sregs.ss.dpl, sregs.cs.dpl);
00796         sregs.ss.dpl = sregs.cs.dpl;
00797     }

00799     // Do checks after fixing up the state to avoid getting excessive
00800     // amounts of warnings.

00802     if (!rflags_nocc.vm) {
00803         // Do segment verification if the CPU isn't entering virtual
00804         // 8086 mode.  We currently assume that unrestricted guest
00805         // mode is available.
00806 
00807 #define APPLY_SEGMENT(kreg, idx) \
00808         checkSeg(# kreg, idx + MISCREG_SEG_SEL_BASE, sregs.kreg, sregs)
00809 
00810         FOREACH_SEGMENT();
00811 #undef APPLY_SEGMENT
00812     }
00813 
00814     setSpecialRegisters(sregs);
00815 }

01132 void
01133 X86KvmCPU::deliverInterrupts()
01134 {
01135     Fault fault;
01136 
01137     syncThreadContext();
01138 
01139     {
01140         // Migrate to the interrupt controller's thread to get the
01141         // interrupt. Even though the individual methods are safe to
01142         // call across threads, we might still lose interrupts unless
01143         // they are getInterrupt() and updateIntrInfo() are called
01144         // atomically.
01145         EventQueue::ScopedMigration migrate(interrupts->eventQueue());
01146         fault = interrupts->getInterrupt(tc);
01147         interrupts->updateIntrInfo(tc);
01148     }
01149 
01150     X86Interrupt *x86int(dynamic_cast<X86Interrupt *>(fault.get()));
01151     if (dynamic_cast<NonMaskableInterrupt *>(fault.get())) {
01152         DPRINTF(KvmInt, "Delivering NMI\n");
01153         kvmNonMaskableInterrupt();
01154     } else if (dynamic_cast<InitInterrupt *>(fault.get())) {
01155         DPRINTF(KvmInt, "INIT interrupt\n");
01156         fault.get()->invoke(tc);
01157         // Delay the kvm state update since we won't enter KVM on this
01158         // tick.
01159         threadContextDirty = true;
01160         // HACK: gem5 doesn't actually have any BIOS code, which means
01161         // that we need to halt the thread and wait for a startup
01162         // interrupt before restarting the thread. The simulated CPUs
01163         // use the same kind of hack using a microcode routine.
01164         thread->suspend();
01165     } else if (dynamic_cast<StartupInterrupt *>(fault.get())) {
01166         DPRINTF(KvmInt, "STARTUP interrupt\n");
01167         fault.get()->invoke(tc);
01168         // The kvm state is assumed to have been updated when entering

01186 X86KvmCPU::kvmRun(Tick ticks)
01187 {
01188     struct kvm_run &kvm_run(*getKvmRunState());
01189 
01190     if (interrupts->checkInterruptsRaw()) {
01191         if (interrupts->hasPendingUnmaskable()) {
01192             DPRINTF(KvmInt,
01193                     "Delivering unmaskable interrupt.\n");
01194             syncThreadContext();
01195             deliverInterrupts();
01196         } else if (kvm_run.ready_for_interrupt_injection) {
01197             // KVM claims that it is ready for an interrupt. It might
01198             // be lying if we just updated rflags and disabled
01199             // interrupts (e.g., by doing a CPU handover). Let's sync
01200             // the thread context and check if there are /really/
01201             // interrupts that should be delivered now.
01202             syncThreadContext();

01204                 DPRINTF(KvmInt,
01205                         "M5 has pending interrupts, delivering interrupt.\n");
01206

01209                 DPRINTF(KvmInt,
01210                         "Interrupt delivery delayed due to KVM confusion.\n");
01211                 kvm_run.request_interrupt_window = 1;
01212             }
01213         } else if (!kvm_run.request_interrupt_window) {
01214             DPRINTF(KvmInt,
01215                     "M5 has pending interrupts, requesting interrupt "
01216                     "window.\n");
01217             kvm_run.request_interrupt_window = 1;
01218         }
01219     } else {
01220         kvm_run.request_interrupt_window = 0;
01221     }
01222 
01223     // The CPU might have been suspended as a result of the INIT
01224     // interrupt delivery hack. In that case, don't enter into KVM.
01225     if (_status == Idle)
01226         return 0;
01227     else

01306 Tick
01307 X86KvmCPU::handleKvmExitIO()
01308 {
01309     struct kvm_run &kvm_run(*getKvmRunState());
01310     bool isWrite(kvm_run.io.direction == KVM_EXIT_IO_OUT);
01311     unsigned char *guestData(getGuestData(kvm_run.io.data_offset));
01312     Tick delay(0);
01313     uint16_t port(kvm_run.io.port);
01314     Addr pAddr;
01315     const int count(kvm_run.io.count);
01316 
01317     assert(kvm_run.io.direction == KVM_EXIT_IO_IN ||
01318            kvm_run.io.direction == KVM_EXIT_IO_OUT);
01319 
01320     DPRINTF(KvmIO, "KVM-x86: Handling IO instruction (%s) (port: 0x%x)\n",
01321             (isWrite ? "out" : "in"), kvm_run.io.port);
01322 
01323     /* Vanilla gem5 handles PCI discovery in the TLB(!). Since we
01324      * don't use the TLB component, we need to intercept and handle
01325      * the PCI configuration space IO ports here.
01326      *
01327      * The IO port PCI discovery mechanism uses one address register
01328      * and one data register. We map the address register to a misc
01329      * reg and use that to re-route data register accesses to the
01330      * right location in the PCI configuration space.
01331      */
01332     if (port == IO_PCI_CONF_ADDR) {
01333         handleIOMiscReg32(MISCREG_PCI_CONFIG_ADDRESS);
01334         return 0;
01335     } else if ((port & ~0x3) == IO_PCI_CONF_DATA_BASE) {
01336         Addr pciConfigAddr(tc->readMiscRegNoEffect(MISCREG_PCI_CONFIG_ADDRESS));
01337         if (pciConfigAddr & 0x80000000) {
01338             pAddr = X86ISA::x86PciConfigAddress((pciConfigAddr & 0x7ffffffc) |
01339                                                 (port & 0x3));
01340         } else {
01341             pAddr = X86ISA::x86IOAddress(port);
01342         }
01343     } else {
01344         pAddr = X86ISA::x86IOAddress(port);
01345     }
01346 
01347     Request io_req(pAddr, kvm_run.io.size, Request::UNCACHEABLE,
01348                    dataMasterId());
01349     io_req.setThreadContext(tc->contextId(), 0);
01350 
01351     const MemCmd cmd(isWrite ? MemCmd::WriteReq : MemCmd::ReadReq);
01352     // Temporarily lock and migrate to the event queue of the
01353     // VM. This queue is assumed to "own" all devices we need to
01354     // access if running in multi-core mode.
01355     EventQueue::ScopedMigration migrate(vm.eventQueue());
01356     for (int i = 0; i < count; ++i) {
01357         Packet pkt(&io_req, cmd);
01358 
01359         pkt.dataStatic(guestData);
01360         delay += dataPort.sendAtomic(&pkt);
01361 
01362         guestData += kvm_run.io.size;
01363     }
01364 
01365     return delay;
01366 }

00310     int index_op = 1;
00311     int index_val = 2;
00312     int index_timeout = 3;
00313 
00314     uint64_t uaddr = process->getSyscallArg(tc, index_uaddr);
00315     int op = process->getSyscallArg(tc, index_op);
00316     int val = process->getSyscallArg(tc, index_val);
00317     uint64_t timeout = process->getSyscallArg(tc, index_timeout);
00318 
00319     std::map<uint64_t, std::list<ThreadContext *> * >
00320         &futex_map = tc->getSystemPtr()->futexMap;
00321 
00322     DPRINTF(SyscallVerbose, "In sys_futex: Address=%llx, op=%d, val=%d\n",
00323             uaddr, op, val);
00324 
00325     op &= ~OS::TGT_FUTEX_PRIVATE_FLAG;
00326 
00327     if (op == OS::TGT_FUTEX_WAIT) {
00328         if (timeout != 0) {
00329             warn("sys_futex: FUTEX_WAIT with non-null timeout unimplemented;"
00330                  "we'll wait indefinitely");
00331         }
00332 
00333         uint8_t *buf = new uint8_t[sizeof(int)];
00334         tc->getMemProxy().readBlob((Addr)uaddr, buf, (int)sizeof(int));
00335         int mem_val = *((int *)buf);
00336         delete buf;
00337 
00338         if(val != mem_val) {
00339             DPRINTF(SyscallVerbose, "sys_futex: FUTEX_WAKE, read: %d, "
00340                                     "expected: %d\n", mem_val, val);
00341             return -OS::TGT_EWOULDBLOCK;
00342         }
00343 
00344         // Queue the thread context
00345         std::list<ThreadContext *> * tcWaitList;
00346         if (futex_map.count(uaddr)) {
00347             tcWaitList = futex_map.find(uaddr)->second;
00348         } else {
00349             tcWaitList = new std::list<ThreadContext *>();
00350             futex_map.insert(std::pair< uint64_t,
00351                             std::list<ThreadContext *> * >(uaddr, tcWaitList));
00352         }
00353         tcWaitList->push_back(tc);
00354         DPRINTF(SyscallVerbose, "sys_futex: FUTEX_WAIT, suspending calling "
00355                                 "thread context\n");
00356         tc->suspend();
00357         return 0;
00358     } else if (op == OS::TGT_FUTEX_WAKE){
00359         int wokenUp = 0;
00360         std::list<ThreadContext *> * tcWaitList;
00361         if (futex_map.count(uaddr)) {
00362             tcWaitList = futex_map.find(uaddr)->second;
00363             while (tcWaitList->size() > 0 && wokenUp < val) {
00364                 tcWaitList->front()->activate();
00365                 tcWaitList->pop_front();
00366                 wokenUp++;
00367             }
00368             if(tcWaitList->empty()) {
00369                 futex_map.erase(uaddr);
00370                 delete tcWaitList;
00371             }
00372         }
00373         DPRINTF(SyscallVerbose, "sys_futex: FUTEX_WAKE, activated %d waiting "
00374                                 "thread contexts\n", wokenUp);
00375         return wokenUp;
00376     } else {
00377         warn("sys_futex: op %d is not implemented, just returning...", op);
00378         return 0;
00379     }

00833 template <class OS>
00834 SyscallReturn
00835 mremapFunc(SyscallDesc *desc, int callnum, LiveProcess *process, ThreadContext *tc)
00836 {
00837     int index = 0;
00838     Addr start = process->getSyscallArg(tc, index);
00839     uint64_t old_length = process->getSyscallArg(tc, index);
00840     uint64_t new_length = process->getSyscallArg(tc, index);
00841     uint64_t flags = process->getSyscallArg(tc, index);
00842     uint64_t provided_address = 0;
00843     bool use_provided_address = flags & OS::TGT_MREMAP_FIXED;
00844 
00845     if (use_provided_address)
00846         provided_address = process->getSyscallArg(tc, index);
00847 
00848     if ((start % TheISA::PageBytes != 0) ||
00849         (provided_address % TheISA::PageBytes != 0)) {
00850         warn("mremap failing: arguments not page aligned");
00851         return -EINVAL;
00852     }
00853 
00854     new_length = roundUp(new_length, TheISA::PageBytes);
00855 
00856     if (new_length > old_length) {
00857         if ((start + old_length) == process->mmap_end &&
00858             (!use_provided_address || provided_address == start)) {
00859             uint64_t diff = new_length - old_length;
00860             process->allocateMem(process->mmap_end, diff);
00861             process->mmap_end += diff;
00862             return start;
00863         } else {
00864             if (!use_provided_address && !(flags & OS::TGT_MREMAP_MAYMOVE)) {
00865                 warn("can't remap here and MREMAP_MAYMOVE flag not set\n");
00866                 return -ENOMEM;
00867             } else {
00868                 uint64_t new_start = use_provided_address ?
00869                     provided_address : process->mmap_end;
00870                 process->pTable->remap(start, old_length, new_start);
00871                 warn("mremapping to new vaddr %08p-%08p, adding %d\n",
00872                      new_start, new_start + new_length,
00873                      new_length - old_length);
00874                 // add on the remaining unallocated pages
00875                 process->allocateMem(new_start + old_length,
00876                                      new_length - old_length,
00877                                      use_provided_address /* clobber */);
00878                 if (!use_provided_address)
00879                     process->mmap_end += new_length;
00880                 if (use_provided_address &&
00881                     new_start + new_length > process->mmap_end) {
00882                     // something fishy going on here, at least notify the user
00883                     // @todo: increase mmap_end?
00884                     warn("mmap region limit exceeded with MREMAP_FIXED\n");
00885                 }
00886                 warn("returning %08p as start\n", new_start);
00887                 return new_start;
00888             }
00889         }
00890     } else {
00891         if (use_provided_address && provided_address != start)
00892             process->pTable->remap(start, new_length, provided_address);
00893         process->pTable->unmap(start + new_length, old_length - new_length);
00894         return use_provided_address ? provided_address : start;
00895     }
00896 }

01222 template <class OS>
01223 SyscallReturn
01224 mmapFunc(SyscallDesc *desc, int num, LiveProcess *p, ThreadContext *tc)
01225 {
01226     int index = 0;
01227     Addr start = p->getSyscallArg(tc, index);
01228     uint64_t length = p->getSyscallArg(tc, index);
01229     index++; // int prot = p->getSyscallArg(tc, index);
01230     int flags = p->getSyscallArg(tc, index);
01231     int tgt_fd = p->getSyscallArg(tc, index);
01232     int offset = p->getSyscallArg(tc, index);
01233 
01234     if (length > 0x100000000ULL)
01235         warn("mmap length argument %#x is unreasonably large.\n", length);
01236 
01237     if (!(flags & OS::TGT_MAP_ANONYMOUS)) {
01238         Process::FdMap *fd_map = p->sim_fd_obj(tgt_fd);
01239         if (!fd_map || fd_map->fd < 0) {
01240             warn("mmap failing: target fd %d is not valid\n", tgt_fd);
01241             return -EBADF;
01242         }
01243 
01244         if (fd_map->filename != "/dev/zero") {
01245             // This is very likely broken, but leave a warning here
01246             // (rather than panic) in case /dev/zero is known by
01247             // another name on some platform
01248             warn("allowing mmap of file %s; mmap not supported on files"
01249                  " other than /dev/zero\n", fd_map->filename);
01250         }
01251     }
01252 
01253     length = roundUp(length, TheISA::PageBytes);
01254 
01255     if ((start  % TheISA::PageBytes) != 0 ||
01256         (offset % TheISA::PageBytes) != 0) {
01257         warn("mmap failing: arguments not page-aligned: "
01258              "start 0x%x offset 0x%x",
01259              start, offset);
01260         return -EINVAL;
01261     }
01262 
01263     // are we ok with clobbering existing mappings?  only set this to
01264     // true if the user has been warned.
01265     bool clobber = false;
01266 
01267     // try to use the caller-provided address if there is one
01268     bool use_provided_address = (start != 0);
01269 
01270     if (use_provided_address) {
01271         // check to see if the desired address is already in use
01272         if (!p->pTable->isUnmapped(start, length)) {
01273             // there are existing mappings in the desired range
01274             // whether we clobber them or not depends on whether the caller
01275             // specified MAP_FIXED
01276             if (flags & OS::TGT_MAP_FIXED) {
01277                 // MAP_FIXED specified: map attempt fails
01278                 return -EINVAL;
01279             } else {
01280                 // MAP_FIXED not specified: ignore suggested start address
01281                 warn("mmap: ignoring suggested map address 0x%x\n", start);
01282                 use_provided_address = false;
01283             }
01284         }
01285     }
01286 
01287     if (!use_provided_address) {
01288         // no address provided, or provided address unusable:
01289         // pick next address from our "mmap region"
01290         if (OS::mmapGrowsDown()) {
01291             start = p->mmap_end - length;
01292             p->mmap_end = start;
01293         } else {
01294             start = p->mmap_end;
01295             p->mmap_end += length;
01296         }
01297     }
01298 
01299     p->allocateMem(start, length, clobber);
01300 
01301     return start;
01302 }

00109     assert(!responseQueue.empty());
00110 
00111     DPRINTF(DRAMSim2, "Attempting to send response\n");
00112 
00113     bool success = port.sendTimingResp(responseQueue.front());
00114     if (success) {
00115         responseQueue.pop_front();

00121         if (!responseQueue.empty() && !sendResponseEvent.scheduled())

00182 bool
00183 DRAMSim2::recvTimingReq(PacketPtr pkt)
00184 {
00185     // we should never see a new request while in retry
00186     assert(!retryReq);
00187 
00188     // @todo temporary hack to deal with memory corruption issues until
00189     // 4-phase transactions are complete
00190     for (int x = 0; x < pendingDelete.size(); x++)
00191         delete pendingDelete[x];
00192     pendingDelete.clear();
00193 
00194     if (pkt->memInhibitAsserted()) {
00195         // snooper will supply based on copy of packet
00196         // still target's responsibility to delete packet
00197         pendingDelete.push_back(pkt);
00198         return true;
00199     }
00200 
00201     // if we cannot accept we need to send a retry once progress can
00202     // be made
00203     bool can_accept = nbrOutstanding() < wrapper.queueSize();
00204 
00205     // keep track of the transaction
00206     if (pkt->isRead()) {
00207         if (can_accept) {
00208             outstandingReads[pkt->getAddr()].push(pkt);
00209 
00210             // we count a transaction as outstanding until it has left the
00211             // queue in the controller, and the response has been sent
00212             // back, note that this will differ for reads and writes
00213             ++nbrOutstandingReads;
00214         }
00215     } else if (pkt->isWrite()) {
00216         if (can_accept) {
00217             outstandingWrites[pkt->getAddr()].push(pkt);
00218 
00219             ++nbrOutstandingWrites;
00220 
00221             // perform the access for writes
00222             accessAndRespond(pkt);
00223         }
00224     } else {
00225         // keep it simple and just respond if necessary
00226         accessAndRespond(pkt);
00227         return true;
00228     }
00229 
00230     if (can_accept) {
00231         // we should never have a situation when we think there is space,
00232         // and there isn't
00233         assert(wrapper.canAccept());
00234 
00235         DPRINTF(DRAMSim2, "Enqueueing address %lld\n", pkt->getAddr());
00236 
00237         // @todo what about the granularity here, implicit assumption that
00238         // a transaction matches the burst size of the memory (which we
00239         // cannot determine without parsing the ini file ourselves)
00240         wrapper.enqueue(pkt->isWrite(), pkt->getAddr());
00241 
00242         return true;
00243     } else {
00244         retryReq = true;
00245         return false;
00246     }
00247 }

00259 void
00260 DRAMSim2::accessAndRespond(PacketPtr pkt)
00261 {
00262     DPRINTF(DRAMSim2, "Access for address %lld\n", pkt->getAddr());
00263 
00264     bool needsResponse = pkt->needsResponse();
00265 
00266     // do the actual memory access which also turns the packet into a
00267     // response
00268     access(pkt);
00269 
00270     // turn packet around to go back to requester if response expected
00271     if (needsResponse) {
00272         // access already turned the packet into a response
00273         assert(pkt->isResponse());
00274         // Here we pay for xbar additional delay and to process the payload
00275         // of the packet.
00276         Tick time = curTick() + pkt->headerDelay + pkt->payloadDelay;
00277         // Reset the timings of the packet
00278         pkt->headerDelay = pkt->payloadDelay = 0;
00279 
00280         DPRINTF(DRAMSim2, "Queuing response for address %lld\n",
00281                 pkt->getAddr());
00282 
00283         // queue it to be sent back
00284         responseQueue.push_back(pkt);
00285 
00286         // if we are not already waiting for a retry, or are scheduled
00287         // to send a response, schedule an event
00288         if (!retryResp && !sendResponseEvent.scheduled())
00289             schedule(sendResponseEvent, time);
00290     } else {
00291         // @todo the packet is going to be deleted, and the DRAMPacket
00292         // is still having a pointer to it
00293         pendingDelete.push_back(pkt);
00294     }
00295 }

00297 void DRAMSim2::readComplete(unsigned id, uint64_t addr, uint64_t cycle)
00298 {
00299     assert(cycle == divCeil(curTick() - startTick,
00300                             wrapper.clockPeriod() * SimClock::Int::ns));
00301 
00302     DPRINTF(DRAMSim2, "Read to address %lld complete\n", addr);
00303 
00304     // get the outstanding reads for the address in question
00305     auto p = outstandingReads.find(addr);
00306     assert(p != outstandingReads.end());
00307 
00308     // first in first out, which is not necessarily true, but it is
00309     // the best we can do at this point
00310     PacketPtr pkt = p->second.front();
00311     p->second.pop();
00312 
00313     if (p->second.empty())
00314         outstandingReads.erase(p);
00315 
00316     // no need to check for drain here as the next call will add a
00317     // response to the response queue straight away
00318     assert(nbrOutstandingReads != 0);
00319     --nbrOutstandingReads;
00320 
00321     // perform the actual memory access
00322     accessAndRespond(pkt);
00323 }

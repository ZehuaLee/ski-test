00163     assert(pkt->isRequest());
00164 
00165     assert(blk && blk->isValid());

00167     // satisfying a string of Read and ReadEx requests from
00168     // upper-level caches, a Read will mark the block as shared but we
00169     // can satisfy a following ReadEx anyway since we can rely on the
00170     // Read requester(s) to have buffered the ReadEx snoop and to
00171     // invalidate their blocks after receiving them.
00172     // assert(!pkt->needsExclusive() || blk->isWritable());
00173     assert(pkt->getOffset(blkSize) + pkt->getSize() <= blkSize);
00174 
00175     // Check RMW operations first since both isRead() and
00176     // isWrite() will be true for them
00177     if (pkt->cmd == MemCmd::SwapReq) {
00178         cmpAndSwap(blk, pkt);
00179     } else if (pkt->isWrite() &&
00180                (!pkt->isWriteInvalidate() || isTopLevel)) {
00181         assert(blk->isWritable());
00182         // Write or WriteInvalidate at the first cache with block in Exclusive
00183         if (blk->checkWrite(pkt)) {
00184             pkt->writeDataToBlock(blk->data, blkSize);
00185         }
00186         // Always mark the line as dirty even if we are a failed
00187         // StoreCond so we supply data to any snoops that have
00188         // appended themselves to this cache before knowing the store
00189         // will fail.
00190         blk->status |= BlkDirty;
00191         DPRINTF(Cache, "%s for %s addr %#llx size %d (write)\n", __func__,
00192                 pkt->cmdString(), pkt->getAddr(), pkt->getSize());
00193     } else if (pkt->isRead()) {
00194         if (pkt->isLLSC()) {
00195             blk->trackLoadLocked(pkt);
00196         }
00197         pkt->setDataFromBlock(blk->data, blkSize);
00198         if (pkt->getSize() == blkSize) {
00199             // special handling for coherent block requests from
00200             // upper-level caches
00201             if (pkt->needsExclusive()) {
00202                 // if we have a dirty copy, make sure the recipient
00203                 // keeps it marked dirty
00204                 if (blk->isDirty()) {
00205                     pkt->assertMemInhibit();
00206                 }
00207                 // on ReadExReq we give up our copy unconditionally
00208                 if (blk != tempBlock)
00209                     tags->invalidate(blk);
00210                 blk->invalidate();
00211             } else if (blk->isWritable() && !pending_downgrade
00212                       && !pkt->sharedAsserted() && !pkt->req->isInstFetch()) {
00213                 // we can give the requester an exclusive copy (by not
00214                 // asserting shared line) on a read request if:
00215                 // - we have an exclusive copy at this level (& below)
00216                 // - we don't have a pending snoop from below
00217                 //   signaling another read request
00218                 // - no other cache above has a copy (otherwise it
00219                 //   would have asseretd shared line on request)
00220                 // - we are not satisfying an instruction fetch (this
00221                 //   prevents dirty data in the i-cache)
00222 
00223                 if (blk->isDirty()) {
00224                     // special considerations if we're owner:
00225                     if (!deferred_response && !isTopLevel) {
00226                         // if we are responding immediately and can
00227                         // signal that we're transferring ownership
00228                         // along with exclusivity, do so
00229                         pkt->assertMemInhibit();
00230                         blk->status &= ~BlkDirty;
00231                     } else {
00232                         // if we're responding after our own miss,
00233                         // there's a window where the recipient didn't
00234                         // know it was getting ownership and may not
00235                         // have responded to snoops correctly, so we
00236                         // can't pass off ownership *or* exclusivity
00237                         pkt->assertShared();
00238                     }
00239                 }
00240             } else {
00241                 // otherwise only respond with a shared copy
00242                 pkt->assertShared();
00243             }
00244         }
00245     } else {
00246         // Upgrade or WriteInvalidate at a different cache than received it.
00247         // Since we have it Exclusively (E or M), we ack then invalidate.
00248         assert(pkt->isUpgrade() ||
00249                (pkt->isWriteInvalidate() && !isTopLevel));
00250         assert(blk != tempBlock);
00251         tags->invalidate(blk);
00252         blk->invalidate();
00253         DPRINTF(Cache, "%s for %s addr %#llx size %d (invalidation)\n",
00254                 __func__, pkt->cmdString(), pkt->getAddr(), pkt->getSize());
00255     }

00623         // uncacheable request
00624         MSHR *mshr = pkt->req->isUncacheable() ? nullptr :
00625             mshrQueue.findMatch(blk_addr, pkt->isSecure());
00626 
00627         // Software prefetch handling:
00628         // To keep the core from waiting on data it won't look at
00629         // anyway, send back a response with dummy data. Miss handling
00630         // will continue asynchronously. Unfortunately, the core will
00631         // insist upon freeing original Packet/Request, so we have to
00632         // create a new pair with a different lifecycle. Note that this
00633         // processing happens before any MSHR munging on the behalf of
00634         // this request because this new Request will be the one stored
00635         // into the MSHRs, not the original.
00636         if (pkt->cmd.isSWPrefetch() && isTopLevel) {
00637             assert(needsResponse);
00638             assert(pkt->req->hasPaddr());
00639             assert(!pkt->req->isUncacheable());
00640 
00641             // There's no reason to add a prefetch as an additional target
00642             // to an existing MSHR. If an outstanding request is already
00643             // in progress, there is nothing for the prefetch to do.
00644             // If this is the case, we don't even create a request at all.
00645             PacketPtr pf = nullptr;
00646 
00647             if (!mshr) {
00648                 // copy the request and create a new SoftPFReq packet
00649                 RequestPtr req = new Request(pkt->req->getPaddr(),
00650                                              pkt->req->getSize(),
00651                                              pkt->req->getFlags(),
00652                                              pkt->req->masterId());
00653                 pf = new Packet(req, pkt->cmd);
00654                 pf->allocate();
00655                 assert(pf->getAddr() == pkt->getAddr());
00656                 assert(pf->getSize() == pkt->getSize());
00657             }
00658 
00659             pkt->makeTimingResponse();
00660             // for debugging, set all the bits in the response data
00661             // (also keeps valgrind from complaining when debugging settings
00662             //  print out instruction results)
00663             std::memset(pkt->getPtr<uint8_t>(), 0xFF, pkt->getSize());
00664             // request_time is used here, taking into account lat and the delay
00665             // charged if the packet comes from the xbar.
00666             cpuSidePort->schedTimingResp(pkt, request_time);
00667 
00668             // If an outstanding request is in progress (we found an
00669             // MSHR) this is set to null
00670             pkt = pf;
00671         }
00672 
00673         if (mshr) {
00677 
00678             //@todo remove hw_pf here
00679 
00680             // Coalesce unless it was a software prefetch (see above).
00681             if (pkt) {
00682                 DPRINTF(Cache, "%s coalescing MSHR for %s addr %#llx size %d\n",
00683                         __func__, pkt->cmdString(), pkt->getAddr(),
00684                         pkt->getSize());
00685 
00686                 assert(pkt->req->masterId() < system->maxMasters());
00687                 mshr_hits[pkt->cmdToIndex()][pkt->req->masterId()]++;
00688                 if (mshr->threadNum != 0/*pkt->req->threadId()*/) {
00689                     mshr->threadNum = -1;
00690                 }
00691                 // We use forward_time here because it is the same
00692                 // considering new targets. We have multiple requests for the
00693                 // same address here. It specifies the latency to allocate an
00694                 // internal buffer and to schedule an event to the queued
00695                 // port and also takes into account the additional delay of
00696                 // the xbar.
00697                 mshr->allocateTarget(pkt, forward_time, order++);
00698                 if (mshr->getNumTargets() == numTarget) {
00699                     noTargetMSHR = mshr;
00700                     setBlocked(Blocked_NoTargets);
00701                     // need to be careful with this... if this mshr isn't
00702                     // ready yet (i.e. time > curTick()), we don't want to
00703                     // move it ahead of mshrs that are ready
00704                     // mshrQueue.moveToFront(mshr);
00705                 }
00706 
00707                 // We should call the prefetcher reguardless if the request is
00708                 // satisfied or not, reguardless if the request is in the MSHR or
00709                 // not.  The request could be a ReadReq hit, but still not
00710                 // satisfied (potentially because of a prior write to the same
00711                 // cache line.  So, even when not satisfied, tehre is an MSHR
00712                 // already allocated for this, we need to let the prefetcher know
00713                 // about the request
00714                 if (prefetcher) {
00715                     // Don't notify on SWPrefetch
00716                     if (!pkt->cmd.isSWPrefetch())
00717                         next_pf_time = prefetcher->notify(pkt);
00718                 }
00719             }
00720         } else {
00721             // no MSHR
00722             assert(pkt->req->masterId() < system->maxMasters());
00723             if (pkt->req->isUncacheable()) {
00724                 mshr_uncacheable[pkt->cmdToIndex()][pkt->req->masterId()]++;
00725             } else {
00726                 mshr_misses[pkt->cmdToIndex()][pkt->req->masterId()]++;
00727             }
00728 
00729             if (pkt->cmd == MemCmd::Writeback ||
00730                 (pkt->req->isUncacheable() && pkt->isWrite())) {
00731                 // We use forward_time here because there is an
00732                 // uncached memory write, forwarded to WriteBuffer. It
00733                 // specifies the latency to allocate an internal buffer and to
00734                 // schedule an event to the queued port and also takes into
00735                 // account the additional delay of the xbar.
00736                 allocateWriteBuffer(pkt, forward_time, true);
00737             } else {
00738                 if (blk && blk->isValid()) {
00739                     // should have flushed and have no valid block
00740                     assert(!pkt->req->isUncacheable());
00741 
00742                     // If we have a write miss to a valid block, we
00743                     // need to mark the block non-readable.  Otherwise
00744                     // if we allow reads while there's an outstanding
00745                     // write miss, the read could return stale data
00746                     // out of the cache block... a more aggressive
00747                     // system could detect the overlap (if any) and
00748                     // forward data out of the MSHRs, but we don't do
00749                     // that yet.  Note that we do need to leave the
00750                     // block valid so that it stays in the cache, in
00751                     // case we get an upgrade response (and hence no
00752                     // new data) when the write miss completes.
00753                     // As long as CPUs do proper store/load forwarding
00754                     // internally, and have a sufficiently weak memory
00755                     // model, this is probably unnecessary, but at some
00756                     // point it must have seemed like we needed it...
00757                     assert(pkt->needsExclusive());
00758                     assert(!blk->isWritable());
00759                     blk->status &= ~BlkReadable;
00760                 }
00761                 // Here we are using forward_time, modelling the latency of
00762                 // a miss (outbound) just as forwardLatency, neglecting the
00763                 // lookupLatency component. In this case this latency value
00764                 // specifies the latency to allocate an internal buffer and to
00765                 // schedule an event to the queued port, when a cacheable miss
00766                 // is forwarded to MSHR queue.
00767                 // We take also into account the additional delay of the xbar.
00768                 allocateMissBuffer(pkt, forward_time, true);
00769             }
00770 
00771             if (prefetcher) {
00772                 // Don't notify on SWPrefetch
00773                 if (!pkt->cmd.isSWPrefetch())
00774                     next_pf_time = prefetcher->notify(pkt);
00775             }
00776         }
00777     }
00778     // Here we condiser just forward_time.
00779     if (next_pf_time != MaxTick)
00780         requestMemSideBus(Request_PF, std::max(clockEdge(forwardLatency),
00781                                                 next_pf_time));
00782 
00783     return true;
00784 }

00787 // See comment in cache.hh.
00788 PacketPtr
00789 Cache::getBusPacket(PacketPtr cpu_pkt, CacheBlk *blk,
00790                     bool needsExclusive) const
00791 {
00792     bool blkValid = blk && blk->isValid();
00793 
00794     if (cpu_pkt->req->isUncacheable()) {
00795         // note that at the point we see the uncacheable request we
00796         // flush any block, but there could be an outstanding MSHR,
00797         // and the cache could have filled again before we actually
00798         // send out the forwarded uncacheable request (blk could thus
00799         // be non-null)
00800         return NULL;
00801     }
00802 
00803     if (!blkValid &&
00804         (cpu_pkt->cmd == MemCmd::Writeback || cpu_pkt->isUpgrade())) {
00805         // Writebacks that weren't allocated in access() and upgrades
00806         // from upper-level caches that missed completely just go
00807         // through.
00808         return NULL;
00809     }
00810 
00811     assert(cpu_pkt->needsResponse());
00812 
00813     MemCmd cmd;
00814     // @TODO make useUpgrades a parameter.
00815     // Note that ownership protocols require upgrade, otherwise a
00816     // write miss on a shared owned block will generate a ReadExcl,
00817     // which will clobber the owned copy.
00818     const bool useUpgrades = true;
00819     if (blkValid && useUpgrades) {
00820         // only reason to be here is that blk is shared
00821         // (read-only) and we need exclusive
00822         assert(needsExclusive);
00823         assert(!blk->isWritable());
00824         cmd = cpu_pkt->isLLSC() ? MemCmd::SCUpgradeReq : MemCmd::UpgradeReq;
00825     } else if (cpu_pkt->cmd == MemCmd::SCUpgradeFailReq ||
00826                cpu_pkt->cmd == MemCmd::StoreCondFailReq) {
00827         // Even though this SC will fail, we still need to send out the
00828         // request and get the data to supply it to other snoopers in the case
00829         // where the determination the StoreCond fails is delayed due to
00830         // all caches not being on the same local bus.
00831         cmd = MemCmd::SCUpgradeFailReq;
00832     } else if (cpu_pkt->isWriteInvalidate()) {
00833         cmd = cpu_pkt->cmd;
00834     } else {
00835         // block is invalid
00836         cmd = needsExclusive ? MemCmd::ReadExReq : MemCmd::ReadReq;
00837     }
00838     PacketPtr pkt = new Packet(cpu_pkt->req, cmd, blkSize);
00839 
00840     // if there are sharers in the upper levels, pass that info downstream
00841     if (cpu_pkt->sharedAsserted()) {
00842         // note that cpu_pkt may have spent a considerable time in the
00843         // MSHR queue and that the information could possibly be out
00844         // of date, however, there is no harm in conservatively
00845         // assuming the block is shared
00846         pkt->assertShared();
00847         DPRINTF(Cache, "%s passing shared from %s to %s addr %#llx size %d\n",
00848                 __func__, cpu_pkt->cmdString(), pkt->cmdString(),
00849                 pkt->getAddr(), pkt->getSize());
00850     }
00851 
00852     // the packet should be block aligned
00853     assert(pkt->getAddr() == blockAlign(pkt->getAddr()));
00854 
00855     pkt->allocate();
00856     DPRINTF(Cache, "%s created %s addr %#llx size %d\n",
00857             __func__, pkt->cmdString(), pkt->getAddr(), pkt->getSize());
00858     return pkt;
00859 }

00862 Tick
00863 Cache::recvAtomic(PacketPtr pkt)
00864 {
00865     // We are in atomic mode so we pay just for lookupLatency here.
00866     Cycles lat = lookupLatency;
00867     // @TODO: make this a parameter
00868     bool last_level_cache = false;
00869 
00870     // Forward the request if the system is in cache bypass mode.
00871     if (system->bypassCaches())
00872         return ticksToCycles(memSidePort->sendAtomic(pkt));
00873 
00874     promoteWholeLineWrites(pkt);
00875 
00876     if (pkt->memInhibitAsserted()) {
00877         // have to invalidate ourselves and any lower caches even if
00878         // upper cache will be responding
00879         if (pkt->isInvalidate()) {
00880             CacheBlk *blk = tags->findBlock(pkt->getAddr(), pkt->isSecure());
00881             if (blk && blk->isValid()) {
00882                 tags->invalidate(blk);
00883                 blk->invalidate();
00884                 DPRINTF(Cache, "rcvd mem-inhibited %s on %#llx (%s):"
00885                         " invalidating\n",
00886                         pkt->cmdString(), pkt->getAddr(),
00887                         pkt->isSecure() ? "s" : "ns");
00888             }
00889             if (!last_level_cache) {
00890                 DPRINTF(Cache, "forwarding mem-inhibited %s on %#llx (%s)\n",
00891                         pkt->cmdString(), pkt->getAddr(),
00892                         pkt->isSecure() ? "s" : "ns");
00893                 lat += ticksToCycles(memSidePort->sendAtomic(pkt));
00894             }
00895         } else {
00896             DPRINTF(Cache, "rcvd mem-inhibited %s on %#llx: not responding\n",
00897                     pkt->cmdString(), pkt->getAddr());
00898         }
00899 
00900         return lat * clockPeriod();
00901     }
00902 
00903     // should assert here that there are no outstanding MSHRs or
00904     // writebacks... that would mean that someone used an atomic
00905     // access in timing mode
00906 
00907     CacheBlk *blk = NULL;
00908     PacketList writebacks;
00909     bool satisfied = access(pkt, blk, lat, writebacks);
00910 
00911     // handle writebacks resulting from the access here to ensure they
00912     // logically proceed anything happening below
00913     while (!writebacks.empty()){
00914         PacketPtr wbPkt = writebacks.front();
00915         memSidePort->sendAtomic(wbPkt);
00916         writebacks.pop_front();
00917         delete wbPkt;
00918     }
00919 
00920     if (!satisfied) {
00921         // MISS
00922 
00923         PacketPtr bus_pkt = getBusPacket(pkt, blk, pkt->needsExclusive());
00924 
00925         bool is_forward = (bus_pkt == NULL);
00926 
00927         if (is_forward) {
00928             // just forwarding the same request to the next level
00929             // no local cache operation involved
00930             bus_pkt = pkt;
00931         }
00932 
00933         DPRINTF(Cache, "Sending an atomic %s for %#llx (%s)\n",
00934                 bus_pkt->cmdString(), bus_pkt->getAddr(),
00935                 bus_pkt->isSecure() ? "s" : "ns");
00936 
00937 #if TRACING_ON
00938         CacheBlk::State old_state = blk ? blk->status : 0;
00939 #endif
00940 
00941         lat += ticksToCycles(memSidePort->sendAtomic(bus_pkt));
00942 
00943         // We are now dealing with the response handling
00944         DPRINTF(Cache, "Receive response: %s for addr %#llx (%s) in state %i\n",
00945                 bus_pkt->cmdString(), bus_pkt->getAddr(),
00946                 bus_pkt->isSecure() ? "s" : "ns",
00947                 old_state);
00948 
00949         // If packet was a forward, the response (if any) is already
00950         // in place in the bus_pkt == pkt structure, so we don't need
00951         // to do anything.  Otherwise, use the separate bus_pkt to
00952         // generate response to pkt and then delete it.
00953         if (!is_forward) {
00954             if (pkt->needsResponse()) {
00955                 assert(bus_pkt->isResponse());
00956                 if (bus_pkt->isError()) {
00957                     pkt->makeAtomicResponse();
00958                     pkt->copyError(bus_pkt);
00959                 } else if (pkt->isWriteInvalidate()) {
00960                     // note the use of pkt, not bus_pkt here.
00961                     if (isTopLevel) {
00962                         blk = handleFill(pkt, blk, writebacks);
00963                         satisfyCpuSideRequest(pkt, blk);
00964                     } else if (blk) {
00965                         satisfyCpuSideRequest(pkt, blk);
00966                     }
00967                 } else if (bus_pkt->isRead() ||
00968                            bus_pkt->cmd == MemCmd::UpgradeResp) {
00969                     // we're updating cache state to allow us to
00970                     // satisfy the upstream request from the cache
00971                     blk = handleFill(bus_pkt, blk, writebacks);
00972                     satisfyCpuSideRequest(pkt, blk);
00973                 } else {
00974                     // we're satisfying the upstream request without
00975                     // modifying cache state, e.g., a write-through
00976                     pkt->makeAtomicResponse();
00977                 }
00978             }
00979             delete bus_pkt;
00980         }
00981     }
00982 
00983     // Note that we don't invoke the prefetcher at all in atomic mode.
00984     // It's not clear how to do it properly, particularly for
00985     // prefetchers that aggressively generate prefetch candidates and
00986     // rely on bandwidth contention to throttle them; these will tend
00987     // to pollute the cache in atomic mode since there is no bandwidth
00988     // contention.  If we ever do want to enable prefetching in atomic
00989     // mode, though, this is the place to do it... see timingAccess()
00990     // for an example (though we'd want to issue the prefetch(es)
00991     // immediately rather than calling requestMemSideBus() as we do
00992     // there).
00993 
00994     // Handle writebacks (from the response handling) if needed
00995     while (!writebacks.empty()){
00996         PacketPtr wbPkt = writebacks.front();
00997         memSidePort->sendAtomic(wbPkt);
00998         writebacks.pop_front();
00999         delete wbPkt;
01000     }
01001 
01002     if (pkt->needsResponse()) {
01003         pkt->makeAtomicResponse();
01004     }
01005     //new start
01006     //    *frac << curTick()/500 << ',' << "access" << ',' << pkt->getAddr() << std::endl;
01007     //    *frac << "access" << ',' << pkt->req << ',' << pkt->getAddr() << std::endl; 
01008       //new end
01009     return lat * clockPeriod();
01010 }

01013 void
01014 Cache::functionalAccess(PacketPtr pkt, bool fromCpuSide)
01015 {
01016     if (system->bypassCaches()) {
01017         // Packets from the memory side are snoop request and
01018         // shouldn't happen in bypass mode.
01019         assert(fromCpuSide);
01020 
01021         // The cache should be flushed if we are in cache bypass mode,
01022         // so we don't need to check if we need to update anything.
01023         memSidePort->sendFunctional(pkt);
01024         return;
01025     }
01026 
01027     Addr blk_addr = blockAlign(pkt->getAddr());
01028     bool is_secure = pkt->isSecure();
01029     CacheBlk *blk = tags->findBlock(pkt->getAddr(), is_secure);
01030     MSHR *mshr = mshrQueue.findMatch(blk_addr, is_secure);
01031 
01032     pkt->pushLabel(name());
01033 
01034     CacheBlkPrintWrapper cbpw(blk);
01035 
01036     // Note that just because an L2/L3 has valid data doesn't mean an
01037     // L1 doesn't have a more up-to-date modified copy that still
01038     // needs to be found.  As a result we always update the request if
01039     // we have it, but only declare it satisfied if we are the owner.
01040 
01041     // see if we have data at all (owned or otherwise)
01042     bool have_data = blk && blk->isValid()
01043         && pkt->checkFunctional(&cbpw, blk_addr, is_secure, blkSize,
01044                                 blk->data);
01045 
01046     // data we have is dirty if marked as such or if valid & ownership
01047     // pending due to outstanding UpgradeReq
01048     bool have_dirty =
01049         have_data && (blk->isDirty() ||
01050                       (mshr && mshr->inService && mshr->isPendingDirty()));
01051 
01052     bool done = have_dirty
01053         || cpuSidePort->checkFunctional(pkt)
01054         || mshrQueue.checkFunctional(pkt, blk_addr)
01055         || writeBuffer.checkFunctional(pkt, blk_addr)
01056         || memSidePort->checkFunctional(pkt);
01057 
01058     DPRINTF(Cache, "functional %s %#llx (%s) %s%s%s\n",
01059             pkt->cmdString(), pkt->getAddr(), is_secure ? "s" : "ns",
01060             (blk && blk->isValid()) ? "valid " : "",
01061             have_data ? "data " : "", done ? "done " : "");
01062 
01063     // We're leaving the cache, so pop cache->name() label
01064     pkt->popLabel();
01065 
01066     if (done) {
01067         pkt->makeResponse();
01068     } else {
01069         // if it came as a request from the CPU side then make sure it
01070         // continues towards the memory side
01071         if (fromCpuSide) {
01072             memSidePort->sendFunctional(pkt);
01073         } else if (forwardSnoops && cpuSidePort->isSnooping()) {
01074             // if it came from the memory side, it must be a snoop request
01075             // and we should only forward it if we are forwarding snoops
01076             cpuSidePort->sendFunctionalSnoop(pkt);
01077         }
01078     }
01079 }

01089 void
01090 Cache::recvTimingResp(PacketPtr pkt)
01091 {
01092     assert(pkt->isResponse());
01093 
01094     MSHR *mshr = dynamic_cast<MSHR*>(pkt->senderState);
01095     bool is_error = pkt->isError();
01096 
01097     assert(mshr);
01098 
01099     if (is_error) {
01100         DPRINTF(Cache, "Cache received packet with error for addr %#llx (%s), "
01101                 "cmd: %s\n", pkt->getAddr(), pkt->isSecure() ? "s" : "ns",
01102                 pkt->cmdString());
01103     }
01104 
01105     DPRINTF(Cache, "Handling response %s for addr %#llx size %d (%s)\n",
01106             pkt->cmdString(), pkt->getAddr(), pkt->getSize(),
01107             pkt->isSecure() ? "s" : "ns");
01108 
01109     MSHRQueue *mq = mshr->queue;
01110     bool wasFull = mq->isFull();
01111 
01112     if (mshr == noTargetMSHR) {
01113         // we always clear at least one target
01114         clearBlocked(Blocked_NoTargets);
01115         noTargetMSHR = NULL;
01116     }
01117 
01118     // Initial target is used just for stats
01119     MSHR::Target *initial_tgt = mshr->getTarget();
01120     CacheBlk *blk = tags->findBlock(pkt->getAddr(), pkt->isSecure());
01121     int stats_cmd_idx = initial_tgt->pkt->cmdToIndex();
01122     Tick miss_latency = curTick() - initial_tgt->recvTime;
01123     PacketList writebacks;
01124     // We need forward_time here because we have a call of
01125     // allocateWriteBuffer() that need this parameter to specify the
01126     // time to request the bus.  In this case we use forward latency
01127     // because there is a writeback.  We pay also here for headerDelay
01128     // that is charged of bus latencies if the packet comes from the
01129     // bus.
01130     Tick forward_time = clockEdge(forwardLatency) + pkt->headerDelay;
01131 
01132     if (pkt->req->isUncacheable()) {
01133         assert(pkt->req->masterId() < system->maxMasters());
01134         mshr_uncacheable_lat[stats_cmd_idx][pkt->req->masterId()] +=
01135             miss_latency;
01136     } else {
01137         assert(pkt->req->masterId() < system->maxMasters());
01138         mshr_miss_latency[stats_cmd_idx][pkt->req->masterId()] +=
01139             miss_latency;
01140     }
01141 
01142     bool is_fill = !mshr->isForward &&
01143         (pkt->isRead() || pkt->cmd == MemCmd::UpgradeResp);
01144 
01145     if (is_fill && !is_error) {
01146         DPRINTF(Cache, "Block for addr %#llx being updated in Cache\n",
01147                 pkt->getAddr());
01148 
01149         // give mshr a chance to do some dirty work
01150         mshr->handleFill(pkt, blk);
01151 
01152         blk = handleFill(pkt, blk, writebacks);
01153         assert(blk != NULL);
01154     }
01155 
01156     // First offset for critical word first calculations
01157     int initial_offset = initial_tgt->pkt->getOffset(blkSize);
01158 
01159     while (mshr->hasTargets()) {
01160         MSHR::Target *target = mshr->getTarget();
01161         Packet *tgt_pkt = target->pkt;
01162 
01163         switch (target->source) {
01164           case MSHR::Target::FromCPU:
01165             Tick completion_time;
01166             // Here we charge on completion_time the delay of the xbar if the
01167             // packet comes from it, charged on headerDelay.
01168             completion_time = pkt->headerDelay;
01169 
01170             // Software prefetch handling for cache closest to core
01171             if (tgt_pkt->cmd.isSWPrefetch() && isTopLevel) {
01172                 // a software prefetch would have already been ack'd immediately
01173                 // with dummy data so the core would be able to retire it.
01174                 // this request completes right here, so we deallocate it.
01175                 delete tgt_pkt->req;
01176                 delete tgt_pkt;
01177                 break; // skip response
01178             }
01179 
01180             // unlike the other packet flows, where data is found in other
01181             // caches or memory and brought back, write invalidates always
01182             // have the data right away, so the above check for "is fill?"
01183             // cannot actually be determined until examining the stored MSHR
01184             // state. We "catch up" with that logic here, which is duplicated
01185             // from above.
01186             if (tgt_pkt->isWriteInvalidate() && isTopLevel) {
01187                 assert(!is_error);
01188 
01189                 // NB: we use the original packet here and not the response!
01190                 mshr->handleFill(tgt_pkt, blk);
01191                 blk = handleFill(tgt_pkt, blk, writebacks);
01192                 assert(blk != NULL);
01193 
01194                 is_fill = true;
01195             }
01196 
01197             if (is_fill) {
01198                 satisfyCpuSideRequest(tgt_pkt, blk,
01199                                       true, mshr->hasPostDowngrade());
01200 
01201                 // How many bytes past the first request is this one
01202                 int transfer_offset =
01203                     tgt_pkt->getOffset(blkSize) - initial_offset;
01204                 if (transfer_offset < 0) {
01205                     transfer_offset += blkSize;
01206                 }
01207 
01208                 // If not critical word (offset) return payloadDelay.
01209                 // responseLatency is the latency of the return path
01210                 // from lower level caches/memory to an upper level cache or
01211                 // the core.
01212                 completion_time += clockEdge(responseLatency) +
01213                     (transfer_offset ? pkt->payloadDelay : 0);
01214 
01215                 assert(!tgt_pkt->req->isUncacheable());
01216 
01217                 assert(tgt_pkt->req->masterId() < system->maxMasters());
01218                 missLatency[tgt_pkt->cmdToIndex()][tgt_pkt->req->masterId()] +=
01219                     completion_time - target->recvTime;
01220             } else if (pkt->cmd == MemCmd::UpgradeFailResp) {
01221                 // failed StoreCond upgrade
01222                 assert(tgt_pkt->cmd == MemCmd::StoreCondReq ||
01223                        tgt_pkt->cmd == MemCmd::StoreCondFailReq ||
01224                        tgt_pkt->cmd == MemCmd::SCUpgradeFailReq);
01225                 // responseLatency is the latency of the return path
01226                 // from lower level caches/memory to an upper level cache or
01227                 // the core.
01228                 completion_time += clockEdge(responseLatency) +
01229                     pkt->payloadDelay;
01230                 tgt_pkt->req->setExtraData(0);
01231             } else {
01232                 // not a cache fill, just forwarding response
01233                 // responseLatency is the latency of the return path
01234                 // from lower level cahces/memory to the core.
01235                 completion_time += clockEdge(responseLatency) +
01236                     pkt->payloadDelay;
01237                 if (pkt->isRead() && !is_error) {
01238                     // sanity check
01239                     assert(pkt->getAddr() == tgt_pkt->getAddr());
01240                     assert(pkt->getSize() >= tgt_pkt->getSize());
01241 
01242                     tgt_pkt->setData(pkt->getConstPtr<uint8_t>());
01243                 }
01244             }
01245             tgt_pkt->makeTimingResponse();
01246             // if this packet is an error copy that to the new packet
01247             if (is_error)
01248                 tgt_pkt->copyError(pkt);
01249             if (tgt_pkt->cmd == MemCmd::ReadResp &&
01250                 (pkt->isInvalidate() || mshr->hasPostInvalidate())) {
01251                 // If intermediate cache got ReadRespWithInvalidate,
01252                 // propagate that.  Response should not have
01253                 // isInvalidate() set otherwise.
01254                 tgt_pkt->cmd = MemCmd::ReadRespWithInvalidate;
01255                 DPRINTF(Cache, "%s updated cmd to %s for addr %#llx\n",
01256                         __func__, tgt_pkt->cmdString(), tgt_pkt->getAddr());
01257             }
01258             // Reset the bus additional time as it is now accounted for
01259             tgt_pkt->headerDelay = tgt_pkt->payloadDelay = 0;
01260             cpuSidePort->schedTimingResp(tgt_pkt, completion_time);
01261             break;
01262 
01263           case MSHR::Target::FromPrefetcher:
01264             assert(tgt_pkt->cmd == MemCmd::HardPFReq);
01265             if (blk)
01266                 blk->status |= BlkHWPrefetched;
01267             delete tgt_pkt->req;
01268             delete tgt_pkt;
01269             break;
01270 
01271           case MSHR::Target::FromSnoop:
01272             // I don't believe that a snoop can be in an error state
01273             assert(!is_error);
01274             // response to snoop request
01275             DPRINTF(Cache, "processing deferred snoop...\n");
01276             assert(!(pkt->isInvalidate() && !mshr->hasPostInvalidate()));
01277             handleSnoop(tgt_pkt, blk, true, true, mshr->hasPostInvalidate());
01278             break;
01279 
01280           default:
01281             panic("Illegal target->source enum %d\n", target->source);
01282         }
01283 
01284         mshr->popTarget();
01285     }
01286 
01287     if (blk && blk->isValid()) {
01288         if ((pkt->isInvalidate() || mshr->hasPostInvalidate()) &&
01289             (!pkt->isWriteInvalidate() || !isTopLevel)) {
01290             assert(blk != tempBlock);
01291             tags->invalidate(blk);
01292             blk->invalidate();
01293         } else if (mshr->hasPostDowngrade()) {
01294             blk->status &= ~BlkWritable;
01295         }
01296     }
01297 
01298     if (mshr->promoteDeferredTargets()) {
01299         // avoid later read getting stale data while write miss is
01300         // outstanding.. see comment in timingAccess()
01301         if (blk) {
01302             blk->status &= ~BlkReadable;
01303         }
01304         mq = mshr->queue;
01305         mq->markPending(mshr);
01306         requestMemSideBus((RequestCause)mq->index, clockEdge() +
01307                           pkt->payloadDelay);
01308     } else {
01309         mq->deallocate(mshr);
01310         if (wasFull && !mq->isFull()) {
01311             clearBlocked((BlockedCause)mq->index);
01312         }
01313 
01314         // Request the bus for a prefetch if this deallocation freed enough
01315         // MSHRs for a prefetch to take place
01316         if (prefetcher && mq == &mshrQueue && mshrQueue.canPrefetch()) {
01317             Tick next_pf_time = std::max(prefetcher->nextPrefetchReadyTime(),
01318                                          curTick());
01319             if (next_pf_time != MaxTick)
01320                 requestMemSideBus(Request_PF, next_pf_time);
01321         }
01322     }
01323     // reset the xbar additional timinig  as it is now accounted for
01324     pkt->headerDelay = pkt->payloadDelay = 0;
01325 
01326     // copy writebacks to write buffer
01327     while (!writebacks.empty()) {
01328         PacketPtr wbPkt = writebacks.front();
01329         allocateWriteBuffer(wbPkt, clockEdge(forwardLatency), true);
01330         writebacks.pop_front();
01331     }
01332     // if we used temp block, clear it out
01333     if (blk == tempBlock) {
01334         if (blk->isDirty()) {
01335             // We use forwardLatency here because we are copying
01336             // writebacks to write buffer. It specifies the latency to
01337             // allocate an internal buffer and to schedule an event to the
01338             // queued port.
01339             allocateWriteBuffer(writebackBlk(blk), forward_time, true);
01340         }
01341         blk->invalidate();
01342     }
01343 
01344     DPRINTF(Cache, "Leaving %s with %s for addr %#llx\n", __func__,
01345             pkt->cmdString(), pkt->getAddr());
01346     delete pkt;
01347 }

01470 // Note that the reason we return a list of writebacks rather than
01471 // inserting them directly in the write buffer is that this function
01472 // is called by both atomic and timing-mode accesses, and in atomic
01473 // mode we don't mess with the write buffer (we just perform the
01474 // writebacks atomically once the original request is complete).
01475 CacheBlk*
01476 Cache::handleFill(PacketPtr pkt, CacheBlk *blk, PacketList &writebacks)
01477 {
01478     assert(pkt->isResponse() || pkt->isWriteInvalidate());
01479     Addr addr = pkt->getAddr();
01480     bool is_secure = pkt->isSecure();
01481 #if TRACING_ON
01482     CacheBlk::State old_state = blk ? blk->status : 0;
01483 #endif
01484 
01485     if (blk == NULL) {
01486         // better have read new data...
01487         assert(pkt->hasData());
01488 
01489         // only read responses and (original) write invalidate req's have data;
01490         // note that we don't write the data here for write invalidate - that
01491         // happens in the subsequent satisfyCpuSideRequest.
01492         assert(pkt->isRead() || pkt->isWriteInvalidate());
01493 
01494         // need to do a replacement
01495         blk = allocateBlock(addr, is_secure, writebacks);
01496         if (blk == NULL) {
01497             // No replaceable block... just use temporary storage to
01498             // complete the current request and then get rid of it
01499             assert(!tempBlock->isValid());
01500             blk = tempBlock;
01501             tempBlock->set = tags->extractSet(addr);
01502             tempBlock->tag = tags->extractTag(addr);
01503             // @todo: set security state as well...
01504             DPRINTF(Cache, "using temp block for %#llx (%s)\n", addr,
01505                     is_secure ? "s" : "ns");
01506         } else {
01507             tags->insertBlock(pkt, blk);
01508         }
01509 
01510         // we should never be overwriting a valid block
01511         assert(!blk->isValid());
01512     } else {
01513         // existing block... probably an upgrade
01514         assert(blk->tag == tags->extractTag(addr));
01515         // either we're getting new data or the block should already be valid
01516         assert(pkt->hasData() || blk->isValid());
01517         // don't clear block status... if block is already dirty we
01518         // don't want to lose that
01519     }
01520 
01521     if (is_secure)
01522         blk->status |= BlkSecure;
01523     blk->status |= BlkValid | BlkReadable;
01524 
01525     if (!pkt->sharedAsserted()) {
01526         blk->status |= BlkWritable;
01527         // If we got this via cache-to-cache transfer (i.e., from a
01528         // cache that was an owner) and took away that owner's copy,
01529         // then we need to write it back.  Normally this happens
01530         // anyway as a side effect of getting a copy to write it, but
01531         // there are cases (such as failed store conditionals or
01532         // compare-and-swaps) where we'll demand an exclusive copy but
01533         // end up not writing it.
01534         if (pkt->memInhibitAsserted())
01535             blk->status |= BlkDirty;
01536     }
01537 
01538     DPRINTF(Cache, "Block addr %#llx (%s) moving from state %x to %s\n",
01539             addr, is_secure ? "s" : "ns", old_state, blk->print());
01540 
01541     // if we got new data, copy it in (checking for a read response
01542     // and a response that has data is the same in the end)
01543     if (pkt->isRead()) {
01544         // sanity checks
01545         assert(pkt->hasData());
01546         assert(pkt->getSize() == blkSize);
01547 
01548         std::memcpy(blk->data, pkt->getConstPtr<uint8_t>(), blkSize);
01549     }
01550     // We pay for fillLatency here.
01551     blk->whenReady = clockEdge() + fillLatency * clockPeriod() +
01552         pkt->payloadDelay;
01553 
01554     // fill needs read for replacing and write
01555     if (cacti) cacti->fill();
01556 
01557     return blk;
01558 }

01567 void
01568 Cache::doTimingSupplyResponse(PacketPtr req_pkt, const uint8_t *blk_data,
01569                               bool already_copied, bool pending_inval)
01570 {
01571     // sanity check
01572     assert(req_pkt->isRequest());
01573     assert(req_pkt->needsResponse());
01574 
01575     DPRINTF(Cache, "%s for %s addr %#llx size %d\n", __func__,
01576             req_pkt->cmdString(), req_pkt->getAddr(), req_pkt->getSize());
01577     // timing-mode snoop responses require a new packet, unless we
01578     // already made a copy...
01579     PacketPtr pkt = req_pkt;
01580     if (!already_copied)
01581         // do not clear flags, and allocate space for data if the
01582         // packet needs it (the only packets that carry data are read
01583         // responses)
01584         pkt = new Packet(req_pkt, false, req_pkt->isRead());
01585 
01586     assert(req_pkt->req->isUncacheable() || req_pkt->isInvalidate() ||
01587            pkt->sharedAsserted());
01588     pkt->makeTimingResponse();
01589     if (pkt->isRead()) {
01590         pkt->setDataFromBlock(blk_data, blkSize);
01591     }
01592     if (pkt->cmd == MemCmd::ReadResp && pending_inval) {
01593         // Assume we defer a response to a read from a far-away cache
01594         // A, then later defer a ReadExcl from a cache B on the same
01595         // bus as us.  We'll assert MemInhibit in both cases, but in
01596         // the latter case MemInhibit will keep the invalidation from
01597         // reaching cache A.  This special response tells cache A that
01598         // it gets the block to satisfy its read, but must immediately
01599         // invalidate it.
01600         pkt->cmd = MemCmd::ReadRespWithInvalidate;
01601     }
01602     // Here we consider forward_time, paying for just forward latency and
01603     // also charging the delay provided by the xbar.
01604     // forward_time is used as send_time in next allocateWriteBuffer().
01605     Tick forward_time = clockEdge(forwardLatency) + pkt->headerDelay;
01606     // Here we reset the timing of the packet.
01607     pkt->headerDelay = pkt->payloadDelay = 0;
01608     DPRINTF(Cache, "%s created response: %s addr %#llx size %d tick: %lu\n",
01609             __func__, pkt->cmdString(), pkt->getAddr(), pkt->getSize(),
01610             forward_time);
01611     memSidePort->schedTimingSnoopResp(pkt, forward_time, true);
01612 }

01614 void
01615 Cache::handleSnoop(PacketPtr pkt, CacheBlk *blk, bool is_timing,
01616                    bool is_deferred, bool pending_inval)
01617 {
01618     DPRINTF(Cache, "%s for %s addr %#llx size %d\n", __func__,
01619             pkt->cmdString(), pkt->getAddr(), pkt->getSize());
01620     // deferred snoops can only happen in timing mode
01621     assert(!(is_deferred && !is_timing));
01622     // pending_inval only makes sense on deferred snoops
01623     assert(!(pending_inval && !is_deferred));
01624     assert(pkt->isRequest());
01625 
01626     // the packet may get modified if we or a forwarded snooper
01627     // responds in atomic mode, so remember a few things about the
01628     // original packet up front
01629     bool invalidate = pkt->isInvalidate();
01630     bool M5_VAR_USED needs_exclusive = pkt->needsExclusive();
01631 
01632     if (forwardSnoops) {
01633         // first propagate snoop upward to see if anyone above us wants to
01634         // handle it.  save & restore packet src since it will get
01635         // rewritten to be relative to cpu-side bus (if any)
01636         bool alreadyResponded = pkt->memInhibitAsserted();
01637         if (is_timing) {
01638             // copy the packet so that we can clear any flags before
01639             // forwarding it upwards, we also allocate data (passing
01640             // the pointer along in case of static data), in case
01641             // there is a snoop hit in upper levels
01642             Packet snoopPkt(pkt, true, true);
01643             snoopPkt.setExpressSnoop();
01644             snoopPkt.pushSenderState(new ForwardResponseRecord());
01645             // the snoop packet does not need to wait any additional
01646             // time
01647             snoopPkt.headerDelay = snoopPkt.payloadDelay = 0;
01648             cpuSidePort->sendTimingSnoopReq(&snoopPkt);
01649             if (snoopPkt.memInhibitAsserted()) {
01650                 // cache-to-cache response from some upper cache
01651                 assert(!alreadyResponded);
01652                 pkt->assertMemInhibit();
01653             } else {
01654                 // no cache (or anyone else for that matter) will
01655                 // respond, so delete the ForwardResponseRecord here
01656                 delete snoopPkt.popSenderState();
01657             }
01658             if (snoopPkt.sharedAsserted()) {
01659                 pkt->assertShared();
01660             }
01661             // If this request is a prefetch or clean evict and an
01662             // upper level signals block present, make sure to
01663             // propagate the block presence to the requester.
01664             if (snoopPkt.isBlockCached()) {
01665                 pkt->setBlockCached();
01666             }
01667         } else {
01668             cpuSidePort->sendAtomicSnoop(pkt);
01669             if (!alreadyResponded && pkt->memInhibitAsserted()) {
01670                 // cache-to-cache response from some upper cache:
01671                 // forward response to original requester
01672                 assert(pkt->isResponse());
01673             }
01674         }
01675     }
01676 
01677     if (!blk || !blk->isValid()) {
01678         DPRINTF(Cache, "%s snoop miss for %s addr %#llx size %d\n",
01679                 __func__, pkt->cmdString(), pkt->getAddr(), pkt->getSize());
01680         return;
01681     } else {
01682        DPRINTF(Cache, "%s snoop hit for %s for addr %#llx size %d, "
01683                "old state is %s\n", __func__, pkt->cmdString(),
01684                pkt->getAddr(), pkt->getSize(), blk->print());
01685     }
01686 
01687     // we may end up modifying both the block state and the packet (if
01688     // we respond in atomic mode), so just figure out what to do now
01689     // and then do it later.  If we find dirty data while snooping for a
01690     // WriteInvalidate, we don't care, since no merging needs to take place.
01691     // We need the eviction to happen as normal, but the data needn't be
01692     // sent anywhere. nor should the writeback be inhibited at the memory
01693     // controller for any reason.
01694     bool respond = blk->isDirty() && pkt->needsResponse()
01695                                   && !pkt->isWriteInvalidate();
01696     bool have_exclusive = blk->isWritable();
01697 
01698     // Invalidate any prefetch's from below that would strip write permissions
01699     // MemCmd::HardPFReq is only observed by upstream caches.  After missing
01700     // above and in it's own cache, a new MemCmd::ReadReq is created that
01701     // downstream caches observe.
01702     if (pkt->cmd == MemCmd::HardPFReq) {
01703         DPRINTF(Cache, "Squashing prefetch from lower cache %#x\n",
01704                 pkt->getAddr());
01705         pkt->setBlockCached();
01706         return;
01707     }
01708 
01709     if (!pkt->req->isUncacheable() && pkt->isRead() && !invalidate) {
01710         assert(!needs_exclusive);
01711         pkt->assertShared();
01712         int bits_to_clear = BlkWritable;
01713         const bool haveOwnershipState = true; // for now
01714         if (!haveOwnershipState) {
01715             // if we don't support pure ownership (dirty && !writable),
01716             // have to clear dirty bit here, assume memory snarfs data
01717             // on cache-to-cache xfer
01718             bits_to_clear |= BlkDirty;
01719         }
01720         blk->status &= ~bits_to_clear;
01721     }
01722 
01723     if (respond) {
01724         // prevent anyone else from responding, cache as well as
01725         // memory, and also prevent any memory from even seeing the
01726         // request (with current inhibited semantics), note that this
01727         // applies both to reads and writes and that for writes it
01728         // works thanks to the fact that we still have dirty data and
01729         // will write it back at a later point
01730         pkt->assertMemInhibit();
01731         if (have_exclusive) {
01732             // in the case of an uncacheable request there is no need
01733             // to set the exclusive flag, but since the recipient does
01734             // not care there is no harm in doing so
01735             pkt->setSupplyExclusive();
01736         }
01737         if (is_timing) {
01738             doTimingSupplyResponse(pkt, blk->data, is_deferred, pending_inval);
01739         } else {
01740             pkt->makeAtomicResponse();
01741             pkt->setDataFromBlock(blk->data, blkSize);
01742         }
01743     }
01744 
01745     if (!respond && is_timing && is_deferred) {
01746         // if it's a deferred timing snoop then we've made a copy of
01747         // both the request and the packet, and so if we're not using
01748         // those copies to respond and delete them here
01749         DPRINTF(Cache, "Deleting pkt %p and request %p for cmd %s addr: %p\n",
01750                 pkt, pkt->req, pkt->cmdString(), pkt->getAddr());
01751 
01752         // the packets needs a response (just not from us), so we also
01753         // need to delete the request and not rely on the packet
01754         // destructor
01755         assert(pkt->needsResponse());
01756         delete pkt->req;
01757         delete pkt;
01758     }
01759 
01760     // Do this last in case it deallocates block data or something
01761     // like that
01762     if (invalidate) {
01763         if (blk != tempBlock)
01764             tags->invalidate(blk);
01765         blk->invalidate();
01766     }
01767 
01768     // in future, write by snooping should be considered
01769     if (cacti) cacti->read();
01770 
01771     DPRINTF(Cache, "new state is %s\n", blk->print());
01772 }

01775 void
01776 Cache::recvTimingSnoopReq(PacketPtr pkt)
01777 {
01778     DPRINTF(Cache, "%s for %s addr %#llx size %d\n", __func__,
01779             pkt->cmdString(), pkt->getAddr(), pkt->getSize());
01780 
01781     // Snoops shouldn't happen when bypassing caches
01782     assert(!system->bypassCaches());
01783 
01784     // no need to snoop writebacks or requests that are not in range
01785     if (pkt->cmd == MemCmd::Writeback || !inRange(pkt->getAddr())) {
01786         return;
01787     }
01788 
01789     bool is_secure = pkt->isSecure();
01790     CacheBlk *blk = tags->findBlock(pkt->getAddr(), is_secure);
01791 
01792     Addr blk_addr = blockAlign(pkt->getAddr());
01793     MSHR *mshr = mshrQueue.findMatch(blk_addr, is_secure);
01794

01796     if (mshr && pkt->cmd == MemCmd::HardPFReq) {
01797         DPRINTF(Cache, "Setting block present to squash prefetch from"
01798                 "lower cache on mshr hit %#x\n",
01799                 pkt->getAddr());
01800         pkt->setBlockCached();
01801         return;
01802     }
01803 
01804     // Let the MSHR itself track the snoop and decide whether we want
01805     // to go ahead and do the regular cache snoop
01806     if (mshr && mshr->handleSnoop(pkt, order++)) {
01807         DPRINTF(Cache, "Deferring snoop on in-service MSHR to blk %#llx (%s)."
01808                 "mshrs: %s\n", blk_addr, is_secure ? "s" : "ns",
01809                 mshr->print());
01810 
01811         if (mshr->getNumTargets() > numTarget)
01812             warn("allocating bonus target for snoop"); //handle later
01813         return;
01814     }
01815 
01816     //We also need to check the writeback buffers and handle those
01817     std::vector<MSHR *> writebacks;
01818     if (writeBuffer.findMatches(blk_addr, is_secure, writebacks)) {
01819         DPRINTF(Cache, "Snoop hit in writeback to addr %#llx (%s)\n",
01820                 pkt->getAddr(), is_secure ? "s" : "ns");
01821 
01822         // Look through writebacks for any cachable writes.
01823         // We should only ever find a single match
01824         assert(writebacks.size() == 1);
01825         MSHR *wb_entry = writebacks[0];
01826         assert(!wb_entry->isUncacheable());
01827         assert(wb_entry->getNumTargets() == 1);
01828         PacketPtr wb_pkt = wb_entry->getTarget()->pkt;
01829         assert(wb_pkt->cmd == MemCmd::Writeback);
01830 
01831         assert(!pkt->memInhibitAsserted());
01832         pkt->assertMemInhibit();
01833         if (!pkt->needsExclusive()) {
01834             pkt->assertShared();
01835             // the writeback is no longer the exclusive copy in the system
01836             wb_pkt->clearSupplyExclusive();
01837         } else {
01838             // if we're not asserting the shared line, we need to
01839             // invalidate our copy.  we'll do that below as long as
01840             // the packet's invalidate flag is set...
01841             assert(pkt->isInvalidate());
01842         }
01843         doTimingSupplyResponse(pkt, wb_pkt->getConstPtr<uint8_t>(),
01844                                false, false);
01845 
01846         if (pkt->isInvalidate()) {
01847             // Invalidation trumps our writeback... discard here
01848             markInService(wb_entry, false);
01849             delete wb_pkt;
01850         }
01851     }
01852 
01853     // If this was a shared writeback, there may still be
01854     // other shared copies above that require invalidation.
01855     // We could be more selective and return here if the
01856     // request is non-exclusive or if the writeback is
01857     // exclusive.
01858     handleSnoop(pkt, blk, true, false, false);
01859 }

01888 Cache::getNextMSHR()
01889 {
01890     // Check both MSHR queue and write buffer for potential requests,
01891     // note that null does not mean there is no request, it could
01892     // simply be that it is not ready
01893     MSHR *miss_mshr  = mshrQueue.getNextMSHR();
01894     MSHR *write_mshr = writeBuffer.getNextMSHR();
01895 
01896     // If we got a write buffer request ready, first priority is a
01897     // full write buffer, otherwhise we favour the miss requests
01898     if (write_mshr &&
01899         ((writeBuffer.isFull() && writeBuffer.inServiceEntries == 0) ||
01900          !miss_mshr)) {
01901         // need to search MSHR queue for conflicting earlier miss.
01902         MSHR *conflict_mshr =
01903             mshrQueue.findPending(write_mshr->blkAddr,
01904                                   write_mshr->isSecure);
01905 
01906         if (conflict_mshr && conflict_mshr->order < write_mshr->order) {
01907             // Service misses in order until conflict is cleared.
01908             return conflict_mshr;
01909 
01910             // @todo Note that we ignore the ready time of the conflict here
01911         }
01912 
01913         // No conflicts; issue write
01914         return write_mshr;
01915     } else if (miss_mshr) {
01916         // need to check for conflicting earlier writeback
01917         MSHR *conflict_mshr =
01918             writeBuffer.findPending(miss_mshr->blkAddr,
01919                                     miss_mshr->isSecure);
01920         if (conflict_mshr) {
01921             // not sure why we don't check order here... it was in the
01922             // original code but commented out.
01923 
01924             // The only way this happens is if we are
01925             // doing a write and we didn't have permissions
01926             // then subsequently saw a writeback (owned got evicted)
01927             // We need to make sure to perform the writeback first
01928             // To preserve the dirty data, then we can issue the write
01929 
01930             // should we return write_mshr here instead?  I.e. do we
01931             // have to flush writes in order?  I don't think so... not
01932             // for Alpha anyway.  Maybe for x86?
01933             return conflict_mshr;
01934 
01935             // @todo Note that we ignore the ready time of the conflict here
01936         }
01937 
01938         // No conflicts; issue read
01939         return miss_mshr;
01940     }
01941 
01942     // fall through... no pending requests.  Try a prefetch.
01943     assert(!miss_mshr && !write_mshr);
01944     if (prefetcher && mshrQueue.canPrefetch()) {
01945         // If we have a miss queue slot, we can try a prefetch
01946         PacketPtr pkt = prefetcher->getPacket();
01947         if (pkt) {
01948             Addr pf_addr = blockAlign(pkt->getAddr());
01949             if (!tags->findBlock(pf_addr, pkt->isSecure()) &&
01950                 !mshrQueue.findMatch(pf_addr, pkt->isSecure()) &&
01951                 !writeBuffer.findMatch(pf_addr, pkt->isSecure())) {
01952                 // Update statistic on number of prefetches issued
01953                 // (hwpf_mshr_misses)
01954                 assert(pkt->req->masterId() < system->maxMasters());
01955                 mshr_misses[pkt->cmdToIndex()][pkt->req->masterId()]++;
01956                 // Don't request bus, since we already have it
01957                 return allocateMissBuffer(pkt, curTick(), false);
01958             } else {
01959                 // free the request and packet
01960                 delete pkt->req;
01961                 delete pkt;
01962             }
01963         }
01964     }
01965 
01966     return NULL;

01984             tgt_pkt->cmdString(), tgt_pkt->getAddr(), tgt_pkt->getSize());

01987         // no response expected, just forward packet as it is
01988         assert(tags->findBlock(mshr->blkAddr, mshr->isSecure) == NULL);
01989         pkt = tgt_pkt;
01990     } else {
01991         CacheBlk *blk = tags->findBlock(mshr->blkAddr, mshr->isSecure);
01992 
01993         if (tgt_pkt->cmd == MemCmd::HardPFReq && forwardSnoops) {
01994             // We need to check the caches above us to verify that
01995             // they don't have a copy of this block in the dirty state
01996             // at the moment. Without this check we could get a stale
01997             // copy from memory that might get used in place of the
01998             // dirty one.
01999             Packet snoop_pkt(tgt_pkt, true, false);
02000             snoop_pkt.setExpressSnoop();
02001             snoop_pkt.senderState = mshr;
02002             cpuSidePort->sendTimingSnoopReq(&snoop_pkt);
02003 
02004             // Check to see if the prefetch was squashed by an upper cache (to
02005             // prevent us from grabbing the line) or if a Check to see if a
02006             // writeback arrived between the time the prefetch was placed in
02007             // the MSHRs and when it was selected to be sent or if the
02008             // prefetch was squashed by an upper cache.
02009 
02010             // It is important to check msmInhibitAsserted before
02011             // prefetchSquashed. If another cache has asserted MEM_INGIBIT, it
02012             // will be sending a response which will arrive at the MSHR
02013             // allocated ofr this request. Checking the prefetchSquash first
02014             // may result in the MSHR being prematurely deallocated.
02015 
02016             if (snoop_pkt.memInhibitAsserted()) {
02017                 // If we are getting a non-shared response it is dirty
02018                 bool pending_dirty_resp = !snoop_pkt.sharedAsserted();
02019                 markInService(mshr, pending_dirty_resp);
02020                 DPRINTF(Cache, "Upward snoop of prefetch for addr"
02021                         " %#x (%s) hit\n",
02022                         tgt_pkt->getAddr(), tgt_pkt->isSecure()? "s": "ns");
02023                 return NULL;
02024             }
02025 
02026             if (snoop_pkt.isBlockCached() || blk != NULL) {
02027                 DPRINTF(Cache, "Block present, prefetch squashed by cache.  "
02028                                "Deallocating mshr target %#x.\n",
02029                         mshr->blkAddr);
02030 
02031                 // Deallocate the mshr target
02032                 if (mshr->queue->forceDeallocateTarget(mshr)) {
02033                     // Clear block if this deallocation resulted freed an
02034                     // mshr when all had previously been utilized
02035                     clearBlocked((BlockedCause)(mshr->queue->index));
02036                 }
02037                 return NULL;
02038             }
02039 
02040         }
02041 
02042         pkt = getBusPacket(tgt_pkt, blk, mshr->needsExclusive());
02043 
02044         mshr->isForward = (pkt == NULL);
02045 
02046         if (mshr->isForward) {
02047             // not a cache block request, but a response is expected
02048             // make copy of current packet to forward, keep current
02049             // copy for response handling
02050             pkt = new Packet(tgt_pkt, false, true);
02051             if (pkt->isWrite()) {
02052                 pkt->setData(tgt_pkt->getConstPtr<uint8_t>());
02053             }
02054         }
02055     }

02208 void
02209 Cache::CacheReqPacketQueue::sendDeferredPacket()
02210 {
02211     // sanity check
02212     assert(!waitingOnRetry);
02213 
02214     // there should never be any deferred request packets in the
02215     // queue, instead we resly on the cache to provide the packets
02216     // from the MSHR queue or write queue
02217     assert(deferredPacketReadyTime() == MaxTick);
02218 
02219     // check for request packets (requests & writebacks)
02220     PacketPtr pkt = cache.getTimingPacket();
02221     if (pkt == NULL) {
02222         // can happen if e.g. we attempt a writeback and fail, but
02223         // before the retry, the writeback is eliminated because
02224         // we snoop another cache's ReadEx.
02225     } else {
02226         MSHR *mshr = dynamic_cast<MSHR*>(pkt->senderState);
02227         // in most cases getTimingPacket allocates a new packet, and
02228         // we must delete it unless it is successfully sent
02229         bool delete_pkt = !mshr->isForwardNoResponse();
02230 
02231         // let our snoop responses go first if there are responses to
02232         // the same addresses we are about to writeback, note that
02233         // this creates a dependency between requests and snoop
02234         // responses, but that should not be a problem since there is
02235         // a chain already and the key is that the snoop responses can
02236         // sink unconditionally
02237         if (snoopRespQueue.hasAddr(pkt->getAddr())) {
02238             DPRINTF(CachePort, "Waiting for snoop response to be sent\n");
02239             Tick when = snoopRespQueue.deferredPacketReadyTime();
02240             schedSendEvent(when);
02241 
02242             if (delete_pkt)
02243                 delete pkt;
02244 
02245             return;
02246         }
02247 
02248 
02249         waitingOnRetry = !masterPort.sendTimingReq(pkt);
02250 
02251         if (waitingOnRetry) {
02252             DPRINTF(CachePort, "now waiting on a retry\n");
02253             if (delete_pkt) {
02254                 // we are awaiting a retry, but we
02255                 // delete the packet and will be creating a new packet
02256                 // when we get the opportunity
02257                 delete pkt;
02258             }
02259             // note that we have now masked any requestBus and
02260             // schedSendEvent (we will wait for a retry before
02261             // doing anything), and this is so even if we do not
02262             // care about this packet and might override it before
02263             // it gets retried
02264         } else {
02265             // As part of the call to sendTimingReq the packet is
02266             // forwarded to all neighbouring caches (and any
02267             // caches above them) as a snoop. The packet is also
02268             // sent to any potential cache below as the
02269             // interconnect is not allowed to buffer the
02270             // packet. Thus at this point we know if any of the
02271             // neighbouring, or the downstream cache is
02272             // responding, and if so, if it is with a dirty line
02273             // or not.
02274             bool pending_dirty_resp = !pkt->sharedAsserted() &&
02275                 pkt->memInhibitAsserted();
02276 
02277             cache.markInService(mshr, pending_dirty_resp);
02278         }
02279     }

02282     // next send considering when the next MSHR is ready, note that
02283     // snoop responses have their own packet queue and thus schedule
02284     // their own events

00094 void
00095 EthAddr::parse(const std::string &addr)
00096 {
00097     // the hack below is to make sure that ETH_ADDR_LEN is 6 otherwise
00098     // the sscanf function won't work.
00099     int bytes[ETH_ADDR_LEN == 6 ? ETH_ADDR_LEN : -1];
00100     if (sscanf(addr.c_str(), "%x:%x:%x:%x:%x:%x", &bytes[0], &bytes[1],
00101                &bytes[2], &bytes[3], &bytes[4], &bytes[5]) != ETH_ADDR_LEN) {
00102         memset(data, 0xff, ETH_ADDR_LEN);
00103         return;
00104     }
00105 
00106     for (int i = 0; i < ETH_ADDR_LEN; ++i) {
00107         if (bytes[i] & ~0xff) {
00108             memset(data, 0xff, ETH_ADDR_LEN);
00109             return;
00110         }
00111 
00112         data[i] = bytes[i];
00113     }
00114 }

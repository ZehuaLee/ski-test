00099 // be called when something happens that should never ever happen
00100 // regardless of what the user does (i.e., an acutal m5 bug).  panic()

00175 // Only print the message the first time this expression is
00176 // encountered.  i.e.  This doesn't check the string itself and
00177 // prevent duplicate strings, this prevents the statement from
00178 // happening more than once. So, even if the arguments change and that
00179 // would have resulted in a different message thoes messages would be
00180 // supressed.
00181 #define base_message_once(...) do {                     \
00182         static bool once = false;                       \
00183         if (!once) {                                    \
00184             base_message(__VA_ARGS__);                  \
00185             once = true;                                \
00186         }                                               \
00187     } while (0)

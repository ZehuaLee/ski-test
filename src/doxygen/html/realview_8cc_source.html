00113 Addr
00114 RealView::calcPciConfigAddr(int bus, int dev, int func)
00115 {
00116     if (bus != 0)
00117         return ULL(-1);
00118 
00119     Addr cfg_offset = 0;
00120     if (params()->pci_cfg_gen_offsets)
00121         cfg_offset |= ((func & 7) << 12) | ((dev & 0x1f) << 15);
00122     else
00123         cfg_offset |= ((func & 7) << 16) | ((dev & 0x1f) << 19);
00124     return params()->pci_cfg_base | cfg_offset;
00125 }

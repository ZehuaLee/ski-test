00052 SnoopFilter::lookupRequest(const Packet* cpkt, const SlavePort& slave_port)
00053 {
00054     DPRINTF(SnoopFilter, "%s: packet src %s addr 0x%x cmd %s\n",
00055             __func__, slave_port.name(), cpkt->getAddr(), cpkt->cmdString());
00056 
00057     Addr line_addr = cpkt->getAddr() & ~(linesize - 1);
00058     SnoopMask req_port = portToMask(slave_port);
00059     auto sf_it = cachedLocations.find(line_addr);
00060     bool is_hit = (sf_it != cachedLocations.end());
00061     // Create a new element through operator[] and modify in-place
00062     SnoopItem& sf_item = is_hit ? sf_it->second : cachedLocations[line_addr];
00063     SnoopMask interested = sf_item.holder | sf_item.requested;
00064 
00065     totRequests++;
00066     if (is_hit) {
00067         // Single bit set -> value is a power of two
00068         if (isPow2(interested))
00069             hitSingleRequests++;
00070         else
00071             hitMultiRequests++;
00072     }
00073 
00074     DPRINTF(SnoopFilter, "%s:   SF value %x.%x\n",
00075             __func__, sf_item.requested, sf_item.holder);
00076 
00077     if (!cpkt->req->isUncacheable() && cpkt->needsResponse()) {
00078         if (!cpkt->memInhibitAsserted()) {
00079             // Max one request per address per port
00080             panic_if(sf_item.requested & req_port, "double request :( "\
00081                      "SF value %x.%x\n", sf_item.requested, sf_item.holder);
00082 
00083             // Mark in-flight requests to distinguish later on
00084             sf_item.requested |= req_port;
00085         } else {
00086             // NOTE: The memInhibit might have been asserted by a cache closer
00087             // to the CPU, already -> the response will not be seen by this
00088             // filter -> we do not need to keep the in-flight request, but make
00089             // sure that we know that that cluster has a copy
00090             panic_if(!(sf_item.holder & req_port), "Need to hold the value!");
00091             DPRINTF(SnoopFilter, "%s:   not marking request. SF value %x.%x\n",
00092                     __func__,  sf_item.requested, sf_item.holder);
00093         }
00094         DPRINTF(SnoopFilter, "%s:   new SF value %x.%x\n",
00095                 __func__,  sf_item.requested, sf_item.holder);
00096     }
00097     return snoopSelected(maskToPortList(interested & ~req_port), lookupLatency);
00098 }

00123     // will_retry == false
00124     if (!cpkt->needsResponse()) {
00125         // Packets that will not evoke a response but still need updates of the

00127         if (cpkt->cmd == MemCmd::Writeback) {
00128             // make sure that the sender actually had the line
00129             panic_if(sf_item.requested & req_port, "double request :( "\
00130                      "SF value %x.%x\n", sf_item.requested, sf_item.holder);
00131             panic_if(!(sf_item.holder & req_port), "requester %x is not a "\
00132                      "holder :( SF value %x.%x\n", req_port,
00133                      sf_item.requested, sf_item.holder);
00134             // Writebacks -> the sender does not have the line anymore
00135             sf_item.holder &= ~req_port;
00136         } else {
00137             assert(0 == "Handle non-writeback, here");
00138         }
00139         DPRINTF(SnoopFilter, "%s:   new SF value %x.%x\n",

00224     if (cpkt->needsExclusive() || !cpkt->sharedAsserted()) {

00228                 sf_item.requested, sf_item.holder);
00229 
00230         sf_item.holder &= ~rsp_mask;
00231         // The snoop filter does not see any ACKs from non-responding sharers
00232         // that have been invalidated :(  So below assert would be nice, but..
00233         //assert(sf_item.holder == 0);
00234         sf_item.holder = 0;
00235     }
00236     assert(cpkt->cmd != MemCmd::Writeback);
00237     sf_item.holder |=  req_mask;
00238     sf_item.requested &= ~req_mask;

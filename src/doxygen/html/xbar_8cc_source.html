00107 void
00108 BaseXBar::calcPacketTiming(PacketPtr pkt, Tick header_delay)
00109 {
00110     // the crossbar will be called at a time that is not necessarily
00111     // coinciding with its own clock, so start by determining how long
00112     // until the next clock edge (could be zero)
00113     Tick offset = clockEdge() - curTick();
00114 
00115     // the header delay depends on the path through the crossbar, and
00116     // we therefore rely on the caller to provide the actual
00117     // value
00118     pkt->headerDelay += offset + header_delay;
00119 
00120     // note that we add the header delay to the existing value, and
00121     // align it to the crossbar clock
00122 
00123     // do a quick sanity check to ensure the timings are not being
00124     // ignored, note that this specific value may cause problems for
00125     // slower interconnects
00126     panic_if(pkt->headerDelay > SimClock::Int::us,
00127              "Encountered header delay exceeding 1 us\n");
00128 
00129     if (pkt->hasData()) {
00130         // the payloadDelay takes into account the relative time to
00131         // deliver the payload of the packet, after the header delay,
00132         // we take the maximum since the payload delay could already
00133         // be longer than what this parcitular crossbar enforces.
00134         pkt->payloadDelay = std::max<Tick>(pkt->payloadDelay,
00135                                            divCeil(pkt->getSize(), width) *
00136                                            clockPeriod());
00137     }
00138 
00139     // the payload delay is not paying for the clock offset as that is
00140     // already done using the header delay, and the payload delay is
00141     // also used to determine how long the crossbar layer is busy and
00142     // thus regulates throughput
00143 }

00155 {
00156     // ensure the state is busy at this point, as the layer should
00157     // transition from idle as soon as it has decided to forward the

00173 template <typename SrcType, typename DstType>
00174 bool
00175 BaseXBar::Layer<SrcType,DstType>::tryTiming(SrcType* src_port)
00176 {
00177     // if we are in the retry state, we will not see anything but the
00178     // retrying port (or in the case of the snoop ports the snoop
00179     // response port that mirrors the actual slave port) as we leave
00180     // this state again in zero time if the peer does not immediately
00181     // call the layer when receiving the retry
00182 
00183     // first we see if the layer is busy, next we check if the
00184     // destination port is already engaged in a transaction waiting
00185     // for a retry from the peer
00186     if (state == BUSY || waitingForPeer != NULL) {
00187         // the port should not be waiting already
00188         assert(std::find(waitingForLayer.begin(), waitingForLayer.end(),
00189                          src_port) == waitingForLayer.end());
00190 
00191         // put the port at the end of the retry list waiting for the
00192         // layer to be freed up (and in the case of a busy peer, for
00193         // that transaction to go through, and then the layer to free
00194         // up)
00195         waitingForLayer.push_back(src_port);
00196         return false;
00197     }
00198 
00199     state = BUSY;
00200 
00201     return true;
00202 }

00239 void
00240 BaseXBar::Layer<SrcType,DstType>::releaseLayer()
00241 {
00242     // releasing the bus means we should now be idle
00243     assert(state == BUSY);
00244     assert(!releaseEvent.scheduled());
00245 
00246     // update the state
00247     state = IDLE;
00248 
00249     // bus layer is now idle, so if someone is waiting we can retry
00250     if (!waitingForLayer.empty()) {
00251         // there is no point in sending a retry if someone is still
00252         // waiting for the peer
00253         if (waitingForPeer == NULL)
00254             retryWaiting();
00255     } else if (waitingForPeer == NULL && drainManager) {
00256         DPRINTF(Drain, "Crossbar done draining, signaling drain manager\n");
00257         //If we weren't able to drain before, do it now.
00258         drainManager->signalDrainDone();
00259         // Clear the drain event once we're done with it.
00260         drainManager = NULL;
00261     }

00264 template <typename SrcType, typename DstType>
00265 void
00266 BaseXBar::Layer<SrcType,DstType>::retryWaiting()
00267 {
00268     // this should never be called with no one waiting
00269     assert(!waitingForLayer.empty());
00270 
00271     // we always go to retrying from idle
00272     assert(state == IDLE);
00273 
00274     // update the state
00275     state = RETRY;
00276 
00277     // set the retrying port to the front of the retry list and pop it
00278     // off the list
00279     SrcType* retryingPort = waitingForLayer.front();
00280     waitingForLayer.pop_front();
00281 
00282     // tell the port to retry, which in some cases ends up calling the
00283     // layer again
00284     sendRetry(retryingPort);
00285 
00286     // If the layer is still in the retry state, sendTiming wasn't
00287     // called in zero time (e.g. the cache does this when a writeback
00288     // is squashed)
00289     if (state == RETRY) {
00290         // update the state to busy and reset the retrying port, we
00291         // have done our bit and sent the retry
00292         state = BUSY;
00293 
00294         // occupy the crossbar layer until the next clock edge
00295         occupyLayer(xbar.clockEdge());
00296     }

00300 void
00301 BaseXBar::Layer<SrcType,DstType>::recvRetry()
00302 {
00303     // we should never get a retry without having failed to forward
00304     // something to this port
00305     assert(waitingForPeer != NULL);
00306 
00307     // add the port where the failed packet originated to the front of
00308     // the waiting ports for the layer, this allows us to call retry
00309     // on the port immediately if the crossbar layer is idle
00310     waitingForLayer.push_front(waitingForPeer);
00311 
00312     // we are no longer waiting for the peer
00313     waitingForPeer = NULL;
00314 
00315     // if the layer is idle, retry this port straight away, if we
00316     // are busy, then simply let the port wait for its turn
00317     if (state == IDLE) {
00318         retryWaiting();
00319     } else {
00320         assert(state == BUSY);
00321     }

00364 void
00365 BaseXBar::recvRangeChange(PortID master_port_id)
00366 {
00367     DPRINTF(AddrRanges, "Received range change from slave port %s\n",
00368             masterPorts[master_port_id]->getSlavePort().name());
00369 
00370     // remember that we got a range from this master port and thus the
00371     // connected slave module
00372     gotAddrRanges[master_port_id] = true;
00373 
00374     // update the global flag
00375     if (!gotAllAddrRanges) {
00376         // take a logical AND of all the ports and see if we got
00377         // ranges from everyone
00378         gotAllAddrRanges = true;
00379         std::vector<bool>::const_iterator r = gotAddrRanges.begin();
00380         while (gotAllAddrRanges &&  r != gotAddrRanges.end()) {
00381             gotAllAddrRanges &= *r++;
00382         }
00383         if (gotAllAddrRanges)
00384             DPRINTF(AddrRanges, "Got address ranges from all slaves\n");
00385     }
00386 
00387     // note that we could get the range from the default port at any
00388     // point in time, and we cannot assume that the default range is
00389     // set before the other ones are, so we do additional checks once
00390     // all ranges are provided
00391     if (master_port_id == defaultPortID) {
00392         // only update if we are indeed checking ranges for the
00393         // default port since the port might not have a valid range
00394         // otherwise
00395         if (useDefaultRange) {
00396             AddrRangeList ranges = masterPorts[master_port_id]->getAddrRanges();
00397 
00398             if (ranges.size() != 1)
00399                 fatal("Crossbar %s may only have a single default range",
00400                       name());
00401 
00402             defaultRange = ranges.front();
00403         }
00404     } else {
00405         // the ports are allowed to update their address ranges
00406         // dynamically, so remove any existing entries
00407         if (gotAddrRanges[master_port_id]) {
00408             for (auto p = portMap.begin(); p != portMap.end(); ) {
00409                 if (p->second == master_port_id)
00410                     // erasing invalidates the iterator, so advance it
00411                     // before the deletion takes place
00412                     portMap.erase(p++);
00413                 else
00414                     p++;
00415             }
00416         }
00417 
00418         AddrRangeList ranges = masterPorts[master_port_id]->getAddrRanges();
00419 
00420         for (const auto& r: ranges) {
00421             DPRINTF(AddrRanges, "Adding range %s for id %d\n",
00422                     r.to_string(), master_port_id);
00423             if (portMap.insert(r, master_port_id) == portMap.end()) {
00424                 PortID conflict_id = portMap.find(r)->second;
00425                 fatal("%s has two ports responding within range %s:\n\t%s\n\t%s\n",
00426                       name(),
00427                       r.to_string(),
00428                       masterPorts[master_port_id]->getSlavePort().name(),
00429                       masterPorts[conflict_id]->getSlavePort().name());
00430             }
00431         }
00432     }
00433 
00434     // if we have received ranges from all our neighbouring slave
00435     // modules, go ahead and tell our connected master modules in
00436     // turn, this effectively assumes a tree structure of the system
00437     if (gotAllAddrRanges) {
00438         DPRINTF(AddrRanges, "Aggregating address ranges\n");
00439         xbarRanges.clear();
00440 
00441         // start out with the default range
00442         if (useDefaultRange) {
00443             if (!gotAddrRanges[defaultPortID])
00444                 fatal("Crossbar %s uses default range, but none provided",
00445                       name());
00446 
00447             xbarRanges.push_back(defaultRange);
00448             DPRINTF(AddrRanges, "-- Adding default %s\n",
00449                     defaultRange.to_string());
00450         }
00451 
00452         // merge all interleaved ranges and add any range that is not
00453         // a subset of the default range
00454         std::vector<AddrRange> intlv_ranges;
00455         for (const auto& r: portMap) {
00456             // if the range is interleaved then save it for now
00457             if (r.first.interleaved()) {
00458                 // if we already got interleaved ranges that are not
00459                 // part of the same range, then first do a merge
00460                 // before we add the new one
00461                 if (!intlv_ranges.empty() &&
00462                     !intlv_ranges.back().mergesWith(r.first)) {
00463                     DPRINTF(AddrRanges, "-- Merging range from %d ranges\n",
00464                             intlv_ranges.size());
00465                     AddrRange merged_range(intlv_ranges);
00466                     // next decide if we keep the merged range or not
00467                     if (!(useDefaultRange &&
00468                           merged_range.isSubset(defaultRange))) {
00469                         xbarRanges.push_back(merged_range);
00470                         DPRINTF(AddrRanges, "-- Adding merged range %s\n",
00471                                 merged_range.to_string());
00472                     }
00473                     intlv_ranges.clear();
00474                 }
00475                 intlv_ranges.push_back(r.first);
00476             } else {
00477                 // keep the current range if not a subset of the default
00478                 if (!(useDefaultRange &&
00479                       r.first.isSubset(defaultRange))) {
00480                     xbarRanges.push_back(r.first);
00481                     DPRINTF(AddrRanges, "-- Adding range %s\n",
00482                             r.first.to_string());
00483                 }
00484             }
00485         }
00486 
00487         // if there is still interleaved ranges waiting to be merged,
00488         // go ahead and do it
00489         if (!intlv_ranges.empty()) {
00490             DPRINTF(AddrRanges, "-- Merging range from %d ranges\n",
00491                     intlv_ranges.size());
00492             AddrRange merged_range(intlv_ranges);
00493             if (!(useDefaultRange && merged_range.isSubset(defaultRange))) {
00494                 xbarRanges.push_back(merged_range);
00495                 DPRINTF(AddrRanges, "-- Adding merged range %s\n",
00496                         merged_range.to_string());
00497             }
00498         }
00499 
00500         // also check that no range partially overlaps with the
00501         // default range, this has to be done after all ranges are set
00502         // as there are no guarantees for when the default range is
00503         // update with respect to the other ones
00504         if (useDefaultRange) {
00505             for (const auto& r: xbarRanges) {
00506                 // see if the new range is partially
00507                 // overlapping the default range
00508                 if (r.intersects(defaultRange) &&
00509                     !r.isSubset(defaultRange))
00510                     fatal("Range %s intersects the "                    \
00511                           "default range of %s but is not a "           \
00512                           "subset\n", r.to_string(), name());
00513             }
00514         }
00515 
00516         // tell all our neighbouring master ports that our address
00517         // ranges have changed
00518         for (const auto& s: slavePorts)

00520     }
00521 
00522     clearPortCache();
00523 }

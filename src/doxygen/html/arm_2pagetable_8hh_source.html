00199     bool
00200     match(Addr va, uint16_t asn, uint8_t _vmid, bool hypLookUp,
00201           bool secure_lookup, bool ignore_asn, uint8_t target_el) const
00202     {
00203         bool match = false;
00204         Addr v = vpn << N;
00205 
00206         if (valid && va >= v && va <= v + size && (secure_lookup == !nstid) &&
00207             (hypLookUp == isHyp))
00208         {
00209             if (target_el == 2 || target_el == 3)
00210                 match = (el == target_el);
00211             else
00212                 match = (el == 0) || (el == 1);
00213             if (match && !ignore_asn) {
00214                 match = global || (asn == asid);
00215             }
00216             if (match && nstid) {
00217                 match = isHyp || (_vmid == vmid);
00218             }
00219         }
00220         return match;
00221     }

00229     void
00230     updateAttributes()
00231     {
00232         uint64_t mask;
00233         uint64_t newBits;
00234 
00235         // chec bit 11 to determine if its currently LPAE or VMSA format.
00236         if ( attributes & (1 << 11) ) {
00237             newBits = ((outerShareable ? 0x2 :
00238                       shareable         ? 0x3 : 0) << 7);
00239             mask = 0x180;
00240         } else {
00256             newBits = ((outerShareable ? 0:1) << 10) |
00257                       ((shareable ? 1:0) << 7) |
00258                       (innerAttrs << 4) |
00259                       (outerAttrs << 2);
00260                       // TODO: Supersection bit
00261             mask = 0x4FC;
00262         }
00263         // common bits
00264         newBits |= ns << 9;  // NS bit
00265         mask    |= 1  << 9;
00266         // add in the new bits
00267         attributes &= ~mask;
00268         attributes |= newBits;
00269     }

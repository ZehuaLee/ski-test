00063 Fault
00064 Stage2MMU::readDataUntimed(ThreadContext *tc, Addr oVAddr, Addr descAddr,
00065     uint8_t *data, int numBytes, Request::Flags flags, bool isFunctional)
00066 {
00067     Fault fault;
00068 
00069     // translate to physical address using the second stage MMU
00070     Request req = Request();
00071     req.setVirt(0, descAddr, numBytes, flags | Request::PT_WALK, masterId, 0);
00072     if (isFunctional) {
00073         fault = stage2Tlb()->translateFunctional(&req, tc, BaseTLB::Read);
00074     } else {
00075         fault = stage2Tlb()->translateAtomic(&req, tc, BaseTLB::Read);
00076     }
00077 
00078     // Now do the access.
00079     if (fault == NoFault && !req.getFlags().isSet(Request::NO_ACCESS)) {
00080         Packet pkt = Packet(&req, MemCmd::ReadReq);
00081         pkt.dataStatic(data);
00082         if (isFunctional) {
00083             port.sendFunctional(&pkt);
00084         } else {
00085             port.sendAtomic(&pkt);
00086         }
00087         assert(!pkt.isError());
00088     }
00089 
00090     // If there was a fault annotate it with the flag saying the foult occured
00091     // while doing a translation for a stage 1 page table walk.
00092     if (fault != NoFault) {
00093         ArmFault *armFault = reinterpret_cast<ArmFault *>(fault.get());
00094         armFault->annotate(ArmFault::S1PTW, true);
00095         armFault->annotate(ArmFault::OVA, oVAddr);
00096     }
00097     return fault;
00098 }

00120 Stage2MMU::Stage2Translation::finish(const Fault &_fault, RequestPtr req,
00121                                      ThreadContext *tc, BaseTLB::Mode mode)
00122 {
00123     fault = _fault;
00124 
00125     // If there was a fault annotate it with the flag saying the foult occured
00126     // while doing a translation for a stage 1 page table walk.
00127     if (fault != NoFault) {
00128         ArmFault *armFault = reinterpret_cast<ArmFault *>(fault.get());
00129         armFault->annotate(ArmFault::S1PTW, true);
00130         armFault->annotate(ArmFault::OVA, oVAddr);
00131     }
00132 
00133     if (_fault == NoFault && !req->getFlags().isSet(Request::NO_ACCESS)) {
00134         parent.getPort().dmaAction(MemCmd::ReadReq, req->getPaddr(), numBytes,
00135                                    event, data, tc->getCpuPtr()->clockPeriod(),
00136                                    req->getFlags());
00137     } else {
00138         // We can't do the DMA access as there's been a problem, so tell the
00139         // event we're done
00140         event->process();

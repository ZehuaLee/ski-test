00100         addLink(int_idx, ext_idx2, ext_link, LinkDirection_Out);

00124 Topology::createLinks(Network *net)
00125 {
00126     // Find maximum switchID
00127     SwitchID max_switch_id = 0;
00128     for (LinkMap::const_iterator i = m_link_map.begin();
00129          i != m_link_map.end(); ++i) {
00130         std::pair<SwitchID, SwitchID> src_dest = (*i).first;
00131         max_switch_id = max(max_switch_id, src_dest.first);
00132         max_switch_id = max(max_switch_id, src_dest.second);        
00133     }
00134 
00135     // Initialize weight, latency, and inter switched vectors
00136     Matrix topology_weights;
00137     int num_switches = max_switch_id+1;
00138     topology_weights.resize(num_switches);
00139     m_component_latencies.resize(num_switches);
00140     m_component_inter_switches.resize(num_switches);
00141 
00142     for (int i = 0; i < topology_weights.size(); i++) {
00143         topology_weights[i].resize(num_switches);
00144         m_component_latencies[i].resize(num_switches);
00145         m_component_inter_switches[i].resize(num_switches);
00146 
00147         for (int j = 0; j < topology_weights[i].size(); j++) {
00148             topology_weights[i][j] = INFINITE_LATENCY;
00149 
00150             // initialize to invalid values
00151             m_component_latencies[i][j] = -1;
00152 
00153             // initially assume direct connections / no intermediate
00154             // switches between components
00155             m_component_inter_switches[i][j] = 0;
00156         }
00157     }
00158 
00159     // Set identity weights to zero
00160     for (int i = 0; i < topology_weights.size(); i++) {
00161         topology_weights[i][i] = 0;
00162     }
00163 
00164     // Fill in the topology weights and bandwidth multipliers
00165     for (LinkMap::const_iterator i = m_link_map.begin();
00166          i != m_link_map.end(); ++i) {
00167         std::pair<int, int> src_dest = (*i).first;
00168         BasicLink* link = (*i).second.link;
00169         int src = src_dest.first;
00170         int dst = src_dest.second;
00171         m_component_latencies[src][dst] = link->m_latency;
00172         topology_weights[src][dst] = link->m_weight;
00173     }
00174         
00175     // Walk topology and hookup the links
00176     Matrix dist = shortest_path(topology_weights, m_component_latencies,
00177         m_component_inter_switches);
00178     for (int i = 0; i < topology_weights.size(); i++) {
00179         for (int j = 0; j < topology_weights[i].size(); j++) {
00180             int weight = topology_weights[i][j];
00181             if (weight > 0 && weight != INFINITE_LATENCY) {
00182                 NetDest destination_set =
00183                         shortest_path_to_node(i, j, topology_weights, dist);
00184                 makeLink(net, i, j, destination_set);
00185             }
00186         }
00187     }

00211     // Make sure we're not trying to connect two end-point nodes
00212     // directly together
00213     assert(src >= 2 * m_nodes || dest >= 2 * m_nodes);
00214 
00215     std::pair<int, int> src_dest;
00216     LinkEntry link_entry;    
00217 
00218     if (src < m_nodes) {
00219         src_dest.first = src;
00220         src_dest.second = dest;
00221         link_entry = m_link_map[src_dest];
00222         net->makeInLink(src, dest - (2 * m_nodes), link_entry.link,
00223                         link_entry.direction, routing_table_entry);
00224     } else if (dest < 2*m_nodes) {
00225         assert(dest >= m_nodes);
00226         NodeID node = dest - m_nodes;
00227         src_dest.first = src;
00228         src_dest.second = dest;

00231                          link_entry.direction, routing_table_entry);

00233         assert((src >= 2 * m_nodes) && (dest >= 2 * m_nodes));

00243 // The following all-pairs shortest path algorithm is based on the
00244 // discussion from Cormen et al., Chapter 26.1.
00245 void
00246 extend_shortest_path(Matrix& current_dist, Matrix& latencies,
00247     Matrix& inter_switches)
00248 {
00249     bool change = true;
00250     int nodes = current_dist.size();
00251 
00252     while (change) {
00253         change = false;
00254         for (int i = 0; i < nodes; i++) {
00255             for (int j = 0; j < nodes; j++) {
00256                 int minimum = current_dist[i][j];
00257                 int previous_minimum = minimum;
00258                 int intermediate_switch = -1;
00259                 for (int k = 0; k < nodes; k++) {
00260                     minimum = min(minimum,
00261                         current_dist[i][k] + current_dist[k][j]);
00262                     if (previous_minimum != minimum) {
00263                         intermediate_switch = k;
00264                         inter_switches[i][j] =
00265                             inter_switches[i][k] +
00266                             inter_switches[k][j] + 1;
00267                     }
00268                     previous_minimum = minimum;
00269                 }
00270                 if (current_dist[i][j] != minimum) {
00271                     change = true;
00272                     current_dist[i][j] = minimum;
00273                     assert(intermediate_switch >= 0);
00274                     assert(intermediate_switch < latencies[i].size());
00275                     latencies[i][j] = latencies[i][intermediate_switch] +
00276                         latencies[intermediate_switch][j];
00277                 }
00278             }
00279         }
00280     }

00299 shortest_path_to_node(SwitchID src, SwitchID next, const Matrix& weights,
00300     const Matrix& dist)
00301 {
00302     NetDest result;
00303     int d = 0;
00304     int machines;
00305     int max_machines;
00306 
00307     machines = MachineType_NUM;
00308     max_machines = MachineType_base_number(MachineType_NUM);
00309 
00310     for (int m = 0; m < machines; m++) {
00311         for (NodeID i = 0; i < MachineType_base_count((MachineType)m); i++) {
00312             // we use "d+max_machines" below since the "destination"
00313             // switches for the machines are numbered
00314             // [MachineType_base_number(MachineType_NUM)...
00315             //  2*MachineType_base_number(MachineType_NUM)-1] for the
00316             // component network
00317             if (link_is_shortest_path_to_node(src, next, d + max_machines,
00318                     weights, dist)) {
00319                 MachineID mach = {(MachineType)m, i};
00320                 result.add(mach);
00321             }
00322             d++;
00323         }
00324     }
00325 
00326     DPRINTF(RubyNetwork, "Returning shortest path\n"
00327             "(src-(2*max_machines)): %d, (next-(2*max_machines)): %d, "
00328             "src: %d, next: %d, result: %s\n",
00329             (src-(2*max_machines)), (next-(2*max_machines)),
00330             src, next, result);

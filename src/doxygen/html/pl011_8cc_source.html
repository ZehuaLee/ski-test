00073 
00074     DPRINTF(Uart, " read register %#x size=%d\n", daddr, pkt->getSize());
00075 
00076     // use a temporary data since the uart registers are read/written with
00077     // different size operations
00078     //
00079     uint32_t data = 0;
00080 
00081     switch(daddr) {
00082       case UART_DR:
00083         data = 0;
00084         if (term->dataAvailable()) {
00085             data = term->in();

00101         break;
00102       case UART_CR:
00103         data = control;
00104         break;
00105       case UART_IBRD:
00106         data = ibrd;
00107         break;
00108       case UART_FBRD:
00109         data = fbrd;
00110         break;
00111       case UART_LCRH:
00112         data = lcrh;
00113         break;
00114       case UART_IFLS:
00115         data = ifls;
00116         break;
00117       case UART_IMSC:

00061 Tick
00062 QueuedPrefetcher::notify(const PacketPtr &pkt)
00063 {
00064     // Verify this access type is observed by prefetcher
00065     if (observeAccess(pkt)) {
00066         Addr blk_addr = pkt->getAddr() & ~(Addr)(blkSize - 1);
00067         bool is_secure = pkt->isSecure();
00068 
00069         // Squash queued prefetches if demand miss to same line
00070         if (queueSquash) {
00071             auto itr = pfq.begin();
00072             while (itr != pfq.end()) {
00073                 if (itr->pkt->getAddr() == blk_addr &&
00074                     itr->pkt->isSecure() == is_secure) {
00075                     delete itr->pkt->req;
00076                     delete itr->pkt;
00077                     itr = pfq.erase(itr);
00078                 } else {
00079                     ++itr;
00080                 }
00081             }
00082 
00083             if (pfq.empty())
00084                 cache->deassertMemSideBusRequest(BaseCache::Request_PF);
00085         }
00086 
00087         // Calculate prefetches given this access
00088         std::vector<Addr> addresses;
00089         calculatePrefetch(pkt, addresses);
00090 
00091         // Queue up generated prefetches
00092         for (Addr pf_addr : addresses) {
00093 
00094             // Block align prefetch address
00095             pf_addr = pf_addr & ~(Addr)(blkSize - 1);
00096 
00097             pfIdentified++;
00098             DPRINTF(HWPrefetch, "Found a pf candidate addr: %#x, "
00099                     "inserting into prefetch queue.\n", pf_addr);
00100 
00101             if (queueFilter && inPrefetch(pf_addr, is_secure)) {
00102                 pfBufferHit++;
00103                 DPRINTF(HWPrefetch, "Prefetch addr already in "
00104                         "prefetch queue\n");
00105                 continue;
00106             }
00107 
00108             if (cacheSnoop && (inCache(pf_addr, is_secure) ||
00109                         inMissQueue(pf_addr, is_secure))) {
00110                 pfInCache++;
00111                 DPRINTF(HWPrefetch, "Dropping redundant in "
00112                         "cache/MSHR prefetch addr:%#x\n", pf_addr);
00113                 continue;
00114             }
00115 
00116             // Create a prefetch memory request
00117             Request *pf_req =
00118                 new Request(pf_addr, blkSize, 0, masterId);
00119 
00120             if (is_secure) {
00121                 pf_req->setFlags(Request::SECURE);
00122             }
00123             pf_req->taskId(ContextSwitchTaskId::Prefetcher);
00124             PacketPtr pf_pkt = new Packet(pf_req, MemCmd::HardPFReq);
00125             pf_pkt->allocate();
00126 
00127             if (pkt->req->hasContextId()) {
00128                 pf_req->setThreadContext(pkt->req->contextId(),
00129                                          pkt->req->threadId());
00130             }
00131 
00132             if (tagPrefetch && pkt->req->hasPC()) {
00133                 // Tag prefetch packet with  accessing pc
00134                 pf_pkt->req->setPC(pkt->req->getPC());
00135             }
00136 
00137             // Verify prefetch buffer space for request
00138             if (pfq.size() == queueSize) {
00139                 pfRemovedFull++;
00140                 PacketPtr old_pkt = pfq.begin()->pkt;
00141                 DPRINTF(HWPrefetch, "Prefetch queue full, removing "
00142                         "oldest packet addr: %#x", old_pkt->getAddr());
00143                 delete old_pkt->req;
00144                 delete old_pkt;
00145                 pfq.pop_front();
00146             }
00147 
00148             Tick pf_time = curTick() + clockPeriod() * latency;
00149             DPRINTF(HWPrefetch, "Prefetch queued. "
00150                     "addr:%#x tick:%lld.\n", pf_addr, pf_time);
00151 
00152             pfq.emplace_back(DeferredPacket(pf_time, pf_pkt));
00153         }
00154     }
00155 
00156     return pfq.empty() ? MaxTick : pfq.front().tick;
00157 }

00161 {
00162     DPRINTF(HWPrefetch, "Requesting a prefetch to issue.\n");
00163 
00164     if (pfq.empty()) {
00165         DPRINTF(HWPrefetch, "No hardware prefetches available.\n");

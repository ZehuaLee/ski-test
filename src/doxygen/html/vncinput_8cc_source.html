00071 void
00072 VncInput::setFrameBuffer(const FrameBuffer *rfb)

00074     if (!rfb)

00077     fb = rfb;

00079     // create bitmap of the frame with new attributes

00091     const unsigned width(fb->width());

00093 
00094     if (_videoWidth != width || _videoHeight != height) {
00095         DPRINTF(VNC, "Updating video params: width: %d height: %d\n",
00096                 width, height);
00097 
00098         _videoWidth = width;
00099         _videoHeight = height;

00101         frameBufferResized();

00103 
00104      if (captureEnabled)
00105         captureFrameBuffer();

00108 void
00109 VncInput::captureFrameBuffer()
00110 {
00111     assert(captureBitmap);
00112 
00113     // skip identical frames
00114     uint64_t new_hash = fb->getHash();
00115     if (captureLastHash == new_hash)
00116         return;
00117     captureLastHash = new_hash;
00118 
00119     // get the filename for the current frame
00120     char frameFilenameBuffer[64];
00121     snprintf(frameFilenameBuffer, 64, "fb.%06d.%lld.bmp.gz",
00122             captureCurrentFrame, static_cast<long long int>(curTick()));
00123     const string frameFilename(frameFilenameBuffer);
00124 
00125     // create the compressed framebuffer file
00126     ostream *fb_out = simout.create(captureOutputDirectory + frameFilename,
00127                                     true);
00128     captureBitmap->write(*fb_out);
00129     simout.close(fb_out);
00130 
00131     ++captureCurrentFrame;
00132 }

00703 template <class Impl>
00704 void
00705 FullO3CPU<Impl>::activateContext(ThreadID tid)
00706 {
00707     assert(!switchedOut());
00708 
00709     // Needs to set each stage to running as well.
00710     activateThread(tid);
00711 
00712     // We don't want to wake the CPU if it is drained. In that case,
00713     // we just want to flag the thread as active and schedule the tick
00714     // event from drainResume() instead.
00715     if (getDrainState() == Drainable::Drained)
00716         return;
00717 
00718     // If we are time 0 or if the last activation time is in the past,
00719     // schedule the next tick and wake up the fetch unit
00720     if (lastActivatedCycle == 0 || lastActivatedCycle < curTick()) {
00721         scheduleTickEvent(Cycles(0));
00722 
00723         // Be sure to signal that there's some activity so the CPU doesn't
00724         // deschedule itself.
00725         activityRec.activity();
00726         fetch.wakeFromQuiesce();
00727 
00728         Cycles cycles(curCycle() - lastRunningCycle);
00729         // @todo: This is an oddity that is only here to match the stats
00730         if (cycles != 0)
00731             --cycles;
00732         quiesceCycles += cycles;
00733 
00734         lastActivatedCycle = curTick();
00735 
00736         _status = Running;
00737     }
00738 }

01002 template <class Impl>
01003 unsigned int
01004 FullO3CPU<Impl>::drain(DrainManager *drain_manager)
01005 {
01006     // If the CPU isn't doing anything, then return immediately.
01007     if (switchedOut()) {
01008         setDrainState(Drainable::Drained);
01009         return 0;
01010     }
01011 
01012     DPRINTF(Drain, "Draining...\n");
01013     setDrainState(Drainable::Draining);
01014 
01015     // We only need to signal a drain to the commit stage as this
01016     // initiates squashing controls the draining. Once the commit
01017     // stage commits an instruction where it is safe to stop, it'll
01018     // squash the rest of the instructions in the pipeline and force
01019     // the fetch stage to stall. The pipeline will be drained once all
01020     // in-flight instructions have retired.
01021     commit.drain();
01022 
01023     // Wake the CPU and record activity so everything can drain out if
01024     // the CPU was not able to immediately drain.
01025     if (!isDrained())  {
01026         drainManager = drain_manager;
01027 
01028         wakeCPU();
01029         activityRec.activity();
01030 
01031         DPRINTF(Drain, "CPU not drained\n");
01032 
01033         return 1;
01034     } else {
01035         setDrainState(Drainable::Drained);
01036         DPRINTF(Drain, "CPU is already drained\n");
01037         if (tickEvent.scheduled())
01038             deschedule(tickEvent);
01039 
01040         // Flush out any old data from the time buffers.  In
01041         // particular, there might be some data in flight from the
01042         // fetch stage that isn't visible in any of the CPU buffers we
01043         // test in isDrained().
01044         for (int i = 0; i < timeBuffer.getSize(); ++i) {
01045             timeBuffer.advance();
01046             fetchQueue.advance();
01047             decodeQueue.advance();
01048             renameQueue.advance();
01049             iewQueue.advance();
01050         }
01051 
01052         drainSanityCheck();
01053         return 0;
01054     }
01055 }

01135 {

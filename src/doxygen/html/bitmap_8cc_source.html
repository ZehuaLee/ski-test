00059 const Bitmap::CompleteV1Header
00060 Bitmap::getCompleteHeader() const
00061 {
00062     const uint32_t pixel_array_size(sizeof(PixelType) * fb.area());
00063     const uint32_t file_size(sizeof(CompleteV1Header) + pixel_array_size);
00064 
00065     const CompleteV1Header header = {
00066         // File header
00067         {
00068             {'B','M'}, /* Magic */
00069             file_size,
00070             0, 0, /* Reserved */
00071             sizeof(CompleteV1Header) /* Offset to pixel array */
00072         },
00073         // Info/DIB header
00074         {
00075             sizeof(InfoHeaderV1),
00076             fb.width(),
00077             fb.height(),
00078             1, /* Color planes */
00079             32, /* Bits per pixel */
00080             0, /* No compression */
00081             pixel_array_size, /* Image size in bytes */
00082             2835, /* x pixels per meter (assume 72 DPI) */
00083             2835, /* y pixels per meter (assume 72 DPI) */
00084             0, /* Colors in color table */
00085             0 /* Important color count (0 == all are important) */
00086         }
00087     };
00088 
00089     return header;
00090 }

00092 void
00093 Bitmap::write(std::ostream &bmp) const
00094 {
00095     const CompleteV1Header header(getCompleteHeader());
00096 
00097     // 1.  write the header
00098     bmp.write(reinterpret_cast<const char *>(&header), sizeof(header));
00099 
00100     // 2.  write the bitmap data
00101     // BMP start store data left to right starting with the bottom row
00102     // so we need to do some creative flipping
00103     std::vector<PixelType> line_buffer(fb.width());
00104     for (int y = 0; y < fb.height(); ++y) {
00105         for (unsigned x = 0; x < fb.width(); ++x)
00106             line_buffer[x] = fb.pixel(x, fb.height() - y - 1);
00107 
00108         bmp.write(reinterpret_cast<const char *>(line_buffer.data()),
00109                   line_buffer.size() * sizeof(line_buffer[0]));
00110     }
00111 
00112     bmp.flush();
00113 }

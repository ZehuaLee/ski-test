00047 // Unlike gem5, kvm doesn't count the SP as a normal integer register,
00048 // which means we only have 31 normal integer registers.

00050 static_assert(NUM_XREGS == 31, "Unexpected number of aarch64 int. regs.");

00128     inform("Integer registers:\n");
00129     inform("  PC: %s\n", getAndFormatOneReg(INT_REG(regs.pc)));
00130     for (int i = 0; i < NUM_XREGS; ++i)
00131         inform("  X%i: %s\n", i, getAndFormatOneReg(kvmXReg(i)));
00132 
00133     for (int i = 0; i < NUM_QREGS; ++i)
00134         inform("  Q%i: %s\n", i, getAndFormatOneReg(kvmFPReg(i)));
00135 
00136     for (const auto &ri : intRegMap)
00137         inform("  %s: %s\n", ri.name, getAndFormatOneReg(ri.kvm));
00138 
00139     for (const auto &ri : miscRegMap)
00140         inform("  %s: %s\n", ri.name, getAndFormatOneReg(ri.kvm));
00141 
00142     for (const auto &reg : getRegList()) {
00143         const uint64_t arch(reg & KVM_REG_ARCH_MASK);
00144         if (arch != KVM_REG_ARM64) {
00145             inform("0x%x: %s\n", reg, getAndFormatOneReg(reg));
00146             continue;
00147         }
00148 
00149         const uint64_t type(reg & KVM_REG_ARM_COPROC_MASK);
00150         switch (type) {
00151           case KVM_REG_ARM_CORE:
00152             // These have already been printed
00153             break;
00154 
00155           case KVM_REG_ARM64_SYSREG: {
00156               const uint64_t op0(EXTRACT_FIELD(reg, KVM_REG_ARM64_SYSREG_OP0));
00157               const uint64_t op1(EXTRACT_FIELD(reg, KVM_REG_ARM64_SYSREG_OP1));
00158               const uint64_t crn(EXTRACT_FIELD(reg, KVM_REG_ARM64_SYSREG_CRN));
00159               const uint64_t crm(EXTRACT_FIELD(reg, KVM_REG_ARM64_SYSREG_CRM));
00160               const uint64_t op2(EXTRACT_FIELD(reg, KVM_REG_ARM64_SYSREG_OP2));
00161               const MiscRegIndex idx(
00162                   decodeAArch64SysReg(op0, op1, crn, crm, op2));
00163 
00164               inform("  %s (op0: %i, op1: %i, crn: %i, crm: %i, op2: %i): %s",
00165                      miscRegName[idx], op0, op1, crn, crm, op2,
00166                      getAndFormatOneReg(reg));
00167           } break;
00168 
00169           case KVM_REG_ARM_DEMUX: {
00170               const uint64_t id(EXTRACT_FIELD(reg, KVM_REG_ARM_DEMUX_ID));
00171               const uint64_t val(EXTRACT_FIELD(reg, KVM_REG_ARM_DEMUX_VAL));
00172               if (id == KVM_REG_ARM_DEMUX_ID_CCSIDR) {
00173                   inform("  CSSIDR[%i]: %s\n", val,
00174                          getAndFormatOneReg(reg));
00175               } else {
00176                   inform("  UNKNOWN[%i:%i]: %s\n", id, val,
00177                          getAndFormatOneReg(reg));
00178               }
00179           } break;
00180 
00181           default:
00182             inform("0x%x: %s\n", reg, getAndFormatOneReg(reg));
00183         }

00230 ArmV8KvmCPU::updateThreadContext()
00231 {
00232     DPRINTF(KvmContext, "In updateThreadContext():\n");
00233 
00234     // Update core misc regs first as they (particularly PSTATE/CPSR)
00235     // affect how other registers are mapped.
00236     for (const auto &ri : miscRegMap) {
00237         const auto value(getOneRegU64(ri.kvm));
00238         DPRINTF(KvmContext, "  %s := 0x%x\n", ri.name, value);
00239         tc->setMiscRegNoEffect(ri.idx, value);
00240     }
00241 
00242     for (int i = 0; i < NUM_XREGS; ++i) {
00243         const auto value(getOneRegU64(kvmXReg(i)));
00244         DPRINTF(KvmContext, "  X%i := 0x%x\n", i, value);
00245         tc->setIntReg(INTREG_X0 + i, value);
00246     }
00247 
00248     for (const auto &ri : intRegMap) {
00249         const auto value(getOneRegU64(ri.kvm));
00250         DPRINTF(KvmContext, "  %s := 0x%x\n", ri.name, value);
00251         tc->setIntReg(ri.idx, value);
00252     }
00253 
00254     for (int i = 0; i < NUM_QREGS; ++i) {
00255         const RegIndex reg_base(i * FP_REGS_PER_VFP_REG);
00256         KvmFPReg reg;
00257         DPRINTF(KvmContext, "  Q%i: %s\n", i, getAndFormatOneReg(kvmFPReg(i)));
00258         getOneReg(kvmFPReg(i), reg.data);
00259         for (int j = 0; j < FP_REGS_PER_VFP_REG; j++)
00260             tc->setFloatRegBits(reg_base + j, reg.s[j].i);
00261     }
00262 
00263     for (const auto &ri : getSysRegMap()) {
00264         const auto value(getOneRegU64(ri.kvm));
00265         DPRINTF(KvmContext, "  %s := 0x%x\n", ri.name, value);
00266         tc->setMiscRegNoEffect(ri.idx, value);
00267     }
00268 
00269     const CPSR cpsr(tc->readMiscRegNoEffect(MISCREG_CPSR));
00270     PCState pc(getOneRegU64(INT_REG(regs.pc)));
00271     pc.aarch64(inAArch64(tc));
00272     pc.thumb(cpsr.t);
00273     pc.nextAArch64(inAArch64(tc));
00274     // TODO: This is a massive assumption that will break when
00275     // switching to thumb.
00276     pc.nextThumb(cpsr.t);
00277     DPRINTF(KvmContext, "  PC := 0x%x (t: %i, a64: %i)\n",
00278             pc.instAddr(), pc.thumb(), pc.aarch64());
00279     tc->pcState(pc);

00282 const std::vector<ArmV8KvmCPU::MiscRegInfo> &
00283 ArmV8KvmCPU::getSysRegMap() const
00284 {
00285     // Try to use the cached map
00286     if (!sysRegMap.empty())
00287         return sysRegMap;
00288 
00289     for (const auto &reg : getRegList()) {
00290         const uint64_t arch(reg & KVM_REG_ARCH_MASK);
00291         if (arch != KVM_REG_ARM64)
00292             continue;
00293 
00294         const uint64_t type(reg & KVM_REG_ARM_COPROC_MASK);
00295         if (type != KVM_REG_ARM64_SYSREG)
00296             continue;
00297 
00298         const uint64_t op0(EXTRACT_FIELD(reg, KVM_REG_ARM64_SYSREG_OP0));
00299         const uint64_t op1(EXTRACT_FIELD(reg, KVM_REG_ARM64_SYSREG_OP1));
00300         const uint64_t crn(EXTRACT_FIELD(reg, KVM_REG_ARM64_SYSREG_CRN));
00301         const uint64_t crm(EXTRACT_FIELD(reg, KVM_REG_ARM64_SYSREG_CRM));
00302         const uint64_t op2(EXTRACT_FIELD(reg, KVM_REG_ARM64_SYSREG_OP2));
00303         const MiscRegIndex idx(decodeAArch64SysReg(op0, op1, crn, crm, op2));
00304         const auto &info(miscRegInfo[idx]);
00305         const bool writeable(
00306             info[MISCREG_USR_NS_WR] || info[MISCREG_USR_S_WR] ||
00307             info[MISCREG_PRI_S_WR] || info[MISCREG_PRI_NS_WR] ||
00308             info[MISCREG_HYP_WR] ||
00309             info[MISCREG_MON_NS0_WR] || info[MISCREG_MON_NS1_WR]);
00310         const bool implemented(
00311             info[MISCREG_IMPLEMENTED] || info[MISCREG_WARN_NOT_FAIL]);
00312 
00313         // Only add implemented registers that we are going to be able
00314         // to write.
00315         if (implemented && writeable)
00316             sysRegMap.emplace_back(reg, idx, miscRegName[idx]);
00317     }
00318 
00319     return sysRegMap;
00320 }

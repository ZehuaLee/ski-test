00069 inline bool
00070 testPredicate(uint32_t nz, uint32_t c, uint32_t v, ConditionCode code)
00071 {
00072     bool n = (nz & 0x2);
00073     bool z = (nz & 0x1);
00074 
00075     switch (code)
00076     {
00077         case COND_EQ: return  z;
00078         case COND_NE: return !z;
00079         case COND_CS: return  c;
00080         case COND_CC: return !c;
00081         case COND_MI: return  n;
00082         case COND_PL: return !n;
00083         case COND_VS: return  v;
00084         case COND_VC: return !v;
00085         case COND_HI: return  (c && !z);
00086         case COND_LS: return !(c && !z);
00087         case COND_GE: return !(n ^ v);
00088         case COND_LT: return  (n ^ v);
00089         case COND_GT: return !(n ^ v || z);
00090         case COND_LE: return  (n ^ v || z);
00091         case COND_AL: return true;
00092         case COND_UC: return true;
00093         default:
00094             panic("Unhandled predicate condition: %d\n", code);
00095     }
00096 }

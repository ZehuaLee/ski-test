00081 void
00082 CheckTable::addCheck(const Address& address)
00083 {
00084     if (floorLog2(CHECK_SIZE) != 0) {
00085         if (address.bitSelect(0, CHECK_SIZE_BITS - 1) != 0) {
00086             panic("Check not aligned");
00087         }
00088     }
00089 
00090     for (int i = 0; i < CHECK_SIZE; i++) {
00091         if (m_lookup_map.count(Address(address.getAddress()+i))) {
00092             // A mapping for this byte already existed, discard the
00093             // entire check
00094             return;
00095         }
00096     }
00097 
00098     Check* check_ptr = new Check(address, Address(100 + m_check_vector.size()),
00099                                  m_num_writers, m_num_readers, m_tester_ptr);
00100     for (int i = 0; i < CHECK_SIZE; i++) {
00101         // Insert it once per byte
00102         m_lookup_map[Address(address.getAddress() + i)] = check_ptr;
00103     }
00104     m_check_vector.push_back(check_ptr);
00105 }

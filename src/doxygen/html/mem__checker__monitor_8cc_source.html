00097 MemCheckerMonitor::recvFunctional(PacketPtr pkt)
00098 {
00099     Addr addr = pkt->getAddr();
00100     unsigned size = pkt->getSize();
00101 
00102     // Conservatively reset this address-range. Alternatively we could try to
00103     // update the values seen by the memchecker, however, there may be other
00104     // reads/writes to these location from other devices we do not see.
00105     memchecker->reset(addr, size);
00106 
00107     masterPort.sendFunctional(pkt);
00108 
00109     DPRINTF(MemCheckerMonitor,
00110             "Forwarded functional access: addr = %#llx, size = %d\n",

00144 bool
00145 MemCheckerMonitor::recvTimingReq(PacketPtr pkt)
00146 {
00147     // should always see a request
00148     assert(pkt->isRequest());
00149 
00150     // Store relevant fields of packet, because packet may be modified
00151     // or even deleted when sendTiming() is called.
00152     //
00153     // For reads we are only interested in real reads, and not prefetches, as
00154     // it is not guaranteed that the prefetch returns any useful data.
00155     bool is_read = pkt->isRead() && !pkt->req->isPrefetch();
00156     bool is_write = pkt->isWrite();
00157     unsigned size = pkt->getSize();
00158     Addr addr = pkt->getAddr();
00159     bool expects_response = pkt->needsResponse() && !pkt->memInhibitAsserted();
00160     std::unique_ptr<uint8_t> pkt_data;
00161     MemCheckerMonitorSenderState* state = NULL;
00162 
00163     if (expects_response && is_write) {
00164         // On receipt of a request, only need to allocate pkt_data if this is a
00165         // write. For reads, we have no data yet, so it doesn't make sense to
00166         // allocate.
00167         pkt_data.reset(new uint8_t[size]);
00168         memcpy(pkt_data.get(), pkt->getConstPtr<uint8_t*>(), size);
00169     }
00170 
00171     // If a cache miss is served by a cache, a monitor near the memory
00172     // would see a request which needs a response, but this response
00173     // would be inhibited and not come back from the memory. Therefore
00174     // we additionally have to check the inhibit flag.
00175     if (expects_response && (is_read || is_write)) {
00176         state = new MemCheckerMonitorSenderState(0);
00177         pkt->pushSenderState(state);
00178     }
00179 
00180     // Attempt to send the packet (always succeeds for inhibited
00181     // packets)
00182     bool successful = masterPort.sendTimingReq(pkt);
00183 
00184     // If not successful, restore the sender state
00185     if (!successful && expects_response && (is_read || is_write)) {
00186         delete pkt->popSenderState();
00187     }
00188 
00189     if (successful && expects_response) {
00190         if (is_read) {
00191             MemChecker::Serial serial = memchecker->startRead(curTick(),
00192                                                               addr,
00193                                                               size);
00194 
00195             // At the time where we push the sender-state, we do not yet know
00196             // the serial the MemChecker class will assign to this request. We
00197             // cannot call startRead at the time we push the sender-state, as
00198             // the masterPort may not be successful in executing sendTimingReq,
00199             // and in case of a failure, we must not modify the state of the
00200             // MemChecker.
00201             //
00202             // Once we know that sendTimingReq was successful, we can set the
00203             // serial of the newly constructed sender-state. This is legal, as
00204             // we know that nobody else will touch nor is responsible for
00205             // deletion of our sender-state.
00206             state->serial = serial;
00207 
00208             DPRINTF(MemCheckerMonitor,
00209                     "Forwarded read request: serial = %d, addr = %#llx, "
00210                     "size = %d\n",
00211                     serial, addr, size);
00212         } else if (is_write) {
00213             MemChecker::Serial serial = memchecker->startWrite(curTick(),
00214                                                                addr,
00215                                                                size,
00216                                                                pkt_data.get());
00217 
00218             state->serial = serial;
00219 
00220             DPRINTF(MemCheckerMonitor,
00221                     "Forwarded write request: serial = %d, addr = %#llx, "
00222                     "size = %d\n",
00223                     serial, addr, size);
00224         } else {
00225             DPRINTF(MemCheckerMonitor,
00226                     "Forwarded non read/write request: addr = %#llx\n", addr);
00227         }
00228     } else if (successful) {
00229         DPRINTF(MemCheckerMonitor,
00230                 "Forwarded inhibited request: addr = %#llx\n", addr);
00231     }
00232 
00233     return successful;
00234 }

00236 bool
00237 MemCheckerMonitor::recvTimingResp(PacketPtr pkt)
00238 {
00239     // should always see responses
00240     assert(pkt->isResponse());
00241 
00242     // Store relevant fields of packet, because packet may be modified
00243     // or even deleted when sendTiming() is called.
00244     bool is_read = pkt->isRead() && !pkt->req->isPrefetch();
00245     bool is_write = pkt->isWrite();
00246     bool is_failed_LLSC = pkt->isLLSC() && pkt->req->getExtraData() == 0;
00247     unsigned size = pkt->getSize();
00248     Addr addr = pkt->getAddr();
00249     std::unique_ptr<uint8_t> pkt_data;
00250     MemCheckerMonitorSenderState* received_state = NULL;
00251 
00252     if (is_read) {
00253         // On receipt of a response, only need to allocate pkt_data if this is
00254         // a read. For writes, we have already given the MemChecker the data on
00255         // the request, so it doesn't make sense to allocate on write.
00256         pkt_data.reset(new uint8_t[size]);
00257         memcpy(pkt_data.get(), pkt->getConstPtr<uint8_t*>(), size);
00258     }
00259 
00260     if (is_read || is_write) {
00261         received_state =
00262             dynamic_cast<MemCheckerMonitorSenderState*>(pkt->senderState);
00263 
00264         // Restore initial sender state
00265         panic_if(received_state == NULL,
00266                  "Monitor got a response without monitor sender state\n");
00267 
00268         // Restore the state
00269         pkt->senderState = received_state->predecessor;
00270     }
00271 
00272     // Attempt to send the packet
00273     bool successful = slavePort.sendTimingResp(pkt);
00274 
00275     // If packet successfully send, complete transaction in MemChecker
00276     // instance, and delete sender state, otherwise restore state.
00277     if (successful) {
00278         if (is_read) {
00279             DPRINTF(MemCheckerMonitor,
00280                     "Received read response: serial = %d, addr = %#llx, "
00281                     "size = %d\n",
00282                     received_state->serial, addr, size);
00283 
00284             bool result = memchecker->completeRead(received_state->serial,
00285                                                    curTick(),
00286                                                    addr,
00287                                                    size,
00288                                                    pkt_data.get());
00289 
00290             if (!result) {
00291                 warn("%s: read of %#llx @ cycle %d failed:\n%s\n",
00292                      name(),
00293                      addr, curTick(),
00294                      memchecker->getErrorMessage().c_str());
00295 
00296                 panic_if(!warnOnly, "MemChecker violation!");
00297             }
00298 
00299             delete received_state;
00300         } else if (is_write) {
00301             DPRINTF(MemCheckerMonitor,
00302                     "Received write response: serial = %d, addr = %#llx, "
00303                     "size = %d\n",
00304                     received_state->serial, addr, size);
00305 
00306             if (is_failed_LLSC) {
00307                 // The write was not successful, let MemChecker know.
00308                 memchecker->abortWrite(received_state->serial,
00309                                        addr,
00310                                        size);
00311             } else {
00312                 memchecker->completeWrite(received_state->serial,
00313                                           curTick(),
00314                                           addr,
00315                                           size);
00316             }
00317 
00318             delete received_state;
00319         } else {
00320             DPRINTF(MemCheckerMonitor,
00321                     "Received non read/write response: addr = %#llx\n", addr);
00322         }
00323     } else if (is_read || is_write) {
00324         // Don't delete anything and let the packet look like we
00325         // did not touch it
00326         pkt->senderState = received_state;
00327     }
00328 
00329     return successful;
00330 }

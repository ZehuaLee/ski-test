00053 void
00054 ThreadContext::compare(ThreadContext *one, ThreadContext *two)
00055 {
00056     DPRINTF(Context, "Comparing thread contexts\n");
00057 
00058     // First loop through the integer registers.
00059     for (int i = 0; i < TheISA::NumIntRegs; ++i) {
00060         TheISA::IntReg t1 = one->readIntReg(i);
00061         TheISA::IntReg t2 = two->readIntReg(i);
00062         if (t1 != t2)
00063             panic("Int reg idx %d doesn't match, one: %#x, two: %#x",
00064                   i, t1, t2);
00065     }
00066 
00067     // Then loop through the floating point registers.
00068     for (int i = 0; i < TheISA::NumFloatRegs; ++i) {
00069         TheISA::FloatRegBits t1 = one->readFloatRegBits(i);
00070         TheISA::FloatRegBits t2 = two->readFloatRegBits(i);
00071         if (t1 != t2)
00072             panic("Float reg idx %d doesn't match, one: %#x, two: %#x",
00073                   i, t1, t2);
00074     }
00075     for (int i = 0; i < TheISA::NumMiscRegs; ++i) {
00076         TheISA::MiscReg t1 = one->readMiscRegNoEffect(i);
00077         TheISA::MiscReg t2 = two->readMiscRegNoEffect(i);
00078         if (t1 != t2)
00079             panic("Misc reg idx %d doesn't match, one: %#x, two: %#x",
00080                   i, t1, t2);
00081     }
00082 
00083     // loop through the Condition Code registers.
00084     for (int i = 0; i < TheISA::NumCCRegs; ++i) {
00085         TheISA::CCReg t1 = one->readCCReg(i);
00086         TheISA::CCReg t2 = two->readCCReg(i);
00087         if (t1 != t2)
00088             panic("CC reg idx %d doesn't match, one: %#x, two: %#x",
00089                   i, t1, t2);
00090     }
00091     if (!(one->pcState() == two->pcState()))
00092         panic("PC state doesn't match.");
00093     int id1 = one->cpuId();
00094     int id2 = two->cpuId();
00095     if (id1 != id2)
00096         panic("CPU ids don't match, one: %d, two: %d", id1, id2);
00097 
00098     id1 = one->contextId();
00099     id2 = two->contextId();
00100     if (id1 != id2)
00101         panic("Context ids don't match, one: %d, two: %d", id1, id2);

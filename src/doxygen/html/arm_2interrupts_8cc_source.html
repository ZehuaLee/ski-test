00050 ArmISA::Interrupts::takeInt(ThreadContext *tc, InterruptTypes int_type) const
00051 {
00052     // Table G1-17~19 of ARM V8 ARM
00053     InterruptMask mask;
00054     bool highest_el_is_64 = ArmSystem::highestELIs64(tc);
00055 
00056     CPSR cpsr = tc->readMiscReg(MISCREG_CPSR);
00057     SCR scr;
00058     HCR hcr;
00059     hcr = tc->readMiscReg(MISCREG_HCR);
00060     ExceptionLevel el = (ExceptionLevel) ((uint32_t) cpsr.el);
00061     bool cpsr_mask_bit, scr_routing_bit, scr_fwaw_bit, hcr_mask_override_bit;
00062 
00063     if (!highest_el_is_64)
00064         scr = tc->readMiscReg(MISCREG_SCR);
00065     else
00066         scr = tc->readMiscReg(MISCREG_SCR_EL3);
00067 
00068     bool is_secure = inSecureState(scr, cpsr);
00069 
00070     switch(int_type) {
00071       case INT_FIQ:
00072         cpsr_mask_bit = cpsr.f;
00073         scr_routing_bit = scr.fiq;
00074         scr_fwaw_bit = scr.fw;
00075         hcr_mask_override_bit = hcr.fmo;
00076         break;
00077       case INT_IRQ:
00078         cpsr_mask_bit = cpsr.i;
00079         scr_routing_bit = scr.irq;
00080         scr_fwaw_bit = 1;
00081         hcr_mask_override_bit = hcr.imo;
00082         break;
00083       case INT_ABT:
00084         cpsr_mask_bit = cpsr.a;
00085         scr_routing_bit = scr.ea;
00086         scr_fwaw_bit = scr.aw;
00087         hcr_mask_override_bit = hcr.amo;
00088         break;
00089       default:
00090         panic("Unhandled interrupt type!");
00091     }
00092 
00093     if (hcr.tge)
00094         hcr_mask_override_bit = 1;
00095 
00096     if (!highest_el_is_64) {
00097         // AArch32
00098         if (!scr_routing_bit) {
00099             // SCR IRQ == 0
00100             if (!hcr_mask_override_bit)
00101                 mask = INT_MASK_M;
00102             else {
00103                 if (!is_secure && (el == EL0 || el == EL1))
00104                     mask = INT_MASK_T;
00105                 else
00106                     mask = INT_MASK_M;
00107             }
00108         } else {
00109             // SCR IRQ == 1
00110             if ((!is_secure) &&
00111                 (hcr_mask_override_bit ||
00112                     (!scr_fwaw_bit && !hcr_mask_override_bit)))
00113                 mask = INT_MASK_T;
00114             else
00115                 mask = INT_MASK_M;
00116         }
00117     } else {
00118         // AArch64
00119         if (!scr_routing_bit) {
00120             // SCR IRQ == 0
00121             if (!scr.rw) {
00122                 // SCR RW == 0
00123                 if (!hcr_mask_override_bit) {
00124                     if (el == EL3)
00125                         mask = INT_MASK_P;
00126                     else
00127                         mask = INT_MASK_M;
00128                 } else {
00129                     if (el == EL3)
00130                         mask = INT_MASK_T;
00131                     else if (is_secure || el == EL2)
00132                         mask = INT_MASK_M;
00133                     else
00134                         mask = INT_MASK_T;
00135                 }
00136             } else {
00137                 // SCR RW == 1
00138                 if (!hcr_mask_override_bit) {
00139                     if (el == EL3 || el == EL2)
00140                         mask = INT_MASK_P;
00141                     else
00142                         mask = INT_MASK_M;
00143                 } else {
00144                     if (el == EL3)
00145                         mask = INT_MASK_P;
00146                     else if (is_secure || el == EL2)
00147                         mask = INT_MASK_M;
00148                     else
00149                         mask = INT_MASK_T;
00150                 }
00151             }
00152         } else {
00153             // SCR IRQ == 1
00154             if (el == EL3)
00155                 mask = INT_MASK_M;
00156             else
00157                 mask = INT_MASK_T;
00158         }
00159     }
00160 
00161     return ((mask == INT_MASK_T) ||

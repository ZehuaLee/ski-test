00118 DVFSHandler::perfLevel(DomainID domain_id, PerfLevel perf_level)
00119 {
00120     assert(isEnabled());
00121 
00122     DPRINTF(DVFS, "DVFS: setPerfLevel domain %d -> %d\n", domain_id, perf_level);
00123 
00124     auto d = findDomain(domain_id);
00125     if (!d->validPerfLevel(perf_level)) {
00126         warn("DVFS: invalid performance level %d for domain ID %d, request "\
00127              "ignored\n", perf_level, domain_id);
00128         return false;
00129     }
00130 
00131     UpdateEvent *update_event = &updatePerfLevelEvents[domain_id];
00132     // Drop an old DVFS change request once we have established that this is a
00133     // reasonable request
00134     if (update_event->scheduled()) {
00135         DPRINTF(DVFS, "DVFS: Overwriting the previous DVFS event.\n");
00136         deschedule(update_event);
00137     }
00138 
00139     update_event->perfLevelToSet = perf_level;
00140 
00141     // State changes that restore to the current state (and / or overwrite a not
00142     // yet completed in-flight request) will be squashed
00143     if (d->perfLevel() == perf_level) {
00144         DPRINTF(DVFS, "DVFS: Ignoring ineffective performance level change "\
00145                 "%d -> %d\n", d->perfLevel(), perf_level);
00146         return false;
00147     }
00148 
00149     // At this point, a new transition will certainly take place -> schedule
00150     Tick when = curTick() + _transLatency;
00151     DPRINTF(DVFS, "DVFS: Update for perf event scheduled for %ld\n", when);
00152 
00153     schedule(update_event, when);
00154     return true;
00155 }

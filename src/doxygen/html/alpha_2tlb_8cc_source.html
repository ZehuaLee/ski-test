00184         PageTable::const_iterator i = lookupTable.find(vpn);
00185         if (i != lookupTable.end()) {
00186             while (i->first == vpn) {
00187                 int index = i->second;
00188                 TlbEntry *entry = &table[index];
00189                 assert(entry->valid);
00190                 if (vpn == entry->tag && (entry->asma || entry->asn == asn)) {
00191                     retval = updateCache(entry);
00192                     break;
00193                 }
00194 
00195                 ++i;
00196             }
00197         }
00198     }

00207 {
00208     // in Alpha, cacheability is controlled by upper-level bits of the
00209     // physical address
00210 
00211     /*
00212      * We support having the uncacheable bit in either bit 39 or bit
00213      * 40.  The Turbolaser platform (and EV5) support having the bit
00214      * in 39, but Tsunami (which Linux assumes uses an EV6) generates
00215      * accesses with the bit in 40.  So we must check for both, but we
00216      * have debug flags to catch a weird case where both are used,
00217      * which shouldn't happen.
00218      */
00219 
00220 
00221     if (req->getPaddr() & PAddrUncachedBit43) {
00222         // IPR memory space not implemented
00223         if (PAddrIprSpace(req->getPaddr())) {
00224             return std::make_shared<UnimpFault>(
00225                 "IPR memory space not implemented!");
00226         } else {
00227             // mark request as uncacheable
00228             req->setFlags(Request::UNCACHEABLE | Request::STRICT_ORDER);
00229 
00230             // Clear bits 42:35 of the physical address (10-2 in
00231             // Tsunami manual)
00232             req->setPaddr(req->getPaddr() & PAddrUncachedMask);
00233         }
00234         // We shouldn't be able to read from an uncachable address in Alpha as
00235         // we don't have a ROM and we don't want to try to fetch from a device 
00236         // register as we destroy any data that is clear-on-read. 
00237         if (req->isUncacheable() && itb) 
00238             return std::make_shared<UnimpFault>(
00239                 "CPU trying to fetch from uncached I/O");
00240 
00241     }

00373 Fault
00374 TLB::translateInst(RequestPtr req, ThreadContext *tc)
00375 {
00376     //If this is a pal pc, then set PHYSICAL
00377     if (FullSystem && PcPAL(req->getPC()))
00378         req->setFlags(Request::PHYSICAL);
00379 
00380     if (PcPAL(req->getPC())) {
00381         // strip off PAL PC marker (lsb is 1)
00382         req->setPaddr((req->getVaddr() & ~3) & PAddrImplMask);
00383         fetch_hits++;
00384         return NoFault;
00385     }
00386 
00387     if (req->getFlags() & Request::PHYSICAL) {
00388         req->setPaddr(req->getVaddr());
00389     } else {
00390         // verify that this is a good virtual address
00391         if (!validVirtualAddress(req->getVaddr())) {
00392             fetch_acv++;
00393             return std::make_shared<ItbAcvFault>(req->getVaddr());
00394         }
00395 
00396 
00397         // VA<42:41> == 2, VA<39:13> maps directly to PA<39:13> for EV5
00398         // VA<47:41> == 0x7e, VA<40:13> maps directly to PA<40:13> for EV6
00399         if (VAddrSpaceEV6(req->getVaddr()) == 0x7e) {
00400             // only valid in kernel mode
00401             if (ICM_CM(tc->readMiscRegNoEffect(IPR_ICM)) !=
00402                 mode_kernel) {
00403                 fetch_acv++;
00404                 return std::make_shared<ItbAcvFault>(req->getVaddr());
00405             }
00406 
00407             req->setPaddr(req->getVaddr() & PAddrImplMask);
00408 
00409             // sign extend the physical address properly
00410             if (req->getPaddr() & PAddrUncachedBit40)
00411                 req->setPaddr(req->getPaddr() | ULL(0xf0000000000));
00412             else
00413                 req->setPaddr(req->getPaddr() & ULL(0xffffffffff));
00414         } else {
00415             // not a physical address: need to look up pte
00416             int asn = DTB_ASN_ASN(tc->readMiscRegNoEffect(IPR_DTB_ASN));
00417             TlbEntry *entry = lookup(VAddr(req->getVaddr()).vpn(),
00418                               asn);
00419 
00420             if (!entry) {
00421                 fetch_misses++;
00422                 return std::make_shared<ItbPageFault>(req->getVaddr());
00423             }
00424 
00425             req->setPaddr((entry->ppn << PageShift) +
00426                           (VAddr(req->getVaddr()).offset()
00427                            & ~3));
00428 
00429             // check permissions for this access
00430             if (!(entry->xre &
00431                   (1 << ICM_CM(tc->readMiscRegNoEffect(IPR_ICM))))) {
00432                 // instruction access fault
00433                 fetch_acv++;
00434                 return std::make_shared<ItbAcvFault>(req->getVaddr());
00435             }
00436 
00437             fetch_hits++;
00438         }
00439     }
00440 
00441     // check that the physical address is ok (catch bad physical addresses)
00442     if (req->getPaddr() & ~PAddrImplMask) {
00443         return std::make_shared<MachineCheckFault>();
00444     }
00445 
00446     return checkCacheability(req, true);
00447 
00448 }
